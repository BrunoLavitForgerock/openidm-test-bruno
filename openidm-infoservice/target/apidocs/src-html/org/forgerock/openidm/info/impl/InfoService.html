057/**
058 * A system information service to provide an external and internal API to query
059 * OpenIDM state and status.
060 * 
061 * @author aegloff
062 */
063@Component(name = InfoService.PID, policy = ConfigurationPolicy.REQUIRE, metatype = true,
064        description = "OpenIDM Info Service", immediate = true)
065@Properties({
066    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
067    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM Info Service") })
068public class InfoService extends AbstractScriptedService {
069
070    public static final String PID = "org.forgerock.openidm.info";
071
072    /**
073     * Setup logging for the {@link InfoService}.
074     */
075    private static final Logger logger = LoggerFactory.getLogger(InfoService.class);
076
077    /** HealthInfo service. */
078    @Reference(policy = ReferencePolicy.DYNAMIC)
079    private HealthInfo healthInfoSvc;
080
081    private void bindHealthInfoSvc(final HealthInfo service) {
082        healthInfoSvc = service;
083    }
084
085    private void unbindHealthInfoSvc(final HealthInfo service) {
086        healthInfoSvc = null;
087    }
088
089    private ComponentContext context;
090
091    public InfoService() {
092        super(EnumSet.of(RequestType.READ));
093    }
094
095    @Activate
096    protected void activate(ComponentContext context) {
097        this.context = context;
098
099        Dictionary properties = context.getProperties();
100        setProperties(properties);
101        EnhancedConfig config = JSONEnhancedConfig.newInstance();
102
103        String factoryPid = config.getConfigurationFactoryPid(context);
104        if (StringUtils.isBlank(factoryPid)) {
105            throw new IllegalArgumentException("Configuration must have property: "
106                    + ServerConstants.CONFIG_FACTORY_PID);
107        }
108
109        JsonValue configuration = config.getConfigurationAsJson(context);
110        configuration.put(ServerConstants.CONFIG_FACTORY_PID, factoryPid);
111
112        activate(context.getBundleContext(), factoryPid, configuration);
113        logger.info("OpenIDM Info Service component is activated.");
114    }
115
116    @Deactivate
117    protected void deactivate(ComponentContext context) {
118        deactivate();
119        this.context = null;
120        logger.info("OpenIDM Info Service component is deactivated.");
121    }
122
123    protected Object getRouterPrefixes(String factoryPid, JsonValue configuration) {
124        return "/info/" + String.valueOf(factoryPid) + "*";
125    }
126
127    protected BundleContext getBundleContext() {
128        return context.getBundleContext();
129    }
130
131    @Override
132    protected void handleRequest(final ServerContext context, final Request request,
133            final Bindings handler) {
134        super.handleRequest(context, request, handler);
135        handler.put("healthinfo", healthInfoSvc.getHealthInfo().asMap());
136
137    }
138}
