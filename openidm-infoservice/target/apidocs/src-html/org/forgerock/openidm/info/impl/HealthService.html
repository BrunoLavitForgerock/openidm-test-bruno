235    @Activate
236    protected void activate(final ComponentContext context) {
237        this.context = context;
238        requiredBundles = new ArrayList<String>();
239        requiredBundles.addAll(Arrays.asList(defaultRequiredBundles));
240        requiredServices = new ArrayList<String>();
241        requiredServices.addAll(Arrays.asList(defaultRequiredServices));
242        applyPropertyConfig();
243
244        // Set up tracker
245        BundleContext ctx = FrameworkUtil.getBundle(HealthService.class).getBundleContext();
246        tracker = initServiceTracker(ctx);
247
248        // Handle framework changes
249        frameworkListener = new FrameworkListener() {
250            @Override
251            public void frameworkEvent(FrameworkEvent event) {
252                logger.debug("Handle framework event {} {}", event.getType(), event.toString());
253
254                if (event.getType() == FrameworkEvent.STARTED) {
255                    logger.debug("OSGi framework started event.");
256                    frameworkStarted = true;
257                }
258                // Start checking status once framework reported started
259                if (frameworkStarted) {
260                    switch (event.getType()) {
261                    case FrameworkEvent.PACKAGES_REFRESHED: // fall through
262                    case FrameworkEvent.STARTLEVEL_CHANGED: // fall through
263                    case FrameworkEvent.WARNING: // fall trough
264                    case FrameworkEvent.INFO:
265                        // For now do not re-check state for these
266                        break;
267                    default:
268                        checkState();
269                    }
270                }
271                if (event.getType() == FrameworkEvent.STARTED) {
272                    // IF it's not yet ready, give it up to max service startup
273                    // time before reporting failure
274                    if (!stateDetail.state.equals(AppState.ACTIVE_READY)) {
275                        scheduleCheckStartup();
276                    }
277                }
278            }
279        };
280
281        // Handle service changes
282        svcListener = new ServiceListener() {
283            @Override
284            public void serviceChanged(ServiceEvent event) {
285                logger.debug("Handle service event {} {}", event.getType(), event.toString());
286                if (frameworkStarted) {
287                    switch (event.getType()) {
288                    case ServiceEvent.REGISTERED: // fall through
289                    case ServiceEvent.UNREGISTERING: // fall through
290                    case ServiceEvent.MODIFIED:
291                        checkState();
292                        break;
293                    }
294                }
295            }
296        };
297
298        // Handle bundle changes
299        bundleListener = new BundleListener() {
300            @Override
301            public void bundleChanged(BundleEvent event) {
302                logger.debug("Handle bundle event {} {}", event.getType(), event.toString());
303                if (frameworkStarted) {
304                    switch (event.getType()) {
305                    case BundleEvent.STARTED: // fall through
306                    case BundleEvent.STOPPED: // fall through
307                    case BundleEvent.UNRESOLVED:
308                        checkState();
309                        break;
310                    case BundleEvent.RESOLVED:
311                        if (isFragment(event.getBundle())) {
312                            checkState();
313                        }
314                        break;
315                    }
316                }
317            }
318        };
319
320        context.getBundleContext().addServiceListener(svcListener);
321        context.getBundleContext().addBundleListener(bundleListener);
322        context.getBundleContext().addFrameworkListener(frameworkListener);
323
324        logger.info("OpenIDM Health Service component is activated.");
325    }

327    /**
328     * Apply configuration overrides from properties if present
329     */
330    private void applyPropertyConfig() {
331        // Override default requirements
332        String reqBundlesProp =
333                IdentityServer.getInstance().getProperty("openidm.healthservice.reqbundles");
334        if (reqBundlesProp != null) {
335            requiredBundles = parseProp(reqBundlesProp);
336        }
337        String reqServicesProp =
338                IdentityServer.getInstance().getProperty("openidm.healthservice.reqservices");
339        if (reqServicesProp != null) {
340            requiredServices = parseProp(reqServicesProp);
341        }
342        // Optionally add to requirements
343        String additionalReqBundlesProp =
344                IdentityServer.getInstance().getProperty(
345                        "openidm.healthservice.additionalreqbundles");
346        if (additionalReqBundlesProp != null) {
347            requiredBundles.addAll(parseProp(additionalReqBundlesProp));
348        }
349        String additionalReqServicesProp =
350                IdentityServer.getInstance().getProperty(
351                        "openidm.healthservice.additionalreqservices");
352        if (additionalReqServicesProp != null) {
353            requiredServices.addAll(parseProp(additionalReqServicesProp));
354        }
355
356        String serviceStartMaxProp =
357                IdentityServer.getInstance().getProperty("openidm.healthservice.servicestartmax");
358        if (serviceStartMaxProp != null) {
359            serviceStartMax = Long.parseLong(serviceStartMaxProp);
360        }
361    }

363    /**
364     * After the timeout period passes past framework start event, check that
365     * the required services are present. If not, report startup error
366     */
367    private void scheduleCheckStartup() {
368        Runnable command = new Runnable() {
369            @Override
370            public void run() {
371                appStarting = false; // From now on, report not ready rather
372                                     // than starting if something fails
373                checkState();
374                if (!stateDetail.state.equals(AppState.ACTIVE_READY)) {
375                    logger.error("OpenIDM failure during startup, {}: {}", stateDetail.state,
376                            stateDetail.shortDesc);
377                } else {
378                    logger.debug("Startup check found ready state");
379                }
380            }
381        };
382        if (scheduledExecutor.isShutdown()) {
383            scheduledExecutor = Executors.newSingleThreadScheduledExecutor();
384        }
385        scheduledExecutor.schedule(command, serviceStartMax, TimeUnit.MILLISECONDS);
386    }

444    /**
445     * Check and update the application state
446     */
447    private void checkState() {
448        // Check if the required bundles are started or bundle fragments
449        // resolved
450        Bundle[] bundles = context.getBundleContext().getBundles();
451        List<String> missingBundles = new ArrayList<String>(requiredBundles);
452        List<String> bundleFailures = new ArrayList<String>();
453        List<String> fragmentFailures = new ArrayList<String>();
454        for (String req : requiredBundles) {
455            for (Bundle bundle : bundles) {
456                String symbolicName = bundle.getSymbolicName();
457                if (symbolicName != null && symbolicName.matches(req)) {
458                    if (isFragment(bundle)) {
459                        if (bundle.getState() != Bundle.RESOLVED) {
460                            fragmentFailures.add(bundle.getSymbolicName());
461                        }
462                    } else {
463                        if (bundle.getState() != Bundle.ACTIVE) {
464                            bundleFailures.add(bundle.getSymbolicName());
465                        }
466                    }
467                    missingBundles.remove(req);
468                }
469            }
470        }
471
472        // Get currently registered services
473        ServiceReference[] refs = null;
474        try {
475            refs = context.getBundleContext().getAllServiceReferences(null, null);
476        } catch (InvalidSyntaxException e) {
477            // Since we are not passing a filter this should not happen
478            logger.debug("Unexpected failure in getting service references", e);
479        }
480
481        // Scan the registered services for matches to our list of
482        // required services.  Required services can be expressed as a regex,
483        // for example: "org.forgerock.openidm.bootrepo.(orientdb|jdbc)"
484        List<String> missingServices = new ArrayList<String>(requiredServices);
485        for (String req : requiredServices) {
486            for (ServiceReference ref : refs){
487                String pid = (String)ref.getProperty(Constants.SERVICE_PID);
488                if (pid !=null && pid.matches(req)) {
489                    missingServices.remove(req);
490                    break;
491                }
492            }
493        }
494
495        // Ensure state is up to date
496        AppState updatedAppState = null;
497        String updatedShortDesc = null;
498        if (missingBundles.size() > 0 || bundleFailures.size() > 0 || fragmentFailures.size() > 0) {
499            updatedAppState = AppState.ACTIVE_NOT_READY;
500            updatedShortDesc = "Not all modules started " + missingBundles + " " + bundleFailures + " " + fragmentFailures;
501        } else if (missingServices.size() > 0) {
502            updatedAppState = AppState.ACTIVE_NOT_READY;
503            updatedShortDesc = "Required services not all started " + missingServices;
504        } else if (clusterEnabled && !clusterUp) {
505            if (cluster != null && !cluster.isStarted()) {
506                cluster.startClusterManagement();
507            }
508            updatedAppState = AppState.ACTIVE_NOT_READY;
509            updatedShortDesc = "This node can not yet join the cluster";
510        } else {
511            updatedAppState = AppState.ACTIVE_READY;
512            updatedShortDesc = "OpenIDM ready";
513        }
514        setState(updatedAppState, updatedShortDesc);
515    }

517    /**
518     * Process detected state, if it's different than the current state process
519     * the state change and report appropriately
520     *
521     * @param state
522     *            new app state
523     * @param shortDesc
524     *            new short description of state
525     */
526    private void setState(AppState state, String shortDesc) {
527        synchronized (this) {
528            if (!stateDetail.isSameState(state, shortDesc)) {
529                // Whilst we're still not past start-up timeout or successful
530                // start, keep it at starting rather than error
531                if (appStarting) {
532                    if (state == AppState.ACTIVE_READY) {
533                        appStarting = false; // From now on, report not ready
534                                             // rather than starting if
535                                             // something fails
536                    } else {
537                        return;
538                    }
539                }
540
541                StateDetail updatedState = new StateDetail(state, shortDesc);
542
543                // If we're changing from ready to another state, report
544                if (stateDetail.isState(AppState.ACTIVE_READY)
545                        && !updatedState.isState(AppState.ACTIVE_READY)) {
546                    if (updatedState.state == AppState.ACTIVE_NOT_READY) {
547                        // Whilst we do not have a mechanism to detect regular
548                        // system shut down and distinguish, we log as info
549                        logger.info("System changed to a not ready state {}: {}", updatedState
550                                .getState(), updatedState.getShortDesc());
551                    } else {
552                        logger.info("System changed state {}: {}", updatedState.getState(),
553                                updatedState.getShortDesc());
554                    }
555                }
556
557                // IF we're changing to a ready state, report ready
558                if (!stateDetail.isState(AppState.ACTIVE_READY)
559                        && updatedState.isState(AppState.ACTIVE_READY)) {
560                    logger.info("OpenIDM ready");
561                    // Show ready on the system console
562                    System.out.println("OpenIDM ready");
563                }
564
565                stateDetail = updatedState;
566            }
567        }
568    }

580    /**
581     * Translate Bundle state int
582     *
583     * @param bundleState
584     *            bundle state int
585     * @return String version of the state
586     */
587    private String stateToString(int bundleState) {
588        switch (bundleState) {
589        case Bundle.ACTIVE:
590            return "ACTIVE";
591        case Bundle.INSTALLED:
592            return "INSTALLED";
593        case Bundle.RESOLVED:
594            return "RESOLVED";
595        case Bundle.STARTING:
596            return "STARTING";
597        case Bundle.STOPPING:
598            return "STOPPING";
599        case Bundle.UNINSTALLED:
600            return "UNINSTALLED ";
601        }
602        return "UNKNDWN";
603    }

605    /**
606     * Parse the comma delimited property into a list
607     *
608     * @param prop
609     *            comma delimited values
610     * @return properties split by comma
611     */
612    private List<String> parseProp(String prop) {
613        String[] items = new String[0];
614        if (prop != null) {
615            items = prop.split(",\\s*");
616        }
617        return Arrays.asList(items);
618    }

620    @Deactivate
621    protected void deactivate(ComponentContext context) {
622        if (scheduledExecutor != null) {
623            scheduledExecutor.shutdown();
624        }
625        if (frameworkListener != null) {
626            context.getBundleContext().removeFrameworkListener(frameworkListener);
627        }
628        if (svcListener != null) {
629            context.getBundleContext().removeServiceListener(svcListener);
630        }
631        if (bundleListener != null) {
632            context.getBundleContext().removeBundleListener(bundleListener);
633        }
634
635        // For now we have to rely on this bundle stopping as an indicator
636        // that the system may be shutting down
637        // Ideally replace with enhanced detection on regular shutdown initiated
638        frameworkStarted = false;
639        setState(AppState.STOPPING, "OpenIDM stopping");
640        logger.info("OpenIDM Health Service component is deactivated.");
641    }

643    /**
644     * Detailed State
645     *
646     * @author aegloff
647     */
648    private static class StateDetail {
649        JsonValue jsonState;
650
651        public StateDetail(AppState state, String shortDesc) {
652            this.state = state;
653            this.shortDesc = shortDesc;
654            jsonState = new JsonValue(new HashMap<String, Object>());
655            jsonState.put("state", state.name());
656            jsonState.put("shortDesc", shortDesc);
657        }
658
659        private AppState state = AppState.STARTING;
660        private String shortDesc;
661
662        protected AppState getState() {
663            return state;
664        }
665
666        protected String getShortDesc() {
667            return shortDesc;
668        }
669
670        protected boolean isState(AppState compareState) {
671            return state == compareState;
672        }
673
674        protected boolean isSameState(AppState compareState, String compareShortDesc) {
675            return state == compareState
676                    && ((shortDesc == null && compareShortDesc == null) || shortDesc
677                            .equals(compareShortDesc));
678        }
679
680        protected JsonValue toJsonValue() {
681            return jsonState;
682        }
683    }
