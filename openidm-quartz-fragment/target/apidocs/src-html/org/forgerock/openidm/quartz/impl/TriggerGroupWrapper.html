026package org.forgerock.openidm.quartz.impl;
027
028import java.util.ArrayList;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032
033import org.forgerock.json.fluent.JsonValue;
034
035/**
036 * A wrapper the contains all necessary information for a Trigger's group
037 */
038public class TriggerGroupWrapper {
039    
040    private String name;
041    private String revision;
042    private List<String> triggers;
043    private boolean paused;
044    
045    /**
046     * Creates a new TriggerGroupWrapper from a group name
047     * 
048     * @param triggerName the group name
049     */
050    public TriggerGroupWrapper(String triggerName) {
051        triggers = new ArrayList<String>();
052        name = triggerName;
053        paused = false;
054        
055    }
056    
057    /**
058     * Creates a new TriggerGroupWrapper from a JsonValue object
059     * 
060     * @param value the JsonValue object
061     */
062    public TriggerGroupWrapper(JsonValue value) {
063        this(value.asMap());
064    }
065    
066    /**
067     * Creates a new TriggerGroupWrapper from a object map.
068     * 
069     * @param map   the object map
070     */
071    public TriggerGroupWrapper(Map<String, Object> map) {
072        name = (String)map.get("name");
073        if (map.get("paused") != null) {
074            paused = (Boolean)map.get("paused");
075        }
076        if (map.get("triggers") != null) {
077            triggers = (List<String>)map.get("triggers");
078        } else {
079            triggers = new ArrayList<String>();
080        }
081        revision = (String)map.get("_rev");
082    }
083    
084    /**
085     * Returns the name of the group
086     * 
087     * @return the name of the group
088     */
089    public String getName() {
090        return name;
091    }
092    
093    /**
094     * Returns true if the group is in the "paused" state, false otherwise.
095     * 
096     * @return  true if the group is in the "paused" state, false otherwise
097     */
098    public boolean isPaused() {
099        return paused;
100    }
101    
102    /**
103     * Sets the TriggerGroupWrapper in the "paused" state.
104     */
105    public void pause() {
106        setPaused(true);
107    }
108    
109    /**
110     * Resumes the TriggerGroupWrapper form the "paused" state.
111     */
112    public void resume() {
113        setPaused(false);
114    }
115    
116    /**
117     * Sets the "paused" state of the TriggerGroupWrapper
118     * 
119     * @param paused true if "paused", false otherwise
120     */
121    public void setPaused(boolean paused) {
122        this.paused = paused;
123    }
124    
125    /**
126     * Adds a Trigger's ID to the list of Triggers in this group.
127     * 
128     * @param triggerId a Trigger's ID
129     */
130    public void addTrigger(String triggerId) {
131        if (!triggers.contains(triggerId)) {
132            triggers.add(triggerId);
133        }
134    }
135    
136    /**
137     * Removes a Trigger's ID from the list of Triggers in this group.
138     * 
139     * @param triggerId a Trigger's ID
140     */
141    public void removeTrigger(String triggerId) {
142        if (triggers.contains(triggerId)) {
143            triggers.remove(triggerId);
144        }
145    }
146    
147    /**
148     * Returns a list of all Trigger's names (IDs) in this group.
149     * 
150     * @return a list of Trigger's names
151     */
152    public List<String> getTriggerNames() {
153        return triggers;
154    }
155    
156    /**
157     * Returns a JsonValue object wrapper around the object map for the TriggerGroupWrapper.
158     * 
159     * @return a JsonValue object
160     */
161    public JsonValue getValue() {
162        Map<String, Object> valueMap = new HashMap<String, Object>();
163        valueMap.put("triggers", triggers);
164        valueMap.put("name", name);
165        valueMap.put("paused", paused);
166        return new JsonValue(valueMap);
167    }
168
169    /**
170     * Returns the revision, as set by the repo
171     * 
172     * @return the repo revision
173     */
174    public String getRevision() {
175        return revision;
176    }
177}
