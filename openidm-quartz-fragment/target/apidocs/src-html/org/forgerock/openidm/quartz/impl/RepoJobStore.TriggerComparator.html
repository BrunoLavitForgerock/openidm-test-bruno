172    /**
173     * Sets the Repository Service Router and returns true if successful, false otherwise.
174     *
175     * @return  true if successful, false otherwise
176     */
177    public ServerContext getServerContext() throws JobPersistenceException {
178        if (serverContext == null) {
179            BundleContext ctx = FrameworkUtil.getBundle(RouteService.class).getBundleContext();
180            ServiceReference serviceReference = null;
181            try {
182                serviceReference = ctx.getServiceReferences(RouteService.class, "(openidm.router.prefix=/repo*)").iterator().next();
183            } catch (InvalidSyntaxException e) {
184                /* ignore the filter is correct */
185            }
186            RouteService repoService = RouteService.class.cast(ctx.getService(serviceReference));
187            if (repoService != null) {
188                try {
189                    serverContext = repoService.createServerContext();
190                } catch (ResourceException e) {
191                    /* ignore */
192                }
193            }
194            if (serverContext == null) {
195                throw new JobPersistenceException("Repo router is null");
196            }
197        }
198        return serverContext;
199    }

206        if (clusterManager == null) {
207            BundleContext ctx = FrameworkUtil.getBundle(RepoJobStore.class).getBundleContext();

209            clusterManager = ClusterManagementService.class.cast(ctx.getService(serviceReference));

467    @Override
468    public void storeCalendar(SchedulingContext context, String name,
469            Calendar calendar, boolean replaceExisting, boolean updateTriggers)
470            throws ObjectAlreadyExistsException, JobPersistenceException {
471        synchronized (lock) {
472            try {
473                CalendarWrapper cw = new CalendarWrapper(calendar, name);
474                if (retrieveCalendar(context, name) == null) {
475                    // Create Calendar
476                    logger.debug("Creating Calendar: {}", name);
477                    getConnectionFactory().getConnection().create(getServerContext(), getCreateRequest(getCalendarsRepoId(name), cw.getValue().asMap()));
478                } else {
479                    if (!replaceExisting) {
480                        throw new ObjectAlreadyExistsException(name);
481                    }
482                    // Update Calendar
483                    CalendarWrapper oldCw = getCalendarWrapper(name);
484                    logger.debug("Updating Calendar: {}", name);
485                    UpdateRequest r = Requests.newUpdateRequest(getCalendarsRepoId(name), cw.getValue());
486                    r.setRevision(oldCw.getRevision());
487                    getConnectionFactory().getConnection().update(getServerContext(), r);
488                }
489
490                if (updateTriggers) {
491                    List<TriggerWrapper> twList = getTriggerWrappersForCalendar(name);
492                    for (TriggerWrapper tw : twList) {
493                        Trigger t = tw.getTrigger();
494                        boolean removed = removeWaitingTrigger(t);
495                        t.updateWithNewCalendar(calendar, getMisfireThreshold());
496                        tw.updateTrigger(t);
497                        logger.debug("Updating Trigger {} in group {}", new Object[]{tw.getName(),tw.getGroup()});
498                        updateTriggerInRepo(tw.getGroup(), tw.getName(), tw, tw.getRevision());
499                        if (removed) {
500                            addWaitingTrigger(t);
501                        }
502                    }
503                }
504            } catch (Exception e) {
505                logger.warn("Error storing calendar: {}", name, e);
506                throw new JobPersistenceException("Error storing calendar", e);
507            }
508        }
509    }

511    @Override
512    public void storeJob(SchedulingContext context, JobDetail newJob,
513            boolean replaceExisting)
514            throws ObjectAlreadyExistsException, JobPersistenceException {
515        synchronized (lock) {
516            logger.debug("Attempting to store job {} ", newJob.getFullName());
517            String jobName = newJob.getName();
518            String jobGroup = newJob.getGroup();
519            String jobId = getJobsRepoId(jobGroup, jobName);
520            JobGroupWrapper jgw = null;
521            try {
522                // Get job group
523                jgw = getOrCreateJobGroupWrapper(jobGroup);
524            } catch (ResourceException e) {
525                logger.warn("Error storing job", e);
526                throw new JobPersistenceException("Error" + " storing job", e);
527            }
528            List<String> jobNames = jgw.getJobNames();
529            JobWrapper jw = new JobWrapper(newJob, jgw.isPaused());
530            if (jobNames.contains(jobName) && !replaceExisting) {
531                throw new ObjectAlreadyExistsException(newJob);
532            }
533            try {
534                // Check if job name exists
535                if (jobNames.contains(jobName)) {
536                    // Update job
537                    JobWrapper oldJw = getJobWrapper(jobGroup, jobName);
538                    logger.debug("Updating Job {} in group {}", new Object[] {jobName,jobGroup});
539                    UpdateRequest r = Requests.newUpdateRequest(jobId, jw.getValue());
540                    r.setRevision(oldJw.getRevision());
541                    getConnectionFactory().getConnection().update(getServerContext(), r);
542                } else {
543                    // Add job name to list
544                    jgw.addJob(jobName);
545                    // Update job group
546                    UpdateRequest r = Requests.newUpdateRequest(getJobGroupsRepoId(jobGroup), jgw.getValue());
547                    r.setRevision(jgw.getRevision());
548                    getConnectionFactory().getConnection().update(getServerContext(), r);
549
550                    // Create job
551                    logger.debug("Creating Job {} in group {}", new Object[] {jobName,jobGroup});
552                    getConnectionFactory().getConnection().create(getServerContext(), getCreateRequest(jobId, jw.getValue().asMap()));
553                }
554            } catch (ResourceException e) {
555                logger.warn("Error storing job", e);
556                throw new JobPersistenceException("Error" + " storing job", e);
557            }
558        }
559    }

570    @Override
571    public void storeTrigger(SchedulingContext context, Trigger trigger,
572            boolean replaceExisting) throws ObjectAlreadyExistsException, JobPersistenceException {
573        synchronized (lock) {
574            String triggerName = trigger.getKey().getName();
575            String groupName = trigger.getKey().getGroup();
576            String triggerId = getTriggersRepoId(groupName, triggerName);
577            TriggerGroupWrapper tgw = null;
578            try {
579                // Get trigger group
580                tgw = getOrCreateTriggerGroupWrapper(groupName);
581            } catch (ResourceException e) {
582                logger.warn("Error storing trigger", e);
583                throw new JobPersistenceException("Error" + " storing trigger", e);
584            }
585            List<String> triggerNames = tgw.getTriggerNames();
586            TriggerWrapper tw;
587            try {
588                tw = new TriggerWrapper(trigger, tgw.isPaused());
589            } catch (Exception e) {
590                logger.warn("Error storing trigger", e);
591                throw new JobPersistenceException("Error" + " storing trigger", e);
592            }
593            if (triggerNames.contains(triggerName) && !replaceExisting) {
594                throw new ObjectAlreadyExistsException(trigger);
595            }
596            try {
597                // Check if trigger name exists
598                if (triggerNames.contains(triggerName)) {
599                    TriggerWrapper oldTw = getTriggerWrapper(groupName, triggerName);
600                    // Update trigger
601                    logger.debug("Updating Trigger {}", triggerId);
602                    UpdateRequest r = Requests.newUpdateRequest(triggerId, tw.getValue());
603                    r.setRevision(oldTw.getRevision());
604                    getConnectionFactory().getConnection().update(getServerContext(), r);
605                } else {
606                    // Add trigger name to list
607                    tgw.addTrigger(triggerName);
608                    // Update trigger group
609                    UpdateRequest r = Requests.newUpdateRequest(getTriggerGroupsRepoId(groupName), tgw.getValue());
610                    r.setRevision(tgw.getRevision());
611                    getConnectionFactory().getConnection().update(getServerContext(), r);
612
613                    // Create trigger
614                    logger.debug("Creating Trigger {}", triggerId);
615                    getConnectionFactory().getConnection().create(getServerContext(), getCreateRequest(triggerId, tw.getValue().asMap()));
616                }
617            } catch (ResourceException e) {
618                logger.warn("Error storing trigger", e);
619                throw new JobPersistenceException("Error" + " storing trigger", e);
620            }
621            logger.debug("Adding waiting trigger {}", trigger.getName());
622            addWaitingTrigger(trigger);
623        }
624    }

626    @Override
627    public Trigger acquireNextTrigger(SchedulingContext context, long noLaterThan)
628            throws JobPersistenceException {
629        synchronized (lock) {
630            logger.debug("acquiring next trigger");
631            Trigger trigger = null;
632            WaitingTriggers waitingTriggers = null;
633            while (trigger == null) {
634                try {
635                    waitingTriggers = getWaitingTriggers();
636                    trigger = waitingTriggers.getTriggers().first();
637                } catch (NoSuchElementException e1) {
638                    logger.debug("No waiting triggers to acquire");
639                    return null;
640                }
641
642                if (trigger == null) {
643                    logger.debug("No waiting triggers to acquire");
644                    return null;
645                }
646
647
648                Date nextFireTime = trigger.getNextFireTime();
649                if (nextFireTime == null) {
650                    logger.debug("Trigger next fire time = null, removing");
651                    removeWaitingTrigger(trigger);
652                    trigger = null;
653                    continue;
654                }
655
656                if(!removeWaitingTrigger(trigger)) {
657                    trigger = null;
658                    continue;
659                }
660
661                TriggerWrapper tw = getTriggerWrapper(trigger.getGroup(), trigger.getName());
662
663                if (hasTriggerMisfired(trigger)) {
664                    logger.debug("Attempting to process misfired trigger");
665                    processTriggerMisfired(tw);
666                    if (trigger.getNextFireTime() != null) {
667                        addWaitingTrigger(trigger);
668                    }
669                    trigger = null;
670                    continue;
671                }
672
673                if (noLaterThan > 0) {
674                    if (nextFireTime.getTime() > noLaterThan) {
675                        addWaitingTrigger(trigger);
676                        return null;
677                    }
678                }
679
680                tw.setAcquired(true);
681                trigger.setFireInstanceId(getFiredTriggerRecordId());
682                try {
683                    tw.updateTrigger(trigger);
684                } catch (Exception e) {
685                    logger.warn("Error serializing trigger", e);
686                    throw new JobPersistenceException("Error serializing trigger", e);
687                }
688
689                updateTriggerInRepo(trigger.getGroup(), trigger.getName(), tw, tw.getRevision());
690
691                addAcquiredTrigger(trigger, instanceId);
692
693                logger.debug("Acquired next trigger {} to be fired at {}", new Object[]{trigger.getName(), trigger.getNextFireTime()});
694                return (Trigger)trigger.clone();
695            }
696            logger.debug("No waiting triggers to acquire");
697            return null;
698        }
699    }

703    @Override
704    public void releaseAcquiredTrigger(SchedulingContext arg0, Trigger trigger)
705            throws JobPersistenceException {
706        synchronized (lock) {
707            TriggerWrapper tw = getTriggerWrapper(trigger.getGroup(), trigger.getName());
708            if (tw == null) {
709                logger.debug("Cannot release acquired trigger {} in group {}, trigger does not exist",
710                        new Object[] {trigger.getName(),trigger.getGroup()});
711                return;
712            }
713            if (tw.isAcquired()) {
714                tw.setAcquired(false);
715                updateTriggerInRepo(trigger.getGroup(), trigger.getName(), tw, tw.getRevision());
716                addWaitingTrigger(trigger);
717                removeAcquiredTrigger(trigger, instanceId);
718            } else {
719                logger.warn("Cannot release acquired trigger {} in group {}, trigger has not been acquired",
720                        new Object[] {trigger.getName(),trigger.getGroup()});
721            }
722        }
723    }

1439            throws JobPersistenceException {

1441            TriggerWrapper tw;
1442            try {
1443                tw = getTriggerWrapper(trigger.getGroup(), trigger.getName());
1444            } catch (Exception e) {
1445                logger.warn("Error setting trigger fired", e);
1446                throw new JobPersistenceException("Error setting trigger fired", e);
1447            }
1448            if (tw == null) {
1449                logger.warn("Error setting trigger fired, trigger does not exist");
1450                return null;
1451            }
1452            if (!tw.isAcquired()) {
1453                logger.warn("Error setting trigger fired, trigger was not in acquired state");
1454            }
1455            Trigger localTrigger;
1456            try {
1457                localTrigger = tw.getTrigger();
1458            } catch (Exception e) {
1459                logger.warn("Error setting trigger fired", e);
1460                throw new JobPersistenceException("Error setting trigger fired", e);
1461            }
1462            Calendar triggerCalendar = null;
1463            if (localTrigger.getCalendarName() != null) {
1464                CalendarWrapper cw = getCalendarWrapper(localTrigger.getCalendarName());
1465                if (cw == null) {
1466                    logger.warn("Error setting trigger fired, cannot find trigger's calendar");
1467                    return null;
1468                } else {
1469                    try {
1470                        triggerCalendar = cw.getCalendar();
1471                    } catch (Exception e) {
1472                        logger.warn("Error retrieving calendar", e);
1473                        throw new JobPersistenceException("Error retrieving calendar", e);
1474                    }
1475                }
1476            }
1477
1478            Date previousFireTime = trigger.getPreviousFireTime();
1479            removeWaitingTrigger(trigger);
1480
1481            localTrigger.triggered(triggerCalendar);
1482            tw.updateTrigger(localTrigger);
1483            updateTriggerInRepo(localTrigger.getGroup(), localTrigger.getName(), tw, tw.getRevision());
1484
1485            trigger.triggered(triggerCalendar);
1486
1487            // Set trigger into the normal/waiting state
1488            tw.setState(Trigger.STATE_NORMAL);
1489            TriggerFiredBundle tfb = new TriggerFiredBundle(retrieveJob(context, trigger.getJobName(),
1490                    trigger.getJobGroup()),
1491                    trigger,
1492                    triggerCalendar,
1493                    false,
1494                    new Date(),
1495                    trigger.getPreviousFireTime(),
1496                    previousFireTime,
1497                    trigger.getNextFireTime());
1498
1499            JobDetail job = tfb.getJobDetail();
1500
1501            if (job.isStateful()) {
1502                Trigger[] triggers = getTriggersForJob(context, job.getName(), job.getGroup());
1503                for (Trigger t : triggers) {
1504                    TriggerWrapper tmpTw = getTriggerWrapper(t.getGroup(), t.getName());
1505                    if (tmpTw != null) {
1506                        if (tmpTw.getState() == Trigger.STATE_NORMAL || tmpTw.getState() == Trigger.STATE_PAUSED) {
1507                            tmpTw.block();
1508                        }
1509                        // update trigger in repo
1510                        updateTriggerInRepo(t.getGroup(), tmpTw.getName(), tmpTw, tmpTw.getRevision());
1511                        removeWaitingTrigger(t);
1512                    }
1513                }
1514                blockedJobs.add(getJobNameKey(job));
1515            } else if (localTrigger.getNextFireTime() != null) {
1516                addWaitingTrigger(localTrigger);
1517            }
1518            return tfb;
1519        }

1524            JobDetail jobDetail, int triggerInstCode) throws JobPersistenceException {

1531                tw = new TriggerWrapper(triggerValue.asMap());
1532            }
1533
1534            // Remove the acquired trigger (if acquired)

1536
1537            if (jw != null) {

1539                try {
1540                    jd = jw.getJobDetail();
1541                } catch (Exception e) {

1543                }

1556                            if (tmpTw.getState() == Trigger.STATE_BLOCKED) {
1557                                tmpTw.unblock();

1562                                addWaitingTrigger(t);
1563                            }
1564                        }
1565                    }
1566                    schedulerSignaler.signalSchedulingChange(0L);
1567                }
1568            } else {
1569                blockedJobs.remove(jobKey);
1570            }
1571
1572            if (tw != null) {
1573                if (triggerInstCode == Trigger.INSTRUCTION_DELETE_TRIGGER) {
1574                    if (trigger.getNextFireTime() == null) {
1575                        if (tw.getTrigger().getNextFireTime() == null) {

1580                        schedulerSignaler.signalSchedulingChange(0L);
1581                    }
1582                } else if (triggerInstCode == Trigger.INSTRUCTION_SET_TRIGGER_COMPLETE) {
1583                    tw.setState(Trigger.STATE_COMPLETE);
1584                    removeWaitingTrigger(tw.getTrigger());
1585                    schedulerSignaler.signalSchedulingChange(0L);
1586                } else if (triggerInstCode == Trigger.INSTRUCTION_SET_TRIGGER_ERROR) {
1587                    logger.debug("Trigger {} set to ERROR state.", trigger.getFullName());
1588                    tw.setState(Trigger.STATE_ERROR);
1589                    schedulerSignaler.signalSchedulingChange(0L);
1590                } else if (triggerInstCode == Trigger.INSTRUCTION_SET_ALL_JOB_TRIGGERS_ERROR) {
1591                    logger.debug("All triggers of Job {} set to ERROR state.", trigger.getFullJobName());
1592                    setAllTriggersOfJobToState(trigger.getJobName(), trigger.getJobGroup(), Trigger.STATE_ERROR);
1593                    schedulerSignaler.signalSchedulingChange(0L);
1594                } else if (triggerInstCode == Trigger.INSTRUCTION_SET_ALL_JOB_TRIGGERS_COMPLETE) {
1595                    setAllTriggersOfJobToState(trigger.getJobName(), trigger.getJobGroup(), Trigger.STATE_COMPLETE);
1596                    schedulerSignaler.signalSchedulingChange(0L);

1598            }
