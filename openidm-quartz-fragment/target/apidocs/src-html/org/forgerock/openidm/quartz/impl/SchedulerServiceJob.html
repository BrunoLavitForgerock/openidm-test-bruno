026package org.forgerock.openidm.quartz.impl;
027
028import java.util.ArrayList;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032
033import org.forgerock.json.resource.RootContext;
034import org.forgerock.json.resource.SecurityContext;
035import org.forgerock.json.resource.ServerContext;
036import org.forgerock.openidm.config.enhanced.InvalidException;
037import org.forgerock.openidm.util.LogUtil;
038import org.forgerock.openidm.util.LogUtil.LogLevel;
039import org.osgi.framework.BundleContext;
040import org.osgi.framework.Constants;
041import org.osgi.framework.Filter;
042import org.osgi.framework.FrameworkUtil;
043import org.osgi.framework.InvalidSyntaxException;
044import org.osgi.util.tracker.ServiceTracker;
045import org.quartz.Job;
046import org.quartz.JobDataMap;
047import org.quartz.JobExecutionContext;
048import org.quartz.JobExecutionException;
049import org.slf4j.Logger;
050import org.slf4j.LoggerFactory;
051
052/**
053 * Scheduler service job using Quartz
054 *
055 * Invokes the configured ScheduledService in OSGi
056 *
057 * @author aegloff
058 */
059
060public class SchedulerServiceJob implements Job {
061
062    final static Logger logger = LoggerFactory.getLogger(SchedulerServiceJob.class);
063
064    // Default to INFO
065    private LogLevel logLevel = LogLevel.INFO;
066
067    public SchedulerServiceJob() {
068        // Instances of Job must have a public no-argument constructor.
069    }
070
071    /**
072     * Builds the ServerContext
073     * 
074     * @param id  The authentication id
075     * @return the new ServerContext
076     */
077    private ServerContext newScheduledServerContext(String id) {
078        final Map<String, Object> authzid = new HashMap<String, Object>();
079        authzid.put(SecurityContext.AUTHZID_ID, id);
080        List<String> roles = new ArrayList<String>();
081        roles.add("system");
082        authzid.put(SecurityContext.AUTHZID_ROLES, roles);
083        authzid.put(SecurityContext.AUTHZID_COMPONENT, "scheduler");
084        SecurityContext securityContext = new SecurityContext(new RootContext(), id, authzid);
085        return new ServerContext(securityContext);
086    }
087
088    public void execute(JobExecutionContext context) throws JobExecutionException {
089        JobDataMap data = context.getMergedJobDataMap();
090
091        String invokeLogLevel = (String) data.get(ScheduledService.CONFIGURED_INVOKE_LOG_LEVEL);
092        logLevel = LogUtil.asLogLevel(invokeLogLevel);
093
094        String invokeService = (String) data.get(ScheduledService.CONFIGURED_INVOKE_SERVICE);
095        Object invokeContext = data.get(ScheduledService.CONFIGURED_INVOKE_CONTEXT);
096        ServiceTracker scheduledServiceTracker = (ServiceTracker) getServiceTracker(invokeService);
097
098        logger.debug("Job to invoke service with PID {} and invoke context {} with scheduler context {}",
099                new Object[] {invokeService, invokeContext, context});
100        logger.debug("Job to invoke service with PID {} with scheduler context {}", new Object[] {invokeService, context});
101
102        Map<String,Object> scheduledServiceContext = new HashMap<String,Object>();
103        scheduledServiceContext.putAll(data);
104
105        scheduledServiceContext.put(ScheduledService.INVOKER_NAME, "Scheduled " +
106                context.getJobDetail().getName() + "-" + context.getScheduledFireTime());
107        scheduledServiceContext.put(ScheduledService.SCHEDULED_FIRE_TIME, context.getScheduledFireTime());
108        scheduledServiceContext.put(ScheduledService.ACTUAL_FIRE_TIME, context.getFireTime());
109        scheduledServiceContext.put(ScheduledService.NEXT_FIRE_TIME, context.getNextFireTime());
110
111        ScheduledService scheduledService = (ScheduledService) scheduledServiceTracker.getService();
112        if (scheduledService == null) {
113            logger.info("Scheduled service {} to invoke currently not found, not (yet) registered. ", invokeService);
114        } else {
115            try {
116                LogUtil.logAtLevel(logger, logLevel, "Scheduled service \"{}\" found, invoking.", context.getJobDetail().getFullName());
117                scheduledService.execute(newScheduledServerContext((String)scheduledServiceContext.get(ScheduledService.INVOKER_NAME)), scheduledServiceContext);
118                LogUtil.logAtLevel(logger, logLevel, "Scheduled service \"{}\" invoke completed successfully.", context.getJobDetail().getFullName());
119            } catch (Exception ex) {
120                logger.warn("Scheduled service \"{}\" invocation reported failure: {}",
121                        new Object[]{context.getJobDetail().getFullName(), ex.getMessage(), ex});
122            }
123        }
124        scheduledServiceTracker.close();
125    }
126
127    ServiceTracker getServiceTracker(String servicePID) throws InvalidException {
128        Filter filter = null;
129        BundleContext context = null;
130        try {
131            context = FrameworkUtil.getBundle(SchedulerServiceJob.class).getBundleContext();
132            filter = FrameworkUtil.createFilter("(&(" + Constants.OBJECTCLASS + "=" + ScheduledService.class.getName() + ")"
133                    + "(service.pid=" + servicePID + "))");
134        } catch (InvalidSyntaxException ex) {
135            throw new InvalidException("Failure in setting up scheduler to find service to invoke. One possible cause is an invalid "
136                    + "invokeService property. :  " + ex.getMessage(), ex);
137        }
138        ServiceTracker serviceTracker = new ServiceTracker(context, filter, null);
139        serviceTracker.open();
140
141        return serviceTracker;
142    }
143}
