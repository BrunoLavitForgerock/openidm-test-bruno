026package org.forgerock.openidm.quartz.impl;
027
028import java.util.HashMap;
029import java.util.Map;
030
031import org.forgerock.json.fluent.JsonValue;
032import org.quartz.Calendar;
033import org.quartz.JobPersistenceException;
034
035/**
036 * A wrapper that contains the name and serialize form of a calendar.
037 */
038public class CalendarWrapper {
039    private String serialized;
040    private String name;
041    private String revision;
042    
043    /**
044     * Creates a new CalendarWrapper from a specified Calendar object and name.
045     * 
046     * @param cal   the Calendar object
047     * @param name  the name of the calendar
048     * @throws JobPersistenceException
049     */
050    public CalendarWrapper(Calendar cal, String name) throws JobPersistenceException {
051        this.name = name;
052        this.serialized = RepoJobStoreUtils.serialize(cal);
053    }
054    
055    /**
056     * Creates a new CalendarWrapper from an object map.
057     * 
058     * @param map   a object map
059     */
060    public CalendarWrapper(Map<String, Object> map) {
061        serialized = (String)map.get("serialized");
062        name = (String)map.get("name");
063        revision = (String)map.get("_rev");
064    }
065    
066    /**
067     * Deserializes and returns a Calendar object.
068     * 
069     * @return  the deserialized Calendar object
070     * @throws Exception
071     */
072    public Calendar getCalendar() throws Exception {
073        return (Calendar) RepoJobStoreUtils.deserialize(serialized);
074    }
075    
076    /**
077     * Returns the name of the calendar
078     * 
079     * @return  the name of the calendar
080     */
081    public String getName() {
082        return name;
083    }
084    
085    /**
086     * Returns a JsonValue object wrapper around the object map for the calendar.
087     * 
088     * @return a JsonValue object
089     */
090    public JsonValue getValue() {
091        Map<String, Object> map = new HashMap<String, Object>();
092        map.put("serialized", serialized);
093        map.put("name", name);
094        return new JsonValue(map);
095    }
096
097    /**
098     * Returns the revision, as set by the repo
099     * 
100     * @return the repo revision
101     */
102    public String getRevision() {
103        return revision;
104    }
105}
