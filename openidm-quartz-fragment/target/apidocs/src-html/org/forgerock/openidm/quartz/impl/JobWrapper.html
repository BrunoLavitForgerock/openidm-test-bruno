026package org.forgerock.openidm.quartz.impl;
027
028import java.util.HashMap;
029import java.util.Map;
030
031import org.forgerock.json.fluent.JsonValue;
032import org.quartz.JobDetail;
033import org.quartz.JobPersistenceException;
034
035/**
036 * A wrapper that contains all necessary information about a Job.
037 */
038public class JobWrapper {
039    
040    private String serialized;
041    private String key;
042    private String revision;
043    private boolean paused = false;
044    
045    /**
046     * Creates a new JobWrapper from a JobDetail object
047     * 
048     * @param jobDetail a JobDetail object
049     * @param paused    if the job is paused
050     * @throws JobPersistenceException
051     */
052    public JobWrapper(JobDetail jobDetail, boolean paused) throws JobPersistenceException {
053        this.key = jobDetail.getKey().toString();
054        this.serialized = RepoJobStoreUtils.serialize(jobDetail);
055        this.paused = paused;
056    }
057    
058    /**
059     * Creates a new JobWrapper from an object map.
060     * 
061     * @param map an object map
062     */
063    public JobWrapper(Map<String, Object> map) {
064        serialized = (String)map.get("serialized");
065        key = (String)map.get("key");
066        paused = (Boolean)map.get("paused");
067        revision = (String)map.get("_rev");
068    }
069    
070    /**
071     * Returns a JsonValue object representing the JobWrapper
072     * 
073     * @return  a JsonValue object
074     */
075    public JsonValue getValue() {
076        Map<String, Object> map = new HashMap<String, Object>();
077        map.put("serialized", serialized);
078        map.put("key", key);
079        map.put("paused", paused);
080        return new JsonValue(map);
081    }
082    
083    /**
084     * Returns the serialized JobDetail object
085     * 
086     * @return  the serialized JobDetail object
087     */
088    public String getSerialized() {
089        return serialized;
090    }
091    
092    /**
093     * Retuns the Job key
094     * 
095     * @return the Job key
096     */
097    public String getKey() {
098        return key;
099    }
100    
101    /**
102     * Returns the deserialized JobDetail object.
103     * 
104     * @return the JobDetail object
105     * @throws Exception
106     */
107    public JobDetail getJobDetail() throws Exception {
108        return (JobDetail)RepoJobStoreUtils.deserialize(serialized);
109    }
110    
111    /**
112     * Returns true if the JobWrapper is in the "paused" state, false otherwise.
113     * 
114     * @return  true if the JobWrapper is in the "paused" state, false otherwise
115     */
116    public boolean isPaused() {
117        return paused;
118    }
119    
120    /**
121     * Sets the JobWrapper in the "paused" state.
122     * 
123     * @return the paused JobWrapper
124     */
125    public void pause() {
126        setPaused(true);
127    }
128    
129    /**
130     * Resumes the JobWrapper from the paused state
131     * 
132     * @return the resumed JobWrapper
133     */
134    public void resume() {
135        setPaused(false);
136    }
137
138    /**
139     * Sets the "paused" state of the JobWrapper
140     * 
141     * @param paused true if "paused", false otherwise
142     */
143    public void setPaused(boolean paused) {
144        this.paused = paused;
145    }
146
147    /**
148     * Returns the revision, as set by the repo
149     * 
150     * @return the repo revision
151     */
152    public String getRevision() {
153        return revision;
154    }
155}
