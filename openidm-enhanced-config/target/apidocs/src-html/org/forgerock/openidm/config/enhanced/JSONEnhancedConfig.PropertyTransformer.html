098    public String getConfigurationFactoryPid(ComponentContext compContext) {
099        Object o = compContext.getProperties().get(ServerConstants.CONFIG_FACTORY_PID);
100        if (o instanceof String) {
101            return (String) o;
102        }
103        return null;
104    }

106    /**
107     * {@inheritdoc}
108     */
109    public Map<String, Object> getConfiguration(ComponentContext compContext)
110            throws InvalidException, InternalErrorException {
111
112        JsonValue confValue = getConfigurationAsJson(compContext);
113        return confValue.asMap();
114    }
115
116    /**
117     * {@inheritDoc}
118     */
119    public JsonValue getConfigurationAsJson(ComponentContext compContext) throws InvalidException,
120            InternalErrorException {
121
122        Dictionary dict = null;
123        if (compContext != null) {
124            dict = compContext.getProperties();
125        }
126        String servicePid = (String) compContext.getProperties().get(Constants.SERVICE_PID);
127
128        JsonValue conf = getConfiguration(dict, compContext.getBundleContext(), servicePid);
129
130        return conf;
131    }
132
133    /**
134     * {@inheritDoc}
135     */
136    public JsonValue getConfiguration(Dictionary<String, Object> dict, BundleContext context,
137            String servicePid) throws InvalidException, InternalErrorException {
138        return getConfiguration(dict, servicePid, true);
139    }
140
141    /**
142     * {@see getConfiguration(Dictionary<String, Object>, BundleContext, String)
143     * * }
144     *
145     * @param decrypt
146     *            true if any encrypted values should be decrypted in the result
147     */
148    public JsonValue getConfiguration(Dictionary<String, Object> dict, String servicePid,
149            boolean decrypt) throws InvalidException, InternalErrorException {
150        JsonValue jv = new JsonValue(new LinkedHashMap<String, Object>());
151
152        if (dict != null) {
153            String jsonConfig = (String) dict.get(JSON_CONFIG_PROPERTY);
154            logger.trace("Get configuration from JSON config property {}", jsonConfig);
155
156            try {
157                if (jsonConfig != null && jsonConfig.trim().length() > 0) {
158                    jv = JsonUtil.parseStringified(jsonConfig);
159                }
160            } catch (Exception ex) {
161                throw new InvalidException("Configuration for " + servicePid
162                        + " could not be parsed: " + ex.getMessage(), ex);
163            }
164            logger.trace("Parsed configuration {}", jv);
165
166            try {
167                jv.required().expect(Map.class);
168            } catch (JsonValueException ex) {
169                throw new InvalidException("Component configuration for " + servicePid
170                        + " is invalid: " + ex.getMessage(), ex);
171            }
172        }
173        logger.debug("Configuration for {}: {}", servicePid, jv);
174
175        JsonValue decrypted = jv;
176        if (!jv.isNull()) {
177            decrypted.getTransformers().add(new PropertyTransformer(doEscape));
178            decrypted.applyTransformers();
179            decrypted = decrypted.copy();
180        }
181        // todo: different way to handle mock unit tests
182        if (decrypt && dict != null && !jv.isNull()) {
183            decrypted = decrypt(jv);
184        }
185
186        return decrypted;
187    }
188
189    private JsonValue decrypt(JsonValue value) throws JsonException, InternalErrorException {
190        return getCryptoService().decrypt(value); // makes a decrypted copy
191    }
192
193    private CryptoService getCryptoService() throws InternalErrorException {
194        return CryptoServiceFactory.getInstance();
195    }
196
197    public static class PropertyTransformer implements JsonTransformer {
198
199        // Disable the property escaping by default
200        private final boolean doEscape;
201
202        public PropertyTransformer(boolean doEscape) {
203            this.doEscape = doEscape;
204        }
205
206        @Override
207        public void transform(JsonValue value) throws JsonException {
208            if (null != value && value.isString()) {
209                value.setObject(PropertyUtil.substVars(value.asString(), IdentityServer
210                        .getInstance(), doEscape));
211            }
212        }
213    }
