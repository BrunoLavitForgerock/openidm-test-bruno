239    /**
240     * Find the proper target simple class type for the source type.
241     * <p/>
242     * JSON schema has predefined types abd this method maps the {@code clazz} Class to proper representation
243     *
244     * @param clazz
245     * @return
246     */
247    public static String findJSONTypeForClass(Class clazz) {
248        if ((Integer.class.isAssignableFrom(clazz)) || (int.class == clazz)) {
249            return Constants.TYPE_INTEGER;
250        } else if ((Number.class.isAssignableFrom(clazz)) || (double.class == clazz) || (float.class == clazz) || (long.class == clazz)) {
251            return Constants.TYPE_NUMBER;
252        } else if ((Boolean.class.isAssignableFrom(clazz)) || (boolean.class == clazz)) {
253            return Constants.TYPE_BOOLEAN;
254        } else if (List.class.isAssignableFrom(clazz)) {
255            return Constants.TYPE_ARRAY;
256        } else if (Map.class.isAssignableFrom(clazz)) {
257            return Constants.TYPE_OBJECT;
258        } else {
259            return Constants.TYPE_STRING;
260        }
261    }

377    private static Object convertFromConfigurationProperty(ConfigurationProperty configurationProperty) {
378        Object sourceValue = configurationProperty.getValue();
379        if (sourceValue == null) {
380            return null;
381        }
382        boolean isArray = sourceValue.getClass().isArray();
383        Class sourceType = isArray ? sourceValue.getClass().getComponentType() : sourceValue.getClass();
384        Object result = null;
385        if (isArray) {
386            if (sourceType == byte.class) {
387                result = new String((byte[]) sourceValue);
388            } else if (sourceType == char.class) {
389                result = new String((char[]) sourceValue);
390            } else if (sourceType == Character.class) {
391                Character[] characterArray = (Character[]) sourceValue;
392                char[] charArray = new char[characterArray.length];
393                for (int i = 0; i < characterArray.length; i++) {
394                    charArray[i] = characterArray[i];
395                }
396                result = new String(charArray);
397            } else {
398                int length = Array.getLength(sourceValue);
399                List values = new ArrayList(length);
400                for (int i = 0; i < length; i++) {
401                    Object item = Array.get(sourceValue, i);
402                    Object newValue = coercedTypeCasting(item, Object.class);
403                    values.add(newValue);
404                }
405                result = values;
406            }
407        } else {
408            result = coercedTypeCasting(sourceValue, Object.class);
409        }
410        return result;
411    }

414    public static void configureConfigurationProperties(JsonValue source, ConfigurationProperties target) throws JsonValueException {
415        source.required();
416        if (null != target) {
417            List<String> configPropNames = target.getPropertyNames();
418            for (Map.Entry<String, Object> e : source.asMap().entrySet()) {
419                if (!configPropNames.contains(e.getKey())) {
420                    /*
421                    * The connector's Configuration does not define this property.
422                    */
423                    continue;
424                }
425                ConfigurationProperty property = target.getProperty(e.getKey());
426                Class targetType = property.getType();
427                Object propertyValue = null;
428                if (targetType.isArray()) {
429                    Class targetBaseType = targetType.getComponentType();
430                    if (targetBaseType == byte.class || targetBaseType == char.class) {
431                        propertyValue = coercedTypeCasting(e.getValue(), targetType);
432                    } else if (e.getValue() instanceof List) {
433                        List v = (List) e.getValue();
434                        propertyValue = Array.newInstance(targetBaseType, v.size());
435                        for (int i = 0; i < v.size(); i++) {
436                            Array.set(propertyValue, i, coercedTypeCasting(v.get(i), targetBaseType));
437                        }
438                    } else {
439                        propertyValue = Array.newInstance(targetBaseType, 1);
440                        Array.set(propertyValue, 0, coercedTypeCasting(e.getValue(), targetBaseType));
441                    }
442                } else {
443                    propertyValue = coercedTypeCasting(e.getValue(), targetType);
444                }
445                property.setValue(propertyValue);
446            }
447        }
448    }

451    public static void configureDefaultAPIConfiguration(JsonValue source, APIConfiguration target) throws JsonValueException {
452        JsonValue poolConfigOption = source.get(OPENICF_POOL_CONFIG_OPTION);
453        if (poolConfigOption.isMap()) {
454            configureObjectPoolConfiguration(poolConfigOption, target.getConnectorPoolConfiguration());
455        }
456        JsonValue resultsHandlerConfigOption = source.get(OPENICF_RESULTSHANDLER_CONFIG_OPTION);
457        if (resultsHandlerConfigOption.isMap()) {
458            configureResultsHandlerConfiguration(resultsHandlerConfigOption, target.getResultsHandlerConfiguration());
459        }
460        JsonValue operationTimeout = source.get(OPENICF_OPERATION_TIMEOUT);
461        if (operationTimeout.isMap()) {
462            configureTimeout(operationTimeout, target);
463        }
464        JsonValue configurationProperties = source.get(OPENICF_CONFIGURATION_PROPERTIES);
465        configureConfigurationProperties(configurationProperties, target.getConfigurationProperties());
466        if (source.isDefined(BUFFER_SIZE)) {
467            target.setProducerBufferSize(source.get(BUFFER_SIZE).required().asInteger());
468        }
469    }

473        target.put(OPENICF_RESULTSHANDLER_CONFIG_OPTION, getResultsHandlerConfiguration(source.getResultsHandlerConfiguration()));

601        Set<String> objectTypes = configuration.get(OPENICF_OBJECT_TYPES).keys();
602        Map<String, Map<Class<? extends APIOperation>, OperationOptionInfoHelper>> operationOptionConfigurationMap =
603                new HashMap<String, Map<Class<? extends APIOperation>, OperationOptionInfoHelper>>(objectTypes.size());
604
605        JsonValue operationOptions = configuration.get(OPENICF_OPERATION_OPTIONS);
606
607        if (operationOptions.expect(Map.class).isNull()) {
608            for (String type : objectTypes) {
609                Map<Class<? extends APIOperation>, OperationOptionInfoHelper> config = new HashMap<Class<? extends APIOperation>, OperationOptionInfoHelper>(12);
610                operationOptionConfigurationMap.put(type, config);
611                for (OperationType entry : OperationType.values()) {
612                    config.put(entry.getValue(), new OperationOptionInfoHelper());
613                }
614            }
615        } else {
616            for (OperationType entry : OperationType.values()) {
617                OperationOptionInfoHelper defaultOperationOptionInfoHelper = null;
618
619                JsonValue operation = operationOptions.get(entry.name()).expect(Map.class);
620                if (operation.isNull()) {
621                    defaultOperationOptionInfoHelper = new OperationOptionInfoHelper();
622                } else {
623                    defaultOperationOptionInfoHelper = new OperationOptionInfoHelper(operation);
624                }
625
626                JsonValue objectFeatures = operation.get(OPENICF_OBJECT_FEATURES).expect(Map.class);
627                for (String type : objectTypes) {
628                    JsonValue objectFeature = objectFeatures.get(type);
629
630                    Map<Class<? extends APIOperation>, OperationOptionInfoHelper> config = operationOptionConfigurationMap.get(type);
631                    if (null == config) {
632                        config = new HashMap<Class<? extends APIOperation>, OperationOptionInfoHelper>(12);
633                        operationOptionConfigurationMap.put(type, config);
634                    }
635                    config.put(entry.getValue(), new OperationOptionInfoHelper(objectFeature, defaultOperationOptionInfoHelper));
636                }
637            }
638        }
639
640        return operationOptionConfigurationMap;
641    }

659    /**
660     * Build a {@link Map} from the given {@link org.identityconnectors.framework.common.objects.AttributeInfo}
661     * <p/>
662     * The result will look like this:
663     * {
664     * "type"    : "number",
665     * "mapName" : "lastLogin",
666     * "mapType" : "JAVA_TYPE_DOUBLE",
667     * "flags" : [
668     * "REQUIRED",
669     * "MULTIVALUED",
670     * "NOT_CREATABLE",
671     * "NOT_UPDATEABLE",
672     * "NOT_READABLE",
673     * "NOT_RETURNED_BY_DEFAULT"
674     * ]
675     * }
676     *
677     * @param info
678     * @return
679     */
680    public static Map<String, Object> getAttributeInfoMap(AttributeInfo info) {
681        Map<String, Object> schema = new LinkedHashMap<String, Object>();
682        if (info.isMultiValued()) {
683            schema.put(Constants.TYPE, Constants.TYPE_ARRAY);
684            Map<String, Object> itemSchema = new LinkedHashMap<String, Object>(2);
685            itemSchema.put(Constants.TYPE, findJSONTypeForClass(info.getType()));
686            itemSchema.put(ConnectorUtil.OPENICF_NATIVE_TYPE, findNameForClass(info.getType()));
687            schema.put(Constants.ITEMS, itemSchema);
688        } else {
689            schema.put(Constants.TYPE, findJSONTypeForClass(info.getType()));
690        }
691        if (info.isRequired()) {
692            schema.put(Constants.REQUIRED, true);
693        }
694        schema.put(ConnectorUtil.OPENICF_NATIVE_NAME, info.getName());
695        schema.put(ConnectorUtil.OPENICF_NATIVE_TYPE, findNameForClass(info.getType()));
696        if (!info.getFlags().isEmpty()) {
697            List<String> flags = null;
698            for (AttributeInfo.Flags flag : info.getFlags()) {
699                if (AttributeInfo.Flags.MULTIVALUED.equals(flag) || AttributeInfo.Flags.REQUIRED.equals(flag)) {
700                    continue;
701                }
702                if (null == flags) {
703                    flags = new ArrayList<String>(4);
704                }
705                flags.add(flag.name());
706            }
707            if (null != flags) {
708                schema.put(ConnectorUtil.OPENICF_FLAGS, Collections.unmodifiableList(flags));
709            }
710        }
711        return schema;
712    }

791    /**
792     * Coerce the {@code source} object to an object of {@code clazz} type.
793     * <p/>
794     *
795     * @param <T>
796     * @param source
797     * @param clazz
798     * @return
799     * @throws NumberFormatException
800     * @throws URISyntaxException
801     * @throws UnsupportedOperationException
802     */
803    @SuppressWarnings("unchecked")
804    public static <T> T coercedTypeCasting(Object source, Class<T> clazz) throws IllegalArgumentException {
805        if (null == clazz) {
806            throw new IllegalArgumentException("Target Class can not be null");
807        }
808        if (source instanceof JsonValue) {
809            source = ((JsonValue)source).getObject();
810        }
811
812        Class<T> targetClazz = clazz;
813        Class sourceClass = (source == null ? null : source.getClass());
814        boolean coerced = false;
815        T result = null;
816        try {
817            if (source == null) {
818                return null;
819            }
820
821            //Default JSON Type conversion
822            if (targetClazz.equals(Object.class)) {
823                if ((Number.class.isAssignableFrom(sourceClass)) || (int.class == clazz) || (double.class == clazz) || (float.class == clazz) || (long.class == clazz)) {
824                    return (T) source;
825                } else if ((Boolean.class.isAssignableFrom(sourceClass)) || (boolean.class == clazz)) {
826                    return (T) source;
827                } else if (String.class.isAssignableFrom(sourceClass)) {
828                    return (T) source;
829                } else if (Map.class.isAssignableFrom(sourceClass)) {
830                    return (T) source;
831                } else if (List.class.isAssignableFrom(sourceClass)) {
832                    return (T) source;
833                } else if (byte[].class.isAssignableFrom(sourceClass) || Byte[].class.isAssignableFrom(sourceClass) || byte[].class == clazz) {
834                    return (T) source;
835                } else if (sourceClass == QualifiedUid.class) {
836                    //@TODO: Not null safe!!!
837                    Map<String, Object> v = new HashMap<String, Object>(2);
838                    v.put("_id", ((QualifiedUid) source).getUid().getUidValue());
839                    v.put("_type", ((QualifiedUid) source).getObjectClass().getObjectClassValue());
840                    return (T) v;
841                } else if (sourceClass == Script.class) {
842                    Map<String, Object> v = new HashMap<String, Object>(2);
843                    v.put("scriptLanguage", ((Script) source).getScriptLanguage());
844                    v.put("scriptText", ((Script) source).getScriptText());
845                    return (T) v;
846                } else {
847                    targetClazz = (Class<T>) String.class;
848                }
849            }
850
851            if (targetClazz.isAssignableFrom(sourceClass)) {
852                return (T) source;
853            } else if (targetClazz == sourceClass) {
854                return (T) source;
855            } else if (targetClazz.equals(java.math.BigDecimal.class)) {
856                if (Double.class.isAssignableFrom(sourceClass) || sourceClass == double.class) {
857                    result = (T) BigDecimal.valueOf((Double) source);
858                    coerced = true;
859                } else if (Integer.class.isAssignableFrom(sourceClass) || sourceClass == int.class) {
860                    result = (T) BigDecimal.valueOf((Integer) source);
861                    coerced = true;
862                } else if (Long.class.isAssignableFrom(sourceClass) || sourceClass == long.class) {
863                    result = (T) BigDecimal.valueOf((Long) source);
864                    coerced = true;
865                } else if (sourceClass == String.class) {
866                    java.math.BigDecimal v = new java.math.BigDecimal((String) source);
867                    result = targetClazz.cast(v);
868                    coerced = true;
869                }
870            } else if (targetClazz.equals(java.math.BigInteger.class)) {
871                if (Long.class.isAssignableFrom(sourceClass) || sourceClass == long.class) {
872                    result = (T) BigInteger.valueOf((Long) source);
873                    coerced = true;
874                } else if (sourceClass == String.class) {
875                    java.math.BigInteger v = new java.math.BigInteger((String) source);
876                    result = targetClazz.cast(v);
877                    coerced = true;
878                } else {
879                    result = (T) BigInteger.valueOf(coercedTypeCasting(source, Long.class));
880                    coerced = true;
881                }
882            } else if (targetClazz.equals(boolean.class) || targetClazz.equals(Boolean.class)) {
883                if (sourceClass == Boolean.class) {
884                    result = (T) source;
885                    coerced = true;
886                } else if (sourceClass == Integer.class) {
887                    int val = ((Integer) source).intValue();
888                    if (val == 0) {
889                        result = targetClazz.cast(Boolean.FALSE);
890                        coerced = true;
891                    } else if (val == 1) {
892                        result = targetClazz.cast(Boolean.TRUE);
893                        coerced = true;
894                    }
895                } else if (sourceClass == String.class) {
896                    String s = (String) source;
897                    if (s.equalsIgnoreCase("true") || s.equalsIgnoreCase("false")) {
898                        result = (T) Boolean.valueOf((String) source);
899                        coerced = true;
900                    }
901                }
902            } else if (targetClazz.equals(Byte.TYPE) || targetClazz.equals(Byte.class)) {
903                if (sourceClass == Byte.TYPE || sourceClass == Byte.class) {
904                    result = (T) source;
905                    coerced = true;
906                } else if (sourceClass == String.class) {
907                    if (targetClazz.equals(Byte.class)) {
908                        result = (T) new Byte((String) source);
909                        coerced = true;
910                    } else {
911                        result = (T) Byte.valueOf((String) source);
912                        coerced = true;
913                    }
914                } else if (source instanceof Number) {
915                    int sourceInt = ((Number) source).intValue();
916                    if (targetClazz.equals(Byte.TYPE)) {
917                        result = (T) Byte.valueOf((byte) sourceInt);
918                        coerced = true;
919                    } else if (sourceInt > Byte.MIN_VALUE || sourceInt < Byte.MAX_VALUE) {
920                        result = (T) Byte.valueOf((byte) sourceInt);
921                        coerced = true;
922                    }
923                }
924            } else if (targetClazz.equals(byte[].class)) {
925                if (sourceClass == String.class) {
926                    result = targetClazz.cast(((String) source).getBytes());
927                    coerced = true;
928                } else if (sourceClass == String.class) {
929                    result = targetClazz.cast(Base64.decode((String) source));
930                    coerced = true;
931                } else if (sourceClass == GuardedByteArray.class) {
932                    GuardedByteArray gba = (GuardedByteArray) source;
933                    byte[] byteArray = decrypt(gba);
934                    result = targetClazz.cast(byteArray);
935                    coerced = true;
936                }

939                    Character v = ((String) source).charAt(0);
940                    result = (T) v;
941                    coerced = true;
942                }
943            } else if (targetClazz.equals(Character[].class)) {
944                if (sourceClass == String.class) {
945                    char[] charArray = ((String) source).toCharArray();
946                    Character[] characterArray = new Character[charArray.length];
947                    for (int i = 0; i < charArray.length; i++) {
948                        characterArray[i] = new Character(charArray[i]);
949                    }
950                    result = targetClazz.cast(characterArray);
951                    coerced = true;
952                }
953            } else if (targetClazz.equals(char[].class)) {
954                if (sourceClass == String.class) {
955                    char[] charArray = ((String) source).toCharArray();
956                    result = targetClazz.cast(charArray);
957                    coerced = true;

959            } else if (targetClazz.equals(Date.class)) {

961                    //TODO AttributeUtil.getDateValue()

963            } else if (targetClazz.equals(double.class)) {
964                if (sourceClass == Double.class) {
965                    result = (T) source;
966                    coerced = true;
967                } else if (sourceClass == int.class) {
968                    //noinspection BoxingBoxedValue
969                    result = (T) Double.valueOf((((Integer) source).doubleValue()));
970                    coerced = true;
971                } else if (sourceClass == Integer.class) {
972                    result = (T) Double.valueOf(((Integer) source).doubleValue());
973                    coerced = true;
974                } else if (sourceClass == String.class) {
975                    result = targetClazz.cast(Double.valueOf((String) source));

992            } else if (targetClazz.equals(java.io.File.class)) {
993                if (sourceClass == String.class) {
994                    result = (T) new File((String) source);
995                    coerced = true;
996                }
997            } else if (targetClazz.equals(float.class) || targetClazz.equals(Float.class)) {
998                if (sourceClass == Float.class || sourceClass == float.class) {
999                    result = (T) source;

1001                } else if (sourceClass == Double.class || sourceClass == double.class) {
1002                    result = (T) new Float((Double) source);
1003                    coerced = true;

1014            } else if (targetClazz.equals(GuardedByteArray.class)) {

1037                    coerced = true;

1040                    if (l.longValue() <= Integer.MAX_VALUE) {
