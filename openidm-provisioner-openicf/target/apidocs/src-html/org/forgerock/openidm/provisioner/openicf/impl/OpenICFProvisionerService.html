325    @Activate
326    protected void activate(ComponentContext context) {
327        try {
328            jsonConfiguration = JSONEnhancedConfig.newInstance().getConfigurationAsJson(context);
329            systemIdentifier = new SimpleSystemIdentifier(jsonConfiguration);
330
331            if (!jsonConfiguration.get("enabled").defaultTo(true).asBoolean()) {
332                logger.info("OpenICF Provisioner Service {} is disabled, \"enabled\" set to false in configuration",
333                        systemIdentifier.getName());
334                return;
335            }
336
337            loadLocalSystemActions(jsonConfiguration);
338
339            connectorReference = ConnectorUtil.getConnectorReference(jsonConfiguration);
340
341            syncFailureHandler = syncFailureHandlerFactory.create(jsonConfiguration.get("syncFailureHandler"));
342
343            init(jsonConfiguration);
344
345            connectorFacadeCallback = new ConnectorFacadeCallback() {
346                @Override
347                public void addingConnectorInfo(ConnectorInfo connectorInfo,
348                        ConnectorFacadeFactory facadeFactory) {
349                    try {
350                        APIConfiguration config = connectorInfo.createDefaultAPIConfiguration();
351                        ConnectorUtil.configureDefaultAPIConfiguration(jsonConfiguration, config);
352
353                        final ConnectorFacade facade = facadeFactory.newInstance(config);
354
355                        if (null == facade) {
356                            logger.warn("OpenICF ConnectorFacade of {} is not available", connectorReference);
357                        } else {
358                            facade.validate();
359                            if (connectorFacade.compareAndSet(null, facade)) {
360                                if (null != routeEntry) {
361                                    // This should not happen but keep it in case
362                                    routeEntry.removeRoute();
363                                }
364
365                                if (facade.getSupportedOperations().contains(TestApiOp.class)) {
366                                    try {
367                                        facade.test();
368                                        logger.debug("OpenICF connector test of {} succeeded!", systemIdentifier);
369                                        serviceAvailable = true;
370                                    } catch (Exception e) {
371                                        logger.error("OpenICF connector test of {} failed!", systemIdentifier, e);
372                                    }
373                                } else {
374                                    logger.debug("OpenICF connector of {} does not support test.", connectorReference);
375                                    serviceAvailable = true;
376                                }
377                            }
378                        }
379                    } catch (Throwable t) {
380                        logger.warn(t.getMessage());
381                    } finally {
382                        logger.info("OpenICF Provisioner Service component {} is activated{}",
383                                systemIdentifier.getName(),
384                                (null != connectorFacade.get()
385                                    ? "."
386                                    : " although the service is not available yet."));
387                    }
388                }
389
390                @Override
391                public void removedConnectorInfo(ConnectorInfo connectorInfo) {
392                    connectorFacade.set(null);
393                }
394            };
395
396            connectorInfoProvider.addConnectorFacadeCallback(connectorReference, connectorFacadeCallback);
397
398            routeEntry = routerRegistry.addRoute(RouteBuilder.newBuilder()
399                    .withTemplate("/system/" + systemIdentifier.getName())
400                    .withSingletonResourceProvider(this)
401                    .buildNext()
402                    .withModeStartsWith()
403                    .withTemplate("/system/" + systemIdentifier.getName() + ObjectClassRequestHandler.OBJECTCLASS_TEMPLATE)
404                    .withRequestHandler(new ObjectClassRequestHandler())
405                    .seal());
406
407            logger.info("OpenICF Provisioner Service component {} is activated{}", systemIdentifier.getName(),
408                    (null != connectorFacade.get()
409                        ? "."
410                        : " although the service is not available yet."));
411        } catch (Exception e) {
412            logger.error("OpenICF Provisioner Service configuration has errors", e);
413            throw new ComponentException("OpenICF Provisioner Service configuration has errors", e);
414        }
415    }

417    private void init(JsonValue configuration) {
418        ConnectorInfo ci = connectorInfoProvider.findConnectorInfo(connectorReference);
419        if (null != ci) {
420            try {
421                operationHelperBuilder = new OperationHelperBuilder(systemIdentifier.getName(), configuration,
422                        ci.createDefaultAPIConfiguration());
423            } catch (Exception e) {
424                logger.error("OpenICF connector jsonConfiguration of {} has errors.", systemIdentifier, e);
425                throw new ComponentException(
426                        "OpenICF connector jsonConfiguration has errors and the service can not be initiated.", e);
427            }
428        } else {
429            throw new ComponentException(
430                    "Can not find the OpenICF Connector: " + connectorReference);
431        }
432
433        try {
434            // TODO Iterate over the supported type and register
435            boolean allowModification = !configuration.get("readOnly").defaultTo(false).asBoolean();
436            if (!allowModification) {
437                logger.debug("OpenICF Provisioner Service {} is running in read-only mode", systemIdentifier.getName());
438            }
439
440            Map<String, Map<Class<? extends APIOperation>, OperationOptionInfoHelper>> objectOperations =
441                    ConnectorUtil.getOperationOptionConfiguration(configuration);
442
443            objectTypes = ConnectorUtil.getObjectTypes(configuration);
444
445            for (Map.Entry<String, ObjectClassInfoHelper> entry :
446                    objectTypes.entrySet()) {
447
448                objectClassHandlers.put(entry.getKey(),
449                        Resources.newCollection(
450                                new ObjectClassResourceProvider(
451                                        entry.getKey(),
452                                        entry.getValue(),
453                                        objectOperations.get(entry.getKey()),
454                                        allowModification)));
455            }
456
457            // TODO Fix this Map
458            // ValidateApiOp
459            // TestApiOp
460            // ScriptOnConnectorApiOp
461            // ScriptOnResourceApiOp
462            // SchemaApiOp
463            systemOperations = Collections.emptyMap();
464        } catch (Exception e) {
465            logger.error("OpenICF connector jsonConfiguration of {} has errors.", systemIdentifier.getName(), e);
466            throw new ComponentException(
467                    "OpenICF connector jsonConfiguration has errors and the service can not be initiated.", e);
468        }
469        logger.debug("OpenICF connector jsonConfiguration has no errors.");
470    }

502    /**
503     * Handle ConnectorExceptions from ConnectorFacade invocations.  Maps each ConnectorException subtype to the
504     * appropriate {@link ResourceException} for passing to {@code handleError}.  Optionally logs to activity log.
505     *
506     * @param context the ServerContext from the original request
507     * @param request the original request
508     * @param exception the ConnectorException that was thrown by the facade
509     * @param resourceId the resourceId being operated on
510     * @param before the object value "before" the request
511     * @param after the object value "after" the request
512     * @param handler the ResultHandler on which to call handleError
513     * @param connectorExceptionActivityLogger the ActivityLogger to use to log the exception
514     */
515    private void handleConnectorException(ServerContext context, Request request, ConnectorException exception,
516            String resourceContainer, String resourceId, JsonValue before, JsonValue after, ResultHandler<?> handler,
517            ActivityLogger connectorExceptionActivityLogger) {
518
519        // default message
520        String message = MessageFormat.format("Operation {0} failed with {1} on system object: {2}",
521                request.getRequestType(), exception.getClass().getSimpleName(), resourceId);
522
523        try {
524            throw exception;
525        } catch (AlreadyExistsException e) {
526            message = MessageFormat.format("System object {0} already exists", resourceId);
527            handler.handleError(new ConflictException(message, exception));
528        } catch (ConfigurationException e) {
529            message = MessageFormat.format("Operation {0} failed with ConfigurationException on system object: {1}",
530                    request.getRequestType().toString(), resourceId);
531            handler.handleError(new InternalServerErrorException(message, exception));
532        } catch (ConnectionBrokenException e) {
533            message = MessageFormat.format("Operation {0} failed with ConnectionBrokenException on system object: {1}",
534                    request.getRequestType().toString(), resourceId);
535            handler.handleError(new ServiceUnavailableException(message, exception));
536        } catch (ConnectionFailedException e) {
537            message = MessageFormat.format("Connection failed during operation {0} on system object: {1}",
538                    request.getRequestType().toString(), resourceId);
539            handler.handleError(new ServiceUnavailableException(message, exception));
540        } catch (ConnectorIOException e) {
541            message = MessageFormat.format("Operation {0} failed with ConnectorIOException on system object: {1}",
542                    request.getRequestType().toString(), resourceId);
543            handler.handleError(new ServiceUnavailableException(message, exception));
544        } catch (OperationTimeoutException e) {
545            message = MessageFormat.format("Operation {0} Timeout on system object: {1}",
546                    request.getRequestType().toString(), resourceId);
547            handler.handleError(new ServiceUnavailableException(message, exception));
548        } catch (PasswordExpiredException e) {
549            message = MessageFormat.format("Operation {0} failed with PasswordExpiredException on system object: {1}",
550                    request.getRequestType().toString(), resourceId);
551            handler.handleError(new ForbiddenException(message, exception));
552        } catch (InvalidPasswordException e) {
553            message = MessageFormat.format("Invalid password has been provided to operation {0} for system object: {1}",
554                    request.getRequestType().toString(), resourceId);
555            handler.handleError(ResourceException.getException(UNAUTHORIZED_ERROR_CODE, message, exception));
556        } catch (UnknownUidException e) {
557            message = MessageFormat.format("Operation {0} could not find resource {1} on system object: {2}",
558                    request.getRequestType().toString(), resourceId, resourceContainer);
559            handler.handleError(
560                    new NotFoundException(message, exception)
561                            .setDetail(new JsonValue(new HashMap<String, Object>())));
562        } catch (InvalidCredentialException e) {
563            message = MessageFormat.format("Invalid credential has been provided to operation {0} for system object: {1}",
564                    request.getRequestType().toString(), resourceId);
565            handler.handleError(ResourceException.getException(UNAUTHORIZED_ERROR_CODE, message, exception));
566        } catch (PermissionDeniedException e) {
567            message = MessageFormat.format("Permission was denied on {0} operation for system object: {1}",
568                    request.getRequestType().toString(), resourceId);
569            handler.handleError(new ForbiddenException(message, exception));
570        } catch (ConnectorSecurityException e) {
571            message = MessageFormat.format("Operation {0} failed with ConnectorSecurityException on system object: {1}",
572                    request.getRequestType().toString(), resourceId);

574        } catch (InvalidAttributeValueException e) {
575            message = MessageFormat.format("Attribute value conflicts with the attribute''s schema definition on " +
576                    "operation {0} for system object: {1}",
577                    request.getRequestType().toString(), resourceId);
578            handler.handleError(new BadRequestException(message, exception));
579        } catch (PreconditionFailedException e) {
580            message = MessageFormat.format("The resource version for {0} does not match the version provided on " +
581                    "operation {1} for system object: {2}",
582                    resourceId, request.getRequestType().toString(), resourceContainer);
583            handler.handleError(new org.forgerock.json.resource.PreconditionFailedException(message, exception));
584        } catch (PreconditionRequiredException e) {
585            message = MessageFormat.format("No resource version for resource {0} has been provided on operation {1} for system object: {2}",
586                    resourceId , request.getRequestType().toString(), resourceContainer);
587            handler.handleError(new org.forgerock.json.resource.PreconditionRequiredException(message, exception));
588        } catch (RetryableException e) {
589            message = MessageFormat.format("Request temporarily unavailable on operation {0} for system object: {1}",
590                    request.getRequestType().toString(), resourceId);
591            handler.handleError(new ServiceUnavailableException(message, exception));
592        } catch (UnsupportedOperationException e) {
593            message = MessageFormat.format("Operation {0} is no supported for system object: {1}",
594                    request.getRequestType().toString(), resourceId);
595            handler.handleError(new NotFoundException(message, exception));
596        } catch (IllegalArgumentException e) {
597            message = MessageFormat.format("Operation {0} failed with IllegalArgumentException on system object: {1}",
598                    request.getRequestType().toString(), resourceId);
599            handler.handleError(new InternalServerErrorException(message, e));
600        } catch (RemoteWrappedException e) {
601            handleRemoteWrappedException(context, request, exception, resourceContainer, resourceId,
602                    before, after, handler, connectorExceptionActivityLogger);
603        } catch (ConnectorException e) {
604            message = MessageFormat.format("Operation {0} failed with ConnectorException on system object: {1}",
605                    request.getRequestType().toString(), resourceId);
606            handler.handleError(new InternalServerErrorException(message, exception));
607        } finally {
608            // log the ConnectorException
609            logger.debug(message, exception);
610            try {
611                connectorExceptionActivityLogger.log(context, request.getRequestType(), message, resourceId,
612                        before, after, Status.FAILURE);
613            } catch (ResourceException e) {
614                // this means the ActivityLogger couldn't log request; log to error log
615                logger.warn("Failed to write activity log", e);
616            }
617        }

832    private void handleScriptAction(final ActionRequest request, final ResultHandler<JsonValue> handler)
833            throws ResourceException {
834        final String scriptId = request.getAdditionalParameter(SystemAction.SCRIPT_ID);
835        if (StringUtils.isBlank(scriptId)) {
836            handler.handleError(new BadRequestException("Missing required parameter: " + SystemAction.SCRIPT_ID));
837            return;
838        }
839
840        if (!localSystemActionCache.containsKey(scriptId)) {
841            handler.handleError(new BadRequestException("Script ID: " + scriptId + " is not defined."));
842            return;
843        }
844
845        SystemAction action = localSystemActionCache.get(scriptId);
846
847        String systemType = connectorReference.getConnectorKey().getConnectorName();
848        List<ScriptContextBuilder> scriptContextBuilderList = action.getScriptContextBuilders(systemType);
849
850        if (scriptContextBuilderList.isEmpty()) {
851            handler.handleError(new BadRequestException("Script ID: " + scriptId +
852                    " for systemType " + systemType + " is not defined."));
853            return;
854        }
855
856        JsonValue result = new JsonValue(new HashMap<String, Object>());
857
858        boolean onConnector = !"resource".equalsIgnoreCase(
859                request.getAdditionalParameter(SystemAction.SCRIPT_EXECUTE_MODE));
860
861        final ConnectorFacade facade = getConnectorFacade0(handler,
862                onConnector ? ScriptOnConnectorApiOp.class : ScriptOnResourceApiOp.class);
863        if (null == facade) {
864            // getConnectorFacade0 already handles error when returning null
865            return;
866        }
867
868        String variablePrefix = request.getAdditionalParameter(SystemAction.SCRIPT_VARIABLE_PREFIX);
869
870        List<Map<String, Object>> resultList =
871                new ArrayList<Map<String, Object>>(scriptContextBuilderList.size());
872        result.put("actions", resultList);
873
874        for (ScriptContextBuilder contextBuilder : scriptContextBuilderList) {
875            boolean isShell = contextBuilder.getScriptLanguage().equalsIgnoreCase("Shell");
876            for (Map.Entry<String, String> entry : request.getAdditionalParameters().entrySet()) {
877                final String key = entry.getKey();
878                if (SystemAction.SCRIPT_PARAMS.contains(key)) {
879                    continue;
880                }
881                Object value = entry.getValue();
882                Object newValue = value;
883                if (isShell) {
884                    if ("password".equalsIgnoreCase(key)) {
885                        if (value instanceof String) {
886                            newValue = new GuardedString(((String) value).toCharArray());
887                        } else {
888                            throw new BadRequestException("Invalid type for password.");
889                        }
890                    }
891                    if ("username".equalsIgnoreCase(key)) {
892                        if (value instanceof String == false) {
893                            throw new BadRequestException("Invalid type for username.");
894                        }
895                    }
896                    if ("workingdir".equalsIgnoreCase(key)) {
897                        if (value instanceof String == false) {
898                            throw new BadRequestException("Invalid type for workingdir.");
899                        }
900                    }
901                    if ("timeout".equalsIgnoreCase(key)) {
902                        if (!(value instanceof String) && !(value instanceof Number)) {
903                            throw new BadRequestException("Invalid type for timeout.");
904                        }
905                    }
906                    contextBuilder.addScriptArgument(key, newValue);
907                    continue;
908                }
909
910                if (null != value) {
911                    if (value instanceof Collection) {
912                        newValue =
913                                Array.newInstance(Object.class,
914                                        ((Collection) value).size());
915                        int i = 0;
916                        for (Object v : (Collection) value) {
917                            if (null == v || FrameworkUtil.isSupportedAttributeType(v.getClass())) {
918                                Array.set(newValue, i, v);
919                            } else {
920                                // Serializable may not be
921                                // acceptable
922                                Array.set(newValue, i, v instanceof Serializable ? v : v.toString());
923                            }
924                            i++;
925                        }
926
927                    } else if (value.getClass().isArray()) {
928                        // TODO implement the array support later
929                    } else if (!FrameworkUtil.isSupportedAttributeType(value.getClass())) {
930                        // Serializable may not be acceptable
931                        newValue = value instanceof Serializable ? value : value.toString();
932                    }
933                }
934                contextBuilder.addScriptArgument(key, newValue);
935            }
936
937            JsonValue content = request.getContent();
938            // if there is no content(content.isNull()), skip adding content to script arguments
939            if (content.isMap()) {
940                for (Map.Entry<String, Object> entry : content.asMap().entrySet()) {
941                    contextBuilder.addScriptArgument(entry.getKey(), entry.getValue());
942                }
943            } else if (!content.isNull()) {
944                handler.handleError(new BadRequestException("Content is not of type Map"));
945                return;
946            }
947
948            // ScriptContext scriptContext = script.getScriptContextBuilder().build();
949            OperationOptionsBuilder operationOptionsBuilder = new OperationOptionsBuilder();
950
951            // It's necessary to keep the backward compatibility with Waveset IDM
952            if (null != variablePrefix && isShell) {
953                operationOptionsBuilder.setOption("variablePrefix", variablePrefix);
954            }
955
956            Map<String, Object> actionResult = new HashMap<String, Object>(2);
957            try {
958                Object scriptResult = null;
959                if (onConnector) {
960                    scriptResult = facade.runScriptOnConnector(
961                            contextBuilder.build(), operationOptionsBuilder.build());
962                } else {
963                    scriptResult = facade.runScriptOnResource(
964                            contextBuilder.build(), operationOptionsBuilder.build());
965                }
966                actionResult.put("result", ConnectorUtil.coercedTypeCasting(scriptResult, Object.class));
967            } catch (Throwable t) {
968                logger.error("Script execution error.", t);
969                actionResult.put("error", t.getMessage());
970            }
971            resultList.add(actionResult);
972        }
973        handler.handleResult(result);
974    }

1729    private static final QueryFilterVisitor<Filter, ObjectClassInfoHelper> RESOURCE_FILTER =
1730            new QueryFilterVisitor<Filter, ObjectClassInfoHelper>() {
1731
1732                @Override
1733                public Filter visitAndFilter(final ObjectClassInfoHelper helper,
1734                        List<QueryFilter> subFilters) {
1735                    final Iterator<QueryFilter> iterator = subFilters.iterator();
1736                    if (iterator.hasNext()) {
1737                        return buildAnd(helper, iterator.next(), iterator);
1738                    } else {
1739                        throw new IllegalArgumentException("cannot parse 'and' QueryFilter with zero operands");
1740                    }
1741                }
1742
1743                private Filter buildAnd(final ObjectClassInfoHelper helper, final QueryFilter left,
1744                        final Iterator<QueryFilter> iterator) {
1745                    if (iterator.hasNext()) {
1746                        final QueryFilter right = iterator.next();
1747                        return and(left.accept(this, helper), buildAnd(helper, right, iterator));
1748                    } else {
1749                        return left.accept(this, helper);
1750                    }
1751                }
1752
1753                @Override
1754                public Filter visitOrFilter(ObjectClassInfoHelper helper,
1755                        List<QueryFilter> subFilters) {
1756                    final Iterator<QueryFilter> iterator = subFilters.iterator();
1757                    if (iterator.hasNext()) {
1758                        return buildOr(helper, iterator.next(), iterator);
1759                    } else {
1760                        throw new IllegalArgumentException("cannot parse 'or' QueryFilter with zero operands");
1761                    }
1762                }
1763
