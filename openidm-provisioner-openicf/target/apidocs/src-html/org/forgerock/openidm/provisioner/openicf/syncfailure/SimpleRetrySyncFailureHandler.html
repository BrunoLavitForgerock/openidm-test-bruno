016package org.forgerock.openidm.provisioner.openicf.syncfailure;
017
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import java.util.Map;
022
023
024/**
025 * A simple sync failure handler that counts retries.
026 *
027 * @author brmiller
028 */
029public class SimpleRetrySyncFailureHandler implements SyncFailureHandler {
030
031    /** Logger */
032    private static final Logger logger = LoggerFactory.getLogger(SimpleRetrySyncFailureHandler.class);
033
034    /** how many times to retry live sync of a particular sync token */
035    private final int syncFailureRetries;
036
037    /** the handler to call after the retries are exhausted */
038    private final SyncFailureHandler postRetryHandler;
039
040    /** current token being retried */
041    private Object currentSyncToken;
042
043    /** number of retries on current token */
044    private int currentRetries;
045
046    /**
047     * Construct the SyncFailureHandler.
048     *
049     * @param syncFailureRetries the number of retries
050     */
051    public SimpleRetrySyncFailureHandler(int syncFailureRetries, SyncFailureHandler postRetryHandler) {
052        this.syncFailureRetries = syncFailureRetries;
053        this.postRetryHandler = postRetryHandler;
054    }
055
056    /**
057     * Handle sync failure by counting retries on this sync token, passing to
058     * (optional) post-retry handler when retries are exceeded.
059     *
060     * @param syncFailure contains sync failure data
061     * @param failureCause the cause of the sync failure
062     * @throws SyncHandlerException when retries are not exceeded
063     */
064    public void invoke(Map<String, Object> syncFailure, Exception failureCause)
065        throws SyncHandlerException {
066
067        final Object token = syncFailure.get("token");
068
069        if (token != null && token.equals(currentSyncToken)) {
070            currentRetries++;
071        }
072        else {
073            currentSyncToken = token;
074            currentRetries = 0;
075        }
076
077        if (currentRetries >= syncFailureRetries) {
078            logger.info("{}:{} sync retries = {}/{}, retries exhausted", 
079                    syncFailure.get("systemIdentifier"), syncFailure.get("uid"),
080                    currentRetries, syncFailureRetries);
081            postRetryHandler.invoke(syncFailure, failureCause);
082        } else {
083            logger.info("{}:{} sync retries = {}/{}, retrying", 
084                    syncFailure.get("systemIdentifier"), syncFailure.get("uid"),
085                    currentRetries, syncFailureRetries);
086            throw new SyncHandlerException("Failed to synchronize " + syncFailure.get("uid")
087                    + " object on " + syncFailure.get("systemIdentifier") + 
088                    ", retries (" + currentRetries + ") not exhausted.",
089                    failureCause);
090        }
091    }
092}
