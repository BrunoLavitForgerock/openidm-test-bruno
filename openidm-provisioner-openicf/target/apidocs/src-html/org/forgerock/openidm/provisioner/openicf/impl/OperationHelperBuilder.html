050public class OperationHelperBuilder {
051
052    private final APIConfigurationImpl runtimeAPIConfiguration;
053    private Map<String, ObjectClassInfoHelper> supportedObjectTypes;
054    private Map<String, Map<Class<? extends APIOperation>, OperationOptionInfoHelper>> operationOptionHelpers;
055    private String systemName;
056
057    public OperationHelperBuilder(String system, JsonValue jsonConfiguration, APIConfiguration defaultAPIConfiguration) throws JsonValueException {
058        runtimeAPIConfiguration = (APIConfigurationImpl) defaultAPIConfiguration;
059        ConnectorUtil.configureDefaultAPIConfiguration(jsonConfiguration, defaultAPIConfiguration);
060        supportedObjectTypes = ConnectorUtil.getObjectTypes(jsonConfiguration);
061        operationOptionHelpers = ConnectorUtil.getOperationOptionConfiguration(jsonConfiguration);
062        this.systemName = Assertions.blankChecked(system, "systemName");
063    }
064
065    public OperationHelper build(String objectType, JsonValue object, CryptoService cryptoService) throws ResourceException {
066        ObjectClassInfoHelper objectClassInfoHelper = supportedObjectTypes.get(objectType);
067        if (null == objectClassInfoHelper) {
068            throw new BadRequestException("Unsupported object type: " + objectType + " not in supported types" + supportedObjectTypes.keySet());
069        }
070//        APIConfiguration _configuration = getRuntimeAPIConfiguration();
071//
072//        TODO Set custom runtimeAPIConfiguration properties
073//        if (null != object.get("_configuration")) {
074//            ConnectorUtil.configureDefaultAPIConfiguration(null, _configuration);
075//        }
076
077        return new OperationHelperImpl(new Id(systemName, objectType), objectClassInfoHelper, operationOptionHelpers.get(objectType), cryptoService);
078    }
