071    /**
072     * Create a custom object class.
073     *
074     * @param schema string representation for the name of the object class.
075     * @throws IllegalArgumentException when objectClass is null
076     * @throws org.forgerock.json.fluent.JsonValueException
077     *
078     */
079    public ObjectClassInfoHelper(JsonValue schema) {
080        //Expect ObjectClass
081        objectClass = new ObjectClass(schema.get(ConnectorUtil.OPENICF_OBJECT_CLASS).required().asString());
082
083        //Expect Properties Map
084        properties = schema.get(Constants.PROPERTIES).required().expect(Map.class);
085        Set<String> propertyNames = properties.keys();
086
087        Set<AttributeInfoHelper> attributes0 = new HashSet<AttributeInfoHelper>(propertyNames.size());
088        Set<String> defaultAttributes = new HashSet<String>(propertyNames.size());
089        String __NAME__ = null;
090
091        for (String propertyName : propertyNames) {
092            AttributeInfoHelper helper = new AttributeInfoHelper(propertyName, false, properties.get(propertyName));
093            if (helper.getAttributeInfo().getName().equals(Name.NAME)) {
094                __NAME__ = propertyName;
095            }
096            if (helper.getAttributeInfo().isReturnedByDefault()) {
097                defaultAttributes.add(helper.getAttributeInfo().getName());
098            }
099            attributes0.add(helper);
100        }
101        //TODO Should we throw exceptions or ??
102        if (null == __NAME__) {
103            logger.warn("Required __NAME__ attribute definition is not configured. The CREATE operation will be disabled");
104            //throw new IllegalArgumentException("Required __NAME__ attribute is not configured");
105        }
106        nameAttribute = __NAME__;
107        attributes = Collections.unmodifiableSet(attributes0);
108        attributesReturnedByDefault = CollectionUtil.newReadOnlySet(defaultAttributes);
109    }

136    public boolean setAttributesToGet(final OperationOptionsBuilder builder,
137            final List<JsonPointer> fieldFilters) {
138        boolean returnResource = false;
139        if (null != fieldFilters) {
140            for (JsonPointer field : fieldFilters) {
141                if (field.isEmpty() || returnResource || !Resource.FIELD_CONTENT_ID.equals(field.leaf()) || !Resource.FIELD_CONTENT_REVISION.equals(field.leaf())){
142                    returnResource = true;
143                }
144                if (field.isEmpty()) {
145                    builder.setAttributesToGet(attributesReturnedByDefault);
146                    continue;
147                }
148                for (AttributeInfoHelper attribute : attributes) {
149                    if (attribute.getName().equals(field.leaf()) && attribute.getAttributeInfo().isReadable()) {
150                        builder.setAttributesToGet(attribute.getAttributeInfo().getName());
151                        break;
152                    }
153                }
154            }
155        }
156        return returnResource;
157    }

161    public Attribute filterAttribute(JsonPointer field, Object valueAssertion) {
162      if (field.size() != 1){
163          throw new IllegalArgumentException("Only one level JsonPointer supported");
164      }
165        String attributeName = field.leaf();
166
167       for (AttributeInfoHelper ai: attributes){
168          if (ai.getName().equals(attributeName)) {
169              return ai.build(valueAssertion);
170          }
171       }
172       return null;
173    }

175    /**
176     * Get the resourceId from a CreateRequest
177     * @param request CreateRequest
178     * @return resourceId to be used for creating the object
179     */
180    public String getCreateResourceId(final CreateRequest request) {
181        String nameValue = request.getNewResourceId();
182
183        if (null == nameValue) {
184            JsonValue o = request.getContent().get(nameAttribute);
185            if (o.isNull()) {
186                o = request.getContent().get(Resource.FIELD_CONTENT_ID);
187            }
188            if (o.isString()) {
189                nameValue = o.asString();
190            }
191        }
192        return nameValue;
193    }

195    /**
196     * Get the attributes are that are writable on a create
197     * @param request CreateRequest
198     * @param cryptoService encryption and decryption service
199     * @return Set of attributes to that are writable on create
200     * @throws BadRequestException when attribute is missing or has a null value
201     */
202    public Set<Attribute> getCreateAttributes(final CreateRequest request,
203            final CryptoService cryptoService) throws ResourceException {
204        JsonValue content = request.getContent().required().expect(Map.class);
205        String nameValue = getCreateResourceId(request);
206
207        if (StringUtils.isBlank(nameValue)) {
208            throw new BadRequestException("Required '_id' or '" + nameAttribute
209                    + "' attribute is missing");
210        }
211
212        Set<String> keySet = content.keys();
213        Map<String,Attribute> result = new HashMap<String, Attribute>(keySet.size());
214        result.put(Name.NAME, new Name(nameValue));
215        for (AttributeInfoHelper attributeInfo : attributes) {
216            if (Name.NAME.equals(attributeInfo.getAttributeInfo().getName())
217                    || Uid.NAME.equals(attributeInfo.getAttributeInfo().getName())
218                    || (!keySet.contains(attributeInfo.getName()) && !attributeInfo.getAttributeInfo().isRequired())) {
219                continue;
220            }
221            if (attributeInfo.getAttributeInfo().isCreateable()) {
222                JsonValue v = content.get(attributeInfo.getName());
223                if (v.isNull() && attributeInfo.getAttributeInfo().isRequired()) {
224                    throw new BadRequestException("Required attribute '" + attributeInfo.getName()
225                            + "' value is null");
226                }
227                Attribute a = attributeInfo.build(v.getObject(), cryptoService);
228                if (null != a) {
229                    result.put(attributeInfo.getAttributeInfo().getName(), a);
230                }
231            }
232        }
233        if (logger.isTraceEnabled()) {
234            ConnectorObjectBuilder builder = new ConnectorObjectBuilder().addAttributes(result.values());
235            builder.setName(nameValue);
236            builder.setUid(nameValue);
237            logger.trace("Update ConnectorObject: {}", SerializerUtil.serializeXmlObject(builder
238                    .build(), false));
239        }
240        return new HashSet<Attribute>(result.values());
241    }

243    public Set<Attribute> getUpdateAttributes(final UpdateRequest request, final Name newName,
244            final CryptoService cryptoService) throws ResourceException {
245        JsonValue content = request.getContent().required().expect(Map.class);
246        Set<String> keySet = content.keys();
247
248        Map<String,Attribute> result = new HashMap<String, Attribute>(keySet.size());
249        if (null != newName) {
250            result.put(Name.NAME, newName);
251        }
252        for (AttributeInfoHelper attributeInfo : attributes) {
253            if ( Uid.NAME.equals(attributeInfo.getAttributeInfo().getName())
254                    || !keySet.contains(attributeInfo.getName())) {
255                continue;
256            }
257            if (attributeInfo.getAttributeInfo().isUpdateable()) {
258                Object v = content.get(attributeInfo.getName()).getObject();
259                Attribute a = attributeInfo.build(v, cryptoService);
260                if (null != a) {
261                    result.put(attributeInfo.getAttributeInfo().getName(), a);
262                }
263            }
264        }
265        if (logger.isTraceEnabled()) {
266            ConnectorObjectBuilder builder = new ConnectorObjectBuilder().addAttributes(result.values());
267            if (null != newName) {
268                builder.setUid(newName.getName());
269            } else {
270                builder.setName("***");
271                builder.setUid("***");
272            }
273            logger.trace("Update ConnectorObject: {}", SerializerUtil.serializeXmlObject(builder
274                    .build(), false));
275        }
276        return new HashSet<Attribute>(result.values());
277    }

279    /**
280     * @param operation
281     * @param name
282     * @param source
283     * @param cryptoService
284     * @return
285     * @throws JsonResourceException if ID value can not be determined from the {@code source}
286     */
287    public ConnectorObject build(Class<? extends APIOperation> operation, String name, JsonValue source, CryptoService cryptoService) throws Exception {
288        String nameValue = name;
289
290        if (null == nameValue) {
291            JsonValue o = source.get(Resource.FIELD_CONTENT_ID);
292            if (o.isNull()) {
293                o = source.get(nameAttribute);
294            }
295            if (o.isString()) {
296                nameValue = o.asString();
297            }
298        }
299
300        if (null == nameValue) {
301            throw new BadRequestException("Required NAME attribute is missing");
302        } else {
303            nameValue = Id.unescapeUid(nameValue);
304            logger.trace("Build ConnectorObject {} for {}", nameValue, operation.getSimpleName());
305        }
306
307        ConnectorObjectBuilder builder = new ConnectorObjectBuilder();
308        builder.setObjectClass(objectClass);
309        builder.setUid(nameValue);
310        builder.setName(nameValue);
311        Set<String> keySet = source.required().asMap().keySet();
312        if (CreateApiOp.class.isAssignableFrom(operation)) {
313            for (AttributeInfoHelper attributeInfo : attributes) {
314                if (Name.NAME.equals(attributeInfo.getName()) || Uid.NAME.equals(attributeInfo.getName()) ||
315                        !keySet.contains(attributeInfo.getName())) {
316                    continue;
317                }
318                if (attributeInfo.getAttributeInfo().isCreateable()) {
319                    Object v = source.get(attributeInfo.getName()).getObject();
320                    if (null == v && attributeInfo.getAttributeInfo().isRequired()) {
321                        throw new IllegalArgumentException("Required attribute {" + attributeInfo.getName() + "} value is null");
322                    }
323                    builder.addAttribute(attributeInfo.build(v, cryptoService));
324                }
325            }
326        } else if (UpdateApiOp.class.isAssignableFrom(operation)) {
327            for (AttributeInfoHelper attributeInfo : attributes) {
328                if (Name.NAME.equals(attributeInfo.getName()) || Uid.NAME.equals(attributeInfo.getName()) ||
329                        !keySet.contains(attributeInfo.getName())) {
330                    continue;
331                }
332                if (attributeInfo.getAttributeInfo().isUpdateable()) {
333                    Object v = source.get(attributeInfo.getName()).getObject();
334                    builder.addAttribute(attributeInfo.build(v, cryptoService));
335                }
336            }
337        } else {
338            for (AttributeInfoHelper attributeInfo : attributes) {
339                if (Name.NAME.equals(attributeInfo.getName()) || Uid.NAME.equals(attributeInfo.getName()) ||
340                        !keySet.contains(attributeInfo.getName())) {
341                    continue;
342                }
343                Object v = source.get(attributeInfo.getName()).getObject();
344                builder.addAttribute(attributeInfo.build(v, cryptoService));
345            }
346        }
347        ConnectorObject result = builder.build();
348        if (logger.isTraceEnabled()) {
349            logger.trace("ConnectorObject build return: {}", SerializerUtil.serializeXmlObject(result, false));
350        }
351        return result;
352    }

354    public Resource build(ConnectorObject source, CryptoService cryptoService) throws IOException, JsonCryptoException {
355        if (null == source) {
356            return null;
357        }
358        if (logger.isTraceEnabled()) {
359            logger.trace("ConnectorObject source: {}", SerializerUtil.serializeXmlObject(source, false));
360        }
361        JsonValue result = new JsonValue(new LinkedHashMap<String, Object>(source.getAttributes().size()));
362        for (AttributeInfoHelper attributeInfo : attributes) {
363            Attribute attribute = source.getAttributeByName(attributeInfo.getAttributeInfo().getName());
364            if (null != attribute && attributeInfo.getAttributeInfo().isReadable()) {
365                result.put(attributeInfo.getName(), attributeInfo.build(attribute, cryptoService));
366            }
367        }
368        Uid uid = source.getUid();
369        // TODO are we going to escape ids?
370        result.put(Resource.FIELD_CONTENT_ID, /*Id.escapeUid(*/uid.getUidValue()/*)*/);
371        if (null != uid.getRevision()) {
372            //System supports Revision
373            result.put(Resource.FIELD_CONTENT_REVISION, uid.getRevision());
374        }
375        return new Resource(uid.getUidValue(), uid.getRevision(), result );
376    }

378    public Attribute build(String attributeName, Object source, CryptoService cryptoService) throws Exception {
379        for (AttributeInfoHelper attributeInfoHelper : attributes) {
380            if (attributeInfoHelper.getName().equals(attributeName)) {
381                return attributeInfoHelper.build(source, cryptoService);
382            }
383        }
384        if (source instanceof Collection) {
385            return AttributeBuilder.build(attributeName, (Collection) source);
386        } else {
387            return AttributeBuilder.build(attributeName, source);
388        }
389    }
