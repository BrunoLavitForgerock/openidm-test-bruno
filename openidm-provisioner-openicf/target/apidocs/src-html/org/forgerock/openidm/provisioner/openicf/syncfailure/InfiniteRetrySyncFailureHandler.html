016package org.forgerock.openidm.provisioner.openicf.syncfailure;
017
018import java.util.Map;
019
020/**
021 * A retry failure handler which retries infinitely.
022 *
023 * @author brmiller
024 */
025public class InfiniteRetrySyncFailureHandler implements SyncFailureHandler {
026    /** an instance of the InfiniteRetrySyncFailureHandler -- there only needs to be one */
027    public static final SyncFailureHandler INSTANCE = new InfiniteRetrySyncFailureHandler();
028
029    // restrict instantiation to the above INSTANCE
030    private InfiniteRetrySyncFailureHandler() {
031    }
032
033    /** 
034     * Handle sync failure by counting retries on this sync token, passing to
035     * (optional) post-retry handler when retries are exceeded.
036     *
037     * @param syncFailure contains sync failure data
038     * @param failureCause the cause of the sync failure
039     * @throws SyncHandlerException when retries are not exceeded
040     */
041    public void invoke(Map<String, Object> syncFailure, Exception failureCause) 
042        throws SyncHandlerException {
043
044        throw new SyncHandlerException("Failed to synchronize " + syncFailure.get("uid") + " object on "
045                + syncFailure.get("systemIdentifier") + ": " + failureCause.getMessage(), failureCause);
046    }
047}
