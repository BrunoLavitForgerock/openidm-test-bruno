016package org.forgerock.openidm.provisioner.openicf.syncfailure;
017
018import org.apache.felix.scr.annotations.Component;
019import org.apache.felix.scr.annotations.ConfigurationPolicy;
020import org.apache.felix.scr.annotations.Reference;
021import org.apache.felix.scr.annotations.ReferencePolicy;
022import org.apache.felix.scr.annotations.Service;
023import org.forgerock.json.fluent.JsonValue;
024import org.forgerock.json.resource.ConnectionFactory;
025import org.forgerock.json.resource.ResourceException;
026import org.forgerock.json.resource.ServerContext;
027import org.forgerock.openidm.core.ServerConstants;
028import org.forgerock.openidm.router.RouteService;
029import org.forgerock.openidm.util.Accessor;
030import org.forgerock.script.ScriptRegistry;
031
032/**
033 * A factory service to create the SyncFailureHandler strategy from config.
034 *
035 * @author bmiller
036 */
037@Component(name = SyncFailureHandlerFactoryImpl.PID,
038        policy = ConfigurationPolicy.IGNORE,
039        metatype = true,
040        description = "OpenIDM Sync Failure Handler Factory Service",
041        immediate = true
042)
043@Service()
044public class SyncFailureHandlerFactoryImpl implements SyncFailureHandlerFactory {
045    public static final String PID = "org.forgerock.openidm.openicf.syncfailure";
046
047    /* config tokens */
048    protected static final String CONFIG_MAX_RETRIES = "maxRetries";
049    protected static final String CONFING_POST_RETRY = "postRetryAction";
050    protected static final String CONFIG_DEAD_LETTER = "dead-letter-queue";
051    protected static final String CONFIG_LOGGED_IGNORE = "logged-ignore";
052    protected static final String CONFIG_SCRIPT = "script";
053
054    /** Script Registry service. */
055    @Reference(policy = ReferencePolicy.DYNAMIC)
056    protected ScriptRegistry scriptRegistry;
057
058    private void bindScriptRegistry(final ScriptRegistry service) {
059        scriptRegistry = service;
060    }
061
062    private void unbindScriptRegistry(final ScriptRegistry service) {
063        scriptRegistry = null;
064    }
065
066    /** The Connection Factory */
067    @Reference(policy = ReferencePolicy.STATIC, target="(service.pid=org.forgerock.openidm.internal)")
068    protected ConnectionFactory connectionFactory;
069
070    /** the router */
071    @Reference(target = "("+ ServerConstants.ROUTER_PREFIX + "=/*)")
072    RouteService routeService;
073    ServerContext routerContext = null;
074
075    private void bindRouteService(final RouteService service) throws ResourceException {
076        routeService = service;
077        routerContext = service.createServerContext();
078    }
079
080    private void unbindRouteService(final RouteService service) {
081        routeService = null;
082        routerContext = null;
083    }
084
085    /**
086     * Create a <em>SyncFailureHandler</em> from the config.  The config should optionally
087     * describe
088     * <ul>
089     *     <li>number of retries</li>
090     *     <li>what to do upon retry exhaustion:
091     *     <ul>
092     *         <li>whether to save the failed sync item to a dead-letter queue</li>
093     *         <li>whether to log and ignore the failure</li>
094     *         <li>whether to call an external, user-supplied script</li>
095     *     </ul>
096     * </ul>
097     *
098     * @param config the config for the SyncFailureHandler
099     * @return the SyncFailureHandler
100     */
101    public SyncFailureHandler create(JsonValue config) throws Exception {
102
103        if (null == config || config.isNull()) {
104            return InfiniteRetrySyncFailureHandler.INSTANCE;
105        }
106
107        JsonValue maxRetries = config.get(CONFIG_MAX_RETRIES);
108        JsonValue postRetry = config.get(CONFING_POST_RETRY);
109
110        if (maxRetries.isNull() || maxRetries.asInteger() < 0) {
111            return InfiniteRetrySyncFailureHandler.INSTANCE;
112        } else if (maxRetries.asInteger() == 0) {
113            return getPostRetryHandler(postRetry);
114        } else {
115            return new SimpleRetrySyncFailureHandler(maxRetries.asInteger(),
116                    getPostRetryHandler(postRetry));
117        }
118    }
119
120    /**
121     * Create the SycFailureHandler to execute when the retries are exhausted (or if no retries are configured).
122     * The config passed here is the value side of
123     * {code}
124     * "postRetryAction" : <value>
125     * {code}
126     * It may be either a String which indicates which handler to use, or a Map which indicates a handler
127     * with more complex configuration.
128     *
129     * @param config the config that further specifies the sync failure handler
130     * @return the SyncFailureHandler
131     */
132    private SyncFailureHandler getPostRetryHandler(JsonValue config) throws Exception {
133        if (config.isString()) {
134            if (CONFIG_DEAD_LETTER.equals(config.asString())) {
135                return new DeadLetterQueueHandler(
136                        connectionFactory, new Accessor<ServerContext>() {
137                            public ServerContext access() {
138                                return routerContext;
139                            }
140                        });
141            } else if (CONFIG_LOGGED_IGNORE.equals(config.asString())) {
142                return new LoggedIgnoreHandler();
143            }
144        }
145        else if (config.isMap()) {
146            if (config.get(CONFIG_SCRIPT).isMap()) {
147                return new ScriptedSyncFailureHandler(
148                        scriptRegistry,
149                        config.get(CONFIG_SCRIPT),
150                        // pass internal handlers so a script can call them if desired
151                        new LoggedIgnoreHandler(),
152                        new DeadLetterQueueHandler(
153                                connectionFactory, new Accessor<ServerContext>() {
154                                    public ServerContext access() {
155                                        return routerContext;
156                                    }
157                                }));
158            }
159        }
160
161        return NullSyncFailureHandler.INSTANCE;
162    }
163}
