079    public boolean isOperationPermitted(Class<? extends APIOperation> operation) throws ResourceException {
080        OperationOptionInfoHelper operationOptionInfoHelper = operations.get(operation);
081        String reason = "not supported.";
082        if (null != operationOptionInfoHelper
083                && (null == operationOptionInfoHelper.getSupportedObjectTypes()
084                    || operationOptionInfoHelper.getSupportedObjectTypes().contains(
085                        objectClassInfoHelper.getObjectClass().getObjectClassValue()))) {
086
087            if (!operationOptionInfoHelper.isDenied()) {
088                return true;
089            } else if (OperationOptionInfoHelper.OnActionPolicy.ALLOW.equals(operationOptionInfoHelper.getOnActionPolicy())) {
090                return false;
091            }
092            reason = "denied.";
093        }
094        throw new ForbiddenException("Operation " + operation.getCanonicalName() + " is " + reason);
095    }

126    public JsonValue build(ConnectorObject source) throws Exception {
127        JsonValue result = objectClassInfoHelper.build(source, cryptoService).getContent();
128        resetUid(source.getUid(), result);
129        if (null != source.getUid().getRevision()) {
130            //System supports Revision
131            result.put(Resource.FIELD_CONTENT_REVISION, source.getUid().getRevision());
132        }
133        return result;
134    }

143    /**
144     * Generate the fully qualified id from unqualified object {@link org.identityconnectors.framework.common.objects.Uid}
145     * <p/>
146     * The result id will be system/{@code [endSystemName]}/{@code [objectType]}/{@code [escapedObjectId]}
147     *
148     * @param uid original un escaped unique identifier of the object
149     * @return
150     */
151    public URI resolveQualifiedId(Uid uid) {
152        if (null != uid) {
153            try {
154                return systemObjectSetId.resolveLocalId(uid.getUidValue()).getQualifiedId();
155            } catch (ResourceException e) {
156                // Should never happen in a copy constructor.
157                throw new UndeclaredThrowableException(e);
158            }
159        } else {
160            return systemObjectSetId.getQualifiedId();
161        }
162    }
