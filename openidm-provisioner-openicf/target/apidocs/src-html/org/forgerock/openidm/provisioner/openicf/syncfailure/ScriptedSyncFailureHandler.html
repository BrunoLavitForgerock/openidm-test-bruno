016package org.forgerock.openidm.provisioner.openicf.syncfailure;
017
018import java.util.HashMap;
019import java.util.Map;
020
021import org.forgerock.json.fluent.JsonValue;
022import org.forgerock.json.resource.RootContext;
023import org.forgerock.script.Script;
024import org.forgerock.script.ScriptEntry;
025import org.forgerock.script.ScriptRegistry;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029import javax.script.ScriptException;
030
031/**
032 * Handle liveSync failure by calling an external, user-supplied script.
033 *
034 * @author brmiller
035 */
036public class ScriptedSyncFailureHandler implements SyncFailureHandler {
037
038    /** Logger */
039    private static final Logger logger = LoggerFactory.getLogger(ScriptedSyncFailureHandler.class);
040
041    /** the script to call */
042    private final ScriptEntry scriptEntry;
043
044    /** map of references to built-in sync failure handlers made available to user-supplied script */
045    private final Map<String,SyncFailureHandler> builtInHandlers;
046
047    /**
048     * Construct this sync failure handler.
049     *
050     * @param config the config
051     */
052    public ScriptedSyncFailureHandler(ScriptRegistry scriptRegistry, JsonValue config, SyncFailureHandler... builtInHandlers)
053        throws ScriptException {
054
055        this.scriptEntry = scriptRegistry.takeScript(config);
056        this.builtInHandlers = new HashMap<String,SyncFailureHandler>();
057        for (SyncFailureHandler handler : builtInHandlers) {
058            if (handler instanceof LoggedIgnoreHandler) {
059                this.builtInHandlers.put("loggedIgnore", handler);
060            } else if (handler instanceof DeadLetterQueueHandler) {
061                this.builtInHandlers.put("deadLetterQueue", handler);
062            }
063        }
064    }
065
066    /**
067     * Handle sync failure by counting retries on this sync token, passing to
068     * (optional) post-retry handler when retries are exceeded.
069     *
070     * @param syncFailure @throws SyncHandlerException when retries are not exceeded
071     * @param failureCause the cause of the sync failure
072     */
073    public void invoke(Map<String, Object> syncFailure, Exception failureCause)
074        throws SyncHandlerException {
075
076        if (null == scriptEntry) {
077            throw new SyncHandlerException("No script registered");
078        }
079
080        Script script = scriptEntry.getScript(new RootContext());
081        script.put("syncFailure", syncFailure);
082        script.put("failureCause", failureCause);
083        script.put("failureHandlers", builtInHandlers);
084        try {
085            script.eval();
086        } catch (Exception e) {
087            logger.debug("sync failure script on {} encountered exception", syncFailure.get("systemIdentifier"), e);
088            throw new SyncHandlerException("Issue with handling the failure during synchronize "
089                    + syncFailure.get("uid") + " object: " + failureCause.getMessage()
090                    + ". Failure handling reported " + e.getMessage(), e);
091        }
092    }
093}
