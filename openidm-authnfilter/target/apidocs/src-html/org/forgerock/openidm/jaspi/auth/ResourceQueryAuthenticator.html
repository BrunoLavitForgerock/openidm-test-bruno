017package org.forgerock.openidm.jaspi.auth;
018
019import org.eclipse.jetty.plus.jaas.spi.UserInfo;
020import org.eclipse.jetty.util.security.Password;
021import org.forgerock.json.resource.ConnectionFactory;
022import org.forgerock.json.resource.QueryRequest;
023import org.forgerock.json.resource.Requests;
024import org.forgerock.json.resource.Resource;
025import org.forgerock.json.resource.ResourceException;
026import org.forgerock.json.resource.ServerContext;
027import org.forgerock.openidm.crypto.CryptoService;
028import org.forgerock.util.Reject;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import java.util.HashSet;
033import java.util.Set;
034
035/**
036 * Authenticator class which performs authentication against managed/internal user tables using a queryId to fetch
037 * the complete local user data and validates the password locally.
038 */
039public class ResourceQueryAuthenticator implements Authenticator {
040
041    private static final Logger logger = LoggerFactory.getLogger(ResourceQueryAuthenticator.class);
042
043    private final CryptoService cryptoService;
044    private final ConnectionFactory connectionFactory;
045    private final String queryOnResource;
046    private final String queryId;
047    private final String authenticationIdProperty;
048    private final String userCredentialProperty;
049
050    /**
051     * Constructs an instance of the ResourceQueryAuthenticator.
052     *
053     * @param cryptoService The CryptoService.
054     * @param connectionFactory The ConnectionFactory.
055     * @param queryOnResource The query resource.
056     * @param queryId The query id.
057     * @param authenticationIdProperty The user id property.
058     * @param userCredentialProperty The user credential property.
059     */
060    public ResourceQueryAuthenticator(CryptoService cryptoService, ConnectionFactory connectionFactory,
061            String queryOnResource, String queryId,  String authenticationIdProperty, String userCredentialProperty) {
062
063        Reject.ifNull(cryptoService, "CryptoService is null");
064        Reject.ifNull(connectionFactory, "ConnectionFactory is null");
065        Reject.ifNull(queryOnResource, "User query resource was null");
066        Reject.ifNull(queryId, "Credential query was null");
067        Reject.ifNull(authenticationIdProperty, "authenticationId property is not defined");
068        Reject.ifNull(userCredentialProperty, "userCredential property is not defined");
069
070        this.cryptoService = cryptoService;
071        this.connectionFactory = connectionFactory;
072        this.queryOnResource = queryOnResource;
073        this.queryId = queryId;
074        this.authenticationIdProperty = authenticationIdProperty;
075        this.userCredentialProperty = userCredentialProperty;
076    }
077
078    /**
079     * Performs the authentication using the given query id, resource, username and password.
080     *
081     * @param username The username.
082     * @param password The password.
083     * @param context the ServerContext to use
084     * @return True if authentication is successful, otherwise false.
085     */
086    public boolean authenticate(String username, String password, ServerContext context) throws ResourceException {
087
088        Reject.ifNull(username, "Provided username was null");
089        Reject.ifNull(context, "Router context was null");
090
091        final UserInfo userInfo = getRepoUserInfo(username, context);
092        if (userInfo == null) {
093            return false; // getRepoUserInfo already logged why
094        } else if (userInfo.checkCredential(password)) {
095            logger.debug("Authentication succeeded for {}", username);
096            return true;
097        } else {
098            logger.debug("Authentication failed for {} due to invalid credentials", username);
099            return false;
100        }
101    }
102
103    private UserInfo getRepoUserInfo(String username, ServerContext context) throws ResourceException {
104        QueryRequest request = Requests.newQueryRequest(queryOnResource)
105                .setQueryId(queryId)
106                .setAdditionalParameter(authenticationIdProperty, username);
107
108        final Set<Resource> result = new HashSet<Resource>();
109        connectionFactory.getConnection().query(context, request, result);
110
111        if (result.size() == 0) {
112            logger.debug("Query to match user credentials found no user matching {}", username);
113            return null;
114        }
115
116        if (result.size() > 1) {
117            logger.debug("Query to match user credentials found more than one matching user for {}", username);
118            for (Resource entry : result) {
119                logger.debug("Ambiguous matching username for {} found id: {}", username, entry.getId());
120            }
121            throw ResourceException.getException(401, "Access denied, user detail retrieved was ambiguous.");
122        }
123
124        final Resource resource = result.iterator().next();
125        final String retrievedCred =
126                cryptoService.decryptIfNecessary(resource.getContent().get(userCredentialProperty)).asString();
127
128        return new UserInfo(username, new Password(retrievedCred), null);
129    }
130}
