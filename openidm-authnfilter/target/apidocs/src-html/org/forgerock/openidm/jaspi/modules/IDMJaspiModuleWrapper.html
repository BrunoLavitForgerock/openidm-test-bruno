058/**
059 * A Jaspi ServerAuthModule that is designed to wrap any other Jaspi ServerAuthModule. This module provides
060 * IDM specific authentication processing to the authentication mechanism of underlying auth module.
061 * <br/>
062 * This allows IDM to use any common auth module and still benefit from automatic role calculation
063 * and augment security context scripts (providing the authentication.json contains the required configuration).
064 *
065 * @since 3.0.0
066 */
067public class IDMJaspiModuleWrapper implements ServerAuthModule {
068

104     * Constructs a new instance of the IDMJaspiModuleWrapper.
105     */
106    public IDMJaspiModuleWrapper() throws AuthException {
107        this.authnFilterHelper = OSGiAuthnFilterBuilder.getInstance();
108        if (authnFilterHelper == null) {
109            throw new AuthException("OSGiAuthnFilterHelper is not ready.");
110        }
111        this.authModuleConstructor = new AuthModuleConstructor();
112        this.roleCalculatorFactory = new RoleCalculatorFactory();
113        this.augmentationScriptExecutor = new AugmentationScriptExecutor(authnFilterHelper);
114    }

144    /**
145     * Initialises the underlying auth module with the provided parameters and constructs an instance
146     * of the RoleCalculator from the authentication configuration.
147     * <br/>
148     * Required configuration:
149     * <ul>
150     *     <li>connectionFactory - the ConnectionFactory for making an authenticate request on the router</li>
151     *     <li>context - the ServerContext to use when making requests on the router</li>
152     *     <li>queryOnResource - the resource to perform the role calculation query on</li>
153     *     <li>authenticationId - the object attribute that represents the authentication id</li>
154     *     <li>groupMembership - the object attribute representing the group membership</li>
155     *     <li>defaultRoles - the list of default roles</li>
156     *     <li>roleMapping - the mapping between OpenIDM roles and pass-through auth groups</li>
157     *     <li>groupComparison - the method of {@link GroupComparison} to use</li>
158     * </ul>
159     *
160     * @param requestMessagePolicy {@inheritDoc}
161     * @param responseMessagePolicy {@inheritDoc}
162     * @param handler {@inheritDoc}
163     * @param options {@inheritDoc}
164     * @throws AuthException {@inheritDoc}
165     */
166    @Override
167    public void initialize(MessagePolicy requestMessagePolicy, MessagePolicy responseMessagePolicy,
168            CallbackHandler handler, Map options) throws AuthException {
169
170        properties = new JsonValue(options);
171        authModule = authModuleConstructor.construct(properties.get("authModuleClassName").asString());
172        authModule.initialize(requestMessagePolicy, responseMessagePolicy, handler, options);
173
174        logClientIPHeader = properties.get("clientIPHeader").asString();
175
176        queryOnResource = properties.get(QUERY_ON_RESOURCE).asString();
177
178        String queryId = properties.get(QUERY_ID).asString();
179        String authenticationId = properties.get(PROPERTY_MAPPING).get(AUTHENTICATION_ID).asString();
180
181        String userRoles = properties.get(PROPERTY_MAPPING).get(USER_ROLES).asString();
182        String groupMembership = properties.get(PROPERTY_MAPPING).get(GROUP_MEMBERSHIP).asString();
183        List<String> defaultRoles = properties.get(DEFAULT_USER_ROLES)
184                .defaultTo(Collections.emptyList())
185                .asList(String.class);
186        Map<String, List<String>> roleMapping = properties.get(GROUP_ROLE_MAPPING)
187                .defaultTo(Collections.emptyMap())
188                .asMapOfList(String.class);
189        MappingRoleCalculator.GroupComparison groupComparison = properties.get(GROUP_COMPARISON_METHOD)
190                .defaultTo(MappingRoleCalculator.GroupComparison.equals.name())
191                .asEnum(MappingRoleCalculator.GroupComparison.class);
192
193        // a function to perform the user detail query on the router
194        queryExecutor =
195                new Function<QueryRequest, Resource, ResourceException>() {
196                    @Override
197                    public Resource apply(QueryRequest request) throws ResourceException {
198                        if (request == null) {
199                            return null;
200                        }
201                        final List<Resource> resources = new ArrayList<Resource>();
202                        authnFilterHelper.getConnectionFactory().getConnection().query(
203                                authnFilterHelper.getRouter().createServerContext(), request, resources);
204
205                        if (resources.isEmpty()) {
206                            throw ResourceException.getException(401, "Access denied, no user detail could be retrieved.");
207                        } else if (resources.size() > 1) {
208                            throw ResourceException.getException(401, "Access denied, user detail retrieved was ambiguous.");
209                        }
210                        return resources.get(0);
211                    }
212                };
213
214        queryBuilder = new UserDetailQueryBuilder(queryOnResource)
215                .useQueryId(queryId)
216                .withAuthenticationIdProperty(authenticationId);
217
218        roleCalculator = roleCalculatorFactory.create(defaultRoles, userRoles, groupMembership, roleMapping, groupComparison);
219
220        JsonValue scriptConfig =  properties.get(SERVLET_FILTER_AUGMENT_SECURITY_CONTEXT);
221        if (!scriptConfig.isNull()) {
222            augmentScript = getAugmentScript(scriptConfig);
223            logger.debug("Registered script {}", augmentScript);
224        }
225    }

227    /**
228     * Gets the ScriptEntry for the specified script config.
229     *
230     * @param scriptConfig The script config.
231     * @return The ScriptEntry.
232     * @throws JaspiAuthException If there is a problem retrieving the ScriptEntry.
233     */
234    ScriptEntry getAugmentScript(JsonValue scriptConfig) throws JaspiAuthException {
235        try {
236            return authnFilterHelper.getScriptRegistry().takeScript(scriptConfig);
237        } catch (ScriptException e) {
238            logger.error("{} when attempting to register script {}", e.toString(), scriptConfig, e);
239            throw new JaspiAuthException(e.toString(), e);
240        }
241    }

243    /**
244     * Provides IDM specific authentication process handling, by setting whether to log the client's IP address,
245     * and then calls the underlying auth module's validateRequest method. If the auth module returns
246     * SUCCESS, based on the authentication configuration will perform role calculation and, if present, will run the
247     * augment security context script.
248     *
249     * @param messageInfo {@inheritDoc}
250     * @param clientSubject {@inheritDoc}
251     * @param serviceSubject {@inheritDoc}
252     * @return {@inheritDoc}
253     * @throws AuthException {@inheritDoc}
254     */
255    @SuppressWarnings("unchecked")
256    @Override
257    public AuthStatus validateRequest(MessageInfo messageInfo, Subject clientSubject, Subject serviceSubject)
258            throws AuthException {
259
260        Map<String, Object> messageInfoParams = messageInfo.getMap();
261
262        // Add this properties so the AuditLogger knows whether to log the client IP in the header.
263        messageInfoParams.put(IDMAuthenticationAuditLogger.LOG_CLIENT_IP_HEADER_KEY, logClientIPHeader);
264
265        final AuthStatus authStatus = authModule.validateRequest(messageInfo, clientSubject, serviceSubject);
266
267        if (!AuthStatus.SUCCESS.equals(authStatus)) {
268            return authStatus;
269        }
270
271        String principalName = null;
272        for (Principal principal : clientSubject.getPrincipals()) {
273            if (principal.getName() != null) {
274                principalName = principal.getName();
275                break;
276            }
277        }
278
279        if (principalName == null) {
280            // As per Jaspi spec, the module developer MUST ensure that the client
281            // subject's principal is set when the module returns SUCCESS.
282            throw new JaspiAuthException("Underlying Server Auth Module has not set the client subject's principal!");
283        }
284
285        // user is authenticated; populate security context
286
287        try {
288            // Attempt to read the user object; will return null if any of the pieces are null
289            final Resource resource = queryExecutor.apply(queryBuilder.forPrincipal(principalName).build());
290
291            final SecurityContextMapper securityContextMapper = SecurityContextMapper.fromMessageInfo(messageInfo)
292                    .setAuthenticationId(principalName);
293
294            // Calculate (and set) roles if not already set
295            if (securityContextMapper.getRoles() == null || securityContextMapper.getRoles().isEmpty()) {
296                roleCalculator.calculateRoles(principalName, securityContextMapper, resource);
297            }
298            
299            // set "resource" (component) if not already set
300            if (securityContextMapper.getResource() == null) {
301                securityContextMapper.setResource(queryOnResource);
302            }
303
304            // set "user id" (authorization.id) if not already set
305            if (securityContextMapper.getUserId() == null) {
306                if (resource != null) {
307                    // assign authorization id from resource if present
308                    securityContextMapper.setUserId(
309                            resource.getId() != null
310                                    ? resource.getId()
311                                    : resource.getContent().get(FIELD_CONTENT_ID).asString());
312                } else {
313                    // set to principal otherwise
314                    securityContextMapper.setUserId(principalName);
315                }
316            }
317
318            // run the augmentation script, if configured (will no-op if none specified)
319            augmentationScriptExecutor.executeAugmentationScript(augmentScript, properties, securityContextMapper);
320
321        } catch (ResourceException e) {
322            if (logger.isDebugEnabled()) {
323                logger.debug("Failed role calculation for {} on {}.", principalName, queryOnResource, e);
324            }
325            if (e.isServerError()) { // HTTP server-side error; AuthException sadly does not accept cause
326                throw new JaspiAuthException("Failed pass-through authentication of " + principalName + " on "
327                        + queryOnResource + ":" + e.getMessage(), e);
328            }
329            // role calculation failed
330            return AuthStatus.SEND_FAILURE;
331        }
332
333        return authStatus;
334    }

336    /**
337     * If the request contains the X-OpenIDM-NoSession header, sets the skipSession property on the MessageInfo,
338     * and then calls the underlying auth module's secureResponse method.
339     *
340     * @param messageInfo {@inheritDoc}
341     * @param serviceSubject {@inheritDoc}
342     * @return {@inheritDoc}
343     * @throws AuthException {@inheritDoc}
344     */
345    @SuppressWarnings("unchecked")
346    @Override
347    public AuthStatus secureResponse(MessageInfo messageInfo, Subject serviceSubject) throws AuthException {
348
349        final HttpServletRequest request = (HttpServletRequest) messageInfo.getRequestMessage();
350        final String noSession = request.getHeader(NO_SESSION);
351
352        if (Boolean.parseBoolean(noSession)) {
353            messageInfo.getMap().put("skipSession", true);
354        }
355
356        return authModule.secureResponse(messageInfo, serviceSubject);
357    }

371    /**
372     * Constructs Server Auth Modules from a given class name, using the mandatory no-arg constructor.
373     *
374     * @since 3.0.0
375     */
376    static class AuthModuleConstructor {
377
378        /**
379         * Creates an instance of a Server Auth Module for the specified class name.
380         * <br/>
381         * Uses the spec mandated no-arg constructor.
382         *
383         * @param authModuleClassName The ServerAuthModule class name.
384         * @return The ServerAuthModule instance.
385         * @throws JaspiAuthException If there is any problem creating the ServerAuthModule instance.
386         */
387        ServerAuthModule construct(String authModuleClassName) throws JaspiAuthException {
388             try {
389                 return Class.forName(authModuleClassName).asSubclass(ServerAuthModule.class).newInstance();
390             } catch (ClassNotFoundException e) {
391                 logger.error("Failed to construct Auth Module instance", e);
392                 throw new JaspiAuthException("Failed to construct Auth Module instance", e);
393             } catch (InstantiationException e) {
394                 logger.error("Failed to construct Auth Module instance", e);
395                 throw new JaspiAuthException("Failed to construct Auth Module instance", e);
396             } catch (IllegalAccessException e) {
397                 logger.error("Failed to construct Auth Module instance", e);
398                 throw new JaspiAuthException("Failed to construct Auth Module instance", e);
399             }
400        }
401    }

403    /**
404     * QueryRequest Builder class for querying the user object detail.
405     * <p>
406     * If queryId is provided, build() will set additional parameters for the authenticationId
407     * property and principal name.  Otherwise a QueryFilter where "authenticationId property = principal name"
408     * is used.
409     *
410     * @since 3.0.0
411     */
412    private static class UserDetailQueryBuilder {
413        private final String queryOnResource;
414        private String queryId = null;
415        private String authenticationId = null;
416        private String principal = null;
417
418        private UserDetailQueryBuilder(final String queryOnResource) {
419            this.queryOnResource = queryOnResource;
420        }
421
422        UserDetailQueryBuilder useQueryId(String queryId) {
423            this.queryId = queryId;
424            return this;
425        }
426
427        UserDetailQueryBuilder withAuthenticationIdProperty(final String authenticationId) {
428            this.authenticationId = authenticationId;
429            return this;
430        }
431
432        UserDetailQueryBuilder forPrincipal(final String principal) {
433            this.principal = principal;
434            return this;
435        }
436
437        QueryRequest build() throws BadRequestException {
438            if (queryOnResource == null || authenticationId == null || principal == null) {
439                return null;
440            }
441            QueryRequest request = Requests.newQueryRequest(queryOnResource);
442            if (queryId != null) {
443                // if we're using a queryId, provide the additional parameter mapping the authenticationId property
444                // and its value (the principal)
445                request.setQueryId(queryId);
446                request.setAdditionalParameter(authenticationId, principal);
447            } else {
448                // otherwise, use a query filter mapping the authenticationId property to the principal
449                request.setQueryFilter(QueryFilter.equalTo(authenticationId, principal));
450            }
451            return request;
452        }
453    }

455    /**
456     * Internal credential bean to hold username/password pair.
457     *
458     * @since 3.0.0
459     */
460    static class Credential {
461        final String username;
462        final String password;
463
464        Credential(final String username, final String password) {
465            this.username = username;
466            this.password = password;
467        }
468
469        boolean isComplete() {
470            return (!StringUtils.isEmpty(username) && !StringUtils.isEmpty(password));
471        }
472    }

483    /** CredentialHelper to get auth header creds from request */
484    static final CredentialHelper HEADER_AUTH_CRED_HELPER = new CredentialHelper() {
485        /** Authentication username header. */
486        private static final String HEADER_USERNAME = "X-OpenIDM-Username";
487
488        /** Authentication password header. */
489        private static final String HEADER_PASSWORD = "X-OpenIDM-Password";
490
491        @Override
492        public Credential getCredential(HttpServletRequest request) {
493            return new Credential(request.getHeader(HEADER_USERNAME), request.getHeader(HEADER_PASSWORD));
494        }
495    };
496
497    /** CredentialHelper to get Basic-Auth creds from request */
498    static final CredentialHelper BASIC_AUTH_CRED_HELPER  = new CredentialHelper() {
499        /** Basic auth header. */
500        private static final String HEADER_AUTHORIZATION = "Authorization";
501        private static final String AUTHORIZATION_HEADER_BASIC = "Basic";
502
503        @Override
504        public Credential getCredential(HttpServletRequest request) {
505            final String authHeader = request.getHeader(HEADER_AUTHORIZATION);
506            if (authHeader != null) {
507                final String[] authValue = authHeader.split("\\s", 2);
508                if (AUTHORIZATION_HEADER_BASIC.equalsIgnoreCase(authValue[0]) && authValue[1] != null) {
509                    final String[] creds = new String(Base64.decodeBase64(authValue[1].getBytes())).split(":");
510                    if (creds.length == 2) {
511                        return new Credential(creds[0], creds[1]);
512                    }
513                }
514            }
515            return new Credential(null, null);
516        }
517    };
