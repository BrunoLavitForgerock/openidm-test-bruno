060/**
061 * Configures the authentication chains based on configuration obtained through OSGi.
062 *
063 * Example:
064 *
065 * <pre>
066 *     <code>
067 * {
068 *     "serverAuthConfig" : {
069 *         "scriptExtensions" : {
070 *             "augmentSecurityContext" : {
071 *                 "type" : "text/javascript",
072 *                 "file" : "auth/authnPopulateContext.js"
073 *             }
074 *         },
075 *         "sessionModule" : {
076 *             "name" : "JWT_SESSION",
077 *                 "properties" : {
078 *                     "someSetting" : "some-value"
079 *                 }
080 *         },
081 *         "authModules" : [
082 *             {
083 *                 "name" : "IWA",
084 *                 "properties" : {
085 *                     "someSetting" : "some-value"
086 *                 }
087 *             },
088 *             {
089 *                 "name" : "PASSTHROUGH",
090 *                 "properties" : {
091 *                     "someSetting" : "some-value"
092 *                 }
093 *             }
094 *         ]
095 *     }
096 * }
097 *     </code>
098 * </pre>
099 *
100 * Also,
101 * <pre>
102 *     <code>
103 *         OSGiAuthnFilterBuilder.getInstance()
104 *     </code>
105 * </pre>
106 * may be used to inject the OSGi dependencies into other modules as an OSGiAuthnFilterHelper.
107 *
108 * @author Jonathan Scudder
109 * @author Phill Cunnington
110 * @author brmiller
111 * @author ckienle
112 */
113@Component(name = OSGiAuthnFilterBuilder.PID, immediate = true, policy = ConfigurationPolicy.IGNORE)
114@Properties({
115        @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
116        @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM Commons Authentication Filter Configuration")
117})
118public class OSGiAuthnFilterBuilder implements OSGiAuthnFilterHelper {
119
120    /** The PID for this component. */
121    public static final String PID = "org.forgerock.openidm.authnfilterbuilder";
122
123    // config tokens
124    private static final String CONFIG_ADDITIONAL_URL_PATTERNS = "additionalUrlPatterns";
125
126    private static final int DEFAULT_FILTER_ORDER = 100;
127
128    private Logger logger = LoggerFactory.getLogger(this.getClass());
129
130    private static OSGiAuthnFilterBuilder instance;
131
132    // ----- Declarative Service Implementation
133
134    @Reference(
135            name = "AuthenticationConfig",
136            referenceInterface = AuthenticationConfig.class,
137            policy = ReferencePolicy.STATIC,
138            cardinality = ReferenceCardinality.MANDATORY_UNARY,
139            bind = "bindAuthenticationConfig",
140            unbind = "unBindAuthenticationConfig"
141    )
142    private JsonValue config;
143    private void bindAuthenticationConfig(AuthenticationConfig authenticationConfig) {
144        config = authenticationConfig.getConfig();
145    }
146    private void unBindAuthenticationConfig(AuthenticationConfig authenticationConfig) {
147        config = null;
148    }
149
150    @Reference(target = "("+ServerConstants.ROUTER_PREFIX+"=/repo/*)")
151    RouteService repositoryRoute;
152
153    @Reference(policy = ReferencePolicy.DYNAMIC)
154    CryptoService cryptoService;
155
156    @Reference(policy = ReferencePolicy.STATIC, target="(service.pid=org.forgerock.openidm.internal)")
157    ConnectionFactory connectionFactory;
158
159    /** Script Registry service. */
160    @Reference(policy = ReferencePolicy.DYNAMIC)
161    protected ScriptRegistry scriptRegistry;
162
163    @Reference(
164            name = "ref_ServletFilterRegistration",
165            referenceInterface = ServletRegistration.class,
166            policy = ReferencePolicy.STATIC,
167            cardinality = ReferenceCardinality.MANDATORY_UNARY
168    )
169    ServletRegistration servletFilterRegistration;
170
171    private RegisteredFilter filter;
172
173    /**
174     * Configures the commons Authentication Filter with the configuration in the authentication.json file.
175     *
176     * @param context The ComponentContext.
177     */
178    @Activate
179    protected void activate(ComponentContext context) throws Exception {
180        instance = this;
181        JsonValue scriptExtensions = config.get(SERVLET_FILTER_SCRIPT_EXTENSIONS);
182        List<String> additionalUrlPatterns = getAdditionalUrlPatterns(config);
183        configureAuthenticationFilter(config);
184
185        registerAuthnFilter(scriptExtensions, additionalUrlPatterns);
186    }
187
188    /**
189     * Gets the additional url parameters value out of the authentication.json configuration.
190     *
191     * @param config The authentication json configuration.
192     * @return The additional url parameters, if any.
193     */
194    private List<String> getAdditionalUrlPatterns(final JsonValue config) {
195        List<String> additionalUrlPatterns = config
196                .get(SERVER_AUTH_CONTEXT_KEY)
197                .get(CONFIG_ADDITIONAL_URL_PATTERNS)
198                .defaultTo(new ArrayList<String>(0))
199                .asList(String.class);
200        config.get(SERVER_AUTH_CONTEXT_KEY).remove(CONFIG_ADDITIONAL_URL_PATTERNS);
201        return additionalUrlPatterns;
202    }
203
204    /**
205     * Configures the commons Authentication Filter with the given configuration.
206     *
207     * @param jsonConfig The authentication configuration.
208     */
209    private void configureAuthenticationFilter(JsonValue jsonConfig) throws Exception {
210
211        if (jsonConfig == null) {
212            // No configurations found
213            logger.warn("Could not find any configurations for the AuthnFilter, filter will not function");
214            return;
215        }
216        JaspiRuntimeConfigurationFactory.INSTANCE.setModuleConfiguration(jsonConfig, this);
217    }
218
219    /**
220     * Registers the Authentication Filter in OSGi.
221     *
222     * @param scriptExtensions A map of script extensions to register for the filter, namely,
223     *                         the location of the authn augment context script.
224     * @param additionalUrlPatterns additional url patterns to which to apply the filter
225     * @throws Exception If a problem occurs whilst registering the filter.
226     */
227    private void registerAuthnFilter(JsonValue scriptExtensions, List<String> additionalUrlPatterns)
228            throws Exception {
229
230        Map<String, String> initParams = new HashMap<String, String>();
231        initParams.put("module-configuration-factory-class", JaspiRuntimeConfigurationFactory.class.getName());
232        initParams.put("logging-configurator-class", JaspiRuntimeConfigurationFactory.class.getName());
233
234        List<String> urlPatterns = new ArrayList<String>();
235        urlPatterns.add("/openidm/*");
236        urlPatterns.addAll(additionalUrlPatterns);
237
238        Map<String, Object> filterConfig = new HashMap<String, Object>();
239        filterConfig.put(SERVLET_FILTER_CLASS_PATH_URLS, new ArrayList<String>());
240        filterConfig.put(SERVLET_FILTER_SYSTEM_PROPERTIES, new HashMap<String, Object>());
241        filterConfig.put(SERVLET_FILTER_PRE_INVOKE_ATTRIBUTES, new HashMap<String, Object>());
242        filterConfig.put(SERVLET_FILTER_INIT_PARAMETERS, initParams);
243        filterConfig.put(SERVLET_FILTER_URL_PATTERNS, urlPatterns);
244        filterConfig.put(SERVLET_FILTER_CLASS, JaspiRuntimeFilter.class.getCanonicalName());
245        filterConfig.put(FILTER_ORDER, DEFAULT_FILTER_ORDER);
246        if (!scriptExtensions.isNull() && scriptExtensions.isMap()) {
247            filterConfig.put(SERVLET_FILTER_SCRIPT_EXTENSIONS, scriptExtensions.asMap());
248        }
249
250        JsonValue filterConfigJson = new JsonValue(filterConfig);
251
252        filter = servletFilterRegistration.registerFilter(filterConfigJson);
253    }
254
255    /**
256     * Unregisters the authentication filter.
257     *
258     * @param context The ComponentContext.
259     */
260    @Deactivate
261    protected synchronized void deactivate(ComponentContext context) {
262        if (filter != null) {
263            try {
264                servletFilterRegistration.unregisterFilter(filter);
265                JaspiRuntimeConfigurationFactory.INSTANCE.clear();
266                logger.info("Unregistered authentication filter.");
267            } catch (Exception ex) {
268                logger.warn("Failure reported during unregistering of authentication filter: {}", ex.getMessage(), ex);
269            }
270        }
271        instance = null;
272    }
273
274    /**
275     * Get this instance of the OSGiAuthnFilterHelper.
276     *
277     * @return the instance of the OSGiAuthnFilterHelper.
278     */
279    public static OSGiAuthnFilterHelper getInstance() {
280        return instance;
281    }
