017package org.forgerock.openidm.jaspi.modules;
018
019import org.forgerock.auth.common.AuditRecord;
020import org.forgerock.auth.common.AuthResult;
021import org.forgerock.jaspi.logging.JaspiAuditLogger;
022import org.forgerock.jaspi.runtime.JaspiRuntime;
023import org.forgerock.json.fluent.JsonValue;
024import org.forgerock.json.resource.CreateRequest;
025import org.forgerock.json.resource.Requests;
026import org.forgerock.json.resource.ResourceException;
027import org.forgerock.json.resource.SecurityContext;
028import org.forgerock.json.resource.ServerContext;
029import org.forgerock.json.resource.servlet.SecurityContextFactory;
030import org.forgerock.openidm.audit.util.Status;
031import org.forgerock.openidm.jaspi.config.OSGiAuthnFilterHelper;
032import org.forgerock.openidm.util.DateUtil;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036import javax.security.auth.message.MessageInfo;
037import javax.servlet.http.HttpServletRequest;
038import java.util.HashMap;
039import java.util.List;
040import java.util.Map;
041
042/**
043 * Creates audit entries for each authentication attempt.
044 *
045 * @author Phill Cunnington
046 */
047public class IDMAuthenticationAuditLogger implements JaspiAuditLogger {
048
049    private static final Logger LOGGER = LoggerFactory.getLogger(IDMAuthenticationAuditLogger.class);
050
051    private static final DateUtil DATE_UTIL = DateUtil.getDateUtil("UTC");
052    public static final String LOG_CLIENT_IP_HEADER_KEY = "logClientIPHeader";
053
054    private final OSGiAuthnFilterHelper authnFilterHelper;
055
056    public IDMAuthenticationAuditLogger(OSGiAuthnFilterHelper authnFilterHelper) {
057        this.authnFilterHelper = authnFilterHelper;
058    }
059
060    /**
061     * {@inheritDoc}
062     */
063    @SuppressWarnings("unchecked")
064    @Override
065    public void audit(AuditRecord<MessageInfo> auditRecord) {
066
067        MessageInfo messageInfo = auditRecord.getAuditObject();
068
069        HttpServletRequest request = (HttpServletRequest) messageInfo.getRequestMessage();
070        Map<String, Object> messageInfoParams = messageInfo.getMap();
071        Map<String, Object> map = (Map<String, Object>) messageInfoParams.get(JaspiRuntime.ATTRIBUTE_AUTH_CONTEXT);
072
073        String username = (String) messageInfoParams.get(SecurityContextFactory.ATTRIBUTE_AUTHCID);
074        String userId = (String) map.get(SecurityContext.AUTHZID_ID);
075        List<String> roles = (List<String>) map.get(SecurityContext.AUTHZID_ROLES);
076
077        boolean status = AuthResult.SUCCESS.equals(auditRecord.getAuthResult());
078        String logClientIPHeader = (String) messageInfoParams.get(LOG_CLIENT_IP_HEADER_KEY);
079        logAuthRequest(request, username, userId, roles, status, logClientIPHeader);
080    }
081
082    /**
083     * Logs the authentication request.
084     *
085     * @param request The HttpServletRequest that made the authentication request.
086     * @param username The username of the user that made the authentication request.
087     * @param userId The user id of the user that made the authentication request.
088     * @param roles The roles of the user that made the authentication request.
089     * @param status The status of the authentication request, either true for success or false for failure.
090     * @param logClientIPHeader Whether to log the client IP in the header.
091     */
092    protected void logAuthRequest(HttpServletRequest request, String username, String userId, List<String> roles,
093            boolean status, String logClientIPHeader) {
094        try {
095            JsonValue entry = new JsonValue(new HashMap<String, Object>());
096            entry.put("timestamp", DATE_UTIL.now());
097            entry.put("action", "authenticate");
098            entry.put("status", status ? Status.SUCCESS.toString() : Status.FAILURE.toString());
099            entry.put("principal", username);
100            entry.put("userid", userId);
101            entry.put("roles", roles);
102            // check for header sent by load balancer for IPAddr of the client
103            String ipAddress;
104            if (logClientIPHeader == null) {
105                ipAddress = request.getRemoteAddr();
106            } else {
107                ipAddress = request.getHeader(logClientIPHeader);
108                if (ipAddress == null) {
109                    ipAddress = request.getRemoteAddr();
110                }
111            }
112            entry.put("ip", ipAddress);
113            if (authnFilterHelper.getRouter() != null) {
114                // TODO We need Context!!!
115                CreateRequest createRequest = Requests.newCreateRequest("/audit/access", entry);
116                ServerContext ctx = authnFilterHelper.getRouter().createServerContext();
117                authnFilterHelper.getConnectionFactory().getConnection().create(ctx, createRequest);
118            } else {
119                // Filter should have rejected request if router is not available
120                LOGGER.warn("Failed to log entry for {} as router is null.", username);
121            }
122        } catch (ResourceException e) {
123            LOGGER.warn("Failed to log entry for {}", username, e);
124        }
125    }
126}
