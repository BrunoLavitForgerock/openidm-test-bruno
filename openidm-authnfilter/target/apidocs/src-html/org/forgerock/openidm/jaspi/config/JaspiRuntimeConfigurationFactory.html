017package org.forgerock.openidm.jaspi.config;
018
019import org.forgerock.auth.common.DebugLogger;
020import org.forgerock.jaspi.logging.JaspiAuditLogger;
021import org.forgerock.jaspi.logging.JaspiLoggingConfigurator;
022import org.forgerock.jaspi.runtime.context.config.ModuleConfigurationFactory;
023import org.forgerock.json.fluent.JsonValue;
024import org.forgerock.openidm.jaspi.modules.IDMAuthModule;
025import org.forgerock.openidm.jaspi.modules.IDMAuthenticationAuditLogger;
026import org.forgerock.openidm.jaspi.modules.IDMJaspiModuleWrapper;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030import java.lang.reflect.InvocationTargetException;
031import java.util.ArrayList;
032import java.util.HashMap;
033import java.util.List;
034
035/**
036 * A singleton instance that implements both the Jaspi ModuleConfigurationFactory and JaspiLoggingConfigurator
037 * interfaces, that provides all of the configuration information for the Jaspi Runtime to be configured correctly.
038 */
039public enum JaspiRuntimeConfigurationFactory implements ModuleConfigurationFactory, JaspiLoggingConfigurator {
040
041    /**
042     * The Singleton instance of the JaspiRuntimeConfigurationFactory.
043     */
044    INSTANCE;
045
046    private final Logger logger = LoggerFactory.getLogger(JaspiRuntimeConfigurationFactory.class);
047
048    public static final String MODULE_CONFIG_ENABLED = "enabled";
049    private static final String CONFIG_AUDIT_LOGGER = "auditLogger";
050
051
052    private JsonValue moduleConfiguration;
053    private DebugLogger debugLogger;
054    private JaspiAuditLogger auditLogger;
055
056    /**
057     * Constructs the instance of the JaspiRuntimeConfigurationFactory.
058     */
059    private JaspiRuntimeConfigurationFactory() {
060        moduleConfiguration = new JsonValue(new HashMap<String, Object>());
061        debugLogger = new JaspiDebugLogger();
062    }
063
064    /**
065     * Gets the ModuleConfigurationFactory that the Jaspi Runtime will use to configure its authentication
066     * modules.
067     *
068     * @return An instance of a ModuleConfigurationFactory.
069     */
070    public static ModuleConfigurationFactory getModuleConfigurationFactory() {
071        return INSTANCE;
072    }
073
074    /**
075     * Gets the Logging Configurator that the Jaspi Runtime will use to configure its debug and audit logger
076     * instances.
077     *
078     * @return An instance of a JaspiLoggingConfigurator.
079     */
080    public static JaspiLoggingConfigurator getLoggingConfigurator() {
081        return INSTANCE;
082    }
083
084    /**
085     * Sets the module configuration for the Jaspi Runtime to use.
086     * <p>
087     * Takes the given Json configuration and processes it to check for enabled/disabled modules and resolves
088     * IDM auth module aliases to class names, prior to configuring the Jaspi runtime.
089     *
090     * @param moduleConfiguration The module configuration json.
091     * @param authnFilterHelper The OSGiAuthnFilterHelper to provide OSGi dependencies.
092     * @throws Exception If there is an error when constructing the Audit Logger for the Jaspi Runtime.
093     */
094    void setModuleConfiguration(final JsonValue moduleConfiguration, OSGiAuthnFilterHelper authnFilterHelper)
095            throws Exception {
096
097        final JsonValue moduleConfig = new JsonValue(moduleConfiguration);
098
099        JsonValue serverAuthContext = moduleConfig.get(ModuleConfigurationFactory.SERVER_AUTH_CONTEXT_KEY)
100                .required();
101        String auditLoggerClassName = serverAuthContext.get(CONFIG_AUDIT_LOGGER).asString();
102
103        if (auditLoggerClassName == null) {
104            setAuditLogger(new IDMAuthenticationAuditLogger(authnFilterHelper));
105        } else {
106            setAuditLogger(constructAuditLogger(auditLoggerClassName));
107        }
108
109        if (serverAuthContext.isDefined(ModuleConfigurationFactory.SESSION_MODULE_KEY)) {
110            JsonValue sessionModuleConfig = serverAuthContext.get(ModuleConfigurationFactory.SESSION_MODULE_KEY);
111            if (!processModuleConfiguration(sessionModuleConfig)) {
112                serverAuthContext.remove(ModuleConfigurationFactory.SESSION_MODULE_KEY);
113            }
114        }
115
116        JsonValue authModulesConfig = serverAuthContext.get(ModuleConfigurationFactory.AUTH_MODULES_KEY).required();
117
118        List<Integer> toRemove = new ArrayList<Integer>();
119        for (int i = 0; i < authModulesConfig.size(); i++) {
120            JsonValue authModuleConfig = authModulesConfig.get(i);
121            if (!processModuleConfiguration(authModuleConfig)) {
122                toRemove.add(i);
123            }
124        }
125        int i = 0;
126        for (int index : toRemove) {
127            authModulesConfig.remove(index - i);
128            i++;
129        }
130
131        this.moduleConfiguration = moduleConfig;
132    }
133
134    /**
135     * Process the module configuration for a specific module, checking to see if the module is enabled and
136     * resolving the module class name if an alias is used.
137     *
138     * @param moduleConfig The specific module configuration json.
139     * @return Whether the module is enabled or not.
140     */
141    private boolean processModuleConfiguration(JsonValue moduleConfig) {
142
143        if (moduleConfig.isDefined(MODULE_CONFIG_ENABLED) && !moduleConfig.get(MODULE_CONFIG_ENABLED).asBoolean()) {
144            return false;
145        }
146        moduleConfig.remove(MODULE_CONFIG_ENABLED);
147
148        if (moduleConfig.isDefined("name")) {
149            String className = resolveAuthModuleClassName(moduleConfig.get("name").asString());
150            moduleConfig.remove("name");
151            moduleConfig.add("className", className);
152        }
153
154        // set the classname config so the actual auth module gets wrapped in a IDMJaspiModuleWrapper:
155        // replace the resolved auth module classname with IDMJaspiModuleWrapper's classname
156        // and put the resolved auth module classname as a property unless we've already done this;
157        // this check is necessary for OPENIDM-1848
158        String className = moduleConfig.get("className").asString();
159        if (!IDMJaspiModuleWrapper.class.getName().equals(className)) {
160            moduleConfig.put("className", IDMJaspiModuleWrapper.class.getName());
161            if (!moduleConfig.isDefined("properties")) {
162                moduleConfig.put("properties", new HashMap<String, Object>());
163            }
164            moduleConfig.get("properties").put("authModuleClassName", className);
165        }
166
167        return true;
168    }
169
170    /**
171     * Resolves the given auth module name from either a core IDM auth module name from {@link IDMAuthModule} or
172     * a fully qualified class name of the auth module.
173     *
174     * @param authModuleName The auth module name.
175     * @return The auth module class name.
176     */
177    private String resolveAuthModuleClassName(String authModuleName) {
178        try {
179            return IDMAuthModule.valueOf(authModuleName).getAuthModuleClass().getCanonicalName();
180        } catch (IllegalArgumentException e) {
181            return authModuleName;
182        }
183    }
184
185    /**
186     * Given a class name of a JaspiAuditLogger, will construct an instance of it.
187     *
188     * @param className The class name of the Audit Logger.
189     * @param <T> The type of the Audit Logger.
190     * @return An instance of the Jaspi Audit Logger.
191     * @throws Exception If there is an error constructing the Audit Logger.
192     */
193    @SuppressWarnings("unchecked")
194    private <T extends JaspiAuditLogger> T constructAuditLogger(String className) throws Exception {
195        try {
196            final Class<?> cls = Class.forName(className);
197            return (T) cls.getConstructor().newInstance();
198        } catch (ClassNotFoundException e) {
199            logger.error("Failed to construct Audit Logger instance", e);
200            throw e;
201        } catch (InvocationTargetException e) {
202            logger.error("Failed to construct Audit Logger instance", e);
203            throw e;
204        } catch (InstantiationException e) {
205            logger.error("Failed to construct Audit Logger instance", e);
206            throw e;
207        } catch (NoSuchMethodException e) {
208            logger.error("Failed to construct Audit Logger instance", e);
209            throw e;
210        } catch (IllegalAccessException e) {
211            logger.error("Failed to construct Audit Logger instance", e);
212            throw e;
213        }
214    }
215
216    /**
217     * Sets the Audit Logger for the Jaspi Runtime.
218     *
219     * @param auditLogger The Audit Logger instance.
220     */
221    private void setAuditLogger(final JaspiAuditLogger auditLogger) {
222        this.auditLogger = auditLogger;
223    }
224
225    /**
226     * Clears the store module configuration and logger instances.
227     */
228    void clear() {
229        moduleConfiguration = new JsonValue(new HashMap<String, Object>());
230        debugLogger = new JaspiDebugLogger();
231        auditLogger = null;
232    }
233
234    /**
235     * {@inheritDoc}
236     */
237    @Override
238    public JsonValue getConfiguration() {
239        return moduleConfiguration;
240    }
241
242    /**
243     * {@inheritDoc}
244     */
245    @Override
246    public DebugLogger getDebugLogger() {
247        return debugLogger;
248    }
249
250    /**
251     * {@inheritDoc}
252     */
253    @Override
254    public JaspiAuditLogger getAuditLogger() {
255        return auditLogger;
256    }
