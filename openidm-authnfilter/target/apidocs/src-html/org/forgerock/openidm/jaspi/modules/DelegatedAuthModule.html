017package org.forgerock.openidm.jaspi.modules;
018
019import javax.security.auth.Subject;
020import javax.security.auth.callback.CallbackHandler;
021import javax.security.auth.message.AuthException;
022import javax.security.auth.message.AuthStatus;
023import javax.security.auth.message.MessageInfo;
024import javax.security.auth.message.MessagePolicy;
025import javax.security.auth.message.module.ServerAuthModule;
026import javax.servlet.http.HttpServletRequest;
027import javax.servlet.http.HttpServletResponse;
028import java.security.Principal;
029import java.util.Map;
030
031import org.forgerock.jaspi.exceptions.JaspiAuthException;
032import org.forgerock.openidm.jaspi.auth.Authenticator;
033import org.forgerock.openidm.jaspi.auth.AuthenticatorFactory;
034import org.forgerock.openidm.jaspi.config.OSGiAuthnFilterBuilder;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037
038import org.forgerock.json.fluent.JsonValue;
039import org.forgerock.json.resource.ResourceException;
040import org.forgerock.openidm.jaspi.config.OSGiAuthnFilterHelper;
041import org.forgerock.openidm.jaspi.modules.IDMJaspiModuleWrapper.Credential;
042
043import static org.forgerock.openidm.jaspi.modules.IDMJaspiModuleWrapper.QUERY_ON_RESOURCE;
044import static org.forgerock.openidm.jaspi.modules.IDMJaspiModuleWrapper.BASIC_AUTH_CRED_HELPER;
045import static org.forgerock.openidm.jaspi.modules.IDMJaspiModuleWrapper.HEADER_AUTH_CRED_HELPER;
046
047/**
048 * Authentication Filter modules for the JASPI common Authentication Filter. Validates client requests by passing though
049 * to a OpenICF Connector.
050 *
051 * @author Phill Cunnington
052 * @author brmiller
053 */
054public class DelegatedAuthModule implements ServerAuthModule {
055
056    private final static Logger logger = LoggerFactory.getLogger(DelegatedAuthModule.class);
057
058    private final OSGiAuthnFilterHelper authnFilterHelper;
059
060    private String queryOnResource;
061    private Authenticator authenticator;
062
063    /**
064     * Constructor used by the commons Authentication Filter framework to create an instance of this authentication
065     * module.
066     */
067    public DelegatedAuthModule() throws AuthException {
068        authnFilterHelper = OSGiAuthnFilterBuilder.getInstance();
069        if (authnFilterHelper == null) {
070            throw new AuthException("OSGiAuthnFilterHelper is not ready.");
071        }
072    }
073
074    /**
075     * Constructor used by tests to inject dependencies.
076     *
077     * @param authnFilterHelper A mock of an OSGiAuthnFilterHelper
078     * @param authenticator A mock of an PassthroughAuthenticator instance.
079     */
080    DelegatedAuthModule(OSGiAuthnFilterHelper authnFilterHelper, Authenticator authenticator) {
081        this.authnFilterHelper = authnFilterHelper;
082        this.authenticator = authenticator;
083    }
084
085    /**
086     * {@inheritDoc}
087     */
088    @Override
089    public final Class[] getSupportedMessageTypes() {
090        return new Class[]{HttpServletRequest.class, HttpServletResponse.class};
091    }
092
093    /**
094     * Initialises the Passthrough authentication module with the OSGi json configuration.
095     *
096     * @param requestPolicy {@inheritDoc}
097     * @param responsePolicy {@inheritDoc}
098     * @param handler {@inheritDoc}
099     */
100    @Override
101    public void initialize(MessagePolicy requestPolicy, MessagePolicy responsePolicy, CallbackHandler handler,
102            Map options) throws AuthException {
103
104        queryOnResource = new JsonValue(options).get(QUERY_ON_RESOURCE).required().asString();
105        authenticator = new AuthenticatorFactory(
106                authnFilterHelper.getConnectionFactory(), authnFilterHelper.getCryptoService())
107            .apply(new JsonValue(options));
108    }
109
110    /**
111     * Validates the client's request by passing through the request to be authenticated against a OpenICF Connector.
112     *
113     * @param messageInfo {@inheritDoc}
114     * @param clientSubject {@inheritDoc}
115     * @param serviceSubject {@inheritDoc}
116     * @return {@inheritDoc}
117     * @throws AuthException If there is a problem performing the authentication.
118     */
119    @Override
120    public AuthStatus validateRequest(MessageInfo messageInfo, Subject clientSubject, Subject serviceSubject)
121            throws AuthException {
122
123        logger.debug("DelegatedAuthModule: validateRequest START");
124
125        SecurityContextMapper securityContextMapper = SecurityContextMapper.fromMessageInfo(messageInfo);
126        HttpServletRequest request = (HttpServletRequest) messageInfo.getRequestMessage();
127
128        try {
129            logger.debug("DelegatedAuthModule: Delegating call to remote authentication");
130
131            if (authenticate(HEADER_AUTH_CRED_HELPER.getCredential(request), securityContextMapper)
132                    || authenticate(BASIC_AUTH_CRED_HELPER.getCredential(request), securityContextMapper)) {
133
134                logger.debug("DelegatedAuthModule: Authentication successful");
135
136                final String authcid = securityContextMapper.getAuthenticationId();
137                clientSubject.getPrincipals().add(new Principal() {
138                    public String getName() {
139                        return authcid;
140                    }
141                });
142
143                // Auth success will be logged in IDMJaspiModuleWrapper
144                return AuthStatus.SUCCESS;
145            } else {
146                logger.debug("DelegatedAuthModule: Authentication failed");
147                return AuthStatus.SEND_FAILURE;
148            }
149        } finally {
150            logger.debug("DelegatedAuthModule: validateRequest END");
151        }
152    }
153
154    private boolean authenticate(Credential credential, SecurityContextMapper securityContextMapper)
155            throws AuthException {
156
157        if (!credential.isComplete()) {
158            logger.debug("Failed authentication, missing or empty headers");
159            return false;
160        }
161
162        // set the authenticationId of the user that is trying to authenticate
163        securityContextMapper.setAuthenticationId(credential.username);
164
165        try {
166            return authenticator.authenticate(credential.username, credential.password,
167                    authnFilterHelper.getRouter().createServerContext());
168        } catch (ResourceException e) {
169            logger.debug("Failed delegated authentication of {} on {}.", credential.username, queryOnResource, e);
170            if (e.isServerError()) { // HTTP server-side error
171                throw new JaspiAuthException(
172                        "Failed delegated authentication of " + credential.username + " on " + queryOnResource, e);
173            }
174            // authentication failed
175            return false;
176        }
177    }
178
179    /**
180     * No work to do here so always returns AuthStatus.SEND_SUCCESS.
181     *
182     * @param messageInfo {@inheritDoc}
183     * @param serviceSubject {@inheritDoc}
184     * @return {@inheritDoc}
185     */
186    @Override
187    public AuthStatus secureResponse(MessageInfo messageInfo, Subject serviceSubject) {
188        return AuthStatus.SEND_SUCCESS;
189    }
190
191    /**
192     * Nothing to clean up.
193     *
194     * @param messageInfo {@inheritDoc}
195     * @param subject {@inheritDoc}
196     */
197    @Override
198    public void cleanSubject(MessageInfo messageInfo, Subject subject) {
199    }
200}
