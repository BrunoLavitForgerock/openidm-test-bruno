017package org.forgerock.openidm.jaspi.modules;
018
019import javax.security.auth.Subject;
020import javax.security.auth.callback.CallbackHandler;
021import javax.security.auth.message.AuthStatus;
022import javax.security.auth.message.MessageInfo;
023import javax.security.auth.message.MessagePolicy;
024import javax.security.auth.message.module.ServerAuthModule;
025import javax.servlet.ServletRequest;
026import javax.servlet.http.HttpServletRequest;
027import javax.servlet.http.HttpServletResponse;
028import java.security.Principal;
029import java.security.cert.X509Certificate;
030import java.util.ArrayList;
031import java.util.HashSet;
032import java.util.List;
033import java.util.Map;
034import java.util.Set;
035
036import org.forgerock.json.fluent.JsonValue;
037import org.forgerock.openidm.core.IdentityServer;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041
042/**
043 * Authentication Module for authenticating users using a client certificate.
044 *
045 * @author brmiller
046 */
047public class ClientCertAuthModule implements ServerAuthModule {
048
049    /** Logger */
050    private final static Logger logger = LoggerFactory.getLogger(ClientCertAuthModule.class);
051
052    private static final String ALLOWED_AUTHENTICATION_ID_PATTERNS = "allowedAuthenticationIdPatterns";
053
054    /** A list of ports that allow authentication purely based on client certificates (SSL mutual auth) */
055    private final Set<Integer> clientAuthOnly = new HashSet<Integer>();
056
057    /** a list of authenticationId patterns to match against the subjectDN for "successful auth". */
058    private List<String> allowedAuthenticationIdPatterns;
059
060    /**
061     * Initialises the ClientCertAuthModule.
062     *
063     * @param requestPolicy {@inheritDoc}
064     * @param responsePolicy {@inheritDoc}
065     * @param handler {@inheritDoc}
066     */
067    @Override
068    public void initialize(MessagePolicy requestPolicy, MessagePolicy responsePolicy, CallbackHandler handler,
069            Map options) {
070
071        final JsonValue properties = new JsonValue(options);
072
073        String clientAuthOnlyStr = IdentityServer.getInstance().getProperty("openidm.auth.clientauthonlyports");
074        if (clientAuthOnlyStr != null) {
075            String[] split = clientAuthOnlyStr.split(",");
076            for (String entry : split) {
077                clientAuthOnly.add(Integer.valueOf(entry));
078            }
079        }
080        logger.info("Authentication disabled on ports: {}", clientAuthOnly);
081
082        allowedAuthenticationIdPatterns = properties.get(ALLOWED_AUTHENTICATION_ID_PATTERNS)
083                .defaultTo(new ArrayList<String>())
084                .asList(String.class);
085    }
086
087    /**
088     * {@inheritDoc}
089     */
090    @Override
091    public final Class[] getSupportedMessageTypes() {
092        return new Class[]{HttpServletRequest.class, HttpServletResponse.class};
093    }
094
095    /**
096     * Validates the request by authenticating against the client certificate in the request.
097     *
098     * @param messageInfo {@inheritDoc}
099     * @param clientSubject {@inheritDoc}
100     * @param serviceSubject {@inheritDoc}
101     * @return {@inheritDoc}
102     */
103    @Override
104    public AuthStatus validateRequest(MessageInfo messageInfo, Subject clientSubject, Subject serviceSubject) {
105
106        SecurityContextMapper securityContextMapper = SecurityContextMapper.fromMessageInfo(messageInfo);
107
108        HttpServletRequest req = (HttpServletRequest) messageInfo.getRequestMessage();
109
110        // if the request's local port is not an allowed client auth port, we cannot proceed with client auth
111        if (!allowClientCertOnly(req)) {
112            return AuthStatus.SEND_FAILURE;
113        }
114
115        if (authenticateUsingClientCert(req, securityContextMapper)) {
116            final String authcid = securityContextMapper.getAuthenticationId();
117            clientSubject.getPrincipals().add(new Principal() {
118                public String getName() {
119                    return authcid;
120                }
121            });
122            return AuthStatus.SUCCESS;
123        } else {
124            return AuthStatus.SEND_FAILURE;
125        }
126    }
127
128    /**
129     * Whether to allow authentication purely based on client certificates.
130     *
131     * Note that the checking of the certificates MUST be done by setting jetty up for client auth required.
132     *
133     * @return true if authentication via client certificate only is sufficient.
134     */
135    private boolean allowClientCertOnly(ServletRequest request) {
136        return clientAuthOnly.contains(Integer.valueOf(request.getLocalPort()));
137    }
138
139    /**
140     * Authenticates the request using the client certificate from the request.
141     *
142     * @param request The ServletRequest.
143     */
144    // This is currently Jetty specific
145    private boolean authenticateUsingClientCert(ServletRequest request, SecurityContextMapper securityContextMapper) {
146
147        logger.debug("Client certificate authentication request");
148        X509Certificate[] certs = getClientCerts(request);
149
150        if (certs == null || certs.length < 1 || certs[0] == null) {
151            return false;
152        }
153
154        Principal existingPrincipal = request instanceof HttpServletRequest
155                ? ((HttpServletRequest) request).getUserPrincipal()
156                : null;
157        logger.debug("Request {} existing Principal {} has {} certificates", request, existingPrincipal, certs.length);
158        for (X509Certificate cert : certs) {
159            logger.debug("Request {} client certificate subject DN: {}", request, cert.getSubjectDN());
160        }
161
162        // Q: is it possible to pass multiple client certs?
163        String username = certs[0].getSubjectDN().getName();
164        securityContextMapper.setAuthenticationId(username);
165
166        if (!usernameMatchesPatterns(username)) {
167            logger.debug("Client certificate subject {} did not match allowed patterns", username);
168            return false;
169        }
170
171        logger.debug("Authentication client certificate subject {}", username);
172        return true;
173    }
174
175    private boolean usernameMatchesPatterns(String username) {
176        for (String pattern : allowedAuthenticationIdPatterns) {
177            if (username.matches(pattern)) {
178                return true;
179            }
180        }
181        return false;
182    }
183
184    /**
185     * Gets the client certificates from the request.
186     *
187     * @param request The ServletRequest.
188     * @return An array of X509Certificates.
189     */
190    // This is currently Jetty specific
191    private X509Certificate[] getClientCerts(ServletRequest request) {
192        Object checkCerts = request.getAttribute("javax.servlet.request.X509Certificate");
193        if (checkCerts instanceof X509Certificate[]) {
194            return (X509Certificate[]) checkCerts;
195        } else {
196            logger.warn("Unknown certificate type retrieved {}", checkCerts);
197            return null;
198        }
199    }
200
201    /**
202     * No work to do here so always returns AuthStatus.SEND_SUCCESS.
203     *
204     * @param messageInfo {@inheritDoc}
205     * @param serviceSubject {@inheritDoc}
206     * @return {@inheritDoc}
207     */
208    @Override
209    public AuthStatus secureResponse(MessageInfo messageInfo, Subject serviceSubject) {
210        return AuthStatus.SEND_SUCCESS;
211    }
212
213    /**
214     * Nothing to clean up.
215     *
216     * @param messageInfo {@inheritDoc}
217     * @param subject {@inheritDoc}
218     */
219    @Override
220    public void cleanSubject(MessageInfo messageInfo, Subject subject) {
221    }
222}
