024package org.forgerock.openidm.repo.jdbc.impl.pool;
025
026import com.jolbox.bonecp.ConnectionHandle;
027import com.jolbox.bonecp.hooks.AbstractConnectionHook;
028import com.jolbox.bonecp.hooks.AcquireFailConfig;
029import org.forgerock.openidm.util.DateUtil;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033/**
034 * @author $author$
035 * @version $Revision$ $Date$
036 */
037public class DatabaseShutdownHook extends AbstractConnectionHook {
038    private final static Logger logger = LoggerFactory.getLogger(DatabaseShutdownHook.class);
039    // Only used for logging, so it's okay to use the default timezone
040    private final static DateUtil dateUtil = DateUtil.getDateUtil();
041
042    @Override
043    public boolean onConnectionException(ConnectionHandle connection, String state, Throwable t) {
044        // handle notifications here: SNMP or SMTP
045        logger.warn("Database down at {}", dateUtil.now());
046        return super.onConnectionException(connection, state, t);
047    }
048
049    @Override
050    public boolean onAcquireFail(Throwable t, AcquireFailConfig acquireConfig) {
051        // handle notifications here: SNMP or SMTP
052        logger.warn("Failure to acquire connection at {}. Retry attempts remaining : {}", dateUtil.now(), acquireConfig.getAcquireRetryAttempts());
053        return super.onAcquireFail(t, acquireConfig);
054    }
055
056}
