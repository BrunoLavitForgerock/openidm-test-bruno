075    Mapping explicitMapping;

077    // The json pointer (used as names) of the properties to replace the ?
078    // tokens in the prepared statement,
079    // in the order they need populating in create and update queries
080    List<JsonPointer> tokenReplacementPropPointers = new ArrayList<JsonPointer>();

082    ObjectMapper mapper = new ObjectMapper();

091    public MappedTableHandler(String tableName, Map mapping, String dbSchemaName,
092            JsonValue queriesConfig, SQLExceptionHandler sqlExceptionHandler,
093            Accessor<CryptoService> cryptoServiceAccessor) throws InternalServerErrorException {
094
095        // TODO Replace this with a "guarantee" somewhere when/if the provision
096        // of this
097        // accessor becomes more automatic
098        if (cryptoServiceAccessor == null)
099            throw new InternalServerErrorException("No CryptoServiceAccessor found!");
100
101        this.tableName = tableName;
102        this.dbSchemaName = dbSchemaName;
103        // Maintain a stable ordering
104        this.rawMappingConfig = new LinkedHashMap<String, String>();
105        this.rawMappingConfig.putAll(mapping);
106
107        explicitMapping =
108                new Mapping(tableName, new JsonValue(rawMappingConfig), cryptoServiceAccessor);
109        logger.debug("Explicit mapping: {}", explicitMapping);
110
111        if (sqlExceptionHandler == null) {
112            this.sqlExceptionHandler = new DefaultSQLExceptionHandler();
113        } else {
114            this.sqlExceptionHandler = sqlExceptionHandler;
115        }
116
117        queries = new TableQueries(new ExplicitQueryResultMapper(explicitMapping));
118        queries.setConfiguredQueries(tableName, dbSchemaName, queriesConfig, null);
119
120        String mainTable = dbSchemaName == null ? tableName : dbSchemaName + "." + tableName;
121
122        StringBuffer colNames = new StringBuffer();
123        StringBuffer tokenNames = new StringBuffer();
124        StringBuffer prepTokens = new StringBuffer();
125        StringBuffer updateAssign = new StringBuffer();
126        boolean isFirst = true;
127
128        for (ColumnMapping colMapping : explicitMapping.columnMappings) {
129            if (!isFirst) {
130                colNames.append(", ");
131                tokenNames.append(",");
132                prepTokens.append(",");
133                updateAssign.append(", ");
134            }
135            colNames.append(colMapping.dbColName);
136            tokenNames.append("${").append(colMapping.objectColName).append("}");
137            prepTokens.append("?");
138            tokenReplacementPropPointers.add(colMapping.objectColPointer);
139            // updateAssign.append(colMapping.dbColName).append(" = ${").append(colMapping.objectColName).append("}");
140            updateAssign.append(colMapping.dbColName).append(" = ?");
141            isFirst = false;
142        }
143
144        readQueryStr = "SELECT * FROM " + mainTable + " WHERE objectid = ?";
145        readForUpdateQueryStr = "SELECT * FROM " + mainTable + " WHERE objectid = ? FOR UPDATE";
146        createQueryStr =
147                "INSERT INTO " + mainTable + " (" + colNames + ") VALUES ( " + prepTokens + ")";
148        updateQueryStr = "UPDATE " + mainTable + " SET " + updateAssign + " WHERE objectid = ?";
149        deleteQueryStr = "DELETE FROM " + mainTable + " WHERE objectid = ? AND rev = ?";
150
151        logger.debug("Unprepared query strings {} {} {} {} {}", new Object[] { readQueryStr,
152            createQueryStr, updateQueryStr, deleteQueryStr });
153
154    }

156    /**
157     * @see org.forgerock.openidm.repo.jdbc.TableHandler#read(java.lang.String,
158     *      java.lang.String, java.lang.String, java.sql.Connection)
159     */
160    @Override
161    public Resource read(String fullId, String type, String localId, Connection connection)
162            throws NotFoundException, SQLException, IOException, InternalServerErrorException {
163        JsonValue resultValue = null;
164        Resource result = null;
165        PreparedStatement readStatement = null;
166        ResultSet rs = null;
167        try {
168            readStatement = queries.getPreparedStatement(connection, readQueryStr);
169
170            logger.debug("Populating prepared statement {} for {}", readStatement, fullId);
171            readStatement.setString(1, localId);
172
173            logger.debug("Executing: {}", readStatement);
174            rs = readStatement.executeQuery();
175
176            if (rs.next()) {
177                resultValue = explicitMapping.mapToJsonValue(rs, Mapping.getColumnNames(rs));
178                JsonValue rev = resultValue.get("_rev");
179                logger.debug(" full id: {}, rev: {}, obj {}", new Object[] { fullId, rev,
180                    resultValue });
181                result = new Resource(fullId, rev.asString(), resultValue);
182            } else {
183                throw new NotFoundException("Object " + fullId + " not found in " + type);
184            }
185        } finally {
186            CleanupHelper.loggedClose(rs);
187            CleanupHelper.loggedClose(readStatement);
188        }
189
190        return result;
191    }

193    /**
194     * Reads an object with for update locking applied
195     *
196     * Note: statement associated with the returned resultset is not closed upon
197     * return. Aside from taking care to close the resultset it also is the
198     * responsibility of the caller to close the associated statement. Although
199     * the specification specifies that drivers/pools should close the statement
200     * automatically, not all do this reliably.
201     *
202     * @param fullId
203     *            qualified id of component type and id
204     * @param type
205     *            the component type
206     * @param localId
207     *            the id of the object within the component type
208     * @param connection
209     *            the connection to use
210     * @return the row for the requested object, selected FOR UPDATE
211     * @throws NotFoundException
212     *             if the requested object was not found in the DB
213     * @throws java.sql.SQLException
214     *             for general DB issues
215     */
216    ResultSet readForUpdate(String fullId, String type, String localId, Connection connection)
217            throws NotFoundException, SQLException {
218
219        PreparedStatement readForUpdateStatement = null;
220        ResultSet rs = null;
221        try {
222            readForUpdateStatement =
223                    queries.getPreparedStatement(connection, readForUpdateQueryStr);
224            logger.trace("Populating prepared statement {} for {}", readForUpdateStatement, fullId);
225            readForUpdateStatement.setString(1, localId);
226
227            logger.debug("Executing: {}", readForUpdateStatement);
228            rs = readForUpdateStatement.executeQuery();
229            if (rs.next()) {
230                logger.debug("Read for update full id: {}", fullId);
231                return rs;
232            } else {
233                CleanupHelper.loggedClose(rs);
234                CleanupHelper.loggedClose(readForUpdateStatement);
235                throw new NotFoundException("Object " + fullId + " not found in " + type);
236            }
237        } catch (SQLException ex) {
238            CleanupHelper.loggedClose(rs);
239            CleanupHelper.loggedClose(readForUpdateStatement);
240            throw ex;
241        }
242    }

251            Connection connection) throws SQLException, IOException {
252        PreparedStatement createStatement =
253                queries.getPreparedStatement(connection, createQueryStr);
254        try {
255            create(fullId, type, localId, obj, connection, createStatement, false);
256        } finally {
257            CleanupHelper.loggedClose(createStatement);

261    /**
262     * Adds the option to batch more than one create statement
263     *
264     * @param batchCreate
265     *            if true just adds create to batched statements, does not
266     *            execute. false the statement is executed directly
267     * @see org.forgerock.openidm.repo.jdbc.TableHandler#create(java.lang.String,
268     *      java.lang.String, java.lang.String, java.util.Map,
269     *      java.sql.Connection) for the other parameters
270     */
271    protected void create(String fullId, String type, String localId, Map<String, Object> obj,
272            Connection connection, PreparedStatement createStatement, boolean batchCreate)
273            throws SQLException, IOException {
274
275        logger.debug("Create with fullid {}", fullId);
276        String rev = "0";
277        obj.put("_id", localId); // Save the id in the object
278        obj.put("_rev", rev); // Save the rev in the object, and return the
279                              // changed rev from the create.
280        JsonValue objVal = new JsonValue(obj);
281
282        logger.debug("Preparing statement {} with {}, {}, {}", new Object[] { createStatement,
283            type, localId, rev });
284        populatePrepStatementColumns(createStatement, objVal, tokenReplacementPropPointers);
285
286        if (!batchCreate) {
287            logger.debug("Executing: {}", createStatement);
288            int val = createStatement.executeUpdate();
289            logger.debug("Created object for id {} with rev {}", fullId, rev);
290        } else {
291            createStatement.addBatch();
292            logger.debug("Added create for object id {} with rev {} to batch", fullId, rev);
293        }
294    }
295
296    /**
297     * Populates the create or update statement with the token replacement
298     * values in the appropriate order. For update the final objectid is not
299     * part of the declarative mapping and needs to be populated separately.
300     *
301     * @param prepStatement
302     *            the update or create prepared statement
303     * @param tokenPointers
304     *            the token replacement pointers pointing into the object set to
305     *            extract the relevant values
306     * @return the buildNext column position if further populating is desired
307     */
308    int populatePrepStatementColumns(PreparedStatement prepStatement, JsonValue objVal,
309            List<JsonPointer> tokenPointers) throws IOException, SQLException {
310        int colPos = 1;
311        for (JsonPointer propPointer : tokenPointers) {
312            // TODO: support explicit column types/conversion specified in
313            // column mapping
314            // This is currently limited to STRING handling
315            JsonValue rawValue = objVal.get(propPointer);
316            String propValue = null;
317            if (null == rawValue) {
318                propValue = null;
319            } else if (rawValue.isString() || rawValue.isNull()) {
320                propValue = rawValue.asString();
321            } else {
322                if (logger.isTraceEnabled()) {
323                    logger.trace(
324                            "Value for col {} from {} is getting Stringified from type {} to store in a STRING column as value: {}",
325                            new Object[] { colPos, propPointer, rawValue.getClass(), rawValue });
326                }
327                propValue = mapper.writeValueAsString(rawValue.getObject());
328            }
329
330            prepStatement.setString(colPos, propValue);
331            colPos++;
332        }
333        return colPos;
334    }

336    /**
337     * @see org.forgerock.openidm.repo.jdbc.TableHandler#update(java.lang.String,
338     *      java.lang.String, java.lang.String, java.lang.String, java.util.Map,
339     *      java.sql.Connection)
340     */
341    @Override
342    public void update(String fullId, String type, String localId, String rev,
343            Map<String, Object> obj, Connection connection) throws SQLException, IOException,
344            PreconditionFailedException, NotFoundException, InternalServerErrorException {
345        logger.debug("Update with fullid {}", fullId);
346
347        int revInt = Integer.parseInt(rev);
348        ++revInt;
349        String newRev = Integer.toString(revInt);
350        obj.put("_rev", newRev); // Save the rev in the object, and return the
351                                 // changed rev from the create.
352
353        ResultSet rs = null;
354        PreparedStatement updateStatement = null;
355        try {
356            rs = readForUpdate(fullId, type, localId, connection);
357            String existingRev = explicitMapping.getRev(rs);
358            logger.debug("Update existing object {} rev: {} ", fullId, existingRev);
359
360            if (!existingRev.equals(rev)) {
361                throw new PreconditionFailedException("Update rejected as current Object revision "
362                        + existingRev + " is different than expected by caller (" + rev
363                        + "), the object has changed since retrieval.");
364            }
365            updateStatement = queries.getPreparedStatement(connection, updateQueryStr);
366
367            // Support changing object identifier
368            String newLocalId = (String) obj.get("_id");
369            if (newLocalId != null && !localId.equals(newLocalId)) {
370                logger.debug("Object identifier is changing from " + localId + " to " + newLocalId);
371            } else {
372                newLocalId = localId; // If it hasn't changed, use the existing
373                                      // ID
374                obj.put("_id", newLocalId); // Ensure the ID is saved in the
375                                            // object
376            }
377
378            JsonValue objVal = new JsonValue(obj);
379            logger.trace("Populating prepared statement {} for {} {} {}", new Object[] {
380                updateStatement, fullId, newLocalId, newRev });
381            int nextCol =
382                    populatePrepStatementColumns(updateStatement, objVal,
383                            tokenReplacementPropPointers);
384            updateStatement.setString(nextCol, localId);
385            logger.debug("Update statement: {}", updateStatement);
386            int updateCount = updateStatement.executeUpdate();
387            logger.trace("Updated rows: {} for {}", updateCount, fullId);
388            if (updateCount != 1) {
389                throw new InternalServerErrorException(
390                        "Update execution did not result in updating 1 row as expected. Updated rows: "
391                                + updateCount);
392            }
393        } finally {
394            if (rs != null) {
395                // Ensure associated statement also is closed
396                Statement rsStatement = rs.getStatement();
397                CleanupHelper.loggedClose(rs);
398                CleanupHelper.loggedClose(rsStatement);
399            }
400            CleanupHelper.loggedClose(updateStatement);
401        }
402    }

404    /**
405     * @see org.forgerock.openidm.repo.jdbc.TableHandler#delete(java.lang.String,
406     *      java.lang.String, java.lang.String, java.lang.String,
407     *      java.sql.Connection)
408     */
409    @Override
410    public void delete(String fullId, String type, String localId, String rev, Connection connection)
411            throws PreconditionFailedException, InternalServerErrorException, NotFoundException,
412            SQLException, IOException {
413        logger.debug("Delete with fullid {}", fullId);
414
415        // First check if the revision matches and select it for UPDATE
416        ResultSet existing = null;
417        PreparedStatement deleteStatement = null;
418        try {
419            try {
420                existing = readForUpdate(fullId, type, localId, connection);
421            } catch (NotFoundException ex) {
422                throw new NotFoundException("Object does not exist for delete on: " + fullId);
423            }
424            String existingRev = explicitMapping.getRev(existing);
425            if (!"*".equals(rev) && !rev.equals(existingRev)) {
426                throw new PreconditionFailedException("Delete rejected as current Object revision "
427                        + existingRev + " is different than " + "expected by caller " + rev
428                        + ", the object has changed since retrieval.");
429            }
430
431            // Proceed with the valid delete
432            deleteStatement = queries.getPreparedStatement(connection, deleteQueryStr);
433            logger.trace("Populating prepared statement {} for {} {} {} {}", new Object[] {
434                deleteStatement, fullId, type, localId, rev });
435
436            deleteStatement.setString(1, localId);
437            deleteStatement.setString(2, rev);
438            logger.debug("Delete statement: {}", deleteStatement);
439
440            int deletedRows = deleteStatement.executeUpdate();
441            logger.trace("Deleted {} rows for id : {} {}", deletedRows, localId);
442            if (deletedRows < 1) {
443                throw new InternalServerErrorException("Deleting object for " + fullId
444                        + " failed, DB reported " + deletedRows + " rows deleted");
445            } else {
446                logger.debug("delete for id succeeded: {} revision: {}", localId, rev);
447            }
448        } finally {
449            if (existing != null) {
450                // Ensure associated statement also is closed
451                Statement existingStatement = existing.getStatement();
452                CleanupHelper.loggedClose(existing);
453                CleanupHelper.loggedClose(existingStatement);
454            }
455            CleanupHelper.loggedClose(deleteStatement);
456        }
457    }

497/**
498 * Handle the conversion of query results to the object set model
499 */
500class ExplicitQueryResultMapper implements QueryResultMapper {
501    final static Logger logger = LoggerFactory.getLogger(ExplicitQueryResultMapper.class);
502    Mapping explicitMapping;
503
504    public ExplicitQueryResultMapper(Mapping explicitMapping) {
505        this.explicitMapping = explicitMapping;
506    }
507
508    public List<Map<String, Object>> mapQueryToObject(ResultSet rs, String queryId, String type,
509            Map<String, Object> params, TableQueries tableQueries) throws SQLException,
510            InternalServerErrorException {
511
512        List<Map<String, Object>> result = new ArrayList<Map<String, Object>>();
513        Set names = Mapping.getColumnNames(rs);
514        while (rs.next()) {
515            JsonValue obj = explicitMapping.mapToJsonValue(rs, names);
516            result.add(obj.asMap());
517        }
518        return result;
519    }
520}

522/**
523 * Parsed Config handling
524 */
525class Mapping {
526
527    final static Logger logger = LoggerFactory.getLogger(Mapping.class);
528
529    String tableName;
530    Accessor<CryptoService> cryptoServiceAccessor;
531    List<ColumnMapping> columnMappings = new ArrayList();
532    ColumnMapping revMapping; // Quick access to mapping for MVCC revision
533    ObjectMapper mapper = new ObjectMapper();
534
535    public Mapping(String tableName, JsonValue mappingConfig, Accessor<CryptoService> cryptoServiceAccessor) {
536        this.cryptoServiceAccessor = cryptoServiceAccessor;
537        this.tableName = tableName;
538        for (Map.Entry<String, Object> entry : mappingConfig.asMap().entrySet()) {
539            String key = entry.getKey();
540            JsonValue value = mappingConfig.get(key);
541            ColumnMapping colMapping = new ColumnMapping(key, value);
542            columnMappings.add(colMapping);
543            if ("_rev".equals(colMapping.objectColName)) {
544                revMapping = colMapping;
545            }
546        }
547    }
548
549    public JsonValue mapToJsonValue(ResultSet rs, Set columnNames) throws SQLException,
550    InternalServerErrorException {
551        JsonValue mappedResult = new JsonValue(new LinkedHashMap<String, Object>());
552
553        for (ColumnMapping entry : columnMappings) {
554            Object value = null;
555            if (columnNames.contains(entry.dbColName)) {
556                if (ColumnMapping.TYPE_STRING.equals(entry.dbColType)) {
557                    value = rs.getString(entry.dbColName);
558                    if (cryptoServiceAccessor == null || cryptoServiceAccessor.access() == null) {
559                        throw new InternalServerErrorException("CryptoService unavailable");
560                    }
561                    if (JsonUtil.isEncrypted((String) value)) {
562                        value = convertToJson(entry.dbColName, "encrypted", (String)value, Map.class).asMap();
563                    }
564                } else if (ColumnMapping.TYPE_JSON_MAP.equals(entry.dbColType)) {
565                    value = convertToJson(entry.dbColName, entry.dbColType, rs.getString(entry.dbColName), Map.class).asMap();
566                } else if (ColumnMapping.TYPE_JSON_LIST.equals(entry.dbColType)) {
567                    value = convertToJson(entry.dbColName, entry.dbColType, rs.getString(entry.dbColName), List.class).asList();
568                } else {
569                    throw new InternalServerErrorException("Unsupported DB column type " + entry.dbColType);
570                }
571                mappedResult.put(entry.objectColPointer, value);
572            }
573        }
574        logger.debug("Mapped rs {} to {}", rs, mappedResult);
575        return mappedResult;
576    }
577
578    private <T> JsonValue convertToJson(String name, String nameType, String value, Class<T> valueType) throws InternalServerErrorException {
579        if (value != null) {
580            try {
581                return new JsonValue(mapper.readValue(value, valueType));
582            } catch (IOException e) {
583                throw new InternalServerErrorException("Unable to map " + nameType + " value for " + name, e);
584            }
585        }
586        return new JsonValue(null);
587    }
588
589    public String getRev(ResultSet rs) throws SQLException {
590        return rs.getString(revMapping.dbColName);
591    }
592
593    public String toString() {
594        StringBuffer sb = new StringBuffer();
595        sb.append("Explicit table mapping for " + tableName + " :\n");
596        for (ColumnMapping entry : columnMappings) {
597            sb.append(entry.toString());
598        }
599        return sb.toString();
600    }
601
602    public static Set<String> getColumnNames(ResultSet rs) throws SQLException {
603        TreeSet<String> set = new TreeSet<String>(String.CASE_INSENSITIVE_ORDER);
604        for (int i = 1; i <= rs.getMetaData().getColumnCount(); i++) {
605            set.add(rs.getMetaData().getColumnName(i));
606        }
607        return set;
608    }
609}

611/**
612 * Parsed Config handling
613 */
614class ColumnMapping {
615    public static final String DB_COLUMN_NAME = "column";
616    public static final String DB_COLUMN_TYPE = "type";
617
618    public static final String TYPE_STRING = "STRING";
619    public static final String TYPE_JSON_MAP = "JSON_MAP";
620    public static final String TYPE_JSON_LIST = "JSON_LIST";
621   
622    
623    public JsonPointer objectColPointer;
624    public String objectColName; // String representation of the column
625                                 // name/path
626    public String dbColName;
627    public String dbColType;
628
629    public ColumnMapping(String objectColName, JsonValue dbColMappingConfig) {
630        this.objectColName = objectColName;
631        this.objectColPointer = new JsonPointer(objectColName);
632        if (dbColMappingConfig.required().isList()) {
633            if (dbColMappingConfig.asList().size() != 2) {
634                throw new InvalidException("Explicit table mapping has invalid entry for "
635                        + objectColName + ", expecting column name and type but contains "
636                        + dbColMappingConfig.asList());
637            }
638            dbColName = dbColMappingConfig.get(0).required().asString();
639            dbColType = dbColMappingConfig.get(1).required().asString();
640        } else if (dbColMappingConfig.isMap()) {
641            dbColName = dbColMappingConfig.asMap().get(DB_COLUMN_NAME).toString();
642            dbColType = dbColMappingConfig.asMap().get(DB_COLUMN_TYPE).toString();
643        } else {
644            dbColName = dbColMappingConfig.asString();
645            dbColType = TYPE_STRING;
