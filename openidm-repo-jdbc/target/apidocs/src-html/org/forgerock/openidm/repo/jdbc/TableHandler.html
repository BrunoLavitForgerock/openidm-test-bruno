024package org.forgerock.openidm.repo.jdbc;
025
026import java.io.IOException;
027import java.sql.Connection;
028import java.sql.SQLException;
029import java.util.List;
030import java.util.Map;
031
032import org.forgerock.json.resource.BadRequestException;
033import org.forgerock.json.resource.ConflictException;
034import org.forgerock.json.resource.ForbiddenException;
035import org.forgerock.json.resource.InternalServerErrorException;
036import org.forgerock.json.resource.NotFoundException;
037import org.forgerock.json.resource.PreconditionFailedException;
038import org.forgerock.json.resource.Resource;
039import org.forgerock.json.resource.ResourceException;
040
041public interface TableHandler {
042
043    /**
044     * Gets an object from the repository by identifier. The returned object is not validated 
045     * against the current schema and may need processing to conform to an updated schema.
046     * <p>
047     * The object will contain metadata properties, including object identifier {@code _id},
048     * and object version {@code _rev} to enable optimistic concurrency
049     *
050     * @param fullId the qualified identifier of the object to retrieve from the object set.
051     * @param type is the qualifier of the object to retrieve
052     * @param localId the identifier without the qualifier of the object to retrieve
053     * @param connection
054     * @throws NotFoundException if the specified object could not be found.
055     * @throws SQLException if a DB failure was reported
056     * @throws IOException if a failure to convert the JSON model was reported
057     * @throws InternalServerErrorException if the operation failed because of a (possibly transient) failure
058     * @return the requested object.
059     */
060    public abstract Resource read(String fullId, String type,
061            String localId, Connection connection) 
062            throws SQLException, IOException, ResourceException;
063
064    /**
065     * Creates a new object in the object set.
066     * <p>
067     * This method sets the {@code _id} property to the assigned identifier for the object,
068     * and the {@code _rev} property to the revised object version (For optimistic concurrency)
069     *
070     * @param fullId the client-generated identifier to use, or {@code null} if server-generated identifier is requested.
071     * @param type
072     * @param localId
073     * @param obj the contents of the object to create in the object set.
074     * @param connection
075     * @throws NotFoundException if the specified id could not be resolved.
076     * @throws ForbiddenException if access to the object or object set is forbidden.
077     * @throws PreconditionFailedException if an object with the same ID already exists.
078     * @throws InternalServerErrorException if the operation failed because of a (possibly transient) failure
079     * @throws java.io.IOException
080     * @throws java.sql.SQLException
081     */
082    public abstract void create(String fullId, String type, String localId,
083            Map<String, Object> obj, Connection connection)
084            throws SQLException, IOException, ResourceException;
085
086    /**
087     * Updates the specified object in the object set. 
088     * <p>
089     * This implementation requires MVCC and hence enforces that clients state what revision they expect 
090     * to be updating
091     * 
092     * If successful, this method updates metadata properties within the passed object,
093     * including: a new {@code _rev} value for the revised object's version
094     *
095     * @param fullId the identifier of the object to be put, or {@code null} to request a generated identifier.
096     * @param type
097     * @param localId
098     * @param rev the version of the object to update; or {@code null} if not provided.
099     * @param obj the contents of the object to put in the object set.
100     * @param connection
101     * @throws ConflictException if version is required but is {@code null}.
102     * @throws ForbiddenException if access to the object is forbidden.
103     * @throws NotFoundException if the specified object could not be found. 
104     * @throws PreconditionFailedException if version did not match the existing object in the set.
105     * @throws BadRequestException if the passed identifier is invalid
106     * @throws InternalServerErrorException if the operation failed because of a (possibly transient) failure
107     * @throws java.io.IOException
108     * @throws java.sql.SQLException
109     */
110    public abstract void update(String fullId, String type, String localId,
111            String rev, Map<String, Object> obj, Connection connection)
112            throws SQLException, IOException, ResourceException;
113
114    /**
115     * Deletes the specified object from the object set.
116     *
117     * @param fullId the identifier of the object to be deleted.
118     * @param type
119     * @param localId
120     * @param rev the version of the object to delete or {@code null} if not provided.
121     * @param connection
122     * @throws NotFoundException if the specified object could not be found.
123     * @throws ForbiddenException if access to the object is forbidden.
124     * @throws ConflictException if version is required but is {@code null}.
125     * @throws PreconditionFailedException if version did not match the existing object in the set.
126     * @throws InternalServerErrorException if the operation failed because of a (possibly transient) failure
127     * @throws java.io.IOException
128     * @throws java.sql.SQLException
129     */
130    public abstract void delete(String fullId, String type, String localId,
131            String rev, Connection connection)
132            throws SQLException, IOException, ResourceException;
133
134    /**
135     * Performs the query on the specified object and returns the associated results.
136     * <p>
137     * Queries are parametric; a set of named parameters is provided as the query criteria.
138     * The query result is a JSON object structure composed of basic Java types. 
139     * 
140     * The returned map is structured as follow: 
141     * - The top level map contains meta-data about the query, plus an entry with the actual result records.
142     * - The <code>QueryConstants</code> defines the map keys, including the result records (QUERY_RESULT)
143     *
144     * @param type identifies the object to query.
145     * @param params the parameters of the query to perform.
146     * @param connection
147     * @return the query results, which includes meta-data and the result records in JSON object structure format.
148     * @throws NotFoundException if the specified object could not be found. 
149     * @throws BadRequestException if the specified params contain invalid arguments, e.g. a query id that is not
150     * configured, a query expression that is invalid, or missing query substitution tokens.
151     * @throws ForbiddenException if access to the object or specified query is forbidden.
152     * @throws InternalServerErrorException if the operation failed because of a (possibly transient) failure
153     * @throws java.sql.SQLException
154     */
155    
156    public List<Map<String, Object>> query(String type, Map<String, Object> params, Connection connection) 
157                throws SQLException, ResourceException;
158
159    /**
160     * Check if a given queryId exists in our set of known queries
161     *
162     * @param queryId Identifier for the query
163     *
164     * @return true if queryId is available
165     */
166    public boolean queryIdExists(final String queryId);
167    
168    /**
169     * Query if a given exception signifies a well known error type
170     * 
171     * Allows table handlers to abstract database specific differences in reporting errors.
172     * 
173     * @param ex The exception thrown by the database
174     * @param errorType the error type to test against
175     * @return true if the exception matches the error type passed
176     */
177    public boolean isErrorType(SQLException ex, ErrorType errorType);
178    
179    /**
180     * As whether a given exception should be retried
181     * @param ex the exception thrown by the database
182     * @param connection where the failure occured, used for additional context
183     * @return
184     */
185    public boolean isRetryable(SQLException ex, Connection connection);
186}
