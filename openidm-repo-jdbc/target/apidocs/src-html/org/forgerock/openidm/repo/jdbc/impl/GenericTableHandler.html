105    public GenericTableHandler(JsonValue tableConfig, String dbSchemaName, JsonValue queriesConfig, int maxBatchSize, SQLExceptionHandler sqlExceptionHandler) {
106        cfg = GenericTableConfig.parse(tableConfig);
107
108        this.mainTableName = cfg.mainTableName;
109        this.propTableName = cfg.propertiesTableName;
110        this.dbSchemaName = dbSchemaName;
111        if (maxBatchSize < 1) {
112            this.maxBatchSize = 1;
113        } else {
114            this.maxBatchSize = maxBatchSize;
115        }
116
117        if (sqlExceptionHandler == null) {
118            this.sqlExceptionHandler = new DefaultSQLExceptionHandler();
119        } else {
120            this.sqlExceptionHandler = sqlExceptionHandler;
121        }
122
123        queries = new TableQueries(new GenericQueryResultMapper());
124        queryMap = Collections.unmodifiableMap(initializeQueryMap());
125        queries.setConfiguredQueries(mainTableName, propTableName, dbSchemaName, queriesConfig, queryMap);
126
127        // TODO: Consider taking into account DB meta-data rather than just configuration
128        //DatabaseMetaData metadata = connection.getMetaData();
129        //boolean isBatchingSupported = metadata.supportsBatchUpdates();
130        //if (!isBatchingSupported) {
131        //    maxBatchSize = 1;
132        //}
133        enableBatching = (this.maxBatchSize > 1);
134        if (enableBatching) {
135            logger.info("JDBC statement batching enabled, maximum batch size {}", this.maxBatchSize);
136        } else {
137            logger.info("JDBC statement batching disabled.");
138        }

141    protected Map<QueryDefinition, String> initializeQueryMap() {
142        Map<QueryDefinition, String> result = new EnumMap<QueryDefinition, String>(QueryDefinition.class);
143
144        String typeTable = dbSchemaName == null ? "objecttypes" : dbSchemaName + ".objecttypes";
145        String mainTable = dbSchemaName == null ? mainTableName : dbSchemaName + "." + mainTableName;
146        String propertyTable = dbSchemaName == null ? propTableName : dbSchemaName + "." + propTableName;
147
148        // objecttypes table
149        result.put(QueryDefinition.CREATETYPEQUERYSTR, "INSERT INTO " + typeTable + " (objecttype) VALUES (?)");
150        result.put(QueryDefinition.READTYPEQUERYSTR, "SELECT id FROM " + typeTable + " objtype WHERE objtype.objecttype = ?");
151
152        // Main object table
153        result.put(QueryDefinition.READFORUPDATEQUERYSTR, "SELECT obj.* FROM " + mainTable + " obj INNER JOIN " + typeTable + " objtype ON obj.objecttypes_id = objtype.id AND objtype.objecttype = ? WHERE obj.objectid  = ? FOR UPDATE");
154        result.put(QueryDefinition.READQUERYSTR, "SELECT obj.rev, obj.fullobject FROM " + typeTable + " objtype, " + mainTable + " obj WHERE obj.objecttypes_id = objtype.id AND objtype.objecttype = ? AND obj.objectid  = ?");
155        result.put(QueryDefinition.CREATEQUERYSTR, "INSERT INTO " + mainTable + " (objecttypes_id, objectid, rev, fullobject) VALUES (?,?,?,?)");
156        result.put(QueryDefinition.UPDATEQUERYSTR, "UPDATE " + mainTable + " obj SET obj.objectid = ?, obj.rev = ?, obj.fullobject = ? WHERE obj.id = ?");
157        result.put(QueryDefinition.DELETEQUERYSTR, "DELETE obj FROM " + mainTable + " obj INNER JOIN " + typeTable + " objtype ON obj.objecttypes_id = objtype.id AND objtype.objecttype = ? WHERE obj.objectid = ? AND obj.rev = ?");
158
159        /* DB2 Script
160        deleteQueryStr = "DELETE FROM " + dbSchemaName + "." + mainTableName + " obj WHERE EXISTS (SELECT 1 FROM " + dbSchemaName + ".objecttypes objtype WHERE obj.objecttypes_id = objtype.id AND objtype.objecttype = ?) AND obj.objectid = ? AND obj.rev = ?";
161        */
162
163        // Object properties table
164        result.put(QueryDefinition.PROPCREATEQUERYSTR, "INSERT INTO " + propertyTable + " ( " + mainTableName + "_id, propkey, proptype, propvalue) VALUES (?,?,?,?)");
165        result.put(QueryDefinition.PROPDELETEQUERYSTR, "DELETE prop FROM " + propertyTable + " prop INNER JOIN " + mainTable + " obj ON prop." + mainTableName + "_id = obj.id INNER JOIN " + typeTable + " objtype ON obj.objecttypes_id = objtype.id WHERE objtype.objecttype = ? AND obj.objectid = ?");
166        // Default object queries
167        String tableVariable =  dbSchemaName == null ? "${_mainTable}" : "${_dbSchema}.${_mainTable}";
168        result.put(QueryDefinition.QUERYALLIDS, "SELECT obj.objectid FROM " + tableVariable + " obj INNER JOIN " + typeTable + " objtype ON obj.objecttypes_id = objtype.id WHERE objtype.objecttype = ${_resource}");
169
170        return result;
171    }

176    @Override
177    public Resource read(String fullId, String type, String localId, Connection connection)
178            throws ResourceException, SQLException, IOException {
179
180        Resource result = null;
181        Map<String, Object> resultMap = null;
182        PreparedStatement readStatement = null;
183        ResultSet rs = null;
184        try {
185            readStatement = getPreparedStatement(connection, QueryDefinition.READQUERYSTR);
186            logger.trace("Populating prepared statement {} for {}", readStatement, fullId);
187            readStatement.setString(1, type);
188            readStatement.setString(2, localId);
189
190            logger.debug("Executing: {}", readStatement);
191            rs = readStatement.executeQuery();
192            if (rs.next()) {
193                String rev = rs.getString("rev");
194                String objString = rs.getString("fullobject");
195                resultMap = mapper.readValue(objString, typeRef);
196                resultMap.put("_rev", rev);
197                logger.debug(" full id: {}, rev: {}, obj {}", new Object[]{fullId, rev, resultMap});
198                result = new Resource(localId, rev, new JsonValue(resultMap));
199            } else {
200                throw ResourceException.getException(ResourceException.NOT_FOUND,
201                        "Object " + fullId + " not found in " + type);
202            }
203        } finally {
204            CleanupHelper.loggedClose(rs);
205            CleanupHelper.loggedClose(readStatement);
206        }
207
208        return result;
209    }

215    public void create(String fullId, String type, String localId, Map<String, Object> obj, Connection connection)
216            throws SQLException, IOException, InternalServerErrorException {
217
218        long typeId = getTypeId(type, connection); // Note this call can commit and start a new transaction in some cases
219
220        PreparedStatement createStatement = null;
221        try {
222            createStatement = queries.getPreparedStatement(connection, queryMap.get(QueryDefinition.CREATEQUERYSTR), true);
223
224            logger.debug("Create with fullid {}", fullId);
225            String rev = "0";
226            obj.put("_id", localId); // Save the id in the object
227            obj.put("_rev", rev); // Save the rev in the object, and return the changed rev from the create.
228            String objString = mapper.writeValueAsString(obj);
229
230            logger.trace("Populating statement {} with params {}, {}, {}, {}",
231                    new Object[]{createStatement, typeId, localId, rev, objString});
232            createStatement.setLong(1, typeId);
233            createStatement.setString(2, localId);
234            createStatement.setString(3, rev);
235            createStatement.setString(4, objString);
236            logger.debug("Executing: {}", createStatement);
237            int val = createStatement.executeUpdate();
238
239            ResultSet keys = createStatement.getGeneratedKeys();
240            boolean validKeyEntry = keys.next();
241            if (!validKeyEntry) {
242                throw new InternalServerErrorException("Object creation for " + fullId + " failed to retrieve an assigned ID from the DB.");
243            }
244            long dbId = keys.getLong(1);
245
246            logger.debug("Created object for id {} with rev {}", fullId, rev);
247            JsonValue jv = new JsonValue(obj);
248            writeValueProperties(fullId, dbId, localId, jv, connection);
249        } finally {
250            CleanupHelper.loggedClose(createStatement);
251        }
252    }

254    /**
255     * Writes all properties of a given resource to the properties table and links them to the main table record.
256     *
257     * @param fullId the full URI of the resource the belongs to
258     * @param dbId the generated identifier to link the properties table with the main table (foreign key)
259     * @param localId the local identifier of the resource these properties belong to
260     * @param value the JSON value with the properties to write
261     * @param connection the DB connection
262     * @throws SQLException if the insert failed
263     */
264    void writeValueProperties(String fullId, long dbId, String localId, JsonValue value, Connection connection) throws SQLException {
265        if (cfg.hasPossibleSearchableProperties()) {
266            Integer batchingCount = 0;
267            PreparedStatement propCreateStatement = getPreparedStatement(connection, QueryDefinition.PROPCREATEQUERYSTR);
268            try {
269                batchingCount = writeValueProperties(fullId, dbId, localId, value, connection, propCreateStatement, batchingCount);
270                if (enableBatching && batchingCount > 0) {
271                    int[] numUpdates = propCreateStatement.executeBatch();
272                    logger.debug("Batch update of objectproperties updated: {}", numUpdates);
273                    if (logger.isDebugEnabled()) {
274                        logger.debug("Writing batch of objectproperties, updated: {}", Arrays.asList(numUpdates));
275                    }
276                    propCreateStatement.clearBatch();
277                }
278            } finally {
279                CleanupHelper.loggedClose(propCreateStatement);
280            }
281        }
282    }
283    /**
284     * Internal recursive function to add/write properties.
285     * If batching is enabled, prepared statements are added to the batch and only executed if they hit the max limit.
286     * After completion returns the number of properties that have only been added to the batch but not yet executed.
287     * The caller is responsible for executing the batch on remaining items when it deems the batch complete.
288     *
289     * If batching is not enabled, prepared statements are immediately executed.
290     *
291     * @param fullId the full URI of the resource the belongs to
292     * @param dbId the generated identifier to link the properties table with the main table (foreign key)
293     * @param localId the local identifier of the resource these properties belong to
294     * @param value the JSON value with the properties to write
295     * @param connection the DB connection
296     * @param propCreateStatement the prepared properties insert statement
297     * @param batchingCount the current number of statements that have been batched and not yet executed on the prepared statement
298     * @return status of the current batchingCount, i.e. how many statements are not yet executed in the PreparedStatement
299     * @throws SQLException if the insert failed
300     */
301    private int writeValueProperties(String fullId, long dbId, String localId, JsonValue value, Connection connection,
302            PreparedStatement propCreateStatement, int batchingCount) throws SQLException {
303
304        for (JsonValue entry : value) {
305            JsonPointer propPointer = entry.getPointer();
306            if (cfg.isSearchable(propPointer)) {
307                String propkey = propPointer.toString();
308                if (entry.isMap() || entry.isList()) {
309                    batchingCount = writeValueProperties(fullId, dbId, localId, entry, connection, propCreateStatement, batchingCount);
310                } else {
311                    String propvalue = null;
312                    Object val = entry.getObject();
313                    if (val != null) {
314                        propvalue = val.toString(); // TODO: proper type conversions?
315                    }
316                    String proptype = null;
317                    if (propvalue != null) {
318                        proptype = entry.getObject().getClass().getName(); // TODO: proper type info
319                    }
320                    if (logger.isTraceEnabled()) {
321                        logger.trace("Populating statement {} with params {}, {}, {}, {}, {}",
322                                new Object[]{propCreateStatement, dbId, localId, propkey, proptype, propvalue});
323                    }
324                    propCreateStatement.setLong(1, dbId);
325                    propCreateStatement.setString(2, propkey);
326                    propCreateStatement.setString(3, proptype);
327                    propCreateStatement.setString(4, propvalue);
328                    logger.debug("Executing: {}", propCreateStatement);
329                    if (enableBatching) {
330                        propCreateStatement.addBatch();
331                        batchingCount++;
332                    } else {
333                        int numUpdate = propCreateStatement.executeUpdate();
334                    }
335                    if (logger.isTraceEnabled()) {
336                        logger.trace("Inserting objectproperty id: {} propkey: {} proptype: {}, propvalue: {}", new Object[]{fullId, propkey, proptype, propvalue});
337                    }
338                }
339                if (enableBatching && batchingCount >= maxBatchSize) {
340                    int[] numUpdates = propCreateStatement.executeBatch();
341                    if (logger.isDebugEnabled()) {
342                        logger.debug("Batch limit reached, update of objectproperties updated: {}", Arrays.asList(numUpdates));
343                    }
344                    propCreateStatement.clearBatch();
345                    batchingCount = 0;
346                }
347            }
348        }
349
350        return batchingCount;
351    }

367    // Ensure type is in objecttypes table and get its assigned id
368    // Callers should note that this may commit a transaction and start a new one if a new type gets added
369    long getTypeId(String type, Connection connection) throws SQLException, InternalServerErrorException {
370        Exception detectedEx = null;
371        long typeId = readTypeId(type, connection);
372        if (typeId < 0) {
373            connection.setAutoCommit(true); // Commit the new type right away, and have no transaction isolation for read
374            try {
375                createTypeId(type, connection);
376
377            } catch (SQLException ex) {
378                // Rather than relying on DB specific ignore if exists functionality handle it here
379                // Could extend this in the future to more explicitly check for duplicate key error codes, but these again can be DB specific
380                detectedEx = ex;
381            }
382            typeId = readTypeId(type, connection);
383            if (typeId < 0) {
384                throw new InternalServerErrorException("Failed to populate and look up objecttypes table, no id could be retrieved for " + type, detectedEx);
385            }
386            connection.setAutoCommit(false); // Start another transaction
387        }
388        return typeId;
389    }

392     * @param type       the object type URI
393     * @param connection the DB connection
394     * @return the typeId for the given type if exists, or -1 if does not exist
395     * @throws java.sql.SQLException
396     */
397    long readTypeId(String type, Connection connection) throws SQLException {
398        long typeId = -1;
399
400        Map<String, Object> result = null;
401        ResultSet rs = null;
402        PreparedStatement readTypeStatement = null;
403        try {
404            readTypeStatement = getPreparedStatement(connection, QueryDefinition.READTYPEQUERYSTR);
405
406            logger.trace("Populating prepared statement {} for {}", readTypeStatement, type);
407            readTypeStatement.setString(1, type);
408
409            logger.debug("Executing: {}", readTypeStatement);
410            rs = readTypeStatement.executeQuery();
411            if (rs.next()) {
412                typeId = rs.getLong("id");
413                logger.debug("Type: {}, id: {}", new Object[]{type, typeId});
414            }
415        } finally {
416            CleanupHelper.loggedClose(rs);
417            CleanupHelper.loggedClose(readTypeStatement);
418        }
419        return typeId;
420    }

441    /**
442     * Reads an object with for update locking applied
443     *
444     * Note: statement associated with the returned resultset
445     * is not closed upon return.
446     * Aside from taking care to close the resultset it also is
447     * the responsibility of the caller to close the associated
448     * statement. Although the specification specifies that drivers/pools
449     * should close the statement automatically, not all do this reliably.
450     *
451     * @param fullId qualified id of component type and id
452     * @param type the component type
453     * @param localId the id of the object within the component type
454     * @param connection the connection to use
455     * @return the row for the requested object, selected FOR UPDATE
456     * @throws NotFoundException if the requested object was not found in the DB
457     * @throws java.sql.SQLException for general DB issues
458     */
459    public ResultSet readForUpdate(String fullId, String type, String localId, Connection connection)
460            throws NotFoundException, SQLException {
461
462        PreparedStatement readForUpdateStatement = null;
463        ResultSet rs = null;
464        try {
465            readForUpdateStatement = getPreparedStatement(connection, QueryDefinition.READFORUPDATEQUERYSTR);
466            logger.trace("Populating prepared statement {} for {}", readForUpdateStatement, fullId);
467            readForUpdateStatement.setString(1, type);
468            readForUpdateStatement.setString(2, localId);
469
470            logger.debug("Executing: {}", readForUpdateStatement);
471            rs = readForUpdateStatement.executeQuery();
472            if (rs.next()) {
473                logger.debug("Read for update full id: {}", fullId);
474                return rs;
475            } else {
476                CleanupHelper.loggedClose(rs);
477                CleanupHelper.loggedClose(readForUpdateStatement);
478                throw new NotFoundException("Object " + fullId + " not found in " + type);
479            }
480        } catch (SQLException ex) {
481            CleanupHelper.loggedClose(rs);
482            CleanupHelper.loggedClose(readForUpdateStatement);
483            throw ex;
484        }
485    }

487    /* (non-Javadoc)
488    * @see org.forgerock.openidm.repo.jdbc.impl.TableHandler#update(java.lang.String, java.lang.String, java.lang.String, java.lang.String, java.util.Map, java.sql.Connection)
489    */
490    @Override
491    public void update(String fullId, String type, String localId, String rev, Map<String, Object> obj, Connection connection)
492            throws SQLException, IOException, PreconditionFailedException, NotFoundException, InternalServerErrorException {
493        logger.debug("Update with fullid {}", fullId);
494
495        int revInt = Integer.parseInt(rev);
496        ++revInt;
497        String newRev = Integer.toString(revInt);
498        obj.put("_rev", newRev); // Save the rev in the object, and return the changed rev from the create.
499
500        ResultSet rs = null;
501        PreparedStatement updateStatement = null;
502        PreparedStatement deletePropStatement = null;
503        try {
504            rs = readForUpdate(fullId, type, localId, connection);
505            String existingRev = rs.getString("rev");
506            long dbId = rs.getLong("id");
507            long objectTypeDbId = rs.getLong("objecttypes_id");
508            logger.debug("Update existing object {} rev: {} db id: {}, object type db id: {}", new Object[]{fullId, existingRev, dbId, objectTypeDbId});
509
510            if (!existingRev.equals(rev)) {
511                throw new PreconditionFailedException("Update rejected as current Object revision " + existingRev + " is different than expected by caller (" + rev + "), the object has changed since retrieval.");
512            }
513            updateStatement = getPreparedStatement(connection, QueryDefinition.UPDATEQUERYSTR);
514            deletePropStatement = getPreparedStatement(connection, QueryDefinition.PROPDELETEQUERYSTR);
515
516            // Support changing object identifier
517            String newLocalId = (String) obj.get("_id");
518            if (newLocalId != null && !localId.equals(newLocalId)) {
519                logger.debug("Object identifier is changing from " + localId + " to " + newLocalId);
520            } else {
521                newLocalId = localId; // If it hasn't changed, use the existing ID
522                obj.put("_id", newLocalId); // Ensure the ID is saved in the object
523            }
524            String objString = mapper.writeValueAsString(obj);
525
526            logger.trace("Populating prepared statement {} for {} {} {} {} {}", new Object[]{updateStatement, fullId, newLocalId, newRev, objString, dbId});
527            updateStatement.setString(1, newLocalId);
528            updateStatement.setString(2, newRev);
529            updateStatement.setString(3, objString);
530            updateStatement.setLong(4, dbId);
531            logger.debug("Update statement: {}", updateStatement);
532            int updateCount = updateStatement.executeUpdate();
533            logger.trace("Updated rows: {} for {}", updateCount, fullId);
534            if (updateCount != 1) {
535                throw new InternalServerErrorException("Update execution did not result in updating 1 row as expected. Updated rows: " + updateCount);
536            }
537
538            JsonValue jv = new JsonValue(obj);
539            // TODO: only update what changed?
540            logger.trace("Populating prepared statement {} for {} {} {}", new Object[]{deletePropStatement, fullId, type, localId});
541            deletePropStatement.setString(1, type);
542            deletePropStatement.setString(2, localId);
543            logger.debug("Update properties del statement: {}", deletePropStatement);
544            int deleteCount = deletePropStatement.executeUpdate();
545            logger.trace("Deleted child rows: {} for: {}", deleteCount, fullId);
546            writeValueProperties(fullId, dbId, localId, jv, connection);
547        } finally {
548            if (rs != null) {
549                // Ensure associated statement also is closed
550                Statement rsStatement = rs.getStatement();
551                CleanupHelper.loggedClose(rs);
552                CleanupHelper.loggedClose(rsStatement);
553            }
554            CleanupHelper.loggedClose(updateStatement);
555            CleanupHelper.loggedClose(deletePropStatement);
556        }
557    }

559    /**
560     * @see org.forgerock.openidm.repo.jdbc.internal.GenericTableHandler#delete(java.lang.String, java.lang.String, java.lang.String, java.lang.String, java.sql.Connection)
561     */
562    @Override
563    public void delete(String fullId, String type, String localId, String rev, Connection connection)
564            throws PreconditionFailedException, InternalServerErrorException, NotFoundException, SQLException, IOException {
565        logger.debug("Delete with fullid {}", fullId);
566
567        // First check if the revision matches and select it for UPDATE
568        ResultSet existing = null;
569        PreparedStatement deleteStatement = null;
570        try {
571            try {
572                existing = readForUpdate(fullId, type, localId, connection);
573            } catch (NotFoundException ex) {
574                throw new NotFoundException("Object does not exist for delete on: " + fullId);
575            }
576            String existingRev = existing.getString("rev");
577            if (!"*".equals(rev) && !rev.equals(existingRev)) {
578                throw new PreconditionFailedException("Delete rejected as current Object revision " + existingRev + " is different than "
579                        + "expected by caller " + rev + ", the object has changed since retrieval.");
580            }
581
582            // Proceed with the valid delete
583            deleteStatement = getPreparedStatement(connection, QueryDefinition.DELETEQUERYSTR);
584            logger.trace("Populating prepared statement {} for {} {} {} {}", new Object[]{deleteStatement, fullId, type, localId, rev});
585
586            // Rely on ON DELETE CASCADE for connected object properties to be deleted
587            deleteStatement.setString(1, type);
588            deleteStatement.setString(2, localId);
589            deleteStatement.setString(3, rev);
590            logger.debug("Delete statement: {}", deleteStatement);
591
592            int deletedRows = deleteStatement.executeUpdate();
593            logger.trace("Deleted {} rows for id : {} {}", deletedRows, localId);
594            if (deletedRows < 1) {
595                throw new InternalServerErrorException("Deleting object for " + fullId + " failed, DB reported " + deletedRows + " rows deleted");
596            } else {
597                logger.debug("delete for id succeeded: {} revision: {}", localId, rev);
598            }
599        } finally {
600            if (existing != null) {
601                // Ensure associated statement also is closed
602                Statement existingStatement = existing.getStatement();
603                CleanupHelper.loggedClose(existing);
604                CleanupHelper.loggedClose(existingStatement);
605            }
606            CleanupHelper.loggedClose(deleteStatement);
607        }
608    }
609
610    /* (non-Javadoc)
611     * @see org.forgerock.openidm.repo.jdbc.impl.TableHandler#delete(java.lang.String, java.lang.String, java.lang.String, java.lang.String, java.sql.Connection)
612     */
613    @Override
614    public List<Map<String, Object>> query(String type, Map<String, Object> params, Connection connection)
615            throws ResourceException {
616        return queries.query(type, params, connection);
617    }
618
619    @Override
620    public String toString() {
621        return "Generic handler mapped to [" + mainTableName + ", " + propTableName + "]";
622    }
623
624    protected PreparedStatement getPreparedStatement(Connection connection, QueryDefinition queryDefinition) throws SQLException {
625        return queries.getPreparedStatement(connection, queryMap.get(queryDefinition));
626    }
627}

629class GenericQueryResultMapper implements QueryResultMapper {
630    final static Logger logger = LoggerFactory.getLogger(GenericQueryResultMapper.class);
631
632    // Jackson parser
633    ObjectMapper mapper = new ObjectMapper();
634    // Type information for the Jackson parser
635    TypeReference<LinkedHashMap<String,Object>> typeRef = new TypeReference<LinkedHashMap<String,Object>>() {};
636
637    public List<Map<String, Object>> mapQueryToObject(ResultSet rs, String queryId, String type, Map<String, Object> params,  TableQueries tableQueries)
638            throws SQLException, IOException {
639        List<Map<String, Object>> result = new ArrayList<Map<String, Object>>();
640        ResultSetMetaData rsMetaData = rs.getMetaData();
641        boolean hasFullObject = tableQueries.hasColumn(rsMetaData, "fullobject");
642        boolean hasId = false;
643        boolean hasRev = false;
644        boolean hasPropKey = false;
645        boolean hasPropValue = false;
646        boolean hasTotal = false;
647        if (!hasFullObject) {
648            hasId = tableQueries.hasColumn(rsMetaData, "objectid");
649            hasRev = tableQueries.hasColumn(rsMetaData, "rev");
650            hasPropKey = tableQueries.hasColumn(rsMetaData, "propkey");
651            hasPropValue = tableQueries.hasColumn(rsMetaData, "propvalue");
652            hasTotal = tableQueries.hasColumn(rsMetaData, "total");
653        }
654        while (rs.next()) {
655            if (hasFullObject) {
656                String objString = rs.getString("fullobject");
657                Map<String, Object> obj = mapper.readValue(objString, typeRef);
658
659                // TODO: remove data logging
660                logger.trace("Query result for queryId: {} type: {} converted obj: {}", new Object[] {queryId, type, obj});
661
662                result.add(obj);
663            } else {
664                Map<String, Object> obj = new HashMap<String, Object>();
665                if (hasId) {
666                    obj.put("_id", rs.getString("objectid"));
667                }
668                if (hasRev) {
669                    obj.put("_rev", rs.getString("rev"));
670                }
671                if (hasTotal) {
672                    obj.put("total", rs.getInt("total"));
673                }
674                // Results from query on individual searchable property
675                if (hasPropKey && hasPropValue) {
676                    String propKey = rs.getString("propkey");
677                    Object propValue = rs.getObject("propvalue");
678                    JsonPointer pointer = new JsonPointer(propKey);
679                    JsonValue wrapped = new JsonValue(obj);
680                    wrapped.put(pointer, propValue);
681                }
682                result.add(obj);
683            }
684        }
685        return result;
686    }
687}

689class GenericTableConfig {
690    public String mainTableName;
691    public String propertiesTableName;
692    public boolean searchableDefault;
693    public GenericPropertiesConfig properties;
694
695    public boolean isSearchable(JsonPointer propPointer) {
696
697        // More specific configuration takes precedence
698        Boolean explicit = null;
699        while (!propPointer.isEmpty() && explicit == null) {
700            explicit = properties.explicitlySearchable.get(propPointer);
701            propPointer = propPointer.parent();
702        }
703        
704        if (explicit != null) {
705            return explicit.booleanValue();
706        } else {
707            return searchableDefault;
708        }
709    }
710    
711    /**  
712     * @return Approximation on whether this may have searchable properties
713     * It is only an approximation as we do not have an exhaustive list of possible properties
714     * to consider against a default setting of searchable.
715     */
716    public boolean hasPossibleSearchableProperties() {
717        return ((searchableDefault) ? true : properties.explicitSearchableProperties);
718    }
719
720    public static GenericTableConfig parse(JsonValue tableConfig) {
721        GenericTableConfig cfg = new GenericTableConfig();
722        tableConfig.required();
723        cfg.mainTableName = tableConfig.get("mainTable").required().asString();
724        cfg.propertiesTableName = tableConfig.get("propertiesTable").required().asString();
725        cfg.searchableDefault = tableConfig.get("searchableDefault").defaultTo(Boolean.TRUE).asBoolean().booleanValue();
726        cfg.properties = GenericPropertiesConfig.parse(tableConfig.get("properties"));
727
728        return cfg;
729    }
730}

732class GenericPropertiesConfig {
733    public Map<JsonPointer, Boolean> explicitlySearchable = new HashMap<JsonPointer, Boolean>();
734    public String mainTableName;
735    public String propertiesTableName;
736    public boolean searchableDefault;
737    public GenericPropertiesConfig properties;
738    // Whether there are any properties explicitly set to searchable true
739    public boolean explicitSearchableProperties;
740
741    public static GenericPropertiesConfig parse(JsonValue propsConfig) {
742        
743        GenericPropertiesConfig cfg = new GenericPropertiesConfig();
744        if (!propsConfig.isNull()) {
745            for (String propName : propsConfig.keys()) {
746                JsonValue detail = propsConfig.get(propName);
747                boolean propSearchable = detail.get("searchable").asBoolean();
748                cfg.explicitlySearchable.put(new JsonPointer(propName), propSearchable);
749                if (propSearchable) {
750                    cfg.explicitSearchableProperties = true;
751                }
752            }
753        }
754
755        return cfg;
756    }
