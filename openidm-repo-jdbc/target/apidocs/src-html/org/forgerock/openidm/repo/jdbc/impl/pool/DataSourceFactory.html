024package org.forgerock.openidm.repo.jdbc.impl.pool;
025
026import com.jolbox.bonecp.BoneCPDataSource;
027import org.codehaus.jackson.map.DeserializationConfig;
028import org.codehaus.jackson.map.ObjectMapper;
029import org.forgerock.json.fluent.JsonValue;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033import javax.sql.DataSource;
034
035/**
036 * @author $author$
037 * @version $Revision$ $Date$
038 */
039public class DataSourceFactory {
040    private final static Logger logger = LoggerFactory.getLogger(DataSourceFactory.class);
041
042    public static DataSource newInstance(JsonValue config) {
043        //TODO Make CP implementation independent
044        ObjectMapper mapper = new ObjectMapper();
045        mapper.getDeserializationConfig().set(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
046        BoneCPDataSource ds = mapper.convertValue(config.asMap(), BoneCPDataSource.class);
047        ds.setConnectionHook(new DatabaseShutdownHook());
048        ds.setTransactionRecoveryEnabled(true);// Important: This should be enabled
049        ds.setAcquireRetryAttempts(10);//default is 5
050        ds.setReleaseHelperThreads(5);
051
052        // Default if not explicitly set
053        if (ds.getMaxConnectionsPerPartition() < 1) {
054            ds.setMinConnectionsPerPartition(1);
055            ds.setMaxConnectionsPerPartition(20);
056        }
057
058        /*
059        // Settings to enable connection testing with BoneCP 0.8 snapshot
060        ds.setDetectUnclosedStatements(true); // Debug setting
061        ds.setCloseConnectionWatchTimeoutInMs(1000); // Debug setting, not for production
062        ds.setCloseConnectionWatch(true); // Debug setting, not for production
063        ds.setStatementsCacheSize(20); // This caching may already be done by DB driver
064        */
065
066        logger.debug("BoneCPDataSource: {}", ds);
067        return ds;
068    }
069}
