104    /**
105     * Get a prepared statement for the given connection and SQL. May come from
106     * a cache (either local or the host container)
107     *
108     * @param connection
109     *            db connection to get a prepared statement for
110     * @param sql
111     *            the prepared statement SQL
112     * @param autoGeneratedKeys
113     *            whether to return auto-generated keys by the DB
114     * @return the prepared statement
115     * @throws SQLException
116     *             if parsing or retrieving the prepared statement failed
117     */
118    public PreparedStatement getPreparedStatement(Connection connection, String sql,
119            boolean autoGeneratedKeys) throws SQLException {
120        PreparedStatement statement = null;
121        // This is where local prepared statement caching could be added for
122        // stand-alone operation.
123
124        // In the context of a (JavaEE) container rely on its built-in prepared
125        // statement caching
126        // rather than doing it explicitly here.
127        if (autoGeneratedKeys) {
128            statement = connection.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS);
129        } else {
130            statement = connection.prepareStatement(sql);
131        }
132
133        return statement;
134    }

136    /**
137     * Get a prepared statement for the given connection and SQL. Returns the
138     * generated Key This is a function used by OracleTableHandler. Since ORACLE
139     * does not return the auto incremented key but the ROWID on using
140     * getGeneratedKeys(), we have to pass a string array containing the column
141     * that has been auto incremented. I.E. passing 'id' as the only entry of
142     * this array to this method will return the value of the id-column instead
143     * of the ROWID
144     *
145     * @param connection
146     *            db connection to get a prepared statement for
147     * @param sql
148     *            the prepared statement SQL
149     * @param columns
150     *            which column shall be returned as the value of
151     *            PreparedStatement.getGeneratedKeys()
152     * @return the prepared statement
153     * @throws SQLException
154     *             if parsing or retrieving the prepared statement failed
155     */
156    public PreparedStatement getPreparedStatement(Connection connection, String sql,
157            String[] columns) throws SQLException {
158        PreparedStatement statement = null;
159
160        statement = connection.prepareStatement(sql, columns);
161
162        return statement;
163    }

165    /**
166     * Execute a query, either a pre-configured query by using the query ID, or
167     * a query expression passed as part of the params.
168     *
169     * The keys for the input parameters as well as the return map entries are
170     * in QueryConstants.
171     *
172     * @param type
173     *            the resource component name targeted by the URI
174     * @param params
175     *            the parameters which include the query id, or the query
176     *            expression, as well as the token key/value pairs to replace in
177     *            the query
178     * @param con
179     *            a handle to a database connection newBuilder for exclusive use
180     *            by the query method whilst it is executing.
181     * @return The query result, which includes meta-data about the query, and
182     *         the result set itself.
183     * @throws BadRequestException
184     *             if the passed request parameters are invalid, e.g. missing
185     *             query id or query expression or tokens.
186     * @throws InternalServerErrorException
187     *             if the preparing or executing the query fails because of
188     *             configuration or DB issues
189     */
190    public List<Map<String, Object>> query(final String type, Map<String, Object> params, Connection con)
191            throws ResourceException {
192
193        List<Map<String, Object>> result = null;
194        params.put(ServerConstants.RESOURCE_NAME, type);
195
196        // If paged results are requested then decode the cookie in order to determine
197        // the index of the first result to be returned.
198        final int requestPageSize = (Integer) params.get(PAGE_SIZE);
199
200        final String offsetParam;
201        final String pageSizeParam;
202        final String pageClause;
203
204        if (requestPageSize > 0) {
205            offsetParam = String.valueOf((Integer) params.get(PAGED_RESULTS_OFFSET));
206            pageSizeParam = String.valueOf(requestPageSize);
207            pageClause = "SKIP " + offsetParam + " LIMIT " + pageSizeParam;
208        } else {
209            offsetParam = "0";
210            pageSizeParam = String.valueOf(Integer.MAX_VALUE);
211            pageClause = "";
212        }
213
214        params.put(PAGED_RESULTS_OFFSET, offsetParam);
215        params.put(PAGE_SIZE, pageSizeParam);
216
217        String queryExpression = (String) params.get(QUERY_EXPRESSION);
218        String queryId = (String) params.get(QUERY_ID);
219        if (queryId == null && queryExpression == null) {
220            throw new BadRequestException("Either " + QUERY_ID + " or " + QUERY_EXPRESSION
221                    + " to identify/define a query must be passed in the parameters. " + params);
222        }
223        PreparedStatement foundQuery = null;
224        try {
225            if (queryExpression != null) {
226                foundQuery = resolveInlineQuery(con, queryExpression, params);
227            } else {
228                foundQuery = getQuery(con, queryId, type, params);
229            }
230        } catch (SQLException ex) {
231            throw new InternalServerErrorException("DB reported failure preparing query: "
232                    + (foundQuery == null ? "null" : foundQuery.toString()) + " with params: " + params + " error code: " + ex.getErrorCode()
233                    + " sqlstate: " + ex.getSQLState() + " message: " + ex.getMessage(), ex);
234        }
235        if (foundQuery == null) {
236            throw new BadRequestException("The passed query identifier " + queryId
237                    + " does not match any configured queries on the OrientDB repository service.");
238        }
239
240        Name eventName = getEventName(queryId);
241        EventEntry measure = Publisher.start(eventName, foundQuery, null);
242        ResultSet rs = null;
243        try {
244            rs = foundQuery.executeQuery();
245            result = resultMapper.mapQueryToObject(rs, queryId, type, params, this);
246            measure.setResult(result);
247        } catch (SQLException ex) {
248            throw new InternalServerErrorException("DB reported failure executing query "
249                    + foundQuery.toString() + " with params: " + params + " error code: "
250                    + ex.getErrorCode() + " sqlstate: " + ex.getSQLState() + " message: "
251                    + ex.getMessage(), ex);
252        } catch (IOException ex) {
253            throw new InternalServerErrorException("Failed to convert result objects for query "
254                    + foundQuery.toString() + " with params: " + params + " message: "
255                    + ex.getMessage(), ex);
256        } finally {
257            CleanupHelper.loggedClose(rs);
258            CleanupHelper.loggedClose(foundQuery);
259            measure.end();
260        }
261        return result;
262    }

264    /**
265     * Whether a result set contains a given column
266     *
267     * @param rsMetaData
268     *            result set meta data
269     * @param columnName
270     *            name of the column to look for
271     * @return true if it is present
272     * @throws SQLException
273     *             if meta data inspection failed
274     */
275    public boolean hasColumn(ResultSetMetaData rsMetaData, String columnName) throws SQLException {
276        for (int colPos = 1; colPos <= rsMetaData.getColumnCount(); colPos++) {
277            if (columnName.equalsIgnoreCase(rsMetaData.getColumnName(colPos))) {
278                return true;
279            }
280        }
281        return false;
282    }

284    /**
285     * Resolves a full query expression Currently does not support token
286     * replacement
287     *
288     * @param con
289     *            The db connection
290     * @param queryExpression
291     *            the native query string
292     * @param params
293     *            parameters passed to the resource query
294     * @return A resolved statement
295     */
296    PreparedStatement resolveInlineQuery(Connection con, String queryExpression,
297            Map<String, Object> params) throws SQLException, ResourceException {
298        // No token replacement on expressions for now
299        List<String> tokenNames = new ArrayList<String>();
300        QueryInfo info = new QueryInfo(queryExpression, tokenNames);
301        PreparedStatement stmt = resolveQuery(info, con, params);
302        return stmt;
303    }

305    /**
306     * Gets and resolves a query by id, using token substitution
307     *
308     * @param con
309     *            The db connection
310     * @param queryId
311     *            the unique identifier of the paramterized, pre-defined query
312     * @param type
313     *            the resource component name targeted by the URI
314     * @param params
315     *            the paramteris passed into the query call
316     * @return The statement
317     * @throws SQLException
318     *             if resolving the statement failed
319     * @throws BadRequestException
320     *             if no query is defined for the given identifier
321     */
322    PreparedStatement getQuery(Connection con, String queryId, String type,
323            Map<String, Object> params) throws SQLException, ResourceException {
324
325        QueryInfo info = queries.get(queryId);
326        if (info == null) {
327            throw new BadRequestException("No query defined/configured for requested queryId "
328                    + queryId);
329        }
330        PreparedStatement stmt = resolveQuery(info, con, params);
331        return stmt;
332    }

345    /**
346     * Resolves a query, given a QueryInfo
347     *
348     * @param info
349     *            The info encapsulating the query information
350     * @param con
351     *            the db connection
352     * @param params
353     *            the paramters passed to query
354     * @return the resolved query
355     * @throws SQLException
356     *             if resolving the query failed
357     */
358    PreparedStatement resolveQuery(QueryInfo info, Connection con, Map<String, Object> params)
359            throws SQLException, ResourceException {
360        String queryStr = info.getQueryString();
361        List<String> tokenNames = info.getTokenNames();
362
363        // replace ${list:variable} tokens with the correct number of bind variables
364        Map<String, Integer> listReplacements = new HashMap<String, Integer>();
365        for (String tokenName : tokenNames) {
366            String[] tokenParts = tokenName.split(":", 2);
367            if (PREFIX_LIST.equals(tokenParts[0]) && params.containsKey(tokenParts[1])) {
368                listReplacements.put(tokenName, ((String) params.get(tokenParts[1])).split(",").length);
369            }
370        }
371        if (listReplacements.size() > 0) {
372            TokenHandler tokenHandler = new TokenHandler();
373            queryStr = tokenHandler.replaceListTokens(queryStr, listReplacements, "?");
374        }
375
376        // now prepare the statement using the correct number of bind variables
377        PreparedStatement statement = getPreparedStatement(con, queryStr);
378        int count = 1; // DB column count starts at 1
379        for (String tokenName : tokenNames) {
380            String[] tokenParts = tokenName.split(":", 2);
381            if (tokenParts.length == 1) {
382                // handle single value - assume String
383                Object objValue =  params.get(tokenName);
384                String value = null;
385                if (objValue != null) {
386                    value = objValue.toString();
387                } else {
388                    // fail with an exception if token not found
389                    throw new BadRequestException("Missing entry in params passed to query for token " + tokenName);
390                }
391                statement.setString(count, value);
392                count++;
393            }
394            else {
395                Object objValue =  params.get(tokenParts[1]);
396                if (objValue == null) {
397                    // fail with an exception if token not found
398                    throw new BadRequestException("Missing entry in params passed to query for token " + tokenName);
399                }
400                if (PREFIX_INT.equals(tokenParts[0])) {
401                    // handle single integer value
402                    Integer int_value = null;
403                    if (objValue != null) {
404                        int_value = Integer.parseInt(objValue.toString());
405                    }
406                    statement.setInt(count, int_value);
407                    count++;
408                } else if (PREFIX_LIST.equals(tokenParts[0])) {
409                    // handle list of values - presently assumes Strings, TODO support integer lists
410                    if (objValue != null) {
411                        for (String list_value : objValue.toString().split(",")) {
412                            // if list value is surrounded by single quotes remove them
413                            if (list_value != null && list_value.startsWith("'") && list_value.endsWith("'")) {
414                                list_value = list_value.substring(1, list_value.length()-1);
415                            }
416                            statement.setString(count, list_value);
417                            count++;
418                        }
419                    }
420                    else {
421                        statement.setString(count, null);
422                        count++;
423                    }
424                }
425            }
426        }
427        logger.debug("Prepared statement: {}", statement);
428
429        return statement;
430    }

432    /**
433     * Set the pre-configured queries for generic tables, which are identified
434     * by a query identifier and can be invoked using this identifier
435     *
436     * Success to set the queries does not mean they are valid as some can only
437     * be validated at query execution time.
438     *
439     * @param mainTableName
440     *            name of the table containing the object identifier and all
441     *            related info
442     * @param propTableName
443     *            name of the table holding individual properties for searching
444     *            and indexing
445     * @param dbSchemaName
446     *            the database scheme the table is in
447     * @param queriesConfig
448     *            queries configured in configuration (files)
449     * @param defaultQueryMap
450     *            static default queries already defined for handling this table
451     *            type
452     *
453     *            query details
454     */
455    public void setConfiguredQueries(String mainTableName, String propTableName,
456            String dbSchemaName, JsonValue queriesConfig,
457            Map<QueryDefinition, String> defaultQueryMap) {
458        Map<String, String> replacements = new HashMap<String, String>();
459        replacements.put("_mainTable", mainTableName);
460        replacements.put("_propTable", propTableName);
461        replacements.put("_dbSchema", dbSchemaName);
462        setConfiguredQueries(replacements, queriesConfig, defaultQueryMap);
463    }

465    /**
466     * Set the pre-configured queries for explicitly mapped tables, which are
467     * identified by a query identifier and can be invoked using this identifier
468     *
469     * Success to set the queries does not mean they are valid as some can only
470     * be validated at query execution time.
471     *
472     * @param tableName
473     *            name of the explicitly mapped table
474     * @param dbSchemaName
475     *            the database scheme the table is in
476     * @param queriesConfig
477     *            queries configured in configuration (files)
478     * @param defaultQueryMap
479     *            static default queries already defined for handling this table
480     *            type
481     *
482     *            query details
483     */
484    public void setConfiguredQueries(String tableName, String dbSchemaName,
485            JsonValue queriesConfig, Map<QueryDefinition, String> defaultQueryMap) {
486        Map<String, String> replacements = new HashMap<String, String>();
487        replacements.put("_table", tableName);
488        replacements.put("_dbSchema", dbSchemaName);
489        setConfiguredQueries(replacements, queriesConfig, defaultQueryMap);
490    }
491
492    private void setConfiguredQueries(Map<String, String> replacements, JsonValue queriesConfig,
493            Map<QueryDefinition, String> defaultQueryMap) {
494        queries = new HashMap<String, QueryInfo>();
495
496        if (queriesConfig == null || queriesConfig.isNull()) {
497            queriesConfig = new JsonValue(new HashMap());
498        }
499        // Default query-all-ids to allow bootstrapping of configuration
500        if (!queriesConfig.isDefined(ServerConstants.QUERY_ALL_IDS) && defaultQueryMap != null) {
501            queriesConfig.put(ServerConstants.QUERY_ALL_IDS, defaultQueryMap
502                    .get(QueryDefinition.QUERYALLIDS));
503        }
504
505        for (String queryName : queriesConfig.keys()) {
506            String rawQuery = queriesConfig.get(queryName).required().asString();
507
508            TokenHandler tokenHandler = new TokenHandler();
509            // Replace the table name tokens.
510            String tempQueryString = tokenHandler.replaceSomeTokens(rawQuery, replacements);
511
512            // Convert to ? for prepared statement, populate token replacement
513            // info
514            List<String> tokenNames = tokenHandler.extractTokens(tempQueryString);
515            String queryString = tokenHandler.replaceTokens(tempQueryString, "?", PREFIX_LIST);
516
517            QueryInfo queryInfo = new QueryInfo(queryString, tokenNames);
518            queries.put(queryName, queryInfo);
519            logger.info("Configured query converted to JDBC query {} and tokens {}", queryString,
520                    tokenNames);
521        }
522    }

524    /**
525     * @return the smartevent Name for a given query
526     */
527    Name getEventName(String queryId) {
528        if (queryId == null) {
529            return Name.get(EVENT_RAW_QUERY_PREFIX + "_query_expression");

531            return Name.get(EVENT_RAW_QUERY_PREFIX + queryId);
