024package org.forgerock.openidm.repo.jdbc.impl;
025
026import org.forgerock.json.fluent.JsonValue;
027import org.forgerock.openidm.repo.jdbc.SQLExceptionHandler;
028
029import java.sql.Connection;
030import java.sql.SQLException;
031
032import java.util.Map;
033
034/**
035 * @author $author$
036 * @version $Revision$ $Date$
037 */
038public class DB2TableHandler extends GenericTableHandler {
039
040    public DB2TableHandler(JsonValue tableConfig, String dbSchemaName, JsonValue queriesConfig, int maxBatchSize, SQLExceptionHandler sqlExceptionHandler) {
041        super(tableConfig, dbSchemaName, queriesConfig, maxBatchSize, sqlExceptionHandler);
042    }
043
044    protected Map<QueryDefinition, String> initializeQueryMap() {
045        Map<QueryDefinition, String> result = super.initializeQueryMap();
046        String typeTable = dbSchemaName == null ? "objecttypes" : dbSchemaName + ".objecttypes";
047        String mainTable = dbSchemaName == null ? mainTableName : dbSchemaName + "." + mainTableName;
048        String propertyTable = dbSchemaName == null ? propTableName : dbSchemaName + "." + propTableName;
049
050        // Main object table DB2 Script
051        result.put(QueryDefinition.DELETEQUERYSTR, "DELETE FROM " + mainTable + " obj WHERE EXISTS (SELECT 1 FROM " + typeTable + " objtype WHERE obj.objecttypes_id = objtype.id AND objtype.objecttype = ?) AND obj.objectid = ? AND obj.rev = ?");
052        //TODO Fix this CPU killer query
053        result.put(QueryDefinition.PROPDELETEQUERYSTR, "DELETE FROM " + propertyTable + " WHERE " + mainTableName + "_id = (SELECT obj.id FROM " + mainTable + " obj, " + typeTable + " objtype WHERE obj.objecttypes_id = objtype.id AND objtype.objecttype = ? AND obj.objectid  = ?)");
054        return result;
055    }
056
057    public boolean isRetryable(SQLException ex, Connection connection) {
058        // Re-tryable DB2 error codes
059        // -911 indicates DB2 rolled back already and expects a retry
060        // -912 indicates deadlock or timeout.
061        // -904 indicates resource limit was exceeded.
062        if (-911 == ex.getErrorCode() || -912 == ex.getErrorCode() || -904 == ex.getErrorCode()) {
063            return true;
064        } else {
065            return false;
066        }
067    }
068}
