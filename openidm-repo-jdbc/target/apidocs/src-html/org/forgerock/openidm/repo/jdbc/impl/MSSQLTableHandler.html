025package org.forgerock.openidm.repo.jdbc.impl;
026
027import java.io.IOException;
028import java.sql.Connection;
029import java.sql.PreparedStatement;
030import java.sql.ResultSet;
031import java.sql.SQLException;
032import java.sql.Statement;
033import java.util.Map;
034
035import org.forgerock.json.fluent.JsonValue;
036import org.forgerock.openidm.repo.jdbc.SQLExceptionHandler;
037import static org.forgerock.openidm.repo.jdbc.impl.GenericTableHandler.logger;
038
039public class MSSQLTableHandler extends GenericTableHandler {
040
041    public MSSQLTableHandler(JsonValue tableConfig, String dbSchemaName, JsonValue queriesConfig,
042            int maxBatchSize, SQLExceptionHandler sqlExceptionHandler) {
043        super(tableConfig, dbSchemaName, queriesConfig, maxBatchSize, sqlExceptionHandler);
044    }
045
046    protected Map<QueryDefinition, String> initializeQueryMap() {
047        Map<QueryDefinition, String> result = super.initializeQueryMap();
048        String typeTable = dbSchemaName == null ? "objecttypes" : dbSchemaName + ".objecttypes";
049        String mainTable =
050                dbSchemaName == null ? mainTableName : dbSchemaName + "." + mainTableName;
051        String propertyTable =
052                dbSchemaName == null ? propTableName : dbSchemaName + "." + propTableName;
053
054        result.put(
055                QueryDefinition.READFORUPDATEQUERYSTR,
056                "SELECT obj.* FROM "
057                        + mainTable
058                        + " obj INNER JOIN "
059                        + typeTable
060                        + " objtype ON obj.objecttypes_id = objtype.id AND objtype.objecttype = ? WHERE obj.objectid = ?");
061        result.put(QueryDefinition.UPDATEQUERYSTR,
062                "UPDATE obj SET obj.objectid = ?, obj.rev = ?, obj.fullobject = ? FROM "
063                        + mainTable + " obj WHERE obj.id = ? AND obj.rev = ?");
064
065        return result;
066
067    }
068    
069    /* (non-Javadoc)
070    * @see org.forgerock.openidm.repo.jdbc.impl.TableHandler#update(java.lang.String, java.lang.String, java.lang.String, java.lang.String, java.util.Map, java.sql.Connection)
071    */
072    @Override
073    public void update(String fullId, String type, String localId, String rev, Map<String, Object> obj, Connection connection)
074            throws SQLException, IOException, org.forgerock.json.resource.PreconditionFailedException, org.forgerock.json.resource.NotFoundException, org.forgerock.json.resource.InternalServerErrorException {
075        logger.debug("Update with fullid {}", fullId);
076
077        int revInt = Integer.parseInt(rev);
078        ++revInt;
079        String newRev = Integer.toString(revInt);
080        obj.put("_rev", newRev); // Save the rev in the object, and return the changed rev from the create.
081
082        ResultSet rs = null;
083        PreparedStatement updateStatement = null;
084        PreparedStatement deletePropStatement = null;
085        try {
086            rs = readForUpdate(fullId, type, localId, connection);
087            String existingRev = rs.getString("rev");
088            long dbId = rs.getLong("id");
089            long objectTypeDbId = rs.getLong("objecttypes_id");
090            logger.debug("Update existing object {} rev: {} db id: {}, object type db id: {}", new Object[]{fullId, existingRev, dbId, objectTypeDbId});
091
092            if (!existingRev.equals(rev)) {
093                throw new org.forgerock.json.resource.PreconditionFailedException("Update rejected as current Object revision " + existingRev + " is different than expected by caller (" + rev + "), the object has changed since retrieval.");
094            }
095            updateStatement = getPreparedStatement(connection, QueryDefinition.UPDATEQUERYSTR);
096            deletePropStatement = getPreparedStatement(connection, QueryDefinition.PROPDELETEQUERYSTR);
097
098            // Support changing object identifier
099            String newLocalId = (String) obj.get("_id");
100            if (newLocalId != null && !localId.equals(newLocalId)) {
101                logger.debug("Object identifier is changing from " + localId + " to " + newLocalId);
102            } else {
103                newLocalId = localId; // If it hasn't changed, use the existing ID
104                obj.put("_id", newLocalId); // Ensure the ID is saved in the object
105            }
106            String objString = mapper.writeValueAsString(obj);
107
108            logger.trace("Populating prepared statement {} for {} {} {} {} {} {}", new Object[]{updateStatement, fullId, newLocalId, newRev, objString, dbId, existingRev});
109            updateStatement.setString(1, newLocalId);
110            updateStatement.setString(2, newRev);
111            updateStatement.setString(3, objString);
112            updateStatement.setLong(4, dbId);
113            updateStatement.setString(5, existingRev);
114            logger.debug("Update statement: {}", updateStatement);
115            int updateCount = updateStatement.executeUpdate();
116            logger.trace("Updated rows: {} for {}", updateCount, fullId);
117            if (updateCount == 0) {
118                throw new org.forgerock.json.resource.PreconditionFailedException("Update rejected as current Object revision " + existingRev + ", has changed since retrieval.");
119            } else if (updateCount > 1) {
120                throw new org.forgerock.json.resource.InternalServerErrorException("Update execution did not result in updating 1 row as expected. Updated rows: " + updateCount);
121            }
122
123            JsonValue jv = new JsonValue(obj);
124            // TODO: only update what changed?
125            logger.trace("Populating prepared statement {} for {} {} {}", new Object[]{deletePropStatement, fullId, type, localId});
126            deletePropStatement.setString(1, type);
127            deletePropStatement.setString(2, localId);
128            logger.debug("Update properties del statement: {}", deletePropStatement);
129            int deleteCount = deletePropStatement.executeUpdate();
130            logger.trace("Deleted child rows: {} for: {}", deleteCount, fullId);
131            writeValueProperties(fullId, dbId, localId, jv, connection);
132        } finally {
133            if (rs != null) {
134                // Ensure associated statement also is closed
135                Statement rsStatement = rs.getStatement();
136                CleanupHelper.loggedClose(rs);
137                CleanupHelper.loggedClose(rsStatement);
138            }
139            CleanupHelper.loggedClose(updateStatement);
140            CleanupHelper.loggedClose(deletePropStatement);
141        }
142    }    
143}
