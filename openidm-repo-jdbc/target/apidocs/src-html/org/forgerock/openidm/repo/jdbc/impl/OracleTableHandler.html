025package org.forgerock.openidm.repo.jdbc.impl;
026
027import java.io.IOException;
028import java.sql.Connection;
029import java.sql.PreparedStatement;
030import java.sql.ResultSet;
031import java.sql.SQLException;
032import java.util.Map;
033
034import org.forgerock.json.fluent.JsonValue;
035import org.forgerock.json.resource.InternalServerErrorException;
036import org.forgerock.openidm.repo.jdbc.SQLExceptionHandler;
037import org.slf4j.Logger;
038import org.slf4j.LoggerFactory;
039
040/**
041 * @author $author$
042 * @version $Revision$ $Date$
043 */
044public class OracleTableHandler extends GenericTableHandler {
045    final static Logger logger = LoggerFactory.getLogger(OracleTableHandler.class);
046
047    public OracleTableHandler(JsonValue tableConfig, String dbSchemaName, JsonValue queriesConfig, int maxBatchSize, SQLExceptionHandler sqlExceptionHandler) {
048        super(tableConfig, dbSchemaName, queriesConfig, maxBatchSize, sqlExceptionHandler);
049    }
050
051    @Override
052    public void create(String fullId, String type, String localId, Map<String, Object> obj, Connection connection)
053            throws SQLException, IOException, InternalServerErrorException {
054        connection.setAutoCommit(true);
055        long typeId = getTypeId(type, connection);
056
057        connection.setAutoCommit(false);
058
059        PreparedStatement createStatement = null;
060        try {
061            // Since ORACLE returns the ROWID instead of an autoincremented column, we have to tell the PreparedStatement to
062            // return the value of the "id-column" instead of the rowid. This is done by passing the following array to the PreparedStatement
063            String generatedColumns[] = {"id"};
064            createStatement = queries.getPreparedStatement(connection, queryMap.get(QueryDefinition.CREATEQUERYSTR), generatedColumns);
065
066            logger.debug("Create with fullid {}", fullId);
067            String rev = "0";
068            obj.put("_id", localId); // Save the id in the object
069            obj.put("_rev", rev); // Save the rev in the object, and return the changed rev from the create.
070            String objString = mapper.writeValueAsString(obj);
071
072            logger.trace("Populating statement {} with params {}, {}, {}, {}",
073                    new Object[]{createStatement, typeId, localId, rev, objString});
074            createStatement.setLong(1, typeId);
075            createStatement.setString(2, localId);
076            createStatement.setString(3, rev);
077            createStatement.setString(4, objString);
078            logger.debug("Executing: {}", createStatement);
079            int val = createStatement.executeUpdate();
080
081            ResultSet keys = createStatement.getGeneratedKeys();
082            boolean validKeyEntry = keys.next();
083            if (!validKeyEntry) {
084                throw new InternalServerErrorException("Object creation for " + fullId + " failed to retrieve an assigned ID from the DB.");
085            }
086
087            // Should now contain the value of the autoincremented column
088            long dbId = keys.getLong(1);
089
090            logger.debug("Created object for id {} with rev {}", fullId, rev);
091            JsonValue jv = new JsonValue(obj);
092            writeValueProperties(fullId, dbId, localId, jv, connection);
093        } finally {
094            CleanupHelper.loggedClose(createStatement);
095        }
096    }
097
098    @Override
099    protected Map<QueryDefinition, String> initializeQueryMap() {
100        Map<QueryDefinition, String> result = super.initializeQueryMap();
101        String typeTable = dbSchemaName == null ? "objecttypes" : dbSchemaName + ".objecttypes";
102        String mainTable = dbSchemaName == null ? mainTableName : dbSchemaName + "." + mainTableName;
103        String propertyTable = dbSchemaName == null ? propTableName : dbSchemaName + "." + propTableName;
104
105        // ORACLE is not capable of using the DELETE statements defined in the StandardHandler, therefore we are changing them to
106        // something more ORACLEfriendly (thanks to the one that wrote the DB2 adapter
107        result.put(QueryDefinition.DELETEQUERYSTR, "DELETE FROM " + mainTable + " obj WHERE EXISTS (SELECT 1 FROM " + typeTable + " objtype WHERE obj.objecttypes_id = objtype.id AND objtype.objecttype = ?) AND obj.objectid = ? AND obj.rev = ?");
108        result.put(QueryDefinition.PROPDELETEQUERYSTR, "DELETE FROM " + propertyTable + " WHERE " + mainTableName + "_id = (SELECT obj.id FROM " + mainTable + " obj, " + typeTable + " objtype WHERE obj.objecttypes_id = objtype.id AND objtype.objecttype = ? AND obj.objectid  = ?)");
109
110        return result;
111    }
112}
