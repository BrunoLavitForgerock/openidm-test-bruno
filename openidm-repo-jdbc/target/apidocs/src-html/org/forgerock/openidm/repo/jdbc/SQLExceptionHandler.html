024package org.forgerock.openidm.repo.jdbc;
025
026import java.sql.Connection;
027import java.sql.SQLException;
028
029/**
030 * Interface to help handle SQLExceptions across different DB implementations
031 *
032 * @author aegloff
033 */
034public interface SQLExceptionHandler {
035    
036    /**
037     * Query if a given exception signifies a well known error type
038     * 
039     * Allows table handlers to abstract database specific differences in reporting errors.
040     * 
041     * @param ex The exception thrown by the database
042     * @param errorType the error type to test against
043     * @return true if the exception matches the error type passed
044     */
045    boolean isErrorType(SQLException ex, ErrorType errorType);
046    
047    /**
048     * As whether a given exception should be retried
049     * @param ex the exception thrown by the database
050     * @param connection where the failure occured, used for additional context
051     * @return true if the expectation is that transaction should be retried by the application
052     */
053    boolean isRetryable(SQLException ex, Connection connection);
054}
