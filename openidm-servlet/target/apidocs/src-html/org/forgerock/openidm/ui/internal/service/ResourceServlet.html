075    private static final String CONFIG_RESOURCE_DIR = "resourceDir";

108    @Override
109    protected void doGet(HttpServletRequest req, HttpServletResponse res)
110            throws ServletException, IOException {
111        logger.debug("GET call on {}", req);
112        
113        String target = req.getPathInfo();
114        if (target == null || "/".equals(target)) {
115            res.sendRedirect(req.getServletPath() + "/index.html");
116        } else {
117            target = prependSlash(target);
118
119            File extFile = null;
120            String extFileCanonical = null;
121            String resName = resourceDir + target;
122            String extDir = IdentityServer.getInstance().getProperty("openidm.ui.extension.dir",
123                    "&{launcher.install.location}/ui/extension", true);
124            for (String ext : extFolders) {
125                if (target.startsWith(ext)) {
126                    // Try to find in extensions folder
127                    extFile = new File(extDir + target);
128                    File extDirFile = new File(extDir);
129                    extFileCanonical = extFile.getCanonicalPath();
130                    if (!extFileCanonical.startsWith(extDirFile.getCanonicalPath())) {
131                        extFile = null;
132                    }
133                    break;
134                }
135            }
136            
137            // Look in the bundle rather than the servlet context, as we're using shared servlet contexts
138            URL url = null;
139            if (extFile != null && extFile.exists()) {
140                url = extFile.getCanonicalFile().toURI().toURL();
141            } else {
142                // this handles the case of a servlet request before the BundleListener has associated the
143                // correct bundle to this servlet instance
144                if (bundle == null) {
145                    res.sendError(HttpServletResponse.SC_NOT_FOUND);
146                }
147                url = bundle.getResource(resName);
148            }
149
150            if (url == null) {
151                res.sendError(HttpServletResponse.SC_NOT_FOUND);
152            } else {
153                handle(req, res, url, resName);
154            }
155        }
156    }

158    /**
159     * Initializes the servlet and registers it with the WebContainer.
160     * 
161     * @param context the ComponentContext containing the configuration
162     * @throws ServletException
163     * @throws NamespaceException
164     */
165    private void init(ComponentContext context) throws ServletException, NamespaceException {
166        JsonValue config = new JSONEnhancedConfig().getConfigurationAsJson(context);
167        
168        if (!config.get(CONFIG_ENABLED).isNull() && Boolean.FALSE.equals(config.get(CONFIG_ENABLED).asBoolean())) {
169            logger.info("UI is disabled - not registering UI servlet");
170            return;
171        }
172        else if (config.get(CONFIG_CONTEXT_ROOT) == null || config.get(CONFIG_CONTEXT_ROOT).isNull()) {
173            logger.info("UI does not specify contextRoot - unable to register servlet");
174            return;
175        }
176        else if (config.get(CONFIG_BUNDLE) == null
177                || config.get(CONFIG_BUNDLE).isNull()
178                || !config.get(CONFIG_BUNDLE).isMap()
179                || config.get(CONFIG_BUNDLE).get(CONFIG_NAME) == null
180                || config.get(CONFIG_BUNDLE).get(CONFIG_NAME).isNull()) {
181            logger.info("UI does not specify bundle name - unable to register servlet");
182            return;
183        }
184        else if (config.get(CONFIG_BUNDLE) == null
185                || config.get(CONFIG_BUNDLE).isNull()
186                || !config.get(CONFIG_BUNDLE).isMap()
187                || config.get(CONFIG_BUNDLE).get(CONFIG_RESOURCE_DIR) == null
188                || config.get(CONFIG_BUNDLE).get(CONFIG_RESOURCE_DIR).isNull()) {
189            logger.info("UI does not specify bundle resourceDir - unable to register servlet");
190            return;
191        }
192
193        bundleName = config.get(CONFIG_BUNDLE).get(CONFIG_NAME).asString();
194        resourceDir = prependSlash(config.get(CONFIG_BUNDLE).get(CONFIG_RESOURCE_DIR).asString());
195        contextRoot = prependSlash(config.get(CONFIG_CONTEXT_ROOT).asString());
196
197        if (bundleName != null) {
198            for (Bundle aBundle : context.getBundleContext().getBundles()) {
199                if (bundleName.equals(aBundle.getSymbolicName())) {
200                    this.bundle = aBundle;
201                    break;
202                }
203            }
204        }
205        
206        if (bundle == null) {
207            logger.info("Could not find bundle " + bundleName+ " (not loaded yet?) - will wait for bundle-start");
208        }
209
210        // handle bundle-start events to associate bundle to this ResourceServlet instance;
211        // Felix's filesystem-installer may load the filesystem bundles after this servlet
212        // instance is activated
213        bundleListener = new BundleListener() {
214            public void bundleChanged(BundleEvent event) {
215                if (event == null) {
216                    logger.debug("BundleEvent is null for bundle {}", bundleName);
217                    return;
218                }
219                Bundle bundle = event.getBundle();
220                if (bundle != null && bundle.getSymbolicName() != null && bundle.getSymbolicName().equals(bundleName)) {
221                    if (event.getType() == BundleEvent.STARTED) {
222                        ResourceServlet.this.bundle = bundle;
223                        logger.info("Bundle " + bundleName + " associated with servlet instance");
224                    } else if (event.getType() == BundleEvent.STOPPED) {
225                        ResourceServlet.this.bundle = null;
226                        logger.info("Bundle " + bundleName + " stopped; disassociated with servlet instance");
227                    }
228                }
229            }
230        };
231
232        context.getBundleContext().addBundleListener(bundleListener);
233
234        extFolders = new ArrayList<String>();
235        extFolders.add("/css/");
236        extFolders.add("/images/");
237        extFolders.add("/locales/");
238        extFolders.add("/templates/");
239
240        Dictionary<String, Object> props = new Hashtable<String, Object>();
241        webContainer.registerServlet(contextRoot, this,  props, webContainer.getDefaultSharedHttpContext());
242        logger.debug("Registered UI servlet at {}", contextRoot);
243    }
244    
245    /**
246     * Clears the servlet, unregistering it with the WebContainer and removing the bundle listener.
247     */
248    private void clear() {
249        if (bundleListener != null) {
250            bundle.getBundleContext().removeBundleListener(bundleListener);
251        }
252        webContainer.unregister(contextRoot);
253        logger.debug("Unregistered UI servlet at {}", contextRoot);
254    }

256    private void handle(HttpServletRequest req, HttpServletResponse res, URL url, String resName)
257            throws IOException {
258        String contentType = getServletContext().getMimeType(resName);
259        if (contentType != null) {
260            res.setContentType(contentType);
261        } else {
262            res.setContentType(getMimeType(resName));
263        }
264
265        long lastModified = getLastModified(url);
266        if (lastModified != 0) {
267            res.setDateHeader("Last-Modified", lastModified);
268        }
269
270        if (!resourceModified(lastModified, req.getDateHeader("If-Modified-Since"))) {
271            res.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
272        } else {
273            copyResource(url, res);
274        }
275    }
276
277    private long getLastModified(URL url) {
278        long lastModified = 0;
279
280        try {
281            URLConnection conn = url.openConnection();
282            lastModified = conn.getLastModified();
283        } catch (Exception e) {
284            // Do nothing
285        }
286
287        if (lastModified == 0) {
288            String filepath = url.getPath();
289            if (filepath != null) {
290                File f = new File(filepath);
291                if (f.exists()) {
292                    lastModified = f.lastModified();
293                }
294            }
295        }
296
297        return lastModified;
298    }
299    
300    private String getMimeType(String fileName) {
301        if (fileName.endsWith(".css")) {
302            return "text/css";
303        } else if (fileName.endsWith(".js")) {
304            return "application/javascript";
305        } else if (fileName.endsWith(".png")) {
306            return "image/png";
307        } else if (fileName.endsWith(".html")) {
308            return "text/html";
309        }
310        
311        return null;
312    }
313
314    private boolean resourceModified(long resTimestamp, long modSince) {
315        modSince /= 1000;
316        resTimestamp /= 1000;
317
318        return resTimestamp == 0 || modSince == -1 || resTimestamp > modSince;
319    }
320
321    private void copyResource(URL url, HttpServletResponse res)
322            throws IOException {
323        OutputStream os = null;
324        InputStream is = null;
325
326        try {
327            os = res.getOutputStream();
328            is = url.openStream();
329
330            int len = 0;
331            byte[] buf = new byte[1024];
332            int n;
333
334            while ((n = is.read(buf, 0, buf.length)) >= 0) {
335                os.write(buf, 0, n);
336                len += n;
337            }
338
339            res.setContentLength(len);
340        } finally {
341            if (is != null) {
342                is.close();
343            }
344
345            if (os != null) {
346                os.close();
347            }
348        }
349    }
350
351    private String prependSlash(String path) {
352        return path.startsWith("/") ? path : "/" + path;
353    }
