054/**
055 * Audit logger that logs to a repository
056 *
057 * @author aegloff
058 * @author brmiller
059 */
060public class RepoAuditLogger extends AbstractAuditLogger implements AuditLogger {
061    final static Logger logger = LoggerFactory.getLogger(RepoAuditLogger.class);
062
063    /**
064     * Event names for monitoring audit behavior
065     */
066    public static final Name EVENT_AUDIT_CREATE = Name.get("openidm/internal/audit/repo/create");
067
068    /** provides a reference to the router for real-time query handling */
069    private ConnectionFactory connectionFactory;
070
071    /**
072     * Constructor.
073     *
074     * @param connectionFactory
075     */
076    public RepoAuditLogger(ConnectionFactory connectionFactory) {
077        this.connectionFactory = connectionFactory;
078    }
079
080    public void setConfig(Map config, BundleContext ctx) throws InvalidException {
081        super.setConfig(config, ctx);
082    }
083
084    public void cleanup() {
085    }
086
087    /**
088     * {@inheritDoc}
089     */
090    @Override
091    public Map<String, Object> read(ServerContext context, String type, String id) throws ResourceException {
092        Map<String, String> params = new HashMap<String, String>();
093        params.put("_queryId", "query-all");
094        params.put("fields", "*");
095        Map<String, Object> result = new HashMap<String, Object>();
096
097        List<Map<String, Object>> entries = new ArrayList<Map<String, Object>>();
098        if (id == null) {
099            QueryRequest request = Requests.newQueryRequest(getRepoTarget(type));
100            request.setQueryId("query-all");
101            request.getAdditionalParameters().putAll(params);
102            Set<Resource> r = new HashSet<Resource>();
103            connectionFactory.getConnection().query(context, request, r);
104            for (Resource entry : r) {
105                entries.add(AuditServiceImpl.formatLogEntry(entry.getContent().asMap(), type));
106            }
107            formatActivityList(entries);
108            result.put("entries", entries);
109        } else {
110            ReadRequest request = Requests.newReadRequest(getRepoTarget(type), id);
111            Map<String, Object> entry = connectionFactory.getConnection().read(context, request).getContent().asMap();
112            formatActivityEntry(entry);
113            result = AuditServiceImpl.formatLogEntry(entry, type);
114        }
115
116        return result;
117    }
118
119    /**
120     * {@inheritDoc}
121     */
122    @Override
123    public Map<String, Object> query(ServerContext context, String type, Map<String, String> params) throws ResourceException {
124        String queryId = params.get("_queryId");
125        boolean formatted = true;
126        try {
127            if (params.get("formatted") != null && !AuditServiceImpl.getBoolValue(params.get("formatted"))) {
128                formatted = false;
129            }
130            QueryRequest request = Requests.newQueryRequest(getRepoTarget(type));
131            request.setQueryId(queryId);
132            request.getAdditionalParameters().putAll(params);
133            final List<Map<String, Object>> queryResults = new ArrayList<Map<String, Object>>();
134            connectionFactory.getConnection().query(context, request, new QueryResultHandler() {
135                @Override
136                public void handleError(ResourceException error) {
137                    // Continue
138                }
139
140                @Override
141                public boolean handleResource(Resource resource) {
142                    queryResults.add(resource.getContent().asMap());
143                    return true;
144                }
145
146                @Override
147                public void handleResult(QueryResult result) {
148                    // Ignore
149                }
150            });
151            if (type.equals(AuditServiceImpl.TYPE_RECON)) {
152                return AuditServiceImpl.getReconResults(queryResults, formatted);
153            } else if (type.equals(AuditServiceImpl.TYPE_ACTIVITY)) {
154                formatActivityList(queryResults);
155                return AuditServiceImpl.getActivityResults(queryResults, formatted);
156            } else if (type.equals(AuditServiceImpl.TYPE_ACCESS)) {
157                return AuditServiceImpl.getAccessResults(queryResults, formatted);
158            } else {
159                throw new BadRequestException("Unsupported queryId " +  queryId + " on type " + type);
160            }
161        } catch (Exception e) {
162            throw new BadRequestException(e);
163        }
164    }
165
166    public void formatActivityList(List<Map<String, Object>> entryList) {
167        for (Map<String, Object> entry : entryList) {
168            formatActivityEntry(entry);
169        }
170    }
171
172    public void formatActivityEntry(Map<String, Object> entry) {
173        Object beforeValue = entry.get("before");
174        Object afterValue = entry.get("after");
175        if (beforeValue != null) {
176            entry.put("before", AuditServiceImpl.parseJsonString((String)beforeValue).getObject());
177        }
178        if (afterValue != null) {
179            entry.put("after", AuditServiceImpl.parseJsonString((String)afterValue).getObject());
180        }
181    }
182
183    /**
184     * {@inheritDoc}
185     */
186    @Override
187    public void create(ServerContext context, String type, Map<String, Object> obj) throws ResourceException {
188        EventEntry measure = Publisher.start(EVENT_AUDIT_CREATE, obj, null);
189        try {
190            AuditServiceImpl.preformatLogEntry(type, obj);
191            createImpl(context, type, obj);
192        } finally {
193            measure.end();
194        }
195    }
196
197    private void createImpl(ServerContext context, String type, Map<String, Object> obj) throws ResourceException {
198        CreateRequest request = Requests.newCreateRequest(
199                getRepoTarget(type), (String) obj.get(Resource.FIELD_CONTENT_ID), new JsonValue(obj));
200        connectionFactory.getConnection().create(context, request);
201    }
202
203    private String getRepoTarget(String type) {
204        return new StringBuilder("/repo").append(AuditService.ROUTER_PREFIX).append("/").append(type).toString();
205    }
206}
