080    File auditLogDir;

084    public void setConfig(Map config, BundleContext ctx) throws InvalidException {
085        String location = null;
086        try {
087            super.setConfig(config, ctx);
088            location = (String) config.get(CONFIG_LOG_LOCATION);
089            auditLogDir = IdentityServer.getFileForWorkingPath(location);
090            logger.info("Audit logging to: {}", auditLogDir.getAbsolutePath());
091            auditLogDir.mkdirs();
092            recordDelim = (String) config.get(CONFIG_LOG_RECORD_DELIM);
093            if (recordDelim == null) {
094                recordDelim = "";
095            }
096            recordDelim += ServerConstants.EOL;
097        } catch (Exception ex) {
098            logger.error("ERROR - Configured CSV file location must be a directory and {} is invalid.", auditLogDir.getAbsolutePath(), ex);
099            throw new InvalidException("Configured CSV file location must be a directory and '" + location
100                    + "' is invalid " + ex.getMessage(), ex);
101        }
102    }
103
104    public void cleanup() {
105        for (Map.Entry<String, FileWriter> entry : fileWriters.entrySet()) {
106            try {
107                FileWriter fileWriter = entry.getValue();
108                if (fileWriter != null) {
109                    fileWriter.close();
110                }
111            } catch (Exception ex) {
112                logger.info("File writer close reported failure ", ex);
113            }
114        }
115    }

120    @Override
121    public Map<String, Object> read(ServerContext context, String type, String id) throws ResourceException {
122        try {
123            Map<String, Object> result = new HashMap<String, Object>();
124            List<Map<String, Object>> entriesList = new ArrayList<Map<String, Object>>();
125            List<Map<String, Object>> entryList = getEntryList(type);
126            if (entryList == null) {
127                throw new NotFoundException(type + " audit log not found");
128            }
129            for (Map<String, Object> entry : entryList) {
130                if (id == null) {
131                    entriesList.add(AuditServiceImpl.formatLogEntry(entry, type));
132                } else if (id.equals(entry.get("_id"))) {
133                    return AuditServiceImpl.formatLogEntry(entry, type);
134                }
135            }
136            if (id != null) {
137                throw new NotFoundException("Audit log entry with id " + id + " not found");
138            }
139            result.put("entries", entriesList);
140            return result;
141        } catch (Exception e) {
142            throw new BadRequestException(e);
143        }
144    }

146    /**
147     * Parser the csv file corresponding the the specified type (recon, activity, etc) and returns a list
148     * of all entries in it.
149     *
150     * @param type the audit log type
151     * @return  A list of audit log entries
152     * @throws Exception
153     */
154    private List<Map<String, Object>> getEntryList(String type) throws Exception {
155        List<Map<String, Object>> entryList = new ArrayList<Map<String, Object>>();
156        CellProcessor [] processors = null;
157        if (AuditServiceImpl.TYPE_RECON.equals(type)) {
158            processors = new CellProcessor[] {
159                    new NotNull(), // _id
160                    new Optional(), // action
161                    new Optional(), // actionId
162                    new Optional(), // ambiguousTargetObjectIds
163                    new Optional(), // entryType
164                    new Optional(), // exception
165                    new Optional(), // mapping
166                    new Optional(), // message
167                    new Optional(new ParseJsonValue()), // messageDetail
168                    new Optional(), // reconciling
169                    new NotNull(), // reconId
170                    new Optional(), // rootActionId
171                    new Optional(), // situation
172                    new Optional(), // sourceObjectId
173                    new Optional(), // status
174                    new Optional(), // targetObjectId
175                    new NotNull() // timestamp
176            };
177        } else if (AuditServiceImpl.TYPE_ACTIVITY.equals(type)) {
178            processors = new CellProcessor[] {
179                    new NotNull(), // _id
180                    new Optional(), // action
181                    new Optional(), // activityId
182                    new Optional(new ParseJsonValue()), // after
183                    new Optional(new ParseJsonValue()), // before
184                    new Optional(), // changedFields
185                    new Optional(), // message
186                    new Optional(), // objectId
187                    new Optional(), // parentActionId
188                    new Optional(), // passwordChanged
189                    new Optional(), // requester
190                    new Optional(), // rev
191                    new Optional(), // rootActionId
192                    new Optional(), // status
193                    new NotNull() // timestamp
194            };
195        } else if (AuditServiceImpl.TYPE_ACCESS.equals(type)) {
196            processors = new CellProcessor[] {
197                    new NotNull(), // _id
198                    new Optional(), // action
199                    new Optional(), // ip
200                    new Optional(), // principal
201                    new Optional(new ParseJsonValue()), // roles
202                    new Optional(), // status
203                    new NotNull(), // timestamp
204                    new Optional() // userid
205            };
206        } else {
207            throw new InternalServerErrorException("Error parsing entries: unknown type " + type);
208        }
209
210        File auditFile = getAuditLogFile(type);
211        if (auditFile.exists()) {
212            ICsvMapReader reader = null;
213            try {
214                reader = new CsvMapReader(new FileReader(auditFile), new CsvPreference.Builder('"', ',', recordDelim).build());
215
216                // the header elements are used to map the values to the bean (names must match)
217                final String[] header = reader.getHeader(true);
218
219                Map<String, Object> entryMap;
220                while( (entryMap = reader.read(header, processors)) != null ) {
221                        entryList.add(entryMap);
222                }
223
224            }
225            finally {
226                if( reader != null ) {
227                    reader.close();
228                }
229            }
230        }
231        return entryList;
232    }

234    /**
235     * {@inheritDoc}
236     */
237    @Override
238    public Map<String, Object> query(ServerContext context, String type, Map<String, String> params) throws ResourceException {
239        String queryId = params.get("_queryId");
240        boolean formatted = true;
241        try {
242            if (params.get("formatted") != null && !AuditServiceImpl.getBoolValue(params.get("formatted"))) {
243                formatted = false;
244            }
245
246            List<Map<String, Object>> reconEntryList = getEntryList(type);
247            if (reconEntryList == null) {
248                throw new NotFoundException(type + " audit log not found");
249            }
250
251            String reconId = params.get("reconId");
252            if (AuditServiceImpl.QUERY_BY_RECON_ID.equals(queryId) && type.equals(AuditServiceImpl.TYPE_RECON)) {
253                return AuditServiceImpl.getReconResults(reconEntryList, formatted);
254            } else if (AuditServiceImpl.QUERY_BY_MAPPING.equals(queryId) && type.equals(AuditServiceImpl.TYPE_RECON)) {
255                return getReconQueryResults(reconEntryList, reconId, "mapping", params.get("mappingName"), formatted);
256            } else if (AuditServiceImpl.QUERY_BY_RECON_ID_AND_SITUATION.equals(queryId) && type.equals(AuditServiceImpl.TYPE_RECON)) {
257                return getReconQueryResults(reconEntryList, reconId, "situation", params.get("situation"), formatted);
258            } else if (AuditServiceImpl.QUERY_BY_RECON_ID_AND_TYPE.equals(queryId) && type.equals(AuditServiceImpl.TYPE_RECON)) {
259                return getReconQueryResults(reconEntryList, reconId, "entryType", params.get("entryType"), formatted);
260            } else if (AuditServiceImpl.QUERY_BY_ACTIVITY_PARENT_ACTION.equals(queryId) && type.equals(AuditServiceImpl.TYPE_ACTIVITY)) {
261                String actionId = params.get("parentActionId");
262                List<Map<String, Object>> rawEntryList = new ArrayList<Map<String, Object>>();
263                for (Map<String, Object> entry : reconEntryList) {
264                    if (entry.get("parentActionId").equals(actionId)) {
265                        rawEntryList.add(entry);
266                    }
267                }
268                return AuditServiceImpl.getActivityResults(rawEntryList, formatted);
269            } else {
270                throw new BadRequestException("Unsupported queryId " +  queryId + " on type " + type);
271            }
272        } catch (Exception e) {
273            e.printStackTrace();
274            throw new BadRequestException(e);
275        }
276    }
277
278    private Map<String, Object> getReconQueryResults(List<Map<String, Object>> list, String reconId, String param, String paramValue, boolean formatted) {
279        List<Map<String, Object>> rawEntryList = new ArrayList<Map<String, Object>>();
280        for (Map<String, Object> entry : list) {
281            if ((reconId == null || (entry.get("reconId").equals(reconId))) && (param == null || paramValue.equals(entry.get(param)))) {
282                rawEntryList.add(entry);
283            }
284        }
285        return AuditServiceImpl.getReconResults(rawEntryList, formatted);
286    }

291    @Override
292    public void create(ServerContext context, String type, Map<String, Object> obj) throws ResourceException {
293        EventEntry measure = Publisher.start(EVENT_AUDIT_CREATE, obj, null);
294        // Synchronize writes so that simultaneous writes don't corrupt the file
295        synchronized (lock) {
296            try {
297                AuditServiceImpl.preformatLogEntry(type, obj);
298                createImpl(type, obj);
299            } finally {
300                measure.end();
301            }
302        }
303    }

306    private void createImpl(String type, Map<String, Object> obj) throws ResourceException {
307
308        // Re-try once in case the writer stream became closed for some reason
309        boolean retry = false;
310        int retryCount = 0;
311        do {
312            retry = false;
313            FileWriter fileWriter = null;
314            // TODO: optimize buffered, cached writing
315            try {
316                // TODO: Optimize ordering etc.
317                Collection<String> fieldOrder =
318                        new TreeSet<String>(Collator.getInstance());
319                fieldOrder.addAll(obj.keySet());
320
321                File auditFile = getAuditLogFile(type);
322                // Create header if creating a new file
323                if (!auditFile.exists()) {
324                    synchronized (this) {
325                        FileWriter existingFileWriter = getWriter(type, auditFile, false);
326                        File auditTmpFile = new File(auditLogDir, type + ".tmp");
327                        // This is atomic, so only one caller will succeed with created
328                        boolean created = auditTmpFile.createNewFile();
329                        if (created) {
330                            FileWriter tmpFileWriter = new FileWriter(auditTmpFile, true);
331                            writeHeaders(fieldOrder, tmpFileWriter);
332                            tmpFileWriter.close();
333                            auditTmpFile.renameTo(auditFile);
334                            resetWriter(type, existingFileWriter);
335                        }
336                    }
337                }
338                fileWriter = getWriter(type, auditFile, true);
339                writeEntry(fileWriter, type, auditFile, obj, fieldOrder);
340            } catch (IOException ex) {
341                if (retryCount == 0) {
342                    retry = true;
343                    logger.debug("IOException during entry write, reset writer and re-try {}", ex.getMessage());
344                    synchronized (this) {
345                        resetWriter(type, fileWriter);
346                    }
347                } else {
348                    throw new BadRequestException(ex);
349                }
350            }
351            ++retryCount;
352        } while (retry);
353    }
354
355    private File getAuditLogFile(String type) {
356        return new File(auditLogDir, type + ".csv");
357    }
358
359    private void writeEntry(FileWriter fileWriter, String type, File auditFile, Map<String, Object> obj, Collection<String> fieldOrder)
360            throws IOException{
361
362        String key = null;
363        Iterator<String> iter = fieldOrder.iterator();
364        while (iter.hasNext()) {
365            key = iter.next();
366            Object value = obj.get(key);
367            fileWriter.append("\"");
368            if (value != null) {
369                if (value instanceof Map) {
370                    value = new JsonValue((Map)value).toString();
371                }
372                String rawStr = value.toString();
373                // Escape quotes with double quotes
374                String escapedStr = rawStr.replaceAll("\"", "\"\"");
375                fileWriter.append(escapedStr);
376            }
377            fileWriter.append("\"");
378            if (iter.hasNext()) {
379                fileWriter.append(",");
380            }
381        }
382        fileWriter.append(recordDelim);
383        fileWriter.flush();
384    }
385
386    private void writeHeaders(Collection<String> fieldOrder, FileWriter fileWriter)
387            throws IOException {
388        Iterator iter = fieldOrder.iterator();
389        while (iter.hasNext()) {
390            String key = (String) iter.next();
391            fileWriter.append("\"");
392            String escapedStr = key.replaceAll("\"", "\"\"");
393            fileWriter.append(escapedStr);
394            fileWriter.append("\"");
395            if (iter.hasNext()) {
396                fileWriter.append(",");
397            }
398        }
399        fileWriter.append(recordDelim);
400    }
401
402    private FileWriter getWriter(String type, File auditFile, boolean createIfMissing) throws IOException {
403        // TODO: optimize synchronization strategy
404        synchronized (fileWriters) {
405            FileWriter existingWriter = fileWriters.get(type);
406            if (existingWriter == null && createIfMissing) {
407                existingWriter = new FileWriter(auditFile, true);
408                fileWriters.put(type, existingWriter);
409            }
410            return existingWriter;
411        }
412    }
413
414    // This should only be called if it is known that
415    // the writer is invalid for use or no thread has obtained it / is using it
416    // In other words, it does not synchronize on the use of the writer
417    // If the writerToReset doesn't exist in the fileWriters (anymore) then
418    // another thread already reset it, and no action is taken
419    private void resetWriter(String type, FileWriter writerToReset) {
420        FileWriter existingWriter = null;
421        // TODO: optimize synchronization strategy
422        synchronized (fileWriters) {
423            existingWriter = fileWriters.get(type);
424            if (existingWriter != null && writerToReset != null && existingWriter == writerToReset) {
425                fileWriters.remove(type);
426                // attempt clean-up close
427                try {
428                    existingWriter.close();
429                } catch (Exception ex) {
430                    // Debug level as the writer is expected to potentially be invalid
431                    logger.debug("File writer close in resetWriter reported failure ", ex);
432                }
433            }
434        }
435    }

437    /**
438     * CellProcessor for parsing JsonValue objects from CSV file.
439     */
440    public class ParseJsonValue implements CellProcessor {
441
442        @Override
443        public Object execute(Object value, CsvContext context) {
444            JsonValue jv = null;
445            // Check if value is JSON object
446            if (((String)value).startsWith("{") && ((String)value).endsWith("}")) {
447                try {
448                    jv = AuditServiceImpl.parseJsonString(((String)value));
449                } catch (Exception e) {
450                    logger.debug("Error parsing JSON string: " + e.getMessage());
451                }
452            }
453            if (jv == null) {
454                return value;
455            }
456            return jv.asMap();
457        }
