024package org.forgerock.openidm.audit.impl;
025
026import java.util.Map;
027
028import org.forgerock.json.resource.ResourceException;
029import org.forgerock.json.resource.ResourceName;
030import org.forgerock.json.resource.ServerContext;
031
032import org.osgi.framework.BundleContext;
033
034/**
035 * OpenIDM audit logger
036 * @author aegloff
037 * @author brmiller
038 */
039public interface AuditLogger {
040
041    /**
042     * Set the audit logger configuration which is a logger specific
043     * map
044     * @param config the configuration
045     * @param ctx
046     */
047    void setConfig(Map config, BundleContext ctx);
048
049    /**
050     * Cleanup called when auditlogger no longer needed
051     */
052    void cleanup();
053
054    /**
055     * Whether this audit logger is used for reads/queries, when multiple
056     * loggers are configured.
057     *
058     * @return whether to use this logger for reads/queries.
059     */
060    boolean isUsedForQueries();
061
062    /**
063     * Returns whether to ignore logging failures
064     *
065     * @return whether to ignore logging failures
066     */
067    boolean isIgnoreLoggingFailures();
068
069    /**
070     * Creates a new object in the object set.
071     * <p>
072     * On completion, this method sets the {@code _id} property to the assigned identifier for
073     * the object, and the {@code _rev} property to object version if optimistic concurrency
074     * is supported.
075     *
076     * @param type the requested type of audit object to create.
077     * @param object the contents of the object to create in the object set.
078     * @throws ForbiddenException if access to the object or object set is forbidden.
079     */
080    void create(ServerContext context, String type, Map<String, Object> object) throws ResourceException;
081
082    /**
083     * Reads an object from the object set.
084     * <p>
085     * The returned object will contain metadata properties, including relative object
086     * identifier {@code _id}. If optimistic concurrency is supported, the object version
087     * {@code _rev} will be set in the returned object. If optimistic concurrency is not
088     * supported, then {@code _rev} must be {@code null} or absent.
089     *
090     *
091     *
092     * @param type
093     * @param localId
094     * @throws NotFoundException if the specified object could not be found.
095     * @throws ForbiddenException if access to the object is forbidden.
096     * @return the requested object.
097     */
098    Map<String, Object> read(ServerContext context, String type, String localId) throws ResourceException;
099
100    /**
101     * Performs a query on the specified object and returns the associated result. The
102     * execution of a query is not allowed to incur side effects.
103     * <p>
104     * Queries are parametric; a set of named parameters is provided as the query criteria.
105     * The query result is a JSON object structure composed of basic Java types; its overall
106     * structure is defined by the implementation.
107     *
108     *
109     *
110     * @param type
111     * @param params the parameters of the query to perform.
112     * @return the query result object.
113     * @throws NotFoundException if the specified object could not be found.
114     * @throws ForbiddenException if access to the object or the specified query is forbidden.
115     */
116    Map<String, Object> query(ServerContext context, String type, Map<String, String> params) throws ResourceException;
117
118}
