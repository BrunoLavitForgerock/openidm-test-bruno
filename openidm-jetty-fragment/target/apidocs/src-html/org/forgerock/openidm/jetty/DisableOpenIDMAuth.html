016package org.forgerock.openidm.jetty;
017
018import java.util.HashSet;
019import java.util.Set;
020
021import org.eclipse.jetty.server.Connector;
022import org.eclipse.jetty.server.ssl.SslConnector;
023
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027/**
028 * Provides the Jetty bundle (and in turn the jetty.xml)
029 * a way to mark which connector to not apply the OpenIDM authentication mechanism on.
030 * Allows for just using SSL mutual auth on a given port.
031 *
032 * @author aegloff
033 */
034public class DisableOpenIDMAuth {
035
036    final static Logger logger = LoggerFactory.getLogger(DisableOpenIDMAuth.class);
037    
038    private static Set<Integer> clientAuthOnly = new HashSet<Integer>();
039    
040    /**
041     * @return Requested OpenIDM configuration property
042     */
043    public static void add(Object connector) {
044        int port = -1;
045        if (connector instanceof SslConnector) {
046            SslConnector sslConnector = (SslConnector) connector;
047            port = sslConnector.getPort();
048            boolean needClientAuth = sslConnector.getNeedClientAuth();
049            if (needClientAuth == false) {
050                logger.warn("OpenIDM authentication disabled on port {} without the port requiring SSL mutual authentication.", port);
051            } else {
052                logger.info("Port {} set up to require SSL mutual authentication only, no additional OpenIDM authentication.", port);
053            }
054        } else if (connector instanceof Connector) {
055            Connector plainConnector = (Connector) connector;
056            port = plainConnector.getPort();
057            logger.warn("OpenIDM authentication disabled on port {} without SSL.", port);
058        } else {
059            logger.warn("Connector type not recognized and can not disable authentication on it. {}", connector);
060            return;
061        }
062        clientAuthOnly.add(Integer.valueOf(port));
063        setProperty();
064     }
065    
066    // TODO: Replace properties mechanism
067    private static void setProperty() {
068        StringBuilder sb = new StringBuilder();
069        boolean first = true;
070        for (Integer port : clientAuthOnly) {
071            if (!first) {
072                sb.append(",");
073            }
074            sb.append(port);
075            first = false;
076        }
077        System.setProperty("openidm.auth.clientauthonlyports", sb.toString());
078    }
079    
080}
