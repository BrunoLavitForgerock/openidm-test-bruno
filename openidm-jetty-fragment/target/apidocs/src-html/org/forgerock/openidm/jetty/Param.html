024package org.forgerock.openidm.jetty;
025
026import org.forgerock.openidm.crypto.util.JettyPropertyUtil;
027
028/**
029 * Provides the Jetty bundle (and in turn the jetty.xml)
030 * access to configuration supplied by OpenIDM,
031 * without having to resort to system properties
032 *
033 * @author aegloff
034 */
035public class Param {
036
037    /**
038     * @return Requested OpenIDM configuration property
039     */
040    public static String getProperty(String propertyName) {
041        return JettyPropertyUtil.getProperty(propertyName, false);
042    }
043
044    /**
045     * @return OpenIDM default certAlias
046     */
047    public static String getCertAlias() {
048        return JettyPropertyUtil.getProperty("openidm.https.keystore.cert.alias", false);
049    }
050    
051    /**
052     * @return OpenIDM keystore type
053     */
054    public static String getKeystoreType() {
055        return JettyPropertyUtil.getProperty("openidm.keystore.type", false);
056    }
057
058    /**
059     * @return OpenIDM keystore provider
060     */
061    public static String getKeystoreProvider() {
062        return JettyPropertyUtil.getProperty("openidm.keystore.provider", false);
063    }
064
065    /**
066     * @return OpenIDM keystore location, as absolute path.
067     */
068    public static String getKeystoreLocation() {
069        String loc = JettyPropertyUtil.getPathProperty("openidm.keystore.location");
070        return loc;
071    }
072
073    /**
074     * @return OpenIDM keystore password, obfuscated in Jetty format.
075     */
076    public static String getKeystorePassword() {
077        return getKeystorePassword(true);
078    }
079
080    /**
081     * @return OpenIDM keystore key password, obfuscated in Jetty format.
082     * If no specific key password is set, the keystore password (if present) is used.
083     */
084    public static String getKeystoreKeyPassword() {
085        String obfPwd = JettyPropertyUtil.getProperty("openidm.keystore.key.password", true);
086        if (obfPwd == null) {
087            obfPwd = getKeystorePassword();
088        }
089        return obfPwd;
090    }
091
092    /**
093     * @return the truststore location, as absolute path.
094     * If no truststore setting is set, the keystore setting (if present) is used.
095     */
096    public static String getTruststoreLocation() {
097        String trustLoc = JettyPropertyUtil.getPathProperty("openidm.truststore.location");
098        if (trustLoc == null) {
099            trustLoc = getKeystoreLocation();
100        }
101        return trustLoc;
102    }
103
104    /**
105     * @return the truststore location, as absolute path.
106     * If no truststore setting is set, the keystore setting (if present) is used.
107     */
108    public static String getTruststoreType() {
109        String trustType = JettyPropertyUtil.getProperty("openidm.truststore.type", false);
110        if (trustType == null) {
111            trustType = getKeystoreType();
112        }
113        return trustType;
114    }
115
116    /**
117     * @return  the truststore password, obfuscated in Jetty format.
118     * If no truststore setting is set, the keystore setting (if present) is used.
119     */
120    public static String getTruststorePassword() {
121        String obfPwd = getTruststorePassword(true);
122        if (obfPwd == null) {
123            obfPwd = getKeystorePassword();
124        }
125        return obfPwd;
126    }
127
128    /**
129     * Returns the truststore password in either clear text or obfuscated.
130     * If no truststore setting is set, the keystore setting (if present) is used.
131     *
132     * @param   obfuscated if the password should be obfuscated.
133     * @return  the truststore password.
134     */
135    public static String getTruststorePassword(boolean obfuscated) {
136        String pwd = JettyPropertyUtil.getProperty("openidm.truststore.password", obfuscated);
137        if (pwd == null) {
138            pwd = getKeystorePassword(obfuscated);
139        }
140        return pwd;
141    }
142
143    /**
144     * Returns the keystore password in either clear text or obfuscated.
145     *
146     * @param   obfuscated if the password should be obfuscated.
147     * @return  the keystore password.
148     */
149    public static String getKeystorePassword(boolean obfuscated) {
150        return JettyPropertyUtil.getProperty("openidm.keystore.password", obfuscated);
151    }
152}
