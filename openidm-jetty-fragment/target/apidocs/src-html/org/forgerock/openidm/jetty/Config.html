024package org.forgerock.openidm.jetty;
025
026import java.util.Dictionary;
027import java.util.Enumeration;
028
029import org.osgi.framework.BundleContext;
030import org.osgi.framework.FrameworkUtil;
031import org.osgi.framework.ServiceReference;
032import org.osgi.service.cm.Configuration;
033import org.osgi.service.cm.ConfigurationAdmin;
034
035/**
036 * Provides methods for management of the Jetty configuration
037 * 
038 * @author ckienle
039 */
040public class Config {
041
042    /**
043     * Updates the Jetty configuration with new values for the supplied Dictionary of properties. Any existing
044     * properties not contained in the supplied Dictionary will remain the same.
045     * 
046     * @param propsToUpdate the properties to update
047     * @throws Exception
048     */
049    public static void updateConfig(Dictionary propsToUpdate) throws Exception {
050        BundleContext context = FrameworkUtil.getBundle(Param.class).getBundleContext();
051        if (context != null) {
052            ServiceReference configAdminRef = context.getServiceReference(ConfigurationAdmin.class.getName());
053            if (configAdminRef != null) {
054                ConfigurationAdmin confAdmin = (ConfigurationAdmin) context.getService(configAdminRef);
055                Configuration configuration = confAdmin.getConfiguration("org.ops4j.pax.web");
056
057                if (propsToUpdate != null) {
058                    Dictionary props = configuration.getProperties();
059                    Enumeration keys = propsToUpdate.keys();
060                    while (keys.hasMoreElements()) {
061                        String key = (String)keys.nextElement();
062                        props.put(key, propsToUpdate.get(key));
063                    }
064                    configuration.update(props);
065                } else {
066                    configuration.update();
067                }
068            }  
069        }
070    }
071}
