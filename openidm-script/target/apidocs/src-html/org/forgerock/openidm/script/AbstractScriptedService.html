109    private ScriptedRequestHandler embeddedHandler = null;

132    /**
133     * Get the {@link ServerConstants#ROUTER_PREFIX} value.
134     * <p/>
135     * If it return null then the {@link ServerConstants#ROUTER_PREFIX} won't be
136     * changed in the service registration properties.
137     * 
138     * @param factoryPid
139     * @param configuration
140     * @return null or String or String[]
141     */
142    protected abstract Object getRouterPrefixes(String factoryPid, JsonValue configuration);
143
144    protected abstract BundleContext getBundleContext();

147        return this;

156    }

158    protected Scope activate(final BundleContext context, final String factoryPid,
159            final JsonValue configuration) {
160
161        Dictionary<String, Object> prop = getProperties();
162        if (null != prop) {
163            Object o = getRouterPrefixes(factoryPid, configuration);
164            if (null != o) {
165                prop.put(ServerConstants.ROUTER_PREFIX, o);
166            }
167        }
168
169        try {
170            ScriptEntry scriptEntry = scriptRegistry.takeScript(configuration);
171            if (null == scriptEntry) {
172                logger.error("Failed to get the script {}:{}", configuration
173                        .get(SourceUnit.ATTR_NAME), configuration.get(SourceUnit.ATTR_TYPE));
174                throw new NullPointerException();
175            }
176            scriptEntry.addScriptListener(this);
177            scriptName = scriptEntry.getName();
178
179            embeddedHandler = new ScriptedRequestHandler(scriptEntry, getScriptCustomizer());
180
181            selfRegistration = context.registerService(RequestHandler.class, embeddedHandler, prop);
182
183            return embeddedHandler;
184        } catch (ScriptException e) {
185            throw new ComponentException("Failed to take script: " + factoryPid, e);
186        }
187    }
188
189    protected Scope modified(final String factoryPid, final JsonValue configuration) {
190        try {
191            ScriptEntry scriptEntry = scriptRegistry.takeScript(configuration);
192            if (null == scriptEntry) {
193                logger.error("Failed to get the script {}:{}", configuration
194                        .get(SourceUnit.ATTR_NAME), configuration.get(SourceUnit.ATTR_TYPE));
195                throw new NullPointerException();
196            }
197            if (null != scriptName) {
198                scriptRegistry.deleteScriptListener(scriptName, this);
199            }
200            scriptEntry.addScriptListener(this);
201            scriptName = scriptEntry.getName();
202
203            embeddedHandler.setScriptEntry(scriptEntry);
204            return embeddedHandler;
205        } catch (ScriptException e) {
206            logger.error("Failed to modify the ScriptedService", e);
207            throw new ComponentException("Failed to take script: " + factoryPid, e);
208        }
209    }
210
211    protected void deactivate() {
212        try {
213            if (null != selfRegistration) {
214                selfRegistration.unregister();
215                selfRegistration = null;
216            }
217        } catch (IllegalStateException e) {
218            /* Catch if the service was already removed */
219            selfRegistration = null;
220        } finally {
221            if (null != scriptName) {
222                scriptRegistry.deleteScriptListener(scriptName, this);
223            }
224        }
225        logger.info("OpenIDM Info Service component is deactivated.");
226    }

228    // ----- Implementation of ScriptListener interface
229
230    public void scriptChanged(ScriptEvent event) {
231        if (ScriptEvent.REGISTERED == event.getType()) {
232            if (null == selfRegistration) {
233                synchronized (selfRegistration) {
234                    if (null == selfRegistration) {
235                        final ScriptEntry scriptEntry = event.getScriptLibraryEntry();
236                        scriptEntry.setBindings(bindings);
237                        selfRegistration =
238                                getBundleContext().registerService(
239                                        RequestHandler.class,
240                                        new ScriptedRequestHandler(scriptEntry,
241                                                getScriptCustomizer()), getProperties());
242                    }
243                }
244            }
245        } else if (ScriptEvent.UNREGISTERING == event.getType()) {
246            if (null != selfRegistration) {
247                synchronized (selfRegistration) {
248                    if (null != selfRegistration) {
249                        selfRegistration.unregister();
250                    }
251                }
252            }
253        } else if (ScriptEvent.MODIFIED == event.getType()) {
254
255            /*
256             * if (null != selfRegistration) { selfRegistration.unregister(); }
257             * if (null == selfRegistration) { selfRegistration =
258             * getBundleContext().registerService(RequestHandler.class, new
259             * ScriptedRequestHandler( event.getScriptLibraryEntry(), this),
260             * properties); }
261             */
262
263        }
264    }

268    public void handleAction(final ServerContext context, final ActionRequest request,
269            final Bindings bindings) throws ResourceException {
270        if (!mask.contains(RequestType.ACTION)) {
271            throw new NotSupportedException("Actions are not supported for resource instances");
272        }
273        handleRequest(context, request, bindings);
274    }

324
325    protected void handleRequest(final ServerContext context, final Request request,
326            final Bindings bindings) {
327        if (ScriptUtil.REQUEST_BINDING_2_1.equals(scriptName.getRequestBinding())) {
328            SecurityContext securityContext = context.asContext(SecurityContext.class);
329            bindings.put("security", securityContext.getAuthorizationId());
330            bindings.put("request", ScriptUtil.getRequestMap(request, context));
331            bindings.put("_context", ScriptUtil.getLazyContext(context));
332            bindings.put("context", context);
333        } else {
334            bindings.put("request", request);
335            bindings.put("context", context);
336        }
337    }
