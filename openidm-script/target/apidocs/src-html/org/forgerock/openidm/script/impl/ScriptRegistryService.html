189    @Activate
190    protected void activate(ComponentContext context) throws Exception {
191        JsonValue configuration = JSONEnhancedConfig.newInstance().getConfigurationAsJson(context);
192
193        setConfiguration(configuration.required().asMap());
194
195        HashMap<String, Object> identityServer = new HashMap<String, Object>();
196        for (IdentityServerFunctions f : IdentityServerFunctions.values()) {
197            identityServer.put(f.name(), f);
198        }
199        Map<String, Object> console = new HashMap<String, Object>();
200        console.put("log", new Function<Void>() {
201            @Override
202            public Void call(Parameter scope, Function<?> callback, Object... arguments) throws ResourceException, NoSuchMethodException {
203                if (arguments.length > 0) {
204                    if (arguments[0] instanceof String) {
205                        System.out.println((String) arguments[0]);
206                    }
207                }
208                return null;
209            }
210        });
211        put(PROP_IDENTITY_SERVER, identityServer);
212        put(PROP_CONSOLE, console);
213        put(PROP_OPENIDM, openidm);
214        JsonValue properties = configuration.get("properties");
215        if (properties.isMap()) {
216            for (Map.Entry<String, Object> entry : properties.asMap().entrySet()) {
217                if (PROP_IDENTITY_SERVER.equals(entry.getKey())
218                        || PROP_OPENIDM.equals(entry.getKey())
219                        || PROP_CONSOLE.equals(entry.getKey())) {
220                    continue;
221                }
222                put(entry.getKey(), entry.getValue());
223            }
224        }
225
226        try {
227            JsonValue sources = configuration.get("sources");
228            if (!sources.isNull()) {
229                for (String key : sources.keys()) {
230                    JsonValue source = sources.get(key);
231                    String directory = source.get(SOURCE_DIRECTORY).asString();
232                    URL directoryURL = (new File(directory)).toURI().toURL();
233                    /* TODO: Support addition config properties (currently set to defaults in commons)
234                    JsonValue subDirValue = source.get(SOURCE_SUBDIRECTORIES).defaultTo("auto-true");
235                    boolean subdirectories = true;
236                    if (subDirValue.isBoolean()) {
237                        subdirectories = subDirValue.asBoolean();
238                    } else {
239                        subdirectories = Boolean.parseBoolean(subDirValue.asString());
240                    }
241                    String type = source.get(SOURCE_TYPE).defaultTo("auto-detect").asString();
242                    String visibility = source.get(SOURCE_VISIBILITY).defaultTo("public").asString();
243                    */
244                    DirectoryContainer dc = new DirectoryContainer(key, directoryURL);
245                    addSourceUnit(dc);
246                }
247            }
248        } catch (Exception e) {
249            logger.error("Error loading sources", e);
250            throw e;
251        }
252
253        // OPENIDM-1746 Set the script registry class loader to this class, so that any libraries
254        // bundled as an OSGI-Fragment attached to this bundle will be "seen" by scripts.
255        setRegistryLevelScriptClassLoader(this.getClass().getClassLoader());
256
257        /*
258         * manifestWatcher = new BundleWatcher<ManifestEntry>(context, new
259         * ScriptEngineManifestScanner(), null); manifestWatcher.start();
260         */
261
262        logger.info("OpenIDM Script Service component is activated.");
263    }

265    @Modified
266    protected void modified(ComponentContext context) {
267        JsonValue configuration = JSONEnhancedConfig.newInstance().getConfigurationAsJson(context);
268        setConfiguration(configuration.required().asMap());
269        propertiesCache.clear();
270        Set<String> keys =
271                null != getBindings() ? new HashSet<String>(getBindings().keySet()) : Collections
272                        .<String> emptySet();
273        keys.remove(PROP_OPENIDM);
274        keys.remove(PROP_IDENTITY_SERVER);
275        keys.remove(PROP_CONSOLE);
276        JsonValue properties = configuration.get("properties");
277        if (properties.isMap()) {
278            for (Map.Entry<String, Object> entry : properties.asMap().entrySet()) {
279                if (PROP_IDENTITY_SERVER.equals(entry.getKey())
280                        || PROP_OPENIDM.equals(entry.getKey())
281                        || PROP_CONSOLE.equals(entry.getKey())) {
282                    continue;
283                }
284                put(entry.getKey(), entry.getValue());
285                keys.remove(entry.getKey());
286            }
287        }
288        if (!keys.isEmpty()) {
289            for (String name : keys) {
290                getBindings().remove(name);
291            }
292        }
293        logger.info("OpenIDM Script Service component is modified.");
294    }

332    protected void bindCryptoService(final CryptoService cryptoService) {
333        // encrypt(any value, string cipher, string alias)
334        openidm.put("encrypt", new Function<JsonValue>() {
335
336            static final long serialVersionUID = 1L;
337
338            public JsonValue call(Parameter scope, Function<?> callback, Object... arguments)
339                    throws ResourceException, NoSuchMethodException {
340                if (arguments.length == 3) {
341                    JsonValue value = null;
342                    String cipher = null;
343                    String alias = null;
344                    if (arguments[0] instanceof Map
345                            || arguments[0] instanceof List
346                            || arguments[0] instanceof String
347                            || arguments[0] instanceof Number
348                            || arguments[0] instanceof Boolean) {
349                        value = new JsonValue(arguments[0]);
350                    } else if (arguments[0] instanceof JsonValue) {
351                        value = (JsonValue) arguments[0];
352                    } else {
353                        throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(
354                                "encrypt", arguments));
355                    }
356                    if (arguments[1] instanceof String) {
357                        cipher = (String) arguments[1];
358                    } else if (arguments[0] == null) {
359                        cipher = ServerConstants.SECURITY_CRYPTOGRAPHY_DEFAULT_CIPHER;
360                    } else {
361                        throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(
362                                "encrypt", arguments));
363                    }
364
365                    if (arguments[2] instanceof String) {
366                        alias = (String) arguments[2];
367                    } else {
368                        throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(
369                                "encrypt", arguments));
370                    }
371                    try {
372                        return cryptoService.encrypt(value, cipher, alias);
373                    } catch (JsonCryptoException e) {
374                        throw new InternalServerErrorException(e.getMessage(), e);
375                    }
376                } else {
377                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(
378                            "encrypt", arguments));
379                }
380            }
381        });
382        // decrypt(any value)
383        openidm.put("decrypt", new Function<JsonValue>() {
384
385            static final long serialVersionUID = 1L;
386
387            public JsonValue call(Parameter scope, Function<?> callback, Object... arguments)
388                    throws ResourceException, NoSuchMethodException {
389                if (arguments.length == 1
390                        && (arguments[0] instanceof Map || arguments[0] instanceof JsonValue)) {
391                    return cryptoService
392                            .decrypt(arguments[0] instanceof JsonValue ? (JsonValue) arguments[0]
393                                    : new JsonValue(arguments[0]));
394                } else {
395                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(
396                            "decrypt", arguments));
397                }
398            }
399        });
400        // isEncrypted(any value)
401        openidm.put("isEncrypted", new Function<Boolean>() {
402
403            static final long serialVersionUID = 1L;
404
405            public Boolean call(Parameter scope, Function<?> callback, Object... arguments)
406                    throws ResourceException, NoSuchMethodException {
407                if (arguments == null || arguments.length == 0) {
408                    return false;
409                } else if (arguments.length == 1) {
410                    return JsonCrypto.isJsonCrypto(arguments[0] instanceof JsonValue
411                        ? (JsonValue) arguments[0] 
412                        : new JsonValue(arguments[0]));
413                } else {
414                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(
415                            "isEncrypted", arguments));
416                }
417            }
418        });
419        logger.info("Crypto functions are enabled");
420    }

429    protected void bindFunction(final Function function, Map properties) {
430        Object name = properties.get(SCRIPT_NAME);
431        if (name instanceof String && StringUtils.isNotBlank((String) name)
432                && !reservedNames.contains((String) name)) {
433            openidm.put((String) name, function);
434            logger.info("openidm.{} function is enabled", name);
435        }
436    }
437
438    protected void unbindFunction(final Function function, Map properties) {
439        Object name = properties.get(SCRIPT_NAME);
440        if (name instanceof String && StringUtils.isNotBlank((String) name)
441                && !reservedNames.contains((String) name)) {
442            openidm.remove(name, function);
443            logger.info("openidm.{} function is disabled", name);
444        }
445    }

447    @Override
448    public ScriptEntry takeScript(JsonValue script) throws ScriptException {
449        // Check if "name" is missing and "file" is used instead
450        JsonValue scriptConfig = script.clone();
451        if (scriptConfig.get(SourceUnit.ATTR_NAME).isNull()) {
452            JsonValue file = scriptConfig.get(SOURCE_FILE);
453            if (!file.isNull()) {
454                scriptConfig.put(SourceUnit.ATTR_NAME, file.asString());
455            }
456        }
457        
458        return super.takeScript(scriptConfig);
459    }

461    private static enum IdentityServerFunctions implements Function<Object> {
462        getProperty {
463            public Object call(Parameter scope, Function<?> callback, Object... arguments)
464                    throws ResourceException, NoSuchMethodException {
465                boolean useCache = false;
466                if (arguments.length < 1 || arguments.length > 3) {
467                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(this.name(), arguments));
468                }
469                if (arguments.length == 3) {
470                    useCache = (Boolean) arguments[2];
471                }
472                if (arguments[0] instanceof String) {
473                    String name = (String) arguments[0];
474                    Object result = null;
475                    if (useCache) {
476                        result = propertiesCache.get(name);
477                    }
478                    if (null == result) {
479                        Object defaultValue = arguments.length == 2 ? arguments[1] : null;
480                        result = IdentityServer.getInstance().getProperty(name, defaultValue, Object.class);
481                        propertiesCache.putIfAbsent(name, result);
482                    }
483                    return result;
484                } else {
485                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(this.name(), arguments));
486                }
487            }
488        },
489
490        getWorkingLocation {
491            public Object call(Parameter scope, Function callback, Object... arguments)
492                    throws ResourceException, NoSuchMethodException {
493                if (arguments.length != 0) {
494                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(this.name(), arguments));
495                }
496                return IdentityServer.getInstance().getWorkingLocation().getAbsolutePath();
497            }
498        },
499
500        getProjectLocation {
501            public Object call(Parameter scope, Function callback, Object... arguments)
502                    throws ResourceException, NoSuchMethodException {
503                if (arguments.length != 0) {
504                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(this
505                            .name(), arguments));
506                }
507                return IdentityServer.getInstance().getProjectLocation().getAbsolutePath();
508            }
509        },
510        getInstallLocation {
511            public Object call(Parameter scope, Function callback, Object... arguments)
512                    throws ResourceException, NoSuchMethodException {
513                if (arguments.length != 0) {
514                    throw new NoSuchMethodException(FunctionFactory.getNoSuchMethodMessage(this
515                            .name(), arguments));
516                }
517                return IdentityServer.getInstance().getInstallLocation().getAbsolutePath();
518            }
519        };
520        static final long serialVersionUID = 1L;
521    }

524        if (SourceUnit.ATTR_NAME.equals(name) ||
525                SourceUnit.ATTR_REQUEST_BINDING.equals(name) ||
526                SourceUnit.ATTR_REVISION.equals(name) ||
527                SourceUnit.ATTR_SOURCE.equals(name) ||
528                SourceUnit.ATTR_TYPE.equals(name) ||

533        }
534        return false;
535    }

539    public void handleAction(final ServerContext context, final ActionRequest request,
540            final ResultHandler<JsonValue> handler) {
541        String resourceName = request.getResourceName();
542        JsonValue content = request.getContent();
543        Map<String, Object> bindings = new HashMap<String, Object>();
544        JsonValue config = new JsonValue(new HashMap<String, Object>());
545        ScriptEntry scriptEntry = null;
546        try {
547            if (resourceName == null || "".equals(resourceName)) {
548                for (String key : content.keys()) {
549                    if (isSourceUnit(key)) {
550                        config.put(key, content.get(key).getObject());
551                    } else {
552                        bindings.put(key, content.get(key).getObject());
553                    }
554                }
555                // The script will be in the request content
556                scriptEntry = takeScript(config);
557                // Add any additional parameters to the map of bindings
558                bindings.putAll(request.getAdditionalParameters());
559            } else {
560                throw new NotSupportedException("Actions are not supported for resource instances");
561            }
562            switch (request.getActionAsEnum(Action.class)) {
563                case eval:
564                    if (scriptEntry.isActive()) {
565                        Script script = scriptEntry.getScript(context);
566                        handler.handleResult(new JsonValue(script.eval(new SimpleBindings(bindings))));
567                    } else {
568                        throw new ServiceUnavailableException();
569                    }
570                    break;
571                default:
572                    throw new BadRequestException("Unrecognized action ID " + request.getAction());
573            }
574        } catch (ResourceException e) {
575            handler.handleError(e);
576        } catch (IllegalArgumentException e) { // from getActionAsEnum
577            handler.handleError(new BadRequestException(e.getMessage(), e));
578        } catch (Exception e) {
579            handler.handleError(new InternalServerErrorException(e.getMessage(), e));
580        }
581    }

619    @Override
620    public void execute(ServerContext context, Map<String, Object> scheduledContext) throws ExecutionException {
621        
622        try {
623            String scriptName = (String) scheduledContext.get(CONFIG_NAME);
624            JsonValue params = new JsonValue(scheduledContext).get(CONFIGURED_INVOKE_CONTEXT);
625            JsonValue scriptValue = params.get("script").expect(Map.class).clone();
626            
627            if (scriptValue.get(SourceUnit.ATTR_NAME).isNull()) {
628                if (!scriptValue.get(SOURCE_FILE).isNull()) {
629                    scriptValue.put(SourceUnit.ATTR_NAME, scriptValue.get(SOURCE_FILE).getObject());
630                }
631            }
632            
633            if (!scriptValue.isNull()) {
634                ScriptEntry entry = takeScript(scriptValue);
635                JsonValue input = params.get("input");
636                execScript(context, entry, input);
637            } else {
638                throw new ExecutionException("No valid script '" + scriptName + "' configured in schedule.");
639            }
640        } catch (JsonValueException jve) {
641            throw new ExecutionException(jve);
642        } catch (ScriptException e) {
643            throw new ExecutionException(e);
644        } catch (ResourceException e) {
645            throw new ExecutionException(e);
646        }
647    }
648
649    private void execScript(Context context, ScriptEntry script, JsonValue value) throws ForbiddenException, InternalServerErrorException {
650        if (null != script && script.isActive()) {
651            Script executable = script.getScript(context);
652            executable.put("object", value.getObject());
653            try {
654                executable.eval(); // allows direct modification to the object
655            } catch (ScriptThrownException ste) {
656                throw new ForbiddenException(ste.getValue().toString());
657            } catch (ScriptException se) {
658                throw new InternalServerErrorException("script encountered exception", se);
659            }
660        }
661    }
