025package org.forgerock.openidm.smartevent;
026
027/**
028 * Publish smart events
029 * 
030 * Events can be used both for * monitoring/low level events * application level
031 * events
032 * 
033 * @author aegloff
034 */
035public class Publisher {
036
037    /**
038     * For events that mark/span the beginning and end of something, call this
039     * method to mark the beginning of the event window. Upon reaching the end
040     * of the event window, end() SHOULD be called on the returned EventEntry,
041     * even in failure condiitons. For example call end() in a try/finally; or
042     * at the very least the EventEntry should be unreferenced so it gets
043     * garbage collected. Event entries where end() was not called do not
044     * qualify for inclusion in statistics and may not be in the event history.
045     * 
046     * @param eventName
047     *            the object representing the hierarchical event name and its
048     *            context. See <code>Name.get()</code>
049     * @param payload
050     *            Optional payload to send so target subscribers (and
051     *            monitoring) can act upon it
052     * @param context
053     *            Optional context information to send so target subscribers
054     *            (and monitoring) can act upon it
055     */
056    public final static EventEntry start(Name eventName, Object payload, Object context) {
057        return eventName.publisherImpl.start(eventName, payload, context);
058    }
059

064}
