017package org.forgerock.openidm.smartevent.core;
018
019import java.util.concurrent.Executor;
020import java.util.concurrent.Executors;
021
022import com.lmax.disruptor.dsl.ProducerType;
023import org.forgerock.openidm.smartevent.EventEntry;
024import org.forgerock.openidm.smartevent.Name;
025
026import com.lmax.disruptor.RingBuffer;
027import com.lmax.disruptor.SleepingWaitStrategy;
028import com.lmax.disruptor.dsl.Disruptor;
029
030/**
031 * Publisher that uses the disruptor pattern and RingBuffer to process events in
032 * a highly concurrent manner.
033 * 
034 * Start event issues EventEntry without limitation on the outstanding start/end
035 * events, making it suitable for long(er) time span events/measurements
036 * 
037 * @author aegloff
038 */
039public class DisruptorReferringPublisher implements PluggablePublisher {
040
041    static PluggablePublisher INSTANCE = new DisruptorReferringPublisher();
042
043    /**
044     * Ring size has considerable implications:
045     * 
046     * - Pre-allocation of memory/event type objects - Deliberate backpressure
047     * when ring is full, i.e. consumers can't keep up with producers - When
048     * used for monitoring may keep objects alive for recent history purposes
049     * Ring size needs to be a power of 2
050     */
051    static int RING_SIZE = (1024 * 2);
052
053    // TODO: consider optimizations for number of ring buffers and sizing
054    // To process events with different requirements such as statistics vs.
055    // application events
056    static Executor executor = Executors.newCachedThreadPool();
057    static Disruptor<DisruptorReferringEventEntry> disruptor =
058            new Disruptor<DisruptorReferringEventEntry>(DisruptorReferringEventEntry.EVENT_FACTORY,
059                    RING_SIZE, executor, ProducerType.MULTI, new SleepingWaitStrategy());
060    static RingBuffer<DisruptorReferringEventEntry> ringBuffer;
061
062    static {
063        disruptor.handleEventsWith(new StatisticsHandler(disruptor));
064        ringBuffer = disruptor.start();
065    }
066
067    static long sequence = -1;
068
069    /**
070     * Factory method
071     */
072    public static PluggablePublisher getInstance() {
073        return INSTANCE;
074    }
075
076    /**
077     * @inheritDoc
078     */
079    public final EventEntry start(Name eventName, Object payload, Object context) {
080        // For start event, do not hold a place in the ringbuffer (yet)
081        // to avoid limiting long running measurements
082        EventEntryImpl eventEntry = new EventEntryImpl();
083        eventEntry.eventName = eventName;
084        eventEntry.publisher = this;
085        eventEntry.payload = payload;
086        eventEntry.context = context;
087        eventEntry.start();
088
089        // TODO: consider adding option to monitor outstanding requests
090
091        return eventEntry;
092    }
093
094    /**
095     * @inheritDoc
096     */
097    public final void setResult(Object result, EventEntry delegate) {
098    }
099
100    /**
101     * @inheritDoc
102     */
103    public final void end(Name eventName, EventEntry delegate) {
104        sequence = ringBuffer.next();
105        final DisruptorReferringEventEntry eventEntry = ringBuffer.claimAndGetPreallocated(sequence);
106        eventEntry.publisher = this;
107        EventEntryImpl delegateImpl = (EventEntryImpl) delegate;
108        eventEntry.delegate = delegateImpl;
109        // Copy data into ring buffer for optimization
110        eventEntry.startTime = delegateImpl.startTime;
111        eventEntry.end();
112        ringBuffer.publish(sequence);
113    }
114}
