017package org.forgerock.openidm.smartevent.core;
018
019import com.lmax.disruptor.EventFactory;
020
021/**
022 * @author aegloff
023 */
024
025/**
026 * Ring buffer event entry for the disruptor Used to prepopulate the ring buffer
027 * with a (fixed) number of entries for the purpose of processing the associated
028 * EventEntry
029 * 
030 * @author aegloff
031 * 
032 */
033public class DisruptorReferringEventEntry {
034
035    // The event entry to process
036    EventEntryImpl delegate;
037    // For processing optimization some data may get copied into the ring buffer
038    // entry directly
039    long startTime;
040    long endTime;
041
042    PluggablePublisher publisher;
043
044    public final static EventFactory<DisruptorReferringEventEntry> EVENT_FACTORY =
045            new EventFactory<DisruptorReferringEventEntry>() {
046                public DisruptorReferringEventEntry newInstance() {
047                    return new DisruptorReferringEventEntry();
048                }
049            };
050
051    /**
052     * @inheritDoc
053     */
054    public final void end() {
055        // Disabled when using batched end time
056        // The low latency batching framework will assign the time
057        endTime = System.nanoTime();
058    }
059
060    /**
061     * @return duration of time taken between start() and end() or -1 if the
062     *         measurement is not complete or available
063     */
064    public final long getDuration() {
065        if (endTime != 0 && startTime != 0) {
066            return getRawDuration();
067        } else {
068            return -1;
069        }
070    }
071
072    // Internal optimization when it is known that
073    // both end and start time exist
074    // Result is undefined if this precondition is not true.
075    final long getRawDuration() {
076        return endTime - startTime;
077    }
078
079    /**
080     * @return duration of time taken between start() and end() in formatted
081     *         form
082     */
083    String getFormattedDuration() {
084        return StatisticsHandler.formatNsAsMs(getDuration());
085    }
086
087    public String toString() {
088        if (delegate != null && delegate.eventName != null) {
089            return "Event name: "
090                    + delegate.eventName.asString()
091                    + " duration: "
092                    + getFormattedDuration()
093                    + " payload: "
094                    + delegate.payload
095                    + " context: "
096                    + delegate.context
097                    + " publisher set a result: "
098                    + delegate.publisherResultSet
099                    + (delegate.eventName.getResultHistoryEnabled() ? " result: "
100                            + delegate.publisherResult : " result history is disabled.");
101        } else {
102            return "";
103        }
104    }
105}
