034/**
035 * Event handler for monitoring and statistics Expects a single thread assigned
036 * to gather the statistics
037 * 
038 * @author aegloff
039 */
040public class StatisticsHandler implements EventHandler<DisruptorReferringEventEntry>,
041        StatisticsHandlerMBean {
042
043    private final static Logger logger = LoggerFactory.getLogger(StatisticsHandler.class);
044
045    public final static String MBEAN_NAME = "OpenIDM:type=Statistics";
046
047    final static NumberFormat MILLISEC_FORMAT = new DecimalFormat("###,###,##0.### ms");
048
049    /**
050     * Access to the ring buffer for monitoring/history display purposes
051     */
052    // Disruptor<DisruptorEventEntry> disruptor;
053    Disruptor<DisruptorReferringEventEntry> disruptor;
054
055    /**
056     * Keep track of monitoring data per event Name
057     */
058    public Map<String, MonitoringInfo> map = new HashMap<String, MonitoringInfo>();
059
060    public StatisticsHandler(
061    /* Disruptor<DisruptorEventEntry> disruptor */
062    Disruptor<DisruptorReferringEventEntry> disruptor) {
063
064        this.disruptor = disruptor;
065
066        javax.management.MBeanServer mbs =
067                java.lang.management.ManagementFactory.getPlatformMBeanServer();
068        javax.management.ObjectName statName = null;
069        try {
070            statName = new javax.management.ObjectName(MBEAN_NAME);
071            mbs.registerMBean(this, statName);
072        } catch (Exception ex) {
073            logger.info("Failed to register statistics MBean", ex);
074            ;
075        }
076    }
077
078    /**
079     * @inheritDoc
080     */
081    public Map getTotals() {
082        Map stats = new TreeMap();
083        for (java.util.Map.Entry<String, MonitoringInfo> entry : map.entrySet()) {
084            stats.put(entry.getKey(), entry.getValue().toString());
085        }
086        return stats;
087    }
088
089    /**
090     * @inheritDoc
091     */
092    public Map getRecent() {
093        // TODO: consider adding history for not yet end()-ed events
094        // Present history ordered by start time, with latest start time first
095        Map recent = new TreeMap(Collections.reverseOrder());
096        try {
097            RingBuffer<DisruptorReferringEventEntry> buf =
098                    disruptor.getRingBuffer();
099            if (buf != null) {
100                for (int count = 0; count < buf.getBufferSize(); count++) {
101                    DisruptorReferringEventEntry entry = buf.claimAndGetPreallocated(count);
102                    if (entry != null && entry.startTime > 0) {
103                        recent.put(Long.valueOf(entry.startTime), entry.toString());
104                    }
105                }
106            }
107        } catch (RuntimeException ex) {
108            logger.info("Failure in getting recent event history", ex);
109        }
110        return recent;
111    }
112
113    /**
114     * @inheritDoc
115     */
116    public void resetAllStatistics() {
117        for (MonitoringInfo entry : map.values()) {
118            entry.reset();
119        }
120    }
121
122    /**
123     * @inheritDoc
124     */
125    public void resetStatistics(String eventName) {
126        MonitoringInfo entry = map.get(eventName);
127        if (entry != null) {
128            entry.reset();
129        } else {
130            throw new IllegalArgumentException("Event name " + eventName
131                    + " does not match an existing name.");
132        }
133    }

142            entry.setEventsEnabled(enabled);

144            throw new IllegalArgumentException("Event name " + eventName
145                    + " does not match an existing name.");
146        }
147    }

154        Map<String, Boolean> enabledMap = new TreeMap<String, Boolean>();

156            enabledMap.put(entry.getKey(), Boolean.valueOf(entry.getValue().getEventsEnabled()));

188    /**
189     * TODO: if DisruptorShortEventEntry gets removed, remove as well
190     * 
191     * Consumes the events off the ring buffer
192     */
193    public void onEvent(final DisruptorShortEventEntry eventEntry, final long sequence,
194            final boolean endOfBatch) throws Exception {
195        long diff = eventEntry.endTime - eventEntry.startTime;
196
197        // Oddly enough in initial tests this seems slower than requiring map
198        // lookups
199        /*
200         * MonitoringInfo info = eventEntry.eventName.monitoring; info.totalTime
201         * += diff; ++info.totalInvokes;
202         */
203
204        MonitoringInfo entry = map.get(eventEntry.eventName.asString());
205        if (entry == null) {
206            entry = new MonitoringInfo();
207            map.put(eventEntry.eventName.asString(), entry);
208        }
209        entry.totalTime += diff;
210        entry.totalInvokes++;
211    }
212
213    // TODO: more research on latency of batched end time option
214    // Mark the beginning of processing a batch of events
215    boolean newBatch = true;
216    long batchTime = -1;
217
218    /**
219     * Consumes the events off the ring buffer
220     */
221    public void onEvent(final DisruptorReferringEventEntry eventEntryWrap, final long sequence,
222            final boolean endOfBatch) throws Exception {
223        // TODO: provide switch for batch or individual end time
224        // if (newBatch) {
225        // batchTime = System.nanoTime();
226        // }
227        // eventEntryWrap.endTime = batchTime;
228
229        EventEntryImpl eventEntry = eventEntryWrap.delegate;
230        long diff = eventEntryWrap.getRawDuration();
231
232        MonitoringInfo entry = map.get(eventEntry.eventName.asString());
233        if (entry == null) {
234            entry = new MonitoringInfo();
235            map.put(eventEntry.eventName.asString(), entry);
236        }
237        entry.totalTime += diff;
238        entry.totalInvokes++;
239        if (endOfBatch) {
240            newBatch = true;
241        } else {
242            newBatch = false;
243        }
244    }
245
246    /**
247     * Helper to format nanosecond difference in human readable ms if a negative
248     * value is passed, returns "N/A"
249     */
250    static String formatNsAsMs(long nanoseconds) {
251        // Convert from nanoseconds to milliseconds
252        if (nanoseconds >= 0) {
253            return MILLISEC_FORMAT.format(nanoseconds / 1000000d);
254        } else {
255            return "N/A";
256        }
257    }
