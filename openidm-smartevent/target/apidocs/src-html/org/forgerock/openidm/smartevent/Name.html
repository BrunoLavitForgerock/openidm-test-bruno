025package org.forgerock.openidm.smartevent;
026
027import java.util.Map;
028import java.util.concurrent.ConcurrentHashMap;
029import java.util.concurrent.ConcurrentMap;
030
031import org.forgerock.openidm.smartevent.core.DisabledPublisher;
032import org.forgerock.openidm.smartevent.core.DisruptorReferringPublisher;
033import org.forgerock.openidm.smartevent.core.PluggablePublisher;
034
035/**
036 * @author aegloff
037 */
038
039/**
040 * Represents the name and additional configuration for an event type
041 * 
042 * Allows callers to pre-construct, parse event hierarchies and set additional
043 * configuration by declaring the Names outside of the publishing of the events
044 * 
045 * The fluent API allows for easier set-up directly in the declaration example:
046 * <code>public static final Name MY_EVENT = Name.get("internal/myevent").setTags(new String[] {MONITOR, APP_EVENT}).setXyz("dummy")</code>
047 * 
048 * @author aegloff
049 * 
050 */
051public class Name {
052
053    // Holds the event stringified name to the Name instance mapping
054    static ConcurrentMap<String, Name> names = new ConcurrentHashMap<String, Name>();
055
056    /**
057     * Stringified version of the event name
058     */
059    String stringifiedName;
060
061    /**
062     * Optional tags to associate with this event name
063     */
064    String[] tags;
065
066    /**
067     * Whether event processing should be enabled or not for this event name
068     */
069    boolean eventsEnabled = true;
070
071    /**
072     * Whether result details should be kept in the recent history
073     */
074    boolean resultHistoryEnabled = false;
075
076    PluggablePublisher publisherImpl;
077
078    private Name(String stringifiedName) {
079        this.stringifiedName = stringifiedName;
080        // Default Setting
081        setEventsEnabled(Boolean.valueOf(System.getProperty("openidm.smartevent.enabled",
082                Boolean.FALSE.toString())));
083        // Name parsing can be added here
084    }
085
086    /**
087     * Factory method to get the event Name object
088     * 
089     * @param stringifiedName
090     *            The string representation of the event name
091     * @return the event Name object representing the requested event type
092     */
093    public final static Name get(String stringifiedName) {
094        Name aName = names.get(stringifiedName);
095        if (aName == null) {
096            aName = new Name(stringifiedName);
097            // Ensure first create/insert wins
098            // (Because of avoiding locking, multiple could create and try to
099            // insert this concurrently)
100            Name existing = names.putIfAbsent(stringifiedName, aName);
101            if (existing != null) {
102                aName = existing;
103            }
104        }
105        return aName;
106    }
107
108    /**
109     * Factory method to get the event Name object
110     * 
111     * @param stringifiedName
112     *            The string representation of the event name
113     * @param tags
114     *            optional additional tags to associate with this event type
115     * @return the event Name object representing the requested event type
116     */
117    public final static Name get(String stringifiedName, String[] tags) {
118        Name aName = get(stringifiedName);
119        aName.tags = tags;
120        return aName;
121    }
122
123    /**
124     * Fluent API to set Tags on this event name
125     * 
126     * @param tags
127     *            the tags to set
128     * @return this Name instance for use as a fluent API
129     */
130    public Name setTags(String[] tags) {
131        this.tags = tags;
132        return this;
133    }
134
135    /**
136     * Fluent API to set events publishing behavior
137     * 
138     * @param enabled
139     *            true to enable event publishing for this Name, false to
140     *            disable
141     * @return this Name instance for use as a fluent API
142     */
143    public Name setEventsEnabled(boolean enabled) {
144        eventsEnabled = enabled;
145        if (enabled == true) {
146            publisherImpl = DisruptorReferringPublisher.getInstance();
147        } else {
148            publisherImpl = DisabledPublisher.getInstance();
149        }
150        return this;
151    }
152
153    /**
154     * @return the events publishing state
155     */
156    public boolean getEventsEnabled() {
157        return eventsEnabled;
158    }
159
160    /**
161     * Fluent API to set result history behavior
162     * 
163     * @param enabled
164     *            true to enable keeping results in the recent event history
165     * @return this Name instance for use as a fluent API
166     */
167    public Name setResultHistoryEnabled(boolean enabled) {
168        resultHistoryEnabled = enabled;
169        return this;
170    }
171
172    /**
173     * @return whether the result history is enabled
174     */
175    public boolean getResultHistoryEnabled() {
176        return resultHistoryEnabled;
177    }
178
179    /**
180     * Get all currently registered event names The returned map should not be
181     * directly modified.
182     * 
183     * @return a map pointing from stringified event name representation to the
184     *         the event Name object for each event Name
185     */
186    public final static Map<String, Name> getAllNames() {
187        // TODO: consider making/wrapping as immutable
188        return names;
189    }
190
191    /**
192     * @return the name in String representation
193     */
194    public final String asString() {
195        return stringifiedName;
196    }
197}
