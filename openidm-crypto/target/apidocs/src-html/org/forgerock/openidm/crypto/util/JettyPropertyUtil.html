025package org.forgerock.openidm.crypto.util;
026
027import java.security.GeneralSecurityException;
028
029import org.forgerock.openidm.core.IdentityServer;
030import org.forgerock.openidm.crypto.impl.Main;
031
032/**
033 * Utility for handling Jetty configuration properties, including properties
034 * that may need to be encrypted/obfuscated, and/or may be encrypted/obfuscated
035 *
036 * @author aegloff
037 */
038public class JettyPropertyUtil {
039
040    /**
041     * Gets a Jetty configuration property. If obfuscated is true, it will
042     * return a value obfuscated in Jetty format.
043     *
044     * @param propName
045     *            name of the property.
046     * @param obfuscated
047     *            if value should be obfuscated.
048     * @return the property value.
049     */
050    public static String getProperty(String propName, boolean obfuscated) {
051        String prop = IdentityServer.getInstance().getProperty(propName);
052        if (prop == null) {
053            return null;
054        }
055        try {
056            String clear = new String(Main.unfold(prop));
057            if (obfuscated) {
058                prop = Main.obfuscate(clear);
059            } else {
060                prop = clear;
061            }
062        } catch (GeneralSecurityException ex) {
063            throw new RuntimeException("Failed to obtain property " + propName
064                    + " in Jetty obfuscated format.", ex);
065        }
066        return prop;
067    }
068
069    /**
070     * Get a Jetty configuration property as qualified file path
071     *
072     * @param propName
073     *            name of the property.
074     * @return the property value.
075     */
076    public static String getPathProperty(String propName) {
077        String loc = getProperty(propName, false);
078        if (loc != null) {
079            loc = IdentityServer.getFileForInstallPath(loc).getAbsolutePath();
080        }
081        return loc;
082    }
083}
