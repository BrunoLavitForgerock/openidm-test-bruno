057    public static void main(String[] arg) throws Exception {
058        System.err.println("This utility helps obfuscate passwords to prevent casual observation.");
059        System.err.println("It is not securely encrypted and needs further measures to prevent disclosure.");
060        if (arg.length != 0 && arg.length != 1) {
061            System.err.println("Usage - java  <password>");
062            System.err.println("If the password is not specified, the user will be prompted for the password");
063            System.exit(1);
064        }
065
066        String passwd = arg.length == 1 ? arg[0] : null;
067        if (null == passwd) {
068            Console console = System.console();
069
070            //read the password, without echoing the output
071            char[] password = console.readPassword("Please enter the password: ");
072
073            passwd = new String(password);
074
075            //the javadoc for the Console class recommends "zeroing-out" the password
076            //when finished verifying it :
077            Arrays.fill(password, ' ');
078        }
079
080        System.err.println(obfuscate(passwd));
081        System.err.println(encrypt(passwd));
082    }

084    public static String obfuscate(String s) {
085        StringBuilder buf = new StringBuilder();
086        byte[] b = s.getBytes();
087
088        buf.append(__OBFUSCATE);
089        for (int i = 0; i < b.length; i++) {
090            byte b1 = b[i];
091            byte b2 = b[s.length() - (i + 1)];
092            int i1 = 127 + b1 + b2;
093            int i2 = 127 + b1 - b2;
094            int i0 = i1 * 256 + i2;
095            String x = Integer.toString(i0, 36);
096
097            switch (x.length()) {
098                case 1:
099                    buf.append('0');
100                case 2:
101                    buf.append('0');
102                case 3:
103                    buf.append('0');
104                default:
105                    buf.append(x);
106            }
107        }
108        return buf.toString();
109
110    }

112    public static String deobfuscate(String s) {
113        if (s.startsWith(__OBFUSCATE))
114            s = s.substring(__OBFUSCATE.length());
115
116        byte[] b = new byte[s.length() / 2];
117        int l = 0;
118        for (int i = 0; i < s.length(); i += 4) {
119            String x = s.substring(i, i + 4);
120            int i0 = Integer.parseInt(x, 36);
121            int i1 = (i0 / 256);
122            int i2 = (i0 % 256);
123            b[l++] = (byte) ((i1 + i2 - 254) / 2);
124        }
125
126        return new String(b, 0, l);
127    }

129    public static String encrypt(String s) throws Exception {
130        StringBuilder buf = new StringBuilder(__CRYPT);
131
132        // Instantiate the cipher
133        Cipher cipher = Cipher.getInstance(CRYPT_ALGORITHM);
134        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(NON_SECRET_KEY, CRYPT_ALGORITHM));
135        return buf.append(byteArrayToHexString(cipher.doFinal(s.getBytes()))).toString();
136    }
137
138    public static String decrypt(String s) throws GeneralSecurityException {
139        if (s.startsWith(__CRYPT))
140            s = s.substring(__CRYPT.length());
141
142        // Instantiate the cipher
143        Cipher cipher = Cipher.getInstance(CRYPT_ALGORITHM);
144        cipher.init(Cipher.DECRYPT_MODE, new SecretKeySpec(NON_SECRET_KEY, CRYPT_ALGORITHM));
145        return new String(cipher.doFinal(hexStringToByteArray(s)));
146    }
147
148    public static char[] unfold(String s) throws GeneralSecurityException {
149        char[] passwordCopy = null;
150        if (null != s) {
151            if (s.startsWith(__CRYPT)) {
152                passwordCopy = decrypt(s).toCharArray();
153            } else if (s.startsWith(__OBFUSCATE)) {
154                passwordCopy = deobfuscate(s).toCharArray();
155            } else if (s.equalsIgnoreCase(__CONSOLE)) {
156                char[] passwordArray = System.console().readPassword("Please enter the password: ");
157                passwordCopy = Arrays.copyOf(passwordArray, passwordArray.length);
158                Arrays.fill(passwordArray, ' ');
159            } else {
160                passwordCopy = s.toCharArray();
161            }
162        }
163        return passwordCopy;
164    }

166    /**
167     * @param hex
168     * @return
169     */
170    private static byte[] hexStringToByteArray(String hex) {
171        byte rc[] = new byte[hex.length() / 2];
172        for (int i = 0; i < rc.length; i++) {
173            String h = hex.substring(i * 2, i * 2 + 2);
174            int x = Integer.parseInt(h, 16);
175            rc[i] = (byte) x;
176        }
177        return rc;
178    }
179
180    /**
181     * @param bytes
182     * @return
183     */
184    private static String byteArrayToHexString(byte[] bytes) {
185        StringBuilder rc = new StringBuilder(bytes.length * 2);
186        for (int i = 0; i < bytes.length; i++) {
187            rc.append(HEX_TABLE[0xFF & bytes[i]]);
188        }
189        return rc.toString();
190    }
