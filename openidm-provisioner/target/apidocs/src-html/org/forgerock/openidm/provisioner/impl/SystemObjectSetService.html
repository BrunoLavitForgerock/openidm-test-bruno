113    protected void unbindProvisionerService(ProvisionerService service, Map properties) {
114        for (Map.Entry<SystemIdentifier, ProvisionerService> entry : provisionerServices.entrySet()) {
115            if (service.equals(entry.getValue())) {
116                provisionerServices.remove(entry.getKey());
117                break;
118            }
119        }
120    }

148    @Override
149    public void actionInstance(ServerContext context, ActionRequest request, ResultHandler<JsonValue> handler) {
150        if (ACTION_CREATE_CONFIGURATION.equalsIgnoreCase(request.getAction())) {
151            if (null != configurationService) {
152                try {
153                    handler.handleResult(configurationService.configure(request.getContent()));
154                } catch (ResourceException e) {
155                    handler.handleError(e);
156                } catch (Exception e) {
157                    handler.handleError(new InternalServerErrorException(e));
158                }
159            } else {
160                handler.handleError(new ServiceUnavailableException("The required service is not available"));
161            }
162        } else if (ACTION_TEST_CONFIGURATION.equalsIgnoreCase(request.getAction())) {
163            try {
164                JsonValue config = request.getContent();
165                if (!config.get("id").isNull()) {
166                    throw new BadRequestException("A system ID must not be specified in the request");
167                }
168                ProvisionerService ps = locateServiceForTest(config.get("name"));
169                if (ps == null) {
170                    throw new BadRequestException("Invalid configuration to test: no 'name' specified");
171                }
172                handler.handleResult(new JsonValue(ps.testConfig(config)));
173            } catch (ResourceException e) {
174                handler.handleError(e);
175            } catch (Exception e) {
176                handler.handleError(new InternalServerErrorException(e));
177            }
178        } else if (ACTION_TEST_CONNECTOR.equalsIgnoreCase(request.getAction())) {
179            try {
180                ProvisionerService ps = locateServiceForTest(request.getContent().get("id"));
181                if (ps != null) {
182                    handler.handleResult(new JsonValue(ps.getStatus()));
183                } else {
184                    List<Object> list = new ArrayList<Object>();
185                    for (Map.Entry<SystemIdentifier, ProvisionerService> entry : provisionerServices.entrySet()) {
186                        list.add(entry.getValue().getStatus());
187                    }
188                    handler.handleResult(new JsonValue(list));
189                }
190            } catch (ResourceException e) {
191                handler.handleError(e);
192            } catch (Exception e) {
193                handler.handleError(new InternalServerErrorException(e));
194            }
195        } else if (isLiveSyncAction(request.getAction())) {
196            JsonValue params = new JsonValue(request.getAdditionalParameters());
197            try {
198                String source = params.get("source").asString();
199                if (source == null) {
200                    logger.debug("liveSync requires an explicit source parameter, source is : {}", source );
201                    throw new BadRequestException("liveSync action requires either an explicit source parameter, "
202                                + "or needs to be called on a specific provisioner URI");
203                } else {
204                    logger.debug("liveSync called with explicit source parameter {}", source);
205                }
206                handler.handleResult(liveSync(source, Boolean.valueOf(params.get("detailedFailure").asString())));
207            } catch (ResourceException e) {
208                handler.handleError(e);
209            }
210        } else {
211            handler.handleError(new BadRequestException("Unsupported actionId: " + request.getAction()));
212        }
213    }

236    /**
237     * Invoked by the scheduler when the scheduler triggers.
238     *
239     * @param schedulerContext Context information passed by the scheduler service
240     * @throws org.forgerock.openidm.quartz.impl.ExecutionException
241     *          if execution of the scheduled work failed.
242     *          Implementations can also throw RuntimeExceptions which will get logged.
243     */
244    public void execute(ServerContext context, Map<String, Object> schedulerContext) throws ExecutionException {
245        try {
246            JsonValue params = new JsonValue(schedulerContext).get(CONFIGURED_INVOKE_CONTEXT);
247            String action = params.get("action").asString();
248            if (isLiveSyncAction(action)) {
249                String source = params.get("source").required().asString();
250                liveSync(source, true);
251            }
252        } catch (JsonValueException jve) {
253            throw new ExecutionException(jve);
254        } catch (ResourceException e) {
255            throw new ExecutionException(e);
256        } catch (RuntimeException e) {
257            throw new ExecutionException(e);
258        }
259    }

261    /**
262     * @param action the requested action
263     * @return true if the action string is to live sync
264     */
265    private boolean isLiveSyncAction(String action) {
266        return (ACTION_LIVE_SYNC.equalsIgnoreCase(action) || ACTION_ACTIVE_SYNC.equalsIgnoreCase(action));
267    }

269    /**
270     * Live sync the specified provisioner resource
271     * @param source the URI of the provisioner instance to live sync
272     * @param detailedFailure whether in the case of failures additional details such as the
273     * record content of where it failed should be included in the response
274     */
275    private JsonValue liveSync(String source, boolean detailedFailure) throws ResourceException {
276        JsonValue response;
277        Id id = new Id(source);
278        String previousStageResourceContainer = "repo/synchronisation/pooledSyncStage";
279        String previousStageId = id.toString().replace("/", "").toUpperCase();
280        Resource previousStage = null;
281        try {
282            ReadRequest readRequest = Requests.newReadRequest(previousStageResourceContainer, previousStageId);
283            previousStage = connectionFactory.getConnection().read(routerContext, readRequest);
284
285            response = locateService(id).liveSynchronize(id.getObjectType(),
286                    previousStage != null && previousStage.getContent() != null ? previousStage.getContent() : null);
287            UpdateRequest updateRequest = Requests.newUpdateRequest(previousStageResourceContainer, previousStageId, response);
288            updateRequest.setRevision(previousStage.getRevision());
289            connectionFactory.getConnection().update(routerContext, updateRequest);
290        } catch (ResourceException e) { // NotFoundException?
291            if (null == previousStage) {
292                                response = locateService(id).liveSynchronize(id.getObjectType(), null);
293                CreateRequest createRequest = Requests.newCreateRequest(previousStageResourceContainer, previousStageId, response);
294                connectionFactory.getConnection().create(routerContext, createRequest);
295            }
296            else {
297                throw e;
298            }
299        }
300        if (response != null && !detailedFailure) {
301            // The detailedFailure option handling ideally should move into provisioners
302            response.get("lastException").remove("syncDelta");
303        }
304        return response;
305    }

307    private ProvisionerService locateService(Id identifier) throws ResourceException {
308        for (Map.Entry<SystemIdentifier, ProvisionerService> entry : provisionerServices.entrySet()) {
309            if (entry.getKey().is(identifier)) {
310                return entry.getValue();
311            }
312        }
313        throw new ServiceUnavailableException("System: " + identifier + " is not available.");
314    }
315
316    private ProvisionerService locateServiceForTest(JsonValue requestId) throws ResourceException {
317        if (requestId.isNull()) {
318            return null;
319        }
320
321        Id id = new Id(requestId.asString() + "/test");
322        for (Map.Entry<SystemIdentifier, ProvisionerService> entry : provisionerServices.entrySet()) {
323            if (entry.getKey().is(id)) {
324                return entry.getValue();
325            }
326        }
327
328        throw new NotFoundException("System: " + requestId.asString() + " is not available.");
329    }
