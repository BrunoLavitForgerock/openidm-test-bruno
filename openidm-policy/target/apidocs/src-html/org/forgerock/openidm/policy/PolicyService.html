025package org.forgerock.openidm.policy;
026
027import java.util.ArrayList;
028import java.util.Dictionary;
029import java.util.EnumSet;
030import java.util.HashMap;
031import java.util.List;
032import java.util.Map;
033
034import javax.script.Bindings;
035
036import org.apache.felix.scr.annotations.Activate;
037import org.apache.felix.scr.annotations.Component;
038import org.apache.felix.scr.annotations.ConfigurationPolicy;
039import org.apache.felix.scr.annotations.Deactivate;
040import org.apache.felix.scr.annotations.Modified;
041import org.apache.felix.scr.annotations.Properties;
042import org.apache.felix.scr.annotations.Property;
043import org.forgerock.json.fluent.JsonValue;
044import org.forgerock.json.resource.ActionRequest;
045import org.forgerock.json.resource.ReadRequest;
046import org.forgerock.json.resource.RequestType;

049import org.forgerock.openidm.config.enhanced.JSONEnhancedConfig;

060/**
061 * A Policy Service for policy validation.
062 * 
063 * @author Chad Kienle
064 */
065@Component(name = PolicyService.PID, policy = ConfigurationPolicy.REQUIRE, metatype = true,
066        description = "OpenIDM Policy Service", immediate = true)
067@Properties({
068    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
069    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM Policy Service"),
070    @Property(name = ServerConstants.ROUTER_PREFIX, value = "/policy*") })
071public class PolicyService extends AbstractScriptedService {
072
073    public static final String PID = "org.forgerock.openidm.policy";
074
075    /**
076     * Setup logging for the {@link PolicyService}.
077     */
078    private static final Logger logger = LoggerFactory.getLogger(PolicyService.class);
079
080    private ComponentContext context;
081    
082    private JsonValue configuration;
083
084    public PolicyService() {
085        super(EnumSet.of(RequestType.ACTION, RequestType.READ));
086    }
087
088    public PolicyService(JsonValue configuration) {
089        super(EnumSet.of(RequestType.ACTION, RequestType.READ));
090        init(configuration);
091    }
092
093    @Activate
094    protected void activate(ComponentContext context) {
095        this.context = context;
096
097        Dictionary properties = context.getProperties();
098        setProperties(properties);
099
100        configuration = getConfiguration(context);
101
102        activate(context.getBundleContext(), null, configuration);
103
104        logger.info("OpenIDM Policy Service component is activated.");
105    }
106
107    /**
108     * Configuration modified handling Ensures the service stays registered even
109     * whilst configuration changes
110     */
111    @Modified
112    void modified(ComponentContext context) throws Exception {
113        configuration = getConfiguration(context);
114        modified(null, configuration);
115        logger.info("OpenIDM Policy Service component is modified.");
116    }
117
118    @Deactivate
119    protected void deactivate(ComponentContext context) {
120        deactivate();
121        this.context = null;
122        logger.info("OpenIDM Policy Service component is deactivated.");
123    }
124
125    @Override
126    protected Object getRouterPrefixes(String factoryPid, JsonValue configuration) {
127        return null;
128    }
129
130    protected BundleContext getBundleContext() {
131        return context.getBundleContext();
132    }
133
134    private JsonValue getConfiguration(ComponentContext context) {
135        JsonValue configuration = JSONEnhancedConfig.newInstance().getConfigurationAsJson(context);
136        init(configuration);
137        return configuration;
138    }
139    
140    private void init(JsonValue configuration) {
141        JsonValue additionalPolicies = configuration.get("additionalFiles");
142        if (!additionalPolicies.isNull()) {
143            configuration.remove("additionalFiles");
144            List<String> list = new ArrayList<String>();
145            for (JsonValue policy : additionalPolicies) {
146                try {
147                    list.add(FileUtil.readFile(IdentityServer.getFileForProjectPath(policy.asString())));
148                } catch (Exception e) {
149                    logger.error("Error loading additional policy script " + policy.asString(), e);
150                }
151            }
152            configuration.add("additionalPolicies", list);
153        }
154    }
155    
156    @Override
157    public void handleAction(final ServerContext context, final ActionRequest request,
158            final Bindings handler) throws ResourceException {
159        super.handleAction(context, request, handler);
160        for (Map.Entry<String, String> entry : request.getAdditionalParameters().entrySet()) {
161            if (handler.containsKey(entry.getKey())) {
162                continue;
163            }
164            if (bindings != null) {
165                bindings.put(entry.getKey(), entry.getValue());
166            }
167        }
168
169        handler.put("request", request);
170        handler.put("resources", configuration.get("resources").getObject());
171    }
172    
173    @Override
174    public void handleRead(final ServerContext context, final ReadRequest request,
175            final Bindings handler) throws ResourceException {
176        super.handleRead(context, request, handler);
177
178        handler.put("request", request);
179        handler.put("resources", configuration.get("resources").getObject());
180    }
181}
