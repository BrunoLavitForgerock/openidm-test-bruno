024package org.forgerock.openidm.workflow.activiti.metadata;
025
026import org.forgerock.json.fluent.JsonPointer;
027import org.forgerock.json.fluent.JsonValue;
028import org.forgerock.openidm.workflow.activiti.impl.ActivitiServiceImpl;
029import org.forgerock.openidm.metadata.MetaDataProvider;
030import org.forgerock.openidm.metadata.MetaDataProviderCallback;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034import java.util.ArrayList;
035import java.util.HashMap;
036import java.util.List;
037import java.util.Map;
038
039/**
040 * Meta data provider to describe configuration
041 * requirements of this bundle
042 *
043 * @author aegloff
044 * @author ckienle
045 */
046public class ConfigMeta implements MetaDataProvider {
047    final static Logger logger = LoggerFactory.getLogger(ConfigMeta.class);
048
049    Map<String, List<JsonPointer>> propertiesToEncrypt;
050    private MetaDataProviderCallback callback = null;
051
052    public ConfigMeta() {
053        propertiesToEncrypt = new HashMap<String, List<JsonPointer>>();
054        List<JsonPointer> props = new ArrayList<JsonPointer>();
055        props.add(new JsonPointer("engine/password"));
056        props.add(new JsonPointer("mail/password"));
057        propertiesToEncrypt.put(ActivitiServiceImpl.PID, props);
058    }
059
060    /**
061     * @inheritDoc
062     */
063    public List<JsonPointer> getPropertiesToEncrypt(String pidOrFactory, String instanceAlias, JsonValue config) {
064        if (propertiesToEncrypt.containsKey(pidOrFactory)) {
065            return propertiesToEncrypt.get(pidOrFactory);
066        }
067        return null;
068    }
069
070    /**
071     * @inheritDoc
072     */
073    @Override
074    public void setCallback(MetaDataProviderCallback callback) {
075        this.callback = callback;
076    }
077}
