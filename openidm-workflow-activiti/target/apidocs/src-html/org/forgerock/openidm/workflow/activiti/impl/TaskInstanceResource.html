078    @Override
079    public void actionInstance(ServerContext context, String resourceId, ActionRequest request, ResultHandler<JsonValue> handler) {
080        try {
081            Authentication.setAuthenticatedUserId(context.asContext(SecurityContext.class).getAuthenticationId());
082            TaskService taskService = processEngine.getTaskService();
083            Task task = processEngine.getTaskService().createTaskQuery().taskId(resourceId).singleResult();
084            if (task == null) {
085                handler.handleError(new NotFoundException());
086            } else {
087                if ("claim".equals(request.getAction())) {
088                    taskService.claim(resourceId, request.getContent().expect(Map.class).asMap().get("userId").toString());
089                } else if ("complete".equals(request.getAction())) {
090                    taskService.complete(resourceId, request.getContent().expect(Map.class).asMap());
091                } else {
092                    handler.handleError(new BadRequestException("Unknown action"));
093                }
094                Map<String, String> result = new HashMap<String, String>(1);
095                result.put("Task action performed", request.getAction());
096                handler.handleResult(new JsonValue(result));
097            }
098        } catch (Exception ex) {
099            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
100        }
101    }

118    @Override
119    public void queryCollection(ServerContext context, QueryRequest request, QueryResultHandler handler) {
120        try {
121            Authentication.setAuthenticatedUserId(context.asContext(SecurityContext.class).getAuthenticationId());
122            TaskQuery query = processEngine.getTaskService().createTaskQuery();
123            if (ActivitiConstants.QUERY_FILTERED.equals(request.getQueryId())
124                    || ActivitiConstants.QUERY_ALL_IDS.equals(request.getQueryId())) {
125                if (ActivitiConstants.QUERY_FILTERED.equals(request.getQueryId())) {
126                    setTaskParams(query, request);
127                }
128                List<Task> list = query.list();
129                for (Task taskInstance : list) {
130                    Map value = mapper.convertValue(taskInstance, HashMap.class);
131                    Resource r = new Resource(taskInstance.getId(), null, new JsonValue(value));
132                    if (taskInstance.getDelegationState() == DelegationState.PENDING) {
133                        r.getContent().add(ActivitiConstants.ACTIVITI_DELEGATE, taskInstance.getAssignee());
134                    } else {
135                        r.getContent().add(ActivitiConstants.ACTIVITI_ASSIGNEE, taskInstance.getAssignee());
136                    }
137                    handler.handleResource(r);
138                }
139                handler.handleResult(new QueryResult());
140            } else {
141                handler.handleError(new BadRequestException("Unknown query-id"));
142            }
143        } catch (Exception ex) {
144            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
145        }
146    }

148    @Override
149    public void readInstance(ServerContext context, String resourceId, ReadRequest request, ResultHandler<Resource> handler) {
150        try {
151            Authentication.setAuthenticatedUserId(context.asContext(SecurityContext.class).getAuthenticationId());
152            TaskQuery query = processEngine.getTaskService().createTaskQuery();
153            query.taskId(resourceId);
154            Task task = query.singleResult();
155            if (task == null) {
156                handler.handleError(new NotFoundException());
157            } else {
158                Map value = mapper.convertValue(task, HashMap.class);
159                Resource r = new Resource(task.getId(), null, new JsonValue(value));
160                TaskFormData data = processEngine.getFormService().getTaskFormData(task.getId());
161                List<Map> propertyValues = new ArrayList<Map>();
162                for (FormProperty p : data.getFormProperties()) {
163                    Map<String, String> entry = new HashMap<String, String>();
164                    entry.put(p.getId(), p.getValue());
165                    propertyValues.add(entry);
166                }
167                r.getContent().add(ActivitiConstants.FORMPROPERTIES, propertyValues);
168                if (task.getDelegationState() == DelegationState.PENDING) {
169                    r.getContent().add(ActivitiConstants.ACTIVITI_DELEGATE, task.getAssignee());
170                } else {
171                    r.getContent().add(ActivitiConstants.ACTIVITI_ASSIGNEE, task.getAssignee());
172                }
173                Map<String, Object> variables = new HashMap<String, Object>(processEngine.getTaskService().getVariables(task.getId()));
174                if (variables.containsKey(ActivitiConstants.OPENIDM_CONTEXT)){
175                    variables.remove(ActivitiConstants.OPENIDM_CONTEXT);
176                }
177                r.getContent().add(ActivitiConstants.ACTIVITI_VARIABLES, variables);
178                handler.handleResult(r);
179            }
180        } catch (Exception ex) {
181            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
182        }
183    }

185    @Override
186    public void updateInstance(ServerContext context, String resourceId, UpdateRequest request, ResultHandler<Resource> handler) {
187        try {
188            Authentication.setAuthenticatedUserId(context.asContext(SecurityContext.class).getAuthenticationId());
189            Task task = processEngine.getTaskService().createTaskQuery().taskId(resourceId).singleResult();
190            if (task == null) {
191                handler.handleError(new NotFoundException());
192            } else {
193                Map value = request.getContent().expect(Map.class).asMap();
194                if (value.containsKey(ActivitiConstants.ACTIVITI_ASSIGNEE)) {
195                    task.setAssignee(value.get(ActivitiConstants.ACTIVITI_ASSIGNEE) == null
196                            ? null : value.get(ActivitiConstants.ACTIVITI_ASSIGNEE).toString());
197                }
198                if (value.get(ActivitiConstants.ACTIVITI_DESCRIPTION) != null) {
199                    task.setDescription(value.get(ActivitiConstants.ACTIVITI_DESCRIPTION).toString());
200                }
201                if (value.get(ActivitiConstants.ACTIVITI_NAME) != null) {
202                    task.setName(value.get(ActivitiConstants.ACTIVITI_NAME).toString());
203                }
204                if (value.get(ActivitiConstants.ACTIVITI_OWNER) != null) {
205                    task.setOwner(value.get(ActivitiConstants.ACTIVITI_OWNER).toString());
206                }
207                processEngine.getTaskService().saveTask(task);
208                Map<String, String> result = new HashMap<String, String>(1);
209                result.put("Task updated", resourceId);
210                handler.handleResult(new Resource(resourceId, null, new JsonValue(result)));
211            }
212        } catch (Exception ex) {
213            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
214        }
215    }

217    /**
218     * Process the query parameters of the request and set it on the TaskQuery
219     *
220     * @param query Query to update
221     * @param request incoming request
222     */
223    private void setTaskParams(TaskQuery query, QueryRequest request) {
224        String executionId = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_EXECUTIONID);
225        query = executionId == null ? query : query.executionId(executionId);
226        String processDefinitionId = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_PROCESSDEFINITIONID);
227        query = processDefinitionId == null ? query : query.processDefinitionId(processDefinitionId);
228        String processDefinitionKey = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_PROCESSDEFINITIONKEY);
229        query = processDefinitionKey == null ? query : query.processDefinitionKey(processDefinitionKey);
230        String processInstanceId = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_PROCESSINSTANCEID);
231        query = processInstanceId == null ? query : query.processInstanceId(processInstanceId);
232        String taskAssignee = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_ASSIGNEE);
233        query = taskAssignee == null ? query : query.taskAssignee(taskAssignee);
234        String taskCandidateGroup = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_CANDIDATEGROUP);
235        if (taskCandidateGroup != null) {
236            String[] taskCandidateGroups = taskCandidateGroup.split(",");
237            if (taskCandidateGroups.length > 1) {
238                query.taskCandidateGroupIn(Arrays.asList(taskCandidateGroups));
239            } else {
240                query.taskCandidateGroup(taskCandidateGroup);
241            }
242        }
243        String taskCandidateUser = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_CANDIDATEUSER);
244        query = taskCandidateUser == null ? query : query.taskCandidateUser(taskCandidateUser);
245        String taskId = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_TASKID);
246        query = taskId == null ? query : query.taskId(taskId);
247        String taskName = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_TASKNAME);
248        query = taskName == null ? query : query.taskName(taskName);
249        String taskOwner = ActivitiUtil.getParamFromRequest(request, ActivitiConstants.ACTIVITI_OWNER);
250        query = taskOwner == null ? query : query.taskOwner(taskOwner);
251
252        Map<String, String> wfParams = ActivitiUtil.fetchVarParams(request);
253        Iterator<Map.Entry<String, String>> itWf = wfParams.entrySet().iterator();
254        while (itWf.hasNext()) {
255            Map.Entry<String, String> e = itWf.next();
256            query = query.processVariableValueEquals(e.getKey(), e.getValue());
257        }
258    }
