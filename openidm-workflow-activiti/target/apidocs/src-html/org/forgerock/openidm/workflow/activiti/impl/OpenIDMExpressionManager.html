024package org.forgerock.openidm.workflow.activiti.impl;
025
026import java.util.HashMap;
027import java.util.Map;
028import org.activiti.engine.delegate.JavaDelegate;
029import org.activiti.engine.delegate.VariableScope;
030import org.activiti.engine.impl.bpmn.data.ItemInstance;
031import org.activiti.engine.impl.el.ExpressionManager;
032import org.activiti.engine.impl.el.ReadOnlyMapELResolver;
033import org.activiti.engine.impl.el.VariableScopeElResolver;
034import org.activiti.engine.impl.javax.el.*;
035import org.osgi.service.component.ComponentConstants;
036
037/**
038 * Custom ExpressionManager for resolving 'openidm' variable in expressions
039 *
040 * @author orsolyamebold
041 */
042public class OpenIDMExpressionManager extends ExpressionManager {
043    private Map<String, JavaDelegate> delegateMap = new HashMap<String, JavaDelegate>();
044    
045    public void bindService(JavaDelegate delegate, Map props) {
046        String name = (String) props.get(ComponentConstants.COMPONENT_NAME);
047        if (name == null) { //handle blueprint services as well
048            name = (String) props.get("osgi.service.blueprint.compname");
049        }
050        if (name != null) {
051            delegateMap.put(name, delegate);
052        }
053    }
054
055    public void unbindService(JavaDelegate delegate, Map props) {
056        String name = (String) props.get(ComponentConstants.COMPONENT_NAME);
057        if (delegateMap.containsKey(name)) {
058            delegateMap.remove(name);
059        }
060    }
061
062    @Override
063    protected ELResolver createElResolver(VariableScope variableScope) {
064        CompositeELResolver elResolver = new CompositeELResolver();
065        elResolver.add(new VariableScopeElResolver(variableScope));
066        elResolver.add(new OpenIDMELResolver(delegateMap));
067
068        if(beans != null) {
069        // ACT-1102: Also expose all beans in configuration when using standalone activiti, not
070        // in spring-context
071        elResolver.add(new ReadOnlyMapELResolver(beans));
072        }
073
074        elResolver.add(new ArrayELResolver());
075        elResolver.add(new ListELResolver());
076        elResolver.add(new MapELResolver());
077        elResolver.add(new DynamicBeanPropertyELResolver(ItemInstance.class, "getFieldValue", "setFieldValue")); //TODO: needs verification
078        elResolver.add(new BeanELResolver());
079        return elResolver;
080    }
081}
