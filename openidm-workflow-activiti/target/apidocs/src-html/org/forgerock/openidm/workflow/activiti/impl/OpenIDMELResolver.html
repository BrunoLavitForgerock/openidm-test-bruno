050/**
051 * Custom ExpressionResolver for OpenIDM
052 *
053 * @author $author$
054 * @version $Revision$ $Date$
055 */
056public class OpenIDMELResolver extends ELResolver {
057
058    private static final Logger LOGGER = LoggerFactory.getLogger(OpenIDMELResolver.class);
059    private Map<String, JavaDelegate> delegateMap = new HashMap<String, JavaDelegate>();
060    private PersistenceConfig persistenceConfig;
061    private ScriptRegistry scriptRegistry;
062
063    public OpenIDMELResolver(Map<String, JavaDelegate> delegateMap) {
064        this.delegateMap = delegateMap;
065    }
066    
067    @Override
068    public Object getValue(ELContext context, Object base, Object property) {
069        OpenIDMSession session = Context.getCommandContext().getSession(OpenIDMSession.class);
070        persistenceConfig = session.getOpenIDMPersistenceConfig();
071        scriptRegistry = session.getOpenIDMScriptRegistry();
072        Map<String, String> scriptJson = new HashMap<String, String>(3);
073        Bindings bindings = null;
074        String key = (String) property;
075        try {
076            JsonValue openidmContext = (JsonValue) context.getELResolver().getValue(context, null, ActivitiConstants.OPENIDM_CONTEXT);
077            ServerContext serverContext = new ServerContext(openidmContext, persistenceConfig);
078            ScriptEntry script = scriptRegistry.takeScript(new ScriptName("ActivitiScript", "groovy"));
079            if (script == null) {
080                scriptJson.put("source", "");
081                scriptJson.put("type", "groovy");
082                scriptJson.put("name", "ActivitiScript");
083                script = scriptRegistry.takeScript(new JsonValue(scriptJson));
084            }
085            bindings = script.getScriptBindings(serverContext, null);
086        } catch (Exception ex) {
087            LOGGER.error(ex.getMessage(), ex);
088            throw new ActivitiException(ex.getMessage(), ex);
089        }
090        if (base == null) {
091            // according to javadoc, can only be a String
092            if (bindings.containsKey(key)) {
093                context.setPropertyResolved(true);
094                return bindings.get(key);
095            } else {
096                for (String name : delegateMap.keySet()) {
097                    if (name.equalsIgnoreCase(key)) {
098                        context.setPropertyResolved(true);
099                        return delegateMap.get(name);
100                    }
101                }
102            }
103        }
104        //fetching the openidmcontext sets it to true, we need to set it 
105        //to false again if the property was not found
106        context.setPropertyResolved(false);
107        return null;
108    }
109
110    @Override
111    public boolean isReadOnly(ELContext context, Object base, Object property) {
112        return true;
113    }
114
115    @Override
116    public void setValue(ELContext context, Object base, Object property, Object value) {
117    }
118
119    @Override
120    public Class<?> getCommonPropertyType(ELContext context, Object arg) {
121        return Object.class;
122    }
123
124    @Override
125    public Iterator<FeatureDescriptor> getFeatureDescriptors(ELContext context, Object arg) {
126        return null;
127    }
128
129    @Override
130    public Class<?> getType(ELContext context, Object arg1, Object arg2) {
131        return Object.class;
132    }
133
134    /**
135     * Called when openidm.xxx() is called from an Expression
136     * @return result of the call
137    */
138    @Override
139    public Object invoke(ELContext context, Object base, Object method, Class<?>[] paramTypes, Object[] params) {
140        if (base instanceof LazyMap && ((LazyMap)base).containsKey(method)) {
141            context.setPropertyResolved(true);
142            FunctionClosure function = (FunctionClosure)((LazyMap)base).get(method);
143            return function.doCall(params);
144        }
145        return null;
146    }
147}
