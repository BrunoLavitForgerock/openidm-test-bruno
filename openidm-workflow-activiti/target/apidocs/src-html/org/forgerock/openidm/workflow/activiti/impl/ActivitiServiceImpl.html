214    public enum EngineLocation {
215
216        embedded, local, remote
217    }

219    @Activate
220    void activate(ComponentContext compContext) {
221        logger.debug("Activating Service with configuration {}", compContext.getProperties());
222        try {
223            readConfiguration(compContext);
224            if (enabled) {
225                switch (location) {
226                    case embedded: //start our embedded ProcessEngine
227                        try {
228                            // Data Source configuration
229                            if (jndiName != null && jndiName.trim().length() > 0) {
230                                // Get DB connection via JNDI
231                                logger.info("Using DB connection configured via Driver Manager");
232                                InitialContext ctx = null;
233                                try {
234                                    ctx = new InitialContext();
235                                } catch (NamingException ex) {
236                                    logger.warn("Getting JNDI initial context failed: " + ex.getMessage(), ex);
237                                }
238                                if (ctx == null) {
239                                    throw new InvalidException("Current platform context does not support lookup of repository DB via JNDI. "
240                                            + " Use embedded OpenIDM repository instead.");
241                                }
242                                dataSource = (DataSource) ctx.lookup(jndiName);
243                            }
244                        } catch (RuntimeException ex) {
245                            logger.warn("Configuration invalid, can not start JDBC repository.", ex);
246                            throw new InvalidException("Configuration invalid, can not start JDBC repository.", ex);
247                        } catch (NamingException ex) {
248                            throw new InvalidException("Could not find configured jndiName " + jndiName + " to start repository ", ex);
249                        }
250
251                        //we need a TransactionManager to use this
252                        JtaProcessEngineConfiguration configuration = new JtaProcessEngineConfiguration();
253
254                        if (null == dataSource) {
255                            //initialise the default h2 DataSource
256                            JdbcDataSource jdbcDataSource = new org.h2.jdbcx.JdbcDataSource(); //Implement it here. There are examples in the JDBCRepoService
257                            File root = IdentityServer.getFileForWorkingPath("db/activiti/database");
258                            jdbcDataSource.setURL("jdbc:h2:file:" + URLDecoder.decode(root.getPath(), "UTF-8") + ";DB_CLOSE_ON_EXIT=FALSE;DB_CLOSE_DELAY=1000");
259                            jdbcDataSource.setUser("sa");
260                            configuration.setDatabaseType("h2");
261                            configuration.setDataSource(jdbcDataSource);
262                        } else {
263                            configuration.setDataSource(dataSource);
264                        }
265                        configuration.setIdentityService(identityService);
266
267                        configuration.setTransactionManager(transactionManager);
268                        configuration.setTransactionsExternallyManaged(true);
269                        configuration.setDatabaseSchemaUpdate("true");
270
271                        List<SessionFactory> customSessionFactories = configuration.getCustomSessionFactories();
272                        if (customSessionFactories == null) {
273                            customSessionFactories = new ArrayList<SessionFactory>();
274                        }
275                        customSessionFactories.add(idmSessionFactory);
276                        configuration.setCustomSessionFactories(customSessionFactories);
277                        configuration.setExpressionManager(expressionManager);
278
279                        configuration.setMailServerHost(mailhost);
280                        configuration.setMailServerPort(mailport);
281                        configuration.setMailServerUseTLS(starttls);
282                        if (mailusername != null) {
283                            configuration.setMailServerUsername(mailusername);
284                        }
285                        if (mailpassword != null) {
286                            configuration.setMailServerPassword(mailpassword);
287                        }
288
289                        if (historyLevel != null) {
290                            configuration.setHistory(historyLevel);
291                        }
292
293                        //needed for async workflows
294                        configuration.setJobExecutorActivate(true);
295
296                        processEngineFactory = new ProcessEngineFactory();
297                        processEngineFactory.setProcessEngineConfiguration(configuration);
298                        processEngineFactory.setBundle(compContext.getBundleContext().getBundle());
299                        processEngineFactory.init();
300
301                        //ScriptResolverFactory
302                        List<ResolverFactory> resolverFactories = configuration.getResolverFactories();
303                        resolverFactories.add(new OpenIDMResolverFactory());
304                        configuration.setResolverFactories(resolverFactories);
305                        configuration.getVariableTypes().addType(new JsonValueType());
306                        configuration.setScriptingEngines(new OsgiScriptingEngines(new ScriptBindingsFactory(resolverFactories)));
307
308                        //We are done!!
309                        processEngine = processEngineFactory.getObject();
310                        //We need to register the service because the Activiti-OSGi need this to deploy new BAR or BPMN
311                        Hashtable<String, String> prop = new Hashtable<String, String>();
312                        prop.put(Constants.SERVICE_PID, "org.forgerock.openidm.workflow.activiti.engine");
313                        prop.put("openidm.activiti.engine", "true");
314                        compContext.getBundleContext().registerService(ProcessEngine.class.getName(), processEngine, prop);
315
316                        if (null != configurationAdmin) {
317                            try {
318                                barInstallerConfiguration = configurationAdmin.createFactoryConfiguration("org.apache.felix.fileinstall", null);
319                                Dictionary<String, String> props = barInstallerConfiguration.getProperties();
320                                if (props == null) {
321                                    props = new Hashtable<String, String>();
322                                }
323                                props.put("felix.fileinstall.poll", "2000");
324                                props.put("felix.fileinstall.noInitialDelay", "true");
325                                //TODO java.net.URLDecoder.decode(IdentityServer.getFileForPath("workflow").getAbsolutePath(),"UTF-8")
326                                props.put("felix.fileinstall.dir", IdentityServer.getFileForInstallPath(workflowDir).getAbsolutePath());
327                                props.put("felix.fileinstall.filter", ".*\\.bar|.*\\.xml");
328                                props.put("felix.fileinstall.bundles.new.start", "true");
329                                props.put("config.factory-pid", "activiti");
330                                barInstallerConfiguration.update(props);
331                            } catch (IOException ex) {
332                                java.util.logging.Logger.getLogger(ActivitiServiceImpl.class.getName()).log(Level.SEVERE, null, ex);
333                            }
334                        }
335                        activitiResource = new ActivitiResource(processEngine, persistenceConfig);
336                        logger.debug("Activiti ProcessEngine is enabled");
337                        break;
338                    case local: //ProcessEngine is connected by @Reference
339                        activitiResource = new ActivitiResource(processEngine, persistenceConfig);
340                        break;
341//                    case remote: //fetch remote connection parameters
342//                        activitiResource = new HttpRemoteJsonResource(url, username, password);
343//                        break;
344                    default:
345                        throw new InvalidException(CONFIG_LOCATION + " invalid, can not start workflow service.");
346                }
347            }
348        } catch (RuntimeException ex) {
349            logger.warn("Configuration invalid, can not start Activiti ProcessEngine service.", ex);
350            throw ex;
351        } catch (Exception ex) {
352            logger.warn("Configuration invalid, can not start  Activiti ProcessEngine service.", ex);
353            throw new RuntimeException(ex);
354        }
355    }

357    @Deactivate
358    void deactivate(ComponentContext compContext) {
359        logger.debug("Deactivating Service {}", compContext.getProperties());
360        if (null != barInstallerConfiguration) {
361            try {
362                barInstallerConfiguration.delete();
363            } catch (IOException e) {
364                logger.error("Can not delete org.apache.felix.fileinstall-activiti configuration", e);
365            }
366            barInstallerConfiguration = null;
367        }
368        if ("h2".equals(((ProcessEngineImpl)processEngine).getProcessEngineConfiguration().getDatabaseType() )) {
369            DataSource h2DdataSource = ((ProcessEngineImpl)processEngine).getProcessEngineConfiguration().getDataSource();
370            java.sql.Connection conn = null;
371            try {
372                conn = h2DdataSource.getConnection();
373                Statement stat = conn.createStatement();
374                stat.execute("SHUTDOWN");
375                stat.close();
376            } catch (SQLException ex) {
377                logger.warn("H2 database failed to stop properly", ex);
378            }
379            if (conn != null) {
380                try {
381                    conn.close();
382                } catch (SQLException ex) {
383                    logger.warn("H2 database failed to stop properly", ex);
384                }
385            }
386        }
387        if (null != processEngineFactory) {
388            try {
389                processEngineFactory.destroy();
390            } catch (Exception e) {
391                //Do Something?
392            }
393        }
394        logger.info(" Activiti ProcessEngine stopped.");
395    }

397    /**
398     * Read and process Workflow configuration file
399     *
400     * @param compContext
401     */
402    private void readConfiguration(ComponentContext compContext) {
403        EnhancedConfig enhancedConfig = new JSONEnhancedConfig();
404        JsonValue config = enhancedConfig.getConfigurationAsJson(compContext);
405        if (!config.isNull()) {
406            enabled = config.get(CONFIG_ENABLED).defaultTo(true).asBoolean();
407            location = config.get(CONFIG_LOCATION).defaultTo(EngineLocation.embedded.name()).asEnum(EngineLocation.class);
408            JsonValue connectionConfig = config.get(CONFIG_CONNECTION);
409            jndiName = connectionConfig.get(CONFIG_JNDI_NAME).asString();
410            JsonValue mailconfig = config.get(CONFIG_MAIL);
411            if (!mailconfig.isNull()) {
412                mailhost = mailconfig.get(new JsonPointer(CONFIG_MAIL_HOST)).asString();
413                mailport = mailconfig.get(new JsonPointer(CONFIG_MAIL_PORT)).asInteger();
414                mailusername = mailconfig.get(new JsonPointer(CONFIG_MAIL_USERNAME)).asString();
415                mailpassword = mailconfig.get(new JsonPointer(CONFIG_MAIL_PASSWORD)).asString();
416                starttls = mailconfig.get(new JsonPointer(CONFIG_MAIL_STARTTLS)).asBoolean();
417            }
418            JsonValue engineConfig = config.get(CONFIG_ENGINE);
419            if (!engineConfig.isNull()) {
420                url = config.get(new JsonPointer(CONFIG_ENGINE_URL)).asString();
421                username = config.get(new JsonPointer(CONFIG_ENGINE_USERNAME)).asString();
422                password = config.get(new JsonPointer(CONFIG_ENGINE_PASSWORD)).asString();
423            }
424            historyLevel = config.get(CONFIG_HISTORY).asString();
425            workflowDir = config.get(CONFIG_WORKFLOWDIR).defaultTo("workflow").asString();
426        }
427    }

429    //This method called before activate if there is a ProcessEngine service in the Service Registry
430    protected void bindProcessEngine(ProcessEngine processEngine) {
431        logger.info("Some other process already created the ProcessEngine so we don't need to make our own");
432        if (null == processEngine) {
433            this.processEngine = processEngine;
434            selfMadeProcessEngine = false;
435        }
436    }
437
438    protected void unbindProcessEngine(ProcessEngine processEngine) {
439        if (!selfMadeProcessEngine) {
440            this.processEngine = null;
441            this.activitiResource = null;
442        }
443        logger.info("ProcessEngine stopped.");
444    }
