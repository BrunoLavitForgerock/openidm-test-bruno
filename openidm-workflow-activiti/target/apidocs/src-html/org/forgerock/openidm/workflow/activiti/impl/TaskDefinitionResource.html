054/**
055 * Resource implementation of TaskDefinition related Activiti operations
056 *
057 * @author orsolyamebold
058 */
059public class TaskDefinitionResource implements CollectionResourceProvider {
060
061    private final static ObjectMapper mapper;
062    private ProcessEngine processEngine;
063
064    static {
065        mapper = new ObjectMapper();
066        mapper.getSerializationConfig().addMixInAnnotations(TaskDefinition.class, TaskDefinitionMixIn.class);
067        mapper.getSerializationConfig().addMixInAnnotations(EnumFormType.class, EnumFormTypeMixIn.class);
068        mapper.getSerializationConfig().addMixInAnnotations(DateFormType.class, DateFormTypeMixIn.class);
069        mapper.getSerializationConfig().addMixInAnnotations(FormPropertyHandler.class, FormPropertyHandlerMixIn.class);
070        mapper.configure(SerializationConfig.Feature.FAIL_ON_EMPTY_BEANS, false);
071        mapper.configure(SerializationConfig.Feature.SORT_PROPERTIES_ALPHABETICALLY, true);
072    }
073
074    public TaskDefinitionResource(ProcessEngine processEngine) {
075        this.processEngine = processEngine;
076    }
077
078    public void setProcessEngine(ProcessEngine processEngine) {
079        this.processEngine = processEngine;
080    }

106
107    @Override
108    public void queryCollection(ServerContext context, QueryRequest request, QueryResultHandler handler) {
109        try {
110            Authentication.setAuthenticatedUserId(context.asContext(SecurityContext.class).getAuthenticationId());
111            if (ActivitiConstants.QUERY_ALL_IDS.equals(request.getQueryId())) {
112                String processDefinitionId = ((RouterContext) context).getUriTemplateVariables().get("procdefid");
113                ProcessDefinitionEntity procdef = (ProcessDefinitionEntity) ((RepositoryServiceImpl) processEngine.getRepositoryService()).getDeployedProcessDefinition(processDefinitionId);
114                Map<String, TaskDefinition> taskdefinitions = procdef.getTaskDefinitions();
115                for (TaskDefinition taskDefinition : taskdefinitions.values()) {
116                    DefaultTaskFormHandler taskFormHandler = (DefaultTaskFormHandler) taskDefinition.getTaskFormHandler();
117                    Map value = mapper.convertValue(taskDefinition, HashMap.class);
118                    Resource r = new Resource(taskDefinition.getKey(), null, new JsonValue(value));
119                    r.getContent().add(ActivitiConstants.ACTIVITI_FORMRESOURCEKEY, taskFormHandler.getFormKey());
120                    handler.handleResource(r);
121                }
122                handler.handleResult(new QueryResult());
123            } else {
124                handler.handleError(new BadRequestException("Unknown query-id"));
125            }
126        } catch (IllegalArgumentException ex) {
127            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
128        }
129    }
130
131    @Override
132    public void readInstance(ServerContext context, String resourceId, ReadRequest request, ResultHandler<Resource> handler) {
133        try {
134            Authentication.setAuthenticatedUserId(context.asContext(SecurityContext.class).getAuthenticationId());
135            String processDefinitionId = ((RouterContext) context).getUriTemplateVariables().get("procdefid");
136            ProcessDefinitionEntity procdef = (ProcessDefinitionEntity) ((RepositoryServiceImpl) processEngine.getRepositoryService()).getDeployedProcessDefinition(processDefinitionId);
137            TaskDefinition taskDefinition = procdef.getTaskDefinitions().get(resourceId);
138            if (taskDefinition != null) {
139                Map value = mapper.convertValue(taskDefinition, HashMap.class);
140                Resource r = new Resource(taskDefinition.getKey(), null, new JsonValue(value));
141                FormService formService = processEngine.getFormService();
142                String taskFormKey = formService.getTaskFormKey(processDefinitionId, resourceId);
143                if (taskFormKey != null){
144                    r.getContent().add(ActivitiConstants.ACTIVITI_FORMRESOURCEKEY, taskFormKey);
145                    ByteArrayInputStream startForm = (ByteArrayInputStream) ((RepositoryServiceImpl) processEngine.getRepositoryService()).getResourceAsStream(procdef.getDeploymentId(), taskFormKey);
146                    Reader reader = new InputStreamReader(startForm);
147                    try {
148                        Scanner s = new Scanner(reader).useDelimiter("\\A");
149                        String formTemplate = s.hasNext() ? s.next() : "";
150                        r.getContent().add(ActivitiConstants.ACTIVITI_FORMGENERATIONTEMPLATE, formTemplate);
151                    } finally {
152                        reader.close();
153                    }
154                }
155                handler.handleResult(r);
156            } else {
157                handler.handleError(new NotFoundException("Task definition for " + resourceId + " was not found"));
158            }
159        } catch (ActivitiObjectNotFoundException ex) {
160            handler.handleError(new NotFoundException(ex.getMessage()));
161        } catch (IllegalArgumentException ex) {
162            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
163        } catch (Exception ex) {
164            handler.handleError(new InternalServerErrorException(ex.getMessage(), ex));
165        }
166    }
