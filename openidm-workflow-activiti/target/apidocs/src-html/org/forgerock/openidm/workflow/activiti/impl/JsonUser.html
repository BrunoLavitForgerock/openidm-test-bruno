024package org.forgerock.openidm.workflow.activiti.impl;
025
026import org.activiti.engine.identity.User;
027import org.forgerock.json.fluent.JsonValue;
028import java.util.HashMap;
029import org.forgerock.openidm.crypto.CryptoService;
030
031import static org.forgerock.openidm.workflow.activiti.impl.SharedIdentityService.*;
032
033/**
034 * @author $author$
035 * @version $Revision$ $Date$
036 */
037public class JsonUser extends JsonValue implements User {
038    static final long serialVersionUID = 1L;
039    private CryptoService cryptoService;
040
041    /**
042     * Constructs a JsonUser value object with a given userId.
043     *
044     * @param service OpenIDM CryptoService used for password decryption
045     * @param userId the Java object representing JSON value.
046     */
047    public JsonUser(CryptoService service, String userId) {
048        super(new HashMap());
049        this.cryptoService = service;
050        put(SCIM_USERNAME, userId);
051    }
052
053    public void setCryptoService(CryptoService cryptoService) {
054        this.cryptoService = cryptoService;
055    }
056    
057    public JsonUser(JsonValue value) {
058        super(value);
059    }
060
061    public String getId() {
062        return get(SCIM_USERNAME).required().asString();
063    }
064
065    public void setId(String id) {
066        put(SCIM_USERNAME, id);
067    }
068
069    public String getFirstName() {
070        return get(SCIM_NAME).get(SCIM_NAME_GIVENNAME).asString();
071    }
072
073    public void setFirstName(String firstName) {
074        if (get(SCIM_NAME).isNull()) {
075            put(SCIM_NAME, new HashMap(6));
076        }
077        if (get(SCIM_NAME).isMap()) {
078            get(SCIM_NAME).put(SCIM_NAME_GIVENNAME, firstName);
079        }
080    }
081
082    public void setLastName(String lastName) {
083        if (get(SCIM_NAME).isNull()) {
084            put(SCIM_NAME, new HashMap(6));
085        }
086        if (get(SCIM_NAME).isMap()) {
087            get(SCIM_NAME).put(SCIM_NAME_FAMILYNAME, lastName);
088        }
089    }
090
091    public String getLastName() {
092        return get(SCIM_NAME).get(SCIM_NAME_FAMILYNAME).asString();
093    }
094
095    public void setEmail(String email) {
096
097    }
098
099    public String getEmail() {
100        return null;
101    }
102
103    public String getPassword() {
104        JsonValue password = get(SCIM_PASSWORD);
105        JsonValue decryptedPassword = cryptoService.decrypt(password);
106        return decryptedPassword.asString();
107    }
108
109    public void setPassword(String password) {
110        put(SCIM_PASSWORD, password);
111    }
112}
