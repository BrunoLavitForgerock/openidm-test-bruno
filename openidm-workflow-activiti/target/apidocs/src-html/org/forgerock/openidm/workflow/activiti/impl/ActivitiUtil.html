024package org.forgerock.openidm.workflow.activiti.impl;
025
026import org.forgerock.openidm.workflow.activiti.ActivitiConstants;
027import java.util.HashMap;
028import java.util.Iterator;
029import java.util.Map;
030import java.util.Map.Entry;
031import org.forgerock.json.fluent.JsonException;
032import org.forgerock.json.fluent.JsonTransformer;
033import org.forgerock.json.fluent.JsonValue;
034import org.forgerock.json.resource.CreateRequest;
035import org.forgerock.json.resource.QueryRequest;
036import org.forgerock.openidm.util.DateUtil;
037import org.joda.time.DateTime;
038
039/**
040 * Utility class for Activiti workflow integration
041 * @author orsolyamebold
042 */
043public class ActivitiUtil {
044    /**
045     * Fetch and remove process key from the request
046     * @param request Request to be processed
047     * @return process key
048     */
049    public static String removeKeyFromRequest(CreateRequest request) {
050          return (String) (request.getContent().isNull() ? null : request.getContent().expect(Map.class).asMap().remove("_key"));
051    }
052    
053    /**
054     * Fetch and remove process business key from the request
055     * @param request Request to be processed
056     * @return process business key
057     */
058    public static String removeBusinessKeyFromRequest(CreateRequest request) {
059          return (String) (request.getContent().isNull() ? null : request.getContent().expect(Map.class).asMap().remove("_businessKey"));
060    }
061    
062    /**
063     * Fetch and remove Activiti workflow processDefinitionId if present
064     * @param request Request to be processed
065     * @return processDefinitionId
066     */
067    public static String removeProcessDefinitionIdFromRequest(CreateRequest request) {
068        return (String) (request.getContent().isNull() ? null : request.getContent().expect(Map.class).asMap().remove("_processDefinitionId"));
069    }
070    
071    /**
072     * Fetch the body of the request
073     * @param request Request to be processed
074     * @return request body
075     */
076    public static Map<String, Object> getRequestBodyFromRequest(CreateRequest request) {
077        if (!request.getContent().isNull()) {
078            JsonValue val = request.getContent();
079            val.getTransformers().add(new DatePropertyTransformer());
080            val.applyTransformers();
081            val = val.copy();
082            return new HashMap<String, Object>(val.expect(Map.class).asMap());
083        } else {
084            return new HashMap<String, Object>(1);
085        }
086    }
087    
088    /**
089     * Fetch query parameters from the request
090     * @param request Request to be processed
091     * @param paramName parameter to be fetched
092     * @return 
093     */
094    public static String getParamFromRequest(QueryRequest request, String paramName) {
095        return request.getAdditionalParameters().get(paramName);
096    }
097    
098    private static class DatePropertyTransformer implements JsonTransformer {
099        @Override
100        public void transform(JsonValue value) throws JsonException {
101            if (null != value && value.isString()) {
102                DateTime d = DateUtil.getDateUtil().parseIfDate(value.asString());
103                if (d != null){
104                    value.setObject(d.toDate());
105                }
106            }
107        }
108    }
109    /**
110     * Process the query parameters if they are workflow/task specific
111     * (prefixed: var-...)
112     *
113     * @param request incoming request
114     * @return map of the workflow/task parameters
115     */
116    public static Map<String, String> fetchVarParams(QueryRequest request) {
117        Map<String, String> wfParams = new HashMap<String, String>();
118        Iterator<Entry<String, String>> itAll = request.getAdditionalParameters().entrySet().iterator();
119        while (itAll.hasNext()) {
120            Map.Entry<String, String> e = itAll.next();
121            if ((e.getKey().startsWith(ActivitiConstants.VARIABLE_QUERY_PREFIX))) {
122                wfParams.put(e.getKey().substring(4), e.getValue());
123            }
124        }
125        return wfParams;
126    }
127    
128}
