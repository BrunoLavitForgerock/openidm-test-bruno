025package org.forgerock.openidm.logging.util;
026
027import java.util.logging.Filter;
028import java.util.logging.LogRecord;
029
030public class LogFilter implements Filter {
031
032    @Override
033    public boolean isLoggable(LogRecord rec) {
034        // Filter out well known noise on the console
035
036        // OrientDB RC6 noise
037        if (rec.getLoggerName().startsWith("com.orientechnologies.orient.core.storage.impl.local.")) {
038            if (rec.getMessage().endsWith(
039                    "was not closed correctly last time. Checking segments...")
040                    || rec.getMessage().endsWith("] OK")) {
041                return false;
042            }
043        } else if (rec.getLoggerName().startsWith("org.forgerock.openidm.Framework")) {
044
045            // Felix 4.0.2 shutdown has some noise in trying to unbind
046            // references when services are already gone
047            if (rec.getMessage()
048                    .endsWith(
049                            "[org.forgerock.openidm.managed] Cannot create component instance due to failure to bind reference ref_ManagedObjectService_JsonResourceRouterService")
050                    || rec.getMessage()
051                            .endsWith(
052                                    "[org.forgerock.openidm.sync] Cannot create component instance due to failure to bind reference ref_SynchronizationService_JsonResourceRouterService")
053                    || rec.getMessage()
054                            .endsWith(
055                                    "[org.forgerock.openidm.sync] Component instance could not be created, activation failed")
056                    || rec.getMessage()
057                            .endsWith(
058                                    "[org.forgerock.openidm.authentication] Cannot create component instance due to failure to bind reference httpService")
059                    || rec.getMessage()
060                            .endsWith(
061                                    "[org.forgerock.openidm.authentication] Component instance could not be created, activation failed")
062                    || rec.getMessage()
063                            .endsWith(
064                                    "[org.forgerock.openidm.managed] Component instance could not be created, activation failed")
065                    || (rec.getMessage().contains("org.ops4j.pax.web.pax-web-jetty-bundle [") && rec
066                            .getMessage().endsWith("] FrameworkEvent ERROR"))) {
067                return false;
068            }
069        }
070        return true;
071    }
072}
