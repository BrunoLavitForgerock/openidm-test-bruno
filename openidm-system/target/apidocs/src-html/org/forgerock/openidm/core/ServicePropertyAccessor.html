025package org.forgerock.openidm.core;
026
027import java.util.Map;
028
029import org.osgi.framework.BundleContext;
030import org.osgi.framework.ServiceReference;
031import org.osgi.util.tracker.ServiceTracker;
032
033/**
034 * A ServicePropertyAccessor tracks services with
035 * {@code ServiceTracker<Map<String, Object>, Map<String, Object>>} and gets the
036 * Properties.
037 *
038 * @author Laszlo Hordos
039 */
040public class ServicePropertyAccessor implements PropertyAccessor {
041
042    private final PropertyAccessor delegate;
043    private final BundleContext context;
044    private final ServiceTracker<Map<String, Object>, Map<String, Object>> serviceTracker;
045
046    public ServicePropertyAccessor(final BundleContext context,
047            final ServiceTracker<Map<String, Object>, Map<String, Object>> serviceTracker,
048            final PropertyAccessor delegate) {
049        assert null != context;
050        this.delegate = delegate;
051        this.context = context;
052        this.serviceTracker = serviceTracker;
053    }
054
055    /**
056     * Returns the value of the specified property. If the key is not found in
057     * the tracked services, the {@code delegate} is then requested. The method
058     * returns {@code null} if the property is not found.
059     *
060     * @param key
061     *            The name of the requested property.
062     * @param defaultValue
063     *            The value to return if the property is not found.
064     * @param expected
065     *            The type of the expected value.
066     * @param <T>
067     *            The type of the expected property.
068     * @return The value of the requested property, or {@code null} if the
069     *         property is undefined.
070     * @throws SecurityException
071     *             If the caller does not have the appropriate
072     *             {@code PropertyPermission} to read the property, and the Java
073     *             Runtime Environment supports permissions.
074     */
075    @SuppressWarnings("unchecked")
076    public <T> T getProperty(String key, T defaultValue, Class<T> expected) {
077        T value = null;
078        ServiceReference<Map<String, Object>>[] references = serviceTracker.getServiceReferences();
079        if (null != references) {
080            int i = 0;
081            while (i < references.length) {
082                Object candidateValue = context.getService(references[i]).get(key);
083                if (null != expected && null != candidateValue
084                        && expected.isAssignableFrom(candidateValue.getClass())) {
085                    // We found the value
086                    value = (T) candidateValue;
087                    break;
088                }
089                i++;
090            }
091        }
092        return null != value ? value : (null != delegate) ? delegate.getProperty(key, defaultValue,
093                expected) : null;
094    }
095}
