025package org.forgerock.openidm.core;
026
027import java.util.Stack;
028
029public class PropertyUtil {
030
031    private PropertyUtil() {
032    }
033
034    public enum Delimiter {
035        DOLLAR {
036            @Override
037            char getStartChar() {
038                return '$';
039            }
040
041            @Override
042            String getStartString() {
043                return DELIM_START_DOLLAR;
044            }
045        },
046        AMPERSAND {
047            @Override
048            char getStartChar() {
049                return '&';
050            }
051
052            @Override
053            String getStartString() {
054                return DELIM_START_AMPERSAND;
055            }
056        };
057
058        abstract char getStartChar();
059
060        abstract String getStartString();
061    }
062
063    public static final String DELIM_START_DOLLAR = "${";
064    public static final String DELIM_START_AMPERSAND = "&{";
065    public static final char DELIM_STOP = '}';
066
067    /**
068     * <p>
069     * This method performs property variable substitution on the specified
070     * value. If the specified value contains the syntax
071     * <tt>&{&lt;prop-name&gt;}</tt>, where <tt>&lt;prop-name&gt;</tt> refers to
072     * either a configuration property or a system property, then the
073     * corresponding property value is substituted for the variable placeholder.
074     * Multiple variable placeholders may exist in the specified value as well
075     * as nested variable placeholders, which are substituted from inner most to
076     * outer most. Configuration properties override system properties.
077     * </p>
078     *
079     * @param val
080     *            The string on which to perform property substitution.
081     * @param propertyAccessor
082     *            Set of configuration properties.
083     * @return The value of the specified string after property substitution.
084     **/
085    public static Object substVars(String val, final PropertyAccessor propertyAccessor,
086            boolean doEscape) {
087        return substVars(val, propertyAccessor, Delimiter.AMPERSAND, doEscape);
088
089    }
090
091    public static Object substVars(String val, final PropertyAccessor propertyAccessor,
092            Delimiter delimiter, boolean doEscape) {
093
094        // Assume we have a value that is something like:
095        // "leading &{foo.&{bar}} middle ${baz} trailing"
096
097        int stopDelim = -1;
098        int startDelim = -1;
099
100        if (!doEscape) {
101            stopDelim = val.indexOf(DELIM_STOP, stopDelim + 1);
102            // If there is no stopping delimiter, then just return
103            // the value since there is no variable declared.
104            if (stopDelim < 0) {
105                return val;
106            }
107            startDelim = val.indexOf(delimiter.getStartString());
108            // If there is no starting delimiter, then just return
109            // the value since there is no variable declared.
110            if (startDelim < 0) {
111                return val;
112            }
113        }
114
115        StringBuilder parentBuilder = new StringBuilder(val.length());
116        Stack<StringBuilder> propertyStack = new Stack<StringBuilder>();
117        propertyStack.push(parentBuilder);
118
119        for (int index = 0; index < val.length(); index++) {
120            switch (val.charAt(index)) {
121            case '\\': {
122                if (doEscape) {
123                    index++;
124                    if (index < val.length()) {
125                        propertyStack.peek().append(val.charAt(index));
126                    }
127                } else {
128                    propertyStack.peek().append(val.charAt(index));
129                }
130                break;
131            }
132            case '&': {
133                if ('{' == val.charAt(index + 1) && val.charAt(index) == delimiter.getStartChar()) {
134                    // This is a start of a new property
135                    propertyStack.push(new StringBuilder(val.length()));
136                    index++;
137                } else {
138                    propertyStack.peek().append(val.charAt(index));
139                }
140                break;
141            }
142            case '$': {
143                if ('{' == val.charAt(index + 1) && val.charAt(index) == delimiter.getStartChar()) {
144                    // This is a start of a new property
145                    propertyStack.push(new StringBuilder(val.length()));
146                    index++;
147                } else {
148                    propertyStack.peek().append(val.charAt(index));
149                }
150                break;
151            }
152            case DELIM_STOP: {
153                // End of the actual property
154                if (propertyStack.size() == 1) {
155                    // There is no start delimiter
156                    propertyStack.peek().append(val.charAt(index));
157                } else {
158                    String variable = propertyStack.pop().toString();
159                    if ((index == val.length() - 1) && propertyStack.size() == 1
160                            && parentBuilder.length() == 0) {
161                        // Replace entire value with an Object
162                        Object substValue =
163                                getSubstituteValue(Object.class, variable, propertyAccessor);
164                        if (null != substValue) {
165                            return substValue;
166                        } else {
167                            propertyStack.peek().append(delimiter.getStartString())
168                                    .append(variable).append(DELIM_STOP);
169                            return propertyStack.peek().toString();
170                        }
171                    } else {
172                        String substValue =
173                                getSubstituteValue(String.class, variable, propertyAccessor);
174                        if (null != substValue) {
175                            propertyStack.peek().append(substValue);
176                        } else {
177                            propertyStack.peek().append(delimiter.getStartString())
178                                    .append(variable).append(DELIM_STOP);
179                        }
180                    }
181                }
182                break;
183            }
184            default: {
185                propertyStack.peek().append(val.charAt(index));
186            }
187            }
188        }
189
190        // Close the open &{ tags.
191        for (int index = propertyStack.size(); index > 1; index--) {
192            StringBuilder top = propertyStack.pop();
193            propertyStack.peek().append(delimiter.getStartString()).append(top.toString());
194        }
195        return parentBuilder.toString();
196    }
197
198    @SuppressWarnings("unchecked")
199    private static <T> T getSubstituteValue(Class<T> type, String variable,
200            final PropertyAccessor propertyAccessor) {
201        T substValue = null;
202        if (String.class.isAssignableFrom(type)) {
203            // Get the value of the deepest nested variable
204            // placeholder.
205            // Try to configuration properties first.
206            substValue =
207                    (propertyAccessor != null) ? (T) propertyAccessor.getProperty(variable, null,
208                            String.class) : null;
209        } else {
210            substValue =
211                    (propertyAccessor != null) ? propertyAccessor.getProperty(variable, null, type)
212                            : null;
213        }
214        return substValue;
215    }
