025package org.forgerock.openidm.core.internal;
026
027import java.util.Map;
028
029import org.forgerock.openidm.core.FrameworkPropertyAccessor;
030import org.forgerock.openidm.core.IdentityServer;
031import org.forgerock.openidm.core.ServicePropertyAccessor;
032import org.osgi.framework.BundleActivator;
033import org.osgi.framework.BundleContext;
034import org.osgi.framework.Constants;
035import org.osgi.framework.Filter;
036import org.osgi.util.tracker.ServiceTracker;
037
038/**
039 * An Activator opens {@code ServiceTracker<Map<String, Object>, Map<String,
040 * Object>>} and tracks all service and initialise the {@link IdentityServer}
041 * with {@link org.forgerock.openidm.core.ServicePropertyAccessor}.
042 *
043 * @author Laszlo Hordos
044 */
045public class Activator implements BundleActivator {
046
047    private ServiceTracker<Map<String, Object>, Map<String, Object>> serviceTracker = null;
048
049    public void start(final BundleContext context) throws Exception {
050        String customFilter = context.getProperty("org.forgerock.openidm.core.map.filter");
051        Filter filter = null;
052        if (null != customFilter && customFilter.trim().length() > 4) {
053            filter =
054                    context.createFilter("(&(" + Constants.OBJECTCLASS + "=java.util.Map)"
055                            + customFilter + ")");
056        } else {
057            filter =
058                    context.createFilter("(&(" + Constants.OBJECTCLASS + "=java.util.Map)("
059                            + Constants.SERVICE_PID
060                            + "=org.forgerock.commons.launcher.OSGiFramework))");
061        }
062        serviceTracker =
063                new ServiceTracker<Map<String, Object>, Map<String, Object>>(context, filter, null);
064        serviceTracker.open(true);
065
066        IdentityServer.initInstance(new ServicePropertyAccessor(context, serviceTracker,
067                new FrameworkPropertyAccessor(context, null)));
068    }
069
070    public void stop(BundleContext context) throws Exception {
071        if (null != serviceTracker) {
072            serviceTracker.close();
073        }
074    }
075}
