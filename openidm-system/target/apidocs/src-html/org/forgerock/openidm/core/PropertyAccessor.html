025package org.forgerock.openidm.core;
026
027/**
028 * A NAME does ...
029 *
030 * @author Laszlo Hordos
031 */
032public interface PropertyAccessor {
033    /**
034     * Returns the value of the specified property. If the key is not found in
035     * the Framework properties, the system properties are then searched. The
036     * method returns {@code null} if the property is not found.
037     *
038     * @param key
039     *            The name of the requested property.
040     * @param defaultValue
041     * @param expected
042     * @param <T>
043     *            The type of the expected property.
044     * @return The value of the requested property, or {@code null} if the
045     *         property is undefined.
046     *
047     * @throws SecurityException
048     *             If the caller does not have the appropriate
049     *             {@code PropertyPermission} to read the property, and the Java
050     *             Runtime Environment supports permissions.
051     */
052    <T> T getProperty(String key, T defaultValue, Class<T> expected);
053}
