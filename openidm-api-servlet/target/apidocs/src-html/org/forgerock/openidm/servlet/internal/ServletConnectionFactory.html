079/**
080 * The ConnectionFactory responsible for providing Connections to routing requests initiated
081 * from an external request on the api servlet.
082 *
083 * @author brmiller
084 */
085@Component(name = ServletConnectionFactory.PID, policy = ConfigurationPolicy.OPTIONAL,
086        configurationFactory = false, immediate = true)
087@Service
088@Properties({
089    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
090    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM Common REST Servlet Connection Factory")
091})
092public class ServletConnectionFactory implements ConnectionFactory {
093
094    public static final String PID = "org.forgerock.openidm.router";
095
096    /**
097     * Setup logging for the {@link org.forgerock.openidm.servlet.internal.ServletConnectionFactory}.
098     */
099    private final static Logger logger = LoggerFactory.getLogger(ServletConnectionFactory.class);
100
101    // the created connection factory
102    protected ConnectionFactory connectionFactory;
103
104    /** the Request Handler (Router) */
105    @Reference(target = "(org.forgerock.openidm.router=*)")
106    protected RequestHandler requestHandler = null;
107
108    /** Script Registry service. */
109    @Reference(policy = ReferencePolicy.DYNAMIC)
110    private ScriptRegistry scriptRegistry = null;
111
112    @Activate
113    protected void activate(ComponentContext context) throws ServletException, NamespaceException {
114        logger.debug("Creating servlet router/connection factory");
115        EnhancedConfig config = JSONEnhancedConfig.newInstance();
116        String factoryPid = config.getConfigurationFactoryPid(context);
117        if (StringUtils.isNotBlank(factoryPid)) {
118            throw new IllegalArgumentException(
119                    "Factory configuration not allowed, must not have property: "
120                            + ServerConstants.CONFIG_FACTORY_PID);
121        }
122        try {
123            connectionFactory = Resources.newInternalConnectionFactory(
124                    init(config.getConfigurationAsJson(context), requestHandler));
125        } catch (Throwable t) {
126            logger.error("Failed to configure the Filtered Router service", t);
127        }
128
129        logger.info("Servlet ConnectionFactory created.");
130    }
131
132    @Deactivate
133    protected synchronized void deactivate(ComponentContext context) {
134    }
135
136    /**
137     * Initialize the router with configuration. Supports modifying router configuration.
138     *
139     * @param configuration the router configuration listing filters that are installed
140     * @param handler the request handler (router)
141     * @return the RequestHandler decorated with a FilterChain consisting of any filters that are configured
142     */
143    RequestHandler init(JsonValue configuration, final RequestHandler handler)
144            throws ScriptException {
145        final JsonValue filterConfig = configuration.get("filters").expect(List.class);
146        final List<Filter> filters = new ArrayList<Filter>(filterConfig.size() + 1); // add one for the logging filter
147
148        filters.add(newLoggingFilter());
149
150        for (JsonValue jv : filterConfig) {
151            Filter filter = newFilter(jv);
152            if (null != filter) {
153                filters.add(filter);
154            }
155        }
156
157        // filters will always have at least the logging filter
158        return new FilterChain(handler, filters);
159    }
160
161    /**
162     * Create a Filter from the filter configuration.
163     *
164     * @param config
165     *            the configuration describing a single filter.
166     * @return a Filter
167     * @throws org.forgerock.json.fluent.JsonValueException
168     *             TODO.
169     */
170    public Filter newFilter(JsonValue config) throws JsonValueException, ScriptException {
171
172        FilterCondition filterCondition = null;
173        Pattern pattern = config.get("pattern").asPattern();
174        if (null != pattern) {
175            filterCondition = Filters.matchResourceName(pattern);
176        }
177
178        final EnumSet<RequestType> requestTypes = EnumSet.noneOf(RequestType.class);
179        for (JsonValue method : config.get("methods").expect(List.class)) {
180            requestTypes.add(method.asEnum(RequestType.class));
181        }
182
183        if (!requestTypes.isEmpty()) {
184            if (null == filterCondition) {
185                filterCondition = Filters.matchRequestType(requestTypes);
186            } else {
187                filterCondition =
188                        Filters.and(filterCondition, Filters.matchRequestType(requestTypes));
189            }
190        }
191
192        // TODO add condition to filter condition
193        Pair<JsonPointer, ScriptEntry> condition = getScript(config.get("condition"));
194
195        Pair<JsonPointer, ScriptEntry> onRequest = getScript(config.get("onRequest"));
196        Pair<JsonPointer, ScriptEntry> onResponse = getScript(config.get("onResponse"));
197        Pair<JsonPointer, ScriptEntry> onFailure = getScript(config.get("onFailure"));
198
199        if (null == onRequest && null == onResponse && null == onFailure) {
200            return null;
201        }
202
203        if (null == filterCondition) {
204            return Filters.asFilter(new ScriptedFilter(onRequest, onResponse, onFailure));
205        } else {
206            return Filters.conditionalFilter(filterCondition, Filters.asFilter(
207                    new ScriptedFilter(onRequest, onResponse, onFailure)));
208        }
209    }
210
211    private Pair<JsonPointer, ScriptEntry> getScript(JsonValue scriptJson) throws ScriptException {
212        if (scriptJson.expect(Map.class).isNull()) {
213            return null;
214        }
215
216        ScriptEntry entry = scriptRegistry.takeScript(scriptJson);
217        return Pair.of(scriptJson.getPointer(), entry);
218    }
219
220    private Filter newLoggingFilter() {
221        return Filters.asFilter(
222                new UntypedCrossCutFilter<Void>() {
223                    @Override
224                    public void filterGenericError(ServerContext context, Void state, ResourceException error, ResultHandler<Object> handler) {
225                        int code = error.getCode();
226                        if (logger.isTraceEnabled()) {
227                            logger.trace("Resource exception: {} {}: \"{}\"", new Object[] { error.getCode(), error.getReason(), error.getMessage(), error });
228                        } else if (code >= 500 && code <= 599) { // log server-side errors
229                            logger.warn("Resource exception: {} {}: \"{}\"", new Object[] { error.getCode(), error.getReason(), error.getMessage(), error });
230                        }
231                        handler.handleError(error);
232                    }
233
234                    @Override
235                    public void filterGenericRequest(ServerContext context, Request request, RequestHandler next, CrossCutFilterResultHandler<Void, Object> handler) {
236                        logger.trace("Request: {}", request);
237                        handler.handleContinue(context, null);
238                    }
239
240                    @Override
241                    public <R> void filterGenericResult(ServerContext context, Void state, R result, ResultHandler<R> handler) {
242                        logger.trace("Result: {}", result);
243                        handler.handleResult(result);
244                    }
245
246                    @Override
247                    public void filterQueryResource(ServerContext context, Void state, Resource resource, QueryResultHandler handler) {
248                        logger.trace("Response: {}", resource);
249                        handler.handleResource(resource);
250                    }
251                });
252    }

268    }
