067/**
068 * A NAME does ...
069 * 
070 * @author Laszlo Hordos
071 * @author ckienle
072 * @author brmiller
073 */
074@Component(name = "org.forgerock.openidm.api-servlet",
075        immediate = true,
076        policy = ConfigurationPolicy.IGNORE)
077@Service
078@Properties({
079    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
080    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM Common REST HttpServlet"),
081    @Property(name = EventConstants.EVENT_TOPIC, value = { "org/forgerock/openidm/servlet/*" })
082})
083public class ServletComponent implements EventHandler {
084
085    public static final String PID = "org.forgerock.openidm.router";
086
087    private static final String SERVLET_ALIAS = "/openidm";
088
089    /** Setup logging for the {@link ServletComponent}. */
090    private final static Logger logger = LoggerFactory.getLogger(ServletComponent.class);
091
092    @Reference(policy = ReferencePolicy.DYNAMIC, target = "(service.pid=org.forgerock.openidm.router)")
093    protected ConnectionFactory connectionFactory;
094
095    @Reference
096    private ServletRegistration servletRegistration;
097
098    /** Script Registry service. */
099    @Reference(policy = ReferencePolicy.DYNAMIC)
100    protected ScriptRegistry scriptRegistry;
101
102    // Optional scripts to augment/populate the security context
103    private List<ScriptEntry> augmentSecurityScripts = new CopyOnWriteArrayList<ScriptEntry>();
104
105    // Register script extensions configured
106    @Reference(
107            name = "reference_Servlet_ServletFilterRegistrator",
108            referenceInterface = ServletFilterRegistrator.class,
109            bind = "bindRegistrator",
110            unbind = "unbindRegistrator",
111            cardinality = ReferenceCardinality.OPTIONAL_MULTIPLE,
112            policy = ReferencePolicy.DYNAMIC,
113            strategy = ReferenceStrategy.EVENT
114    )
115    protected Map<ServletFilterRegistrator, ScriptEntry> filterRegistratorMap =
116            new HashMap<ServletFilterRegistrator, ScriptEntry>();
117
118    protected synchronized void bindRegistrator(ServletFilterRegistrator registrator, Map<String, Object> properties) {
119        JsonValue scriptConfig = registrator.getConfiguration()
120                .get(SERVLET_FILTER_SCRIPT_EXTENSIONS)
121                .get(SERVLET_FILTER_AUGMENT_SECURITY_CONTEXT);
122        if (!scriptConfig.isNull() && !scriptConfig.expect(Map.class).isNull()) {
123            try {
124                ScriptEntry augmentScript = scriptRegistry.takeScript(scriptConfig);
125                filterRegistratorMap.put(registrator, augmentScript);
126                augmentSecurityScripts.add(augmentScript);
127                logger.debug("Registered script {}", augmentScript);
128            } catch (ScriptException e) {
129                logger.debug("{} when attempting to registered script {}", new Object[] { e.toString(), scriptConfig, e});
130            }
131        }
132    }
133
134    protected synchronized void unbindRegistrator(ServletFilterRegistrator registrator, Map<String, Object> properties) {
135        ScriptEntry augmentScript = filterRegistratorMap.remove(registrator);
136        if (augmentScript != null) {
137            augmentSecurityScripts.remove(augmentScript);
138            logger.debug("Deregistered script {}", augmentScript);
139        }
140    }
141
142    private HttpServlet servlet;
143
144    @Activate
145    protected void activate(ComponentContext context) throws ServletException, NamespaceException {
146        logger.debug("Try registering servlet at {}", SERVLET_ALIAS);
147        servlet = new HttpServlet(connectionFactory, new IDMSecurityContextFactory(augmentSecurityScripts));
148
149        servletRegistration.registerServlet(SERVLET_ALIAS, servlet, new Hashtable());
150        logger.info("Registered servlet at {}", SERVLET_ALIAS);
151    }
152
153    @Deactivate
154    protected synchronized void deactivate(ComponentContext context) {
155        servletRegistration.unregisterServlet(servlet);
156    }
157
158    // ----- Implementation of EventHandler
159
160    @Override
161    public void handleEvent(Event event) {
162        // TODO: receive the OpenIDM started event and enable the full HTTP
163        // service
164        if (event.getTopic().equals("org/forgerock/openidm/servlet/ACTIVATE")) {
165            try {
166                activate(null);
167            } catch (Exception e) {
168                logger.error("Error activating api-servlet", e);
169            }
170        } else if (event.getTopic().equals("org/forgerock/openidm/servlet/DEACTIVATE")) {
171            deactivate(null);
172        }
173    }
174}
