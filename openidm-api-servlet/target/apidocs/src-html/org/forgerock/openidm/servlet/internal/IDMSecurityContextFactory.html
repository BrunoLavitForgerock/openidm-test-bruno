025package org.forgerock.openidm.servlet.internal;
026
027import org.apache.commons.lang3.StringUtils;
028import org.forgerock.json.fluent.JsonValue;
029import org.forgerock.json.resource.Context;
030import org.forgerock.json.resource.ForbiddenException;
031import org.forgerock.json.resource.ResourceException;
032import org.forgerock.json.resource.SecurityContext;
033import org.forgerock.json.resource.ServerContext;
034import org.forgerock.json.resource.ServiceUnavailableException;
035import org.forgerock.json.resource.servlet.HttpServletContextFactory;
036import org.forgerock.json.resource.servlet.SecurityContextFactory;
037import org.forgerock.script.Script;
038import org.forgerock.script.ScriptEntry;
039import org.forgerock.script.engine.Utils;
040
041import org.forgerock.script.exception.ScriptThrownException;
042import org.slf4j.Logger;
043import org.slf4j.LoggerFactory;
044
045import javax.servlet.http.HttpServletRequest;
046import java.util.List;
047
048/**
049 * An HttpServletContextFactory responsible for creating a SecurityContext for OpenIDM
050 * requests.  Specifically, this class executes each of the augmentation scripts that
051 * were registered with the {@link ServletComponent}.
052 *
053 * @author brmiller
054 */
055public class IDMSecurityContextFactory implements HttpServletContextFactory {
056
057    private final static Logger logger = LoggerFactory.getLogger(IDMSecurityContextFactory.class);
058
059    /** a list of augmentation scripts to run */
060    private final List<ScriptEntry> augmentationScripts;
061
062    /**
063     * Construct the IDMServletContextFactory.
064     *
065     * @param augmentationScripts the list of configured security context augmentation scripts
066     */
067    public IDMSecurityContextFactory(List<ScriptEntry> augmentationScripts) {
068        this.augmentationScripts = augmentationScripts;
069    }
070
071    /**
072     * {@inheritDoc}
073     */
074    @Override
075    public Context createContext(HttpServletRequest request) throws ResourceException {
076
077        final SecurityContextFactory securityContextFactory = SecurityContextFactory.getHttpServletContextFactory();
078        final SecurityContext securityContext = securityContextFactory.createContext(request);
079
080        // execute global security context augmentation scripts
081        for (ScriptEntry augmentScript : augmentationScripts) {
082            augmentContext(augmentScript, securityContext);
083        }
084
085        if (isSecurityContextPopulated(securityContext)) {
086            return securityContext;
087        }
088
089        logger.warn("Rejecting invocation as required context to allow invocation not populated");
090        throw new ServiceUnavailableException(
091                "Rejecting invocation as required context to allow invocation not populated");
092    }
093
094    private boolean isSecurityContextPopulated(SecurityContext context) {
095        try {
096            final JsonValue authorziationId = new JsonValue(context.getAuthorizationId());
097            return !StringUtils.isEmpty(context.getAuthenticationId())
098                    && !StringUtils.isEmpty(authorziationId.get(SecurityContext.AUTHZID_ID).asString())
099                    && !StringUtils.isEmpty(authorziationId.get(SecurityContext.AUTHZID_COMPONENT).asString())
100                    && authorziationId.get(SecurityContext.AUTHZID_ROLES).isList();
101        } catch (NullPointerException e) {
102            return false;
103        }
104    }
105
106    /**
107     * Invoke an augmentation script to modify the SecurityContext as appropriate.
108     *
109     * @param augmentScript the script to execute
110     * @param securityContext the current SecurityContext
111     * @throws ResourceException on failure to execute script
112     */
113    private void augmentContext(ScriptEntry augmentScript, SecurityContext securityContext)
114            throws ResourceException {
115
116        if (!augmentScript.isActive()) {
117            throw new ServiceUnavailableException("Failed to execute inactive script: "
118                    + augmentScript.getName().toString());
119        }
120
121        ServerContext context = new ServerContext(securityContext);
122        final Script script = augmentScript.getScript(context);
123        script.put("security", securityContext);
124
125        try {
126            script.eval();
127        } catch (ScriptThrownException ste) {
128            throw ste.toResourceException(ResourceException.INTERNAL_ERROR,
129                    "Security Context augmentation script '" + augmentScript.getName().toString()
130                    + "' resulted in an error");
131        } catch (Throwable t) {
132            ResourceException re = Utils.adapt(t);
133            logger.warn("augment script {} encountered exception with detail {} " ,
134                    new Object[] { augmentScript.getName().getName(), re.getDetail(), re });
135            throw re;
136        }
137    }
138}
