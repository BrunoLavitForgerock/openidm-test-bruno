024package org.forgerock.openidm.repo;
025
026import java.util.List;
027
028import org.forgerock.json.resource.CreateRequest;
029import org.forgerock.json.resource.DeleteRequest;
030import org.forgerock.json.resource.QueryRequest;
031import org.forgerock.json.resource.ReadRequest;
032import org.forgerock.json.resource.Resource;
033import org.forgerock.json.resource.ResourceException;
034import org.forgerock.json.resource.UpdateRequest;
035
036/**
037 * Common OpenIDM bootstrap repository interface
038 * 
039 * Implementing services provide basic connectivity and access 
040 * to bootstrapping configuration
041 * 
042 * Every repository boot service must provide CRUD access to configuration,
043 * And provide built in query support for queryid query-all-ids
044 * 
045 * @author aegloff
046 * @author ckienle
047 */
048public interface RepoBootService {
049    
050    /**
051     * Creates a new resource in the repository.
052     *
053     * @param request
054     *            the create request
055     * @return the created resource
056     * @throws ResourceException
057     *             if an error was encountered during creation
058     */
059    public Resource create(CreateRequest request) throws ResourceException;
060    
061    /**
062     * Reads a resource from the repository based on the supplied read request.
063     *
064     * @param request
065     *            the read request.
066     * @return the requested resource.
067     * @throws ResourceException
068     *             if an error was encountered during the read.
069     */
070    public Resource read(ReadRequest request) throws ResourceException;
071    
072    /**
073     * Updates a resource in the repository
074     * <p/>
075     * This implementation requires MVCC and hence enforces that clients state
076     * what revision they expect to be updating
077     * <p/>
078     * If successful, this method updates metadata properties within the passed
079     * object, including: a new {@code _rev} value for the revised object's
080     * version
081     *
082     * @param request
083     *            the update request
084     * @return the updated resource
085     * @throws ResourceException
086     *             if an error was encountered while updating
087     */
088    public Resource update(UpdateRequest request) throws ResourceException;
089    
090    /**
091     * Deletes a new resource in the repository.
092     *
093     * @param request
094     *            the delete request
095     * @return an empty resource
096     * @throws ResourceException
097     *             if an error was encountered during the delete
098     */
099    public Resource delete(DeleteRequest request) throws ResourceException;
100    
101    /**
102     * Queries resources in the repository.  Must provide built in query support 
103     * for queryid query-all-ids.
104     *
105     * @param request
106     *            the query request
107     * @return a list containing the query results
108     * @throws ResourceException
109     *             if an error was encountered during query
110     */
111    public List<Resource> query(QueryRequest request) throws ResourceException;
112    
113}
