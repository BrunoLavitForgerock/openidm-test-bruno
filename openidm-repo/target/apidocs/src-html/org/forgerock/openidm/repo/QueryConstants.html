024package org.forgerock.openidm.repo;
025
026import org.forgerock.json.resource.servlet.HttpUtils;
027
028/**
029 * Map key constants for Repository Queries
030 *
031 * @author aegloff
032 */
033public final class QueryConstants {
034    private QueryConstants() {}
035
036    // Keys for the query input
037    
038    /**
039     * Query input key.
040     * 
041     * Query expression key for the query param map to supply a Query expression in-line. 
042     *
043     * The alternative is to specify a query identifier for a pre-configured query. 
044     * 
045     * The query can contain tokens in the form of ${<param-map-key>} which will 
046     * get substituted from the param map passed to the query.
047     * 
048     * When both an expression and ID are present, the query expression takes precedent
049     */
050    public final static String QUERY_EXPRESSION = "_queryExpression";
051    
052    /**
053     * Query input key.
054     * 
055     * Query identifier for the query param map. The query identifier must match a
056     * configured query. 
057     * 
058     * The alternative is to specify a query expression for an in-line query.
059     *
060     * The configured query can contain tokens in the form of ${<param-map-key>}
061     * which will get substituted from the param map passed to the query.
062     * 
063     * When both an expression and ID are present, the query expression takes precedent
064     */    
065    public final static String QUERY_ID = "_queryId";
066    
067    /**
068     * Query filter.
069     */
070    public final static String QUERY_FILTER = "_queryFilter";
071    
072    /**
073     * System populated query input key.
074     * 
075     * Key available for token substitution inside the query expression, identifying the
076     * resource queried by name. Example use: select * from ${_resource} where ...
077     */
078    public final static String RESOURCE_NAME = "_resource";
079
080
081    // Keys in the query output
082    
083    /**
084     * Query output key.
085     * 
086     * Key for record results in JSON object model format, value format List<Map<String, Object>>
087     */
088    public final static String QUERY_RESULT = "result";    
089    
090    /**
091     * Query output key.
092     * 
093     * Key for query result meta-data, indicating how long the query took in ms
094     * Value format is Long.
095     */
096    public final static String STATISTICS_QUERY_TIME = "query-time-ms";
097    
098    /**
099     * Query output key.
100     * 
101     * Key for query result meta-data, indicating how long the conversion of the results records took in ms
102     * Value format is Long.
103     */
104    public final static String STATISTICS_CONVERSION_TIME = "conversion-time-ms";
105    
106    
107    // Well known values for the query map input
108    
109    /**
110     * Query input value for the QUERY_ID input key.
111     * 
112     * Querying with this query id results in querying all object IDs for a given ObjectSet.
113     * 
114     * All ObjectSets supporting query must support the query corresponding to this ID
115     */    
116    public final static String QUERY_ALL_IDS = "query-all-ids";
117
118    /**
119     * Pagination offset requested. Generally used in an OFFSET clause.
120     */
121    public final static String PAGED_RESULTS_OFFSET = HttpUtils.PARAM_PAGED_RESULTS_OFFSET;
122
123    /**
124     * Page size requested. Generally used in a LIMIT clause.
125     */
126    public static final String PAGE_SIZE = HttpUtils.PARAM_PAGE_SIZE;
127}
