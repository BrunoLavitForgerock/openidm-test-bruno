060    /**
061     * Get the DB pool for the given URL. May return an existing pool instance.
062     * Also can initialize/create/update the DB to meet the passed
063     * configuration if setupDB is enabled
064     * 
065     * Do not close the returned pool directly as it may be used by others.
066     * 
067     * To cleanly shut down the application, call closePools at the end
068     * 
069     * @param dbURL the orientdb URL
070     * @param user the orientdb user to connect
071     * @param password the orientdb password to connect
072     * @param minSize the orientdb pool minimum size
073     * @param maxSize the orientdb pool maximum size
074     * @param completeConfig the full configuration for the DB
075     * @param setupDB true if it should also check the DB exists in the state
076     * to match the passed configuration, and to set it up to match 
077     * @return the pool
078     * @throws org.forgerock.openidm.config.enhanced.InvalidException
079     */
080    public synchronized static ODatabaseDocumentPool getPool(String dbURL, String user, String password, 
081            int minSize, int maxSize, JsonValue completeConfig, boolean setupDB) throws InvalidException {
082
083        ODatabaseDocumentTx setupDbConn = null;
084        ODatabaseDocumentPool pool = null;
085        try {
086            // Disable Snappy compression as it does not currently work inside
087            // OSGi containers. Need to do this here before the DB is created.
088            OGlobalConfiguration.STORAGE_COMPRESSION_METHOD.setValue(ONothingCompression.NAME);
089       
090            if (setupDB) {
091                logger.debug("Check DB exists in expected state for pool {}", dbURL);
092                setupDbConn = checkDB(dbURL, user, password, completeConfig);
093            }
094            logger.debug("Getting pool {}", dbURL);
095            pool = pools.get(dbURL);
096            if (pool == null) {
097                pool = initPool(dbURL, user, password, minSize, maxSize, completeConfig);
098                pools.put(dbURL, pool);
099            }
100        } finally {
101            if (setupDbConn != null) {
102                setupDbConn.close();
103            }
104        }
105        
106        return pool;
107    }

109    /**
110     * Updates the username and password for the default admin user
111     * 
112     * @param dbURL the orientdb URL
113     * @param oldUser the old orientdb user to update
114     * @param oldPassword the old orientdb password to update
115     * @param oldUser the new orientdb user
116     * @param oldPassword the new orientdb password
117     * @throws org.forgerock.openidm.config.InvalidException
118     */
119    public synchronized static void updateDbCredentials(String dbURL, String oldUser, String oldPassword, 
120            String newUser, String newPassword) throws InvalidException {
121
122        ODatabaseDocumentTx db = null;
123        try {
124            db = new ODatabaseDocumentTx(dbURL);
125            db.open(oldUser, oldPassword);
126            OSecurity security = db.getMetadata().getSecurity();
127            // Delete the old admin user
128            security.dropUser(oldUser);
129            // Create new admin user with new username and password
130            security.createUser(newUser, newPassword, security.getRole(ORole.ADMIN));
131        } catch (Exception e) {
132            logger.error("Error updating DB credentials", e);
133        } finally {
134            if (db != null) {
135                db.close();
136            }
137        }
138    }

140    /**
141     * Closes all pools managed by this helper
142     * Call at application shut-down to cleanly shut down the pools.
143     */
144    public synchronized static void closePools() {
145        logger.debug("Close DB pools");
146        for (ODatabaseDocumentPool pool : pools.values()) {
147            try {
148                pool.close();
149                logger.trace("Closed pool {}", pool);
150            } catch (Exception ex) {
151                logger.info("Faillure reported in closing pool {}", pool, ex);
152            }
153        }
154        pools = new HashMap(); // release all our closed pool references
155    }

157    /**
158     * Initialize the DB pool.
159     * @param dbURL the orientdb URL
160     * @param user the orientdb user to connect
161     * @param password the orientdb password to connect
162     * @param minSize the orientdb pool minimum size
163     * @param maxSize the orientdb pool maximum size
164     * @param completeConfig
165     * @return the initialized pool
166     * @throws org.forgerock.openidm.config.enhanced.InvalidException
167     */
168    private static ODatabaseDocumentPool initPool(String dbURL, String user, String password, 
169            int minSize, int maxSize, JsonValue completeConfig) throws InvalidException {
170        logger.trace("Initializing DB Pool {}", dbURL);
171        
172        // Enable transaction log
173        OGlobalConfiguration.TX_USE_LOG.setValue(true);
174        
175        // Immediate disk sync for commit 
176        OGlobalConfiguration.TX_COMMIT_SYNCH.setValue(true);
177        
178        // Have the storage closed when the DB is closed.
179        OGlobalConfiguration.STORAGE_KEEP_OPEN.setValue(false);
180        
181        boolean success = false;
182        int maxRetry = 10;
183        int retryCount = 0;
184        ODatabaseDocumentPool pool = null;
185        
186        // Initialize and try to verify the DB. Retry maxRetry times.
187        do {
188            retryCount++;
189            if (pool != null) {
190                pool.close();
191            }
192            pool = ODatabaseDocumentPool.global();
193            // Moving from 0.9.25 to 1.0 RC had to change this
194            //ODatabaseDocumentPool pool = ODatabaseDocumentPool.global();
195            //pool.setup(minSize, maxSize);
196            warmUpPool(pool, dbURL, user, password, 1);
197            
198            boolean finalTry = (retryCount >= maxRetry);
199            success = test(pool, dbURL, user, password, finalTry);
200        } while (!success && retryCount < maxRetry);
201        
202        if (!success) {
203            logger.warn("DB could not be verified.");
204        } else {
205            logger.info("DB verified on try {}", retryCount);
206        }
207        
208        logger.debug("Opened and initialized pool {}", pool);
209
210        return pool;
211    }

213    /**
214     * Perform a basic access on the DB for a rudimentary test 
215     * @return whether the basic access succeeded
216     */
217    private static boolean test(ODatabaseDocumentPool pool, String dbURL, String user,
218            String password, boolean finalTry) {
219        
220        ODatabaseDocumentTx db = null;
221        try {
222            logger.info("Verifying the DB.");
223            db = pool.acquire(dbURL, user, password);
224            java.util.Iterator iter = db.browseClass("config"); // Config always should exist
225            if (iter.hasNext()) {
226                iter.next();
227            }
228        } catch (OException ex) {
229            if (finalTry) {
230                logger.info("Exceptions encountered in verifying the DB", ex);
231            } else {
232                logger.debug("DB exception in testing.", ex);
233            }
234            return false;
235        } finally {
236            if (db != null) {
237                db.close();
238            } 
239        }
240        return true;
241    }

243    /**
244     * Ensure the min size pool entries are initilized.
245     * Cuts down on some (small) initial latency with lazy init
246     * Do not call with a min past the real pool max, it will block.
247     */
248    private static void warmUpPool(ODatabaseDocumentPool pool, String dbURL, String user,
249            String password, int minSize) {
250        
251        logger.trace("Warming up pool up to minSize {}", Integer.valueOf(minSize));
252        List<ODatabaseDocumentTx> list = new ArrayList<ODatabaseDocumentTx>();
253        for (int count=0; count < minSize; count++) {
254            logger.trace("Warming up entry {}", Integer.valueOf(count));
255            try {
256                list.add(pool.acquire(dbURL, user, password));
257            } catch (Exception ex) {
258                logger.warn("Issue in warming up db pool, entry {}", Integer.valueOf(count), ex);
259            }
260        }
261        for (ODatabaseDocumentTx entry : list) {
262            try {
263                if (entry != null) {
264                    entry.close();
265                }
266            } catch (Exception ex) {
267                logger.warn("Issue in connection close during warming up db pool, entry {}", entry, ex);
268            }
269        }
270    }

272    /**
273     * Ensures the DB is present in the expected form.
274     * @return the db reference. The caller MUST reliably close that DB when done with it, e.g. in a finally.
275     * Be aware that an in-memory DB will disappear if there is no connection open to it,
276     * and the keep open setting is not explicitly set to true
277     */
278    private static ODatabaseDocumentTx checkDB(String dbURL, String user, String password, JsonValue completeConfig) 
279            throws InvalidException {
280        
281        // TODO: Creation/opening of db may be not be necessary if we require this managed externally
282        ODatabaseDocumentTx db = new ODatabaseDocumentTx(dbURL);
283        boolean dbExists = false;
284
285        // To add support for remote DB checking/creation one 
286        // would need to use OServerAdmin instead
287        // boolean dbExists = new OServerAdmin(dbURL).connect(user, password).existsDatabase();
288
289        // Local DB we can auto populate 
290        if (isLocalDB(dbURL) || isMemoryDB(dbURL)) {
291            if (db.exists()) {
292                logger.info("Using DB at {}", dbURL);
293                db.open(user, password); 
294                populateSample(db, completeConfig);
295            } else { 
296                logger.info("DB does not exist, creating {}", dbURL);
297                db.create();
298                // Delete default admin user
299                OSecurity security = db.getMetadata().getSecurity();
300                security.dropUser(OUser.ADMIN);
301                // Create new admin user with new username and password
302                security.createUser(user, password, security.getRole(ORole.ADMIN));
303                populateSample(db, completeConfig);
304            } 
305        } else {
306            logger.info("Using remote DB at {}", dbURL);
307        }
308        return db;
309    }

311    /**
312     * Whether the URL represents a local DB
313     * @param dbURL the OrientDB db url
314     * @return true if local, false if remote
315     * @throws InvalidException if the dbURL is null or otherwise known to be invalid
316     */
317    public static boolean isLocalDB(String dbURL) throws InvalidException {
318        if (dbURL == null) {
319            throw new InvalidException("dbURL is not set");
320        }
321        return dbURL.startsWith("local:") || dbURL.startsWith("plocal");
322    }

324    /**
325     * Whether the URL represents a memory DB
326     * @param dbURL the OrientDB db url
327     * @return true if local, false if remote
328     * @throws InvalidException if the dbURL is null or otherwise known to be invalid
329     */
330    public static boolean isMemoryDB(String dbURL) throws InvalidException {
331        if (dbURL == null) {
332                throw new InvalidException("dbURL is not set");
333        }
334        return dbURL.startsWith("memory:");
335    }

337    // TODO: Review the initialization mechanism
338    private static void populateSample(ODatabaseDocumentTx db, JsonValue completeConfig) 
339            throws InvalidException {
340        
341        JsonValue dbStructure = completeConfig.get(OrientDBRepoService.CONFIG_DB_STRUCTURE);
342        if (dbStructure == null) {
343            logger.warn("No database structure defined in the configuration." + completeConfig);
344        } else {
345            JsonValue orientDBClasses = dbStructure.get(OrientDBRepoService.CONFIG_ORIENTDB_CLASS);
346            OSchema schema = db.getMetadata().getSchema();
347            
348            // Default always to create Config class for bootstrapping
349            if (orientDBClasses == null || orientDBClasses.isNull()) {
350                orientDBClasses = new JsonValue(new java.util.HashMap());
351            }
352            
353            Map cfgIndexes = new java.util.HashMap();
354            orientDBClasses.put("config", cfgIndexes);
355                            
356            logger.info("Setting up database");
357            if (orientDBClasses != null) {
358            for (Object key : orientDBClasses.keys()) {
359                String orientClassName = (String) key;
360                JsonValue orientClassConfig = (JsonValue) orientDBClasses.get(orientClassName);
361                if (schema.existsClass(orientClassName)) {
362                        logger.trace("OrientDB class {} already exists, checking indexes", orientClassName);
363                        createOrienDBClassIndexes(schema, orientClassName, orientClassConfig);
364                } else {
365                    createOrientDBClass(db,schema, orientClassName, orientClassConfig);
366                    if ("internal_user".equals(orientClassName)) {
367                        populateDefaultUsers(orientClassName, db, completeConfig);
368                    }
369                }
370            }
371            }
372            schema.save(); 
373            
374        }
375    }

377    private static void createOrientDBClass(ODatabaseDocumentTx db, OSchema schema,
378            String orientClassName, JsonValue orientClassConfig) {
379        
380        logger.info("Creating OrientDB class {}", orientClassName);
381        OClass orientClass = schema.createClass(orientClassName, 
382                db.addCluster(orientClassName, 
383                OStorage.CLUSTER_TYPE.PHYSICAL));
384        
385        JsonValue indexes = orientClassConfig.get(OrientDBRepoService.CONFIG_INDEX);
386        for (JsonValue index : indexes) {
387            String propertyType = index.get(OrientDBRepoService.CONFIG_PROPERTY_TYPE).asString();
388            String indexType = index.get(OrientDBRepoService.CONFIG_INDEX_TYPE).asString();
389            
390            String propertyName = index.get(OrientDBRepoService.CONFIG_PROPERTY_NAME).asString();
391            String[] propertyNames = null; 
392            if (propertyName != null) {
393                propertyNames = new String[] {propertyName};
394            } else {
395                List propNamesList = index.get(OrientDBRepoService.CONFIG_PROPERTY_NAMES).asList();
396                if (propNamesList == null) {
397                    throw new InvalidException("Invalid index configuration. " 
398                            + "Missing property name(s) on index configuration for property type "
399                            + propertyType + " with index type " + indexType + " on " + orientClassName);
400                }
401                propertyNames = (String[]) propNamesList.toArray(new String[0]);
402            }
403
404            // Check if a single property is being defined and create it if so
405            for (String propName : propertyNames) {
406                if (orientClass.getProperty(propName) != null) {
407                    continue;
408                }
409                logger.info("Creating property {} of type {}", new Object[] {propName, propertyType});
410                OType orientPropertyType = null;
411                try {
412                    // Create property type object
413                    orientPropertyType = OType.valueOf(propertyType.toUpperCase());
414                } catch (IllegalArgumentException ex) {
415                    throw new InvalidException("Invalid property type '"
416                            + propertyType + "' in configuration on properties "
417                            + propertyNames + " with index type " + indexType
418                            + " on " + orientClassName + " valid values: " + OType.values() 
419                            + " failure message: " + ex.getMessage(), ex);
420                }
421                // Create property
422                orientClass.createProperty(propName, orientPropertyType);
423            }
424            createIndex(orientClass, indexType, propertyNames, propertyType);
425        }
426    }

428    // Populates the default user, the pwd needs to be changed by the installer
429    private static void populateDefaultUsers(String defaultTableName, ODatabaseDocumentTx db, 
430            JsonValue completeConfig) throws InvalidException {
431        
432        String defaultAdminUser = "openidm-admin";
433        // Default password needs to be replaced after installation
434        String defaultAdminPwd = "openidm-admin";
435        String defaultAdminRoles = "openidm-admin,openidm-authorized";
436        populateDefaultUser(defaultTableName, db, completeConfig, defaultAdminUser, 
437                defaultAdminPwd, defaultAdminRoles);
438        logger.trace("Created default user {}. Please change the assigned default password.", 
439                defaultAdminUser);
440        
441        String anonymousUser = "anonymous";
442        String anonymousPwd = "anonymous";
443        String anonymousRoles = "openidm-reg";
444        populateDefaultUser(defaultTableName, db, completeConfig, anonymousUser, anonymousPwd, anonymousRoles);
445        logger.trace("Created default user {} for registration purposes.", anonymousUser);
446    }

448    private static void populateDefaultUser(String defaultTableName, ODatabaseDocumentTx db, 
449            JsonValue completeConfig, String user, String pwd, String roles) throws InvalidException {        
450        
451        JsonValue defaultAdmin = new JsonValue(new HashMap<String, Object>());
452        defaultAdmin.put("_openidm_id", user);
453        defaultAdmin.put("userName", user);
454        defaultAdmin.put("password", pwd);
455        defaultAdmin.put("roles", roles);
456        
457        try {
458            ODocument newDoc = DocumentUtil.toDocument(defaultAdmin.asMap(), null, db, defaultTableName);
459            newDoc.save();
460        } catch (ConflictException ex) {
461            throw new InvalidException("Unexpected failure during DB set-up of default user", ex);
462        }
463    }

465    private static String uniqueIndexName(String orientClassName, String[] propertyNames) {
466        // Determine a unique name to use for the index
467        // Naming pattern used is <class>!property1[!propertyN]*!Idx
468        StringBuilder sb = new StringBuilder(orientClassName);
469        sb.append("!");
470        for (String entry : propertyNames) {
471            sb.append(entry);
472            sb.append("!");
473        }
474        sb.append("Idx");
475        return sb.toString();
476    }

478    private static void createIndex(OClass orientClass, String indexType, String[] propertyNames, String propertyType) {
479            logger.info("Creating index on properties {} of type {} with index type {} on {} for OrientDB class ", 
480                    new Object[] {propertyNames, propertyType, indexType, orientClass.getName()});
481            try {
482                // Create the index
483                String indexName = uniqueIndexName(orientClass.getName(), propertyNames);
484                OClass.INDEX_TYPE orientIndexType = OClass.INDEX_TYPE.valueOf(indexType.toUpperCase());
485                orientClass.createIndex(indexName, orientIndexType, propertyNames);
486            } catch (IllegalArgumentException ex) {
487                throw new InvalidException("Invalid index type '" + indexType + 
488                        "' in configuration on properties "
489                        + propertyNames + " of type " + propertyType + " on " 
490                        + orientClass.getName() + " valid values: " + OClass.INDEX_TYPE.values() 
491                        + " failure message: " + ex.getMessage(), ex);
492            }
493    }

495    private static void createOrienDBClassIndexes(OSchema schema, String orientClassName, JsonValue orientClassConfig) {
496        logger.info("Checking indexes for OrientDB class {}", orientClassName);
497        OClass orientClass = schema.getClass(orientClassName);
498        if (orientClass == null) {
499            // Should be impossible
500            throw new InvalidException("Invalid index configuration. " 
501                    + "Index defined for  " + orientClassName + " which does not exist");
502        } else {
503            JsonValue indexes = orientClassConfig.get(OrientDBRepoService.CONFIG_INDEX);
504            for (JsonValue index : indexes) {
505                String propertyType = index.get(OrientDBRepoService.CONFIG_PROPERTY_TYPE).asString();
506                String indexType = index.get(OrientDBRepoService.CONFIG_INDEX_TYPE).asString();
507
508                String propertyName = index.get(OrientDBRepoService.CONFIG_PROPERTY_NAME).asString();
509                String[] propertyNames = null; 
510                if (propertyName != null) {
511                    propertyNames = new String[] {propertyName};
512                } else {
513                    List propNamesList = index.get(OrientDBRepoService.CONFIG_PROPERTY_NAMES).asList();
514                    if (propNamesList == null) {
515                        throw new InvalidException("Invalid index configuration. " 
516                                + "Missing property name(s) on index configuration for property type "
517                                + propertyType + " with index type " + indexType + " on " + orientClassName);
518                    }
519                    propertyNames = (String[]) propNamesList.toArray(new String[0]);
520                }
521                
522                String indexName = uniqueIndexName(orientClass.getName(), propertyNames);
523                if (orientClass.getClassIndex(indexName) != null) {
524                    logger.trace("Index {} already exists for properties {} of type {} with index type"
525                            + " {} on {} for OrientDB class, skipping",
526                            new Object[] {indexName, propertyNames, propertyType, indexType, orientClass.getName()});
527                } else {
528                    createIndex(orientClass, indexType, propertyNames, propertyType);
529                }
530            }
531        }
