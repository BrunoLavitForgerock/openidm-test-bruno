049/**
050 * Component for embedded OrientDB server
051 *
052 * @author aegloff
053 */
054public class EmbeddedOServerService {
055    final static Logger logger = LoggerFactory.getLogger(EmbeddedOServerService.class);
056
057    OServer orientDBServer;
058
059    void activate(JsonValue config) throws Exception {
060        logger.trace("Activating Service with configuration {}", config);
061        try {
062            JsonValue enabled = config.get("embeddedServer").get("enabled");
063
064            // Create regardless of whether enabled to ensure proper shutdown handling (observed in RC9)
065            orientDBServer = OServerMain.create();
066            
067            // enabled flag should be Boolean, but allow for (deprecated) String representation for now.
068            if ((enabled.isBoolean() && Boolean.TRUE.equals(enabled.asBoolean())
069                    || enabled.isString() && "true".equalsIgnoreCase(enabled.asString()))) {
070                OServerConfiguration serverConfig = getOrientDBConfig(config);
071                orientDBServer.startup(serverConfig);
072                orientDBServer.activate();
073                logger.info("Embedded DB server started.");
074            }
075        } catch (Exception ex) {
076            logger.warn("Could not start OrientDB embedded server, service disabled.", ex);
077            throw ex;
078        }
079    }
080
081    void deactivate() {
082        if (orientDBServer != null) {
083            orientDBServer.shutdown();
084            logger.debug("Embedded DB server stopped.");
085        }
086    }
087
088    // TODO: make configurable
089    protected OServerConfiguration getOrientDBConfig(JsonValue config) {
090
091        OServerConfiguration configuration = new OServerConfiguration();
092
093        Boolean studioUiEnabled  = config.get("embeddedServer").get("studioUi")
094                .get("enabled").defaultTo(Boolean.FALSE).asBoolean();
095        
096        Boolean clustered  = config.get("embeddedServer").get("clustered").defaultTo(Boolean.FALSE).asBoolean();
097        
098        if (clustered) {
099            configuration.handlers = new ArrayList<OServerHandlerConfiguration>();
100            OServerHandlerConfiguration handler = new OServerHandlerConfiguration();
101            handler.clazz = "com.orientechnologies.orient.server.handler.distributed.ODistributedServerManager";
102            configuration.handlers.add(handler);
103            
104            String clusterName = config.get("embeddedServer").get("clusterName").defaultTo("openidm").asString();
105            String multicastAddress = config.get("embeddedServer").get("clusterAddress").defaultTo("235.1.1.1").asString();
106            String multicastPort = config.get("embeddedServer").get("clusterPort").defaultTo("2424").asString();
107            String clusterSecurityKey = config.get("embeddedServer").get("clusterSecurityKey").defaultTo("hw3CgjSzqm8I/axu").asString();
108            
109            handler.parameters = new OServerParameterConfiguration[]{
110                    new OServerParameterConfiguration("enabled", Boolean.toString(clustered)),
111                    new OServerParameterConfiguration("name", clusterName),
112                    new OServerParameterConfiguration("security.algorithm", "Blowfish"),
113                    new OServerParameterConfiguration("network.multicast.address", multicastAddress),
114                    new OServerParameterConfiguration("network.multicast.port", multicastPort),
115                    new OServerParameterConfiguration("network.multicast.heartbeat", "10"),
116                    new OServerParameterConfiguration("server.update.delay", "5000"),
117                    new OServerParameterConfiguration("server.electedForLeadership", "true"),
118                    new OServerParameterConfiguration("security.key", clusterSecurityKey),
119            };
120            logger.info("OrientDB clustering enabled on {}:{} with cluster name {}", new Object[] {multicastAddress, multicastPort, clusterName});
121        }
122
123        JsonValue automaticBackup  = config.get("embeddedServer").get("automaticBackup");
124        Boolean backupEnabled = automaticBackup.get("enabled").defaultTo(Boolean.FALSE).asBoolean();
125        
126        if (backupEnabled) {
127            configuration.handlers = new ArrayList<OServerHandlerConfiguration>();
128            OServerHandlerConfiguration handler = new OServerHandlerConfiguration();
129            handler.clazz = "com.orientechnologies.orient.server.handler.OAutomaticBackup";
130            configuration.handlers.add(handler);
131            
132            String targetDirectory = automaticBackup.get("targetDirectory").defaultTo("db/backup").asString();
133            String targetFileName = automaticBackup.get("targetFile").defaultTo("${DBNAME}-${DATE:yyyyMMddHHmmss}.zip").asString();
134            String firstTime = automaticBackup.get("firsttime").defaultTo("23:59:00").asString();
135            String delay = automaticBackup.get("delay").defaultTo("1d").asString();
136            String dbExclude = automaticBackup.get("dbExclude").defaultTo("temp").asString();
137            String dbInclude = automaticBackup.get("dbInclude").defaultTo("openidm").asString();
138            
139            handler.parameters = new OServerParameterConfiguration[]{
140                    new OServerParameterConfiguration("enabled", Boolean.toString(backupEnabled)),
141                    new OServerParameterConfiguration("target.directory", targetDirectory),
142                    new OServerParameterConfiguration("target.fileName", targetFileName),
143                    new OServerParameterConfiguration("firsttime", firstTime),
144                    new OServerParameterConfiguration("delay", delay),
145                    new OServerParameterConfiguration("db.exclude", dbExclude),
146                    new OServerParameterConfiguration("db.include", dbInclude)
147            };
148            logger.info("OrientDB automatic backups enabled every {} starting today at {}", new Object[] {delay, firstTime});
149        }
150
151        configuration.network = new OServerNetworkConfiguration();
152        configuration.network.protocols = new ArrayList<OServerNetworkProtocolConfiguration>();
153        OServerNetworkProtocolConfiguration protocol1 = new OServerNetworkProtocolConfiguration();
154        protocol1.name = "binary";
155        protocol1.implementation = "com.orientechnologies.orient.server.network.protocol.binary.ONetworkProtocolBinary";
156        configuration.network.protocols.add(protocol1);
157        OServerNetworkProtocolConfiguration protocol2 = new OServerNetworkProtocolConfiguration();
158        protocol2.name = "http";
159        protocol2.implementation = "com.orientechnologies.orient.server.network.protocol.http.ONetworkProtocolHttpDb";
160        configuration.network.protocols.add(protocol2);
161//        NOTE: No longer exists as of 1.1.0-SNAPSHOT evidently
162//        OServerNetworkProtocolConfiguration protocol3 = new OServerNetworkProtocolConfiguration();
163//        protocol3.name = "distributed";
164//        protocol3.implementation = "com.orientechnologies.orient.server.network.protocol.distributed.ONetworkProtocolDistributed";
165//        configuration.network.protocols.add(protocol3);
166
167        configuration.network.listeners = new ArrayList<OServerNetworkListenerConfiguration>();
168        OServerNetworkListenerConfiguration listener1 = new OServerNetworkListenerConfiguration();
169
170
171        // Explicitly access the restructured config format, intent is to replace with a more general 
172        // default/override mechanism
173        JsonValue binaryConfig = config.get("embeddedServer").get("overrideConfig")
174                .get("network").get("listeners").get("binary");
175
176        listener1.ipAddress = binaryConfig.get("ipAddress").defaultTo("0.0.0.0").asString();
177        listener1.portRange = binaryConfig.get("portRange").defaultTo("2424-2424").asString();
178        if (clustered) {
179            listener1.protocol = "distributed";
180        } else {
181            listener1.protocol = "binary";
182        }
183        configuration.network.listeners.add(listener1);
184        OServerNetworkListenerConfiguration listener2 = new OServerNetworkListenerConfiguration();
185
186        JsonValue httpConfig = config.get("embeddedServer").get("overrideConfig")
187                .get("network").get("listeners").get("http");
188
189        listener2.ipAddress = httpConfig.get("ipAddress").defaultTo("127.0.0.1").asString();
190        listener2.portRange = httpConfig.get("portRange").defaultTo("2480-2480").asString();
191
192        listener2.protocol = "http";
193
194//        NOTE: no longer used in 1.3
195//        OServerCommandConfiguration command1 = new OServerCommandConfiguration();
196//        command1.pattern = "POST|*.action GET|*.action";
197//        command1.implementation = "com.orientechnologies.orient.server.network.protocol.http.command.post.OServerCommandPostAction";
198//        command1.parameters = new OServerEntryConfiguration[0];
199
200        // Access to the studio web app
201        OServerCommandConfiguration command2 = new OServerCommandConfiguration();
202        command2.pattern = "GET|www GET|studio/ GET| GET|*.htm GET|*.html GET|*.xml GET|*.jpeg GET|*.jpg GET|*.png GET|*.gif GET|*.js GET|*.css GET|*.swf GET|*.ico GET|*.txt GET|*.otf GET|*.pjs GET|*.svg";
203        command2.implementation = "com.orientechnologies.orient.server.network.protocol.http.command.get.OServerCommandGetStaticContent";
204        command2.parameters = new OServerEntryConfiguration[2];
205        command2.parameters[0] = new OServerEntryConfiguration("http.cache:*.htm *.html", "Cache-Control: no-cache, no-store, max-age=0, must-revalidate\r\nPragma: no-cache");
206        command2.parameters[1] = new OServerEntryConfiguration("http.cache:default", "Cache-Control: max-age=120");
207
208        listener2.commands = new OServerCommandConfiguration[]{
209//                command1,
210                command2
211        };
212
213        listener2.parameters = new OServerParameterConfiguration[1];
214        // Connection custom parameters. If not specified the global configuration will be taken
215        listener2.parameters[0] = new OServerParameterConfiguration("network.http.charset", "utf-8");
216
217        if (studioUiEnabled) {
218            configuration.network.listeners.add(listener2);
219        }
220
221        OServerStorageConfiguration storage1 = new OServerStorageConfiguration();
222        storage1.name = "temp";
223        storage1.path = "memory:temp";
224        storage1.userName = "admin";
225        storage1.userPassword = "admin";
226        storage1.loadOnStartup = false;
227        String user = config.get(OrientDBRepoService.CONFIG_USER).defaultTo("admin").asString();
228        String pwd = config.get(OrientDBRepoService.CONFIG_PASSWORD).defaultTo("admin").asString();
229        String dbURL = config.get(OrientDBRepoService.CONFIG_DB_URL).asString();
230        if (dbURL == null || dbURL.isEmpty()) {
231            dbURL = "plocal:" + IdentityServer.getFileForWorkingPath("db/openidm").getAbsolutePath();
232        }
233        File dbFolder = IdentityServer.getFileForWorkingPath(dbURL.split(":")[1]);
234
235        OServerStorageConfiguration storage2 = new OServerStorageConfiguration();
236        storage2.name = "openidm";
237        storage2.path = dbURL;
238        storage2.userName = user;
239        storage2.userPassword = pwd;
240        storage2.loadOnStartup = false;
241
242        configuration.storages = new OServerStorageConfiguration[]{
243                storage1,
244                storage2
245        };
246
247        // Defaulted to the same as the regular user
248        String rootPwd = config.get("embeddedServer").get("rootPwd").defaultTo(pwd).asString();
249        configuration.users = new OServerUserConfiguration[]
250        {
251                new OServerUserConfiguration("root", rootPwd, "*"),
252                new OServerUserConfiguration("guest", null, "server.listDatabases")
253        };
254        configuration.properties = new OServerEntryConfiguration[]{
255                new OServerEntryConfiguration("server.cache.staticResources", "false"),
256                new OServerEntryConfiguration("orientdb.www.path", "db/util/orientdb/studio"),
257                new OServerEntryConfiguration("orient.home", dbFolder.getAbsolutePath())
258                //new OServerEntryConfiguration("log.console.level", "info"),
259                //new OServerEntryConfiguration("log.file.level", "fine")
260        };
261        // OrientDB currently logs a warning if this is not set, 
262        // although it should be taking the setting from the config above instead.
263        System.setProperty("ORIENTDB_HOME", dbFolder.getAbsolutePath());
264
265        return configuration;
266    }
