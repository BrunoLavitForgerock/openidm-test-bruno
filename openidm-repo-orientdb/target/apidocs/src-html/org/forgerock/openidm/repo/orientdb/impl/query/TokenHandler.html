024package org.forgerock.openidm.repo.orientdb.impl.query;
025
026import java.util.Arrays;
027import java.util.List;
028import java.util.Map;
029import java.util.regex.Matcher;
030import java.util.regex.Pattern;
031
032import org.forgerock.json.resource.BadRequestException;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036// TODO: replace use of this class with TokenHandler in org.forgerock.openidm.repo.util
037public class TokenHandler {
038
039    public static final String PREFIX_UNQUOTED = "unquoted";
040
041    public static final String PREFIX_DOTNOTATION = "dotnotation";
042
043    public static final String PREFIX_LIST = "list";
044
045    final static Logger logger = LoggerFactory.getLogger(TokenHandler.class);
046    
047    // The OpenIDM query token is of format ${token-name}
048    Pattern tokenPattern = Pattern.compile("\\$\\{(.+?)\\}");
049
050    /**
051     * Replaces a query string with tokens of format ${token-name} with the values from the
052     * passed in map, where the token-name must be the key in the map
053     * 
054     * @param queryString the query with tokens
055     * @param params the parameters to replace the tokens. Values can be String or List.
056     * @return the query with all tokens replace with their found values
057     * @throws BadRequestException if token in the query is not in the passed parameters
058     */
059    String replaceTokensWithValues(String queryString, Map<String, String> params) 
060            throws BadRequestException {
061        java.util.regex.Matcher matcher = tokenPattern.matcher(queryString);
062        StringBuffer buffer = new StringBuffer();
063        while (matcher.find()) {
064            String fullTokenKey = matcher.group(1);
065            String tokenKey = fullTokenKey;
066            String tokenPrefix = null;
067            String[] tokenKeyParts = tokenKey.split(":", 2);
068            // if prefix found
069            if (tokenKeyParts.length == 2) {
070                tokenPrefix = tokenKeyParts[0];
071                tokenKey = tokenKeyParts[1];
072            }
073            if (!params.containsKey(tokenKey)) {
074                // fail with an exception if token not found
075                throw new BadRequestException("Missing entry in params passed to query for token " + tokenKey);
076            } else {
077                Object replacement = params.get(tokenKey);
078
079                if (PREFIX_LIST.equals(tokenPrefix)) {
080                    // quote each element and split on ,
081                    replacement = Arrays.asList(("'" + replacement.toString().replaceAll(",", "','") + "'").split(","));
082                }
083
084                if (replacement instanceof List) {
085                    StringBuffer commaSeparated = new StringBuffer();
086                    boolean first = true;
087                    for (Object entry : ((List) replacement)) {
088                        if (!first) {
089                            commaSeparated.append(",");
090                        } else {
091                            first = false;
092                        }
093                        commaSeparated.append(entry.toString());
094                    }
095                    replacement = commaSeparated.toString();
096                }
097
098                if (replacement == null) {
099                    replacement = "";
100                }
101                
102                // Optional control of representation via prefix
103                if (tokenPrefix != null) {
104                    if (tokenPrefix.equals(PREFIX_UNQUOTED)) {
105                        // Leave replacement unquoted
106                    } else if (tokenPrefix.equals(PREFIX_DOTNOTATION)) {
107                        // Convert Json Pointer to OrientDB dot notation
108                        String dotDelimited = replacement.toString().replace('/', '.');
109                        if (dotDelimited.startsWith(".")) {
110                            replacement = dotDelimited.substring(1);
111                        } else {
112                            replacement = dotDelimited;
113                        }
114                    }
115                } else {
116                    // Default is single quoted string replacement
117                    replacement = "'" + replacement + "'";
118                }
119                
120                matcher.appendReplacement(buffer, "");
121                buffer.append(replacement);
122            }
123        }
124        matcher.appendTail(buffer);
125        return buffer.toString();
126    }
127
128    /**
129     * Replaces a query string with tokens of format ${token-name} with the 
130     * token format in OrientDB, which is of the form :token-name.
131     * 
132     * OrientDB tokens has some limitations, e.g. they can currently only be used
133     * in the where clause, and hence the returned string is not guaranteed to be
134     * valid for use in a prepared statement. If the parsing fails the system may
135     * have to fall back onto non-prepared statements and manual token replacement.
136     * 
137     * @param queryString the query with OpenIDM format tokens ${token}
138     * @return the query with all tokens replaced with the OrientDB style tokens :token
139     * @throws PrepareNotSupported if this method knows a given statement can not be converted into a prepared statement.
140     * That a statement was not rejected here though does not mean it could not fail during the parsing phase later.
141     */
142    String replaceTokensWithOrientToken(String queryString) throws PrepareNotSupported {
143        Matcher matcher = tokenPattern.matcher(queryString);
144        StringBuffer buf = new StringBuffer();
145        while (matcher.find()) {
146            String origToken = matcher.group(1);
147            
148            String tokenKey = origToken;
149            String tokenPrefix = null;
150            String[] tokenKeyParts = tokenKey.split(":", 2);
151            // if prefix found
152            if (tokenKeyParts.length == 2) {
153                tokenPrefix = tokenKeyParts[0];
154                tokenKey = tokenKeyParts[1];
155            }
156            if (tokenPrefix != null && tokenPrefix.equals(PREFIX_DOTNOTATION)) {
157                throw new PrepareNotSupported("Prepared query not supported for params with dotnotation conversion");
158            }
159            if (tokenKey != null && tokenKey.length() > 0) {
160                // OrientDB token is of format :token-name
161                String newToken = ":" + tokenKey;
162                matcher.appendReplacement(buf, "");
163                buf.append(newToken);
164            }
165        }
166        matcher.appendTail(buf);
167        return buf.toString();
168    }
169}
