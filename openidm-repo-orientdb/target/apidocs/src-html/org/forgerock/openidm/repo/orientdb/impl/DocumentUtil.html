061    /**
062     * Convert to JSON object structures (akin to simple binding), composed of
063     * the basic Java types: {@link Map}, {@link List}, {@link String},
064     * {@link Number}, {@link Boolean}.
065     *
066     * @param doc
067     *            the OrientDB document to convert
068     * @return the Resource with the id, rev, and the doc converted into maps, lists, 
069     *         java types; or null if the doc was null
070     * @throws JsonException
071     *             when the JSON can not be parsed
072     */
073    public static Resource toResource(ODocument doc) {
074        Resource result = null;
075        Map<String, Object> map = toMap(doc);
076        if (map != null) {
077            String id = (String) map.get(Resource.FIELD_CONTENT_ID);
078            String rev = (String) map.get(Resource.FIELD_CONTENT_REVISION);
079            result = new Resource(id, rev, new JsonValue(map));
080        }
081        return result;
082    }

084    /**
085     * Convert to JSON object structures (akin to simple binding), 
086     * composed of the basic Java types: {@link Map}, {@link List}, {@link String}, {@link Number}, {@link Boolean}.
087     * @param doc the OrientDB document to convert
088     * @return the doc converted into maps, lists, java types; or null if the doc was null
089     */
090    public static Map<String, Object> toMap(ODocument doc) {
091        return toMap(doc, true);
092    }

094    /**
095     * Convert to JSON object structures (akin to simple binding), 
096     * composed of the basic Java types: {@link Map}, {@link List}, {@link String}, {@link Number}, {@link Boolean}.
097     * This may change the objects in the passed doc, it is not safe to use doc contents after calling this method.
098     * 
099     * @param doc the OrientDB document to convert
100     * @param topLevel if the passed in document represents a top level orientdb class, or false if it is an embedded document
101     * @return the doc converted into maps, lists, java types; or null if the doc was null
102     */
103    private static Map<String, Object> toMap(ODocument doc, boolean topLevel) {        
104        Map<String, Object> result = null;
105        if (doc != null) {
106            result = new LinkedHashMap<String, Object>(); // TODO: As JSON doesn't, do we really want to maintain order?   
107            for (java.util.Map.Entry<String, Object> entry : doc) {
108                Object value = entry.getValue();
109                String key = entry.getKey();
110                if (key.equals(ORIENTDB_PRIMARY_KEY)) {
111                    logger.trace("Setting primary key to value {}", value);
112                    result.put(TAG_ID, value);
113                    String revision = Integer.toString(doc.getVersion());
114                    logger.trace("Setting revision to {}", revision);
115                    result.put(TAG_REV, revision);
116                } else {
117                    // TODO: optimization switch: if we know that no embedded ODocuments are used 
118                    // (i.e. only embedded Maps, Lists) then we would not need to traverse the whole graph
119                    value = asSimpleBinding(value);
120                    logger.trace("Map setting {} to value {}", key, value);
121                    result.put(key, value);
122                }
123            }
124        }
125        logger.trace("Converted document {} to {}", doc, result);
126        return result;
127    }

129    /**
130     * Recursively ensure that the passed type is represented
131     * or converted to JSON object model simple binding types,
132     * i.e. ODocument to Map, Set to List 
133     * 
134     * Modifies the passed in objToClean where possible (List, Map), 
135     * returns new types where it is not (ODocument, Set)
136     * 
137     * @param objToClean
138     * @return the object in JSON object model representation
139     */
140    @SuppressWarnings("rawtypes")
141    private static Object asSimpleBinding(Object objToClean) {
142        if (objToClean instanceof ODocument) {
143            logger.trace("Converting embedded ODocument {} to map ", objToClean);
144            return DocumentUtil.toMap((ODocument) objToClean, false); 
145        } else if (objToClean instanceof List) {
146            logger.trace("Checking embedded list {} ", objToClean);
147            return toSimpleModel((List) objToClean);
148        } else if (objToClean instanceof Set) {
149            logger.trace("Converting embedded Set {} ", objToClean);
150            return toSimpleModel((Set) objToClean);
151        } else if (objToClean instanceof Map) {
152            logger.trace("Checking embedded map {} ", objToClean);
153            return toSimpleModel((Map) objToClean);
154        } else if (objToClean instanceof com.orientechnologies.orient.core.id.ORID) {
155            // OrientDB should have resolved to an ODocument, might indicate a bug in OrientDB
156            logger.warn("Unexpected value of type ORecordId in document. Returning as String.{}", objToClean);
157            return objToClean.toString();
158        } else {
159            return objToClean;
160        }
161    }
162    
163    /**
164     * Iteratively convert contents as necessary to simple model 
165     * @param listToClean list to modify if necessary
166     * @return the modified list
167     */
168    private static List toSimpleModel(List listToClean) {
169        ListIterator<Object> listIter = listToClean.listIterator();
170        while(listIter.hasNext()) {
171            Object listEntry = listIter.next();
172            if (listEntry instanceof ODocument || listEntry instanceof Set) {
173                // Replace the entry with new type
174                listIter.set(asSimpleBinding(listEntry)); 
175            } else {
176                // Replace directly in the entry
177                asSimpleBinding(listEntry);
178            } 
179        }
180        return listToClean;
181    }
182
183    /**
184     * Iteratively convert contents as necessary to simple model 
185     * @param setToClean set to convert to List and modify if necessary
186     * @return the modified list
187     */
188    private static List toSimpleModel(Set setToClean) {
189        // In JSON there are ordered lists, not Set
190        List replacementList = new ArrayList();
191        for (Object setEntry : setToClean) {
192            replacementList.add(asSimpleBinding(setEntry));
193        }
194        return replacementList;
195    }
196
197    /**
198     * Iteratively convert contents as necessary to simple model 
199     * @param mapToClean map to modify if necessary
200     * @return the modified map
201     */
202    private static Map toSimpleModel(Map<String, Object> mapToClean) {
203        for(Map.Entry<String, Object> entry : mapToClean.entrySet()) {
204            entry.setValue(asSimpleBinding(entry.getValue()));
205        }
206        return mapToClean;
207    }

241    /**
242     * Convert from JSON object structures (akin to simple binding), 
243     * composed of the basic Java types: {@link Map}, {@link List}, {@link String}, {@link Number}, {@link Boolean}.
244     * to OrientDB document
245     * @param objModel the JSON object structure to convert
246     * @param docToPopulate an optional existing ODocument to update with new values from {@code objModel}
247     * @param db the database to associate with the ODocument
248     * @param orientDocClass the OrientDB class of the ODocument to create
249     * @param patch whether the objModel passed in is only partial values (replacing and adding values), 
250     * or if false replaces the whole document with the given {@code objModel}
251     * @param topLevel
252     * @return the converted orientdb document, or null if objModel was null
253     * @throws ConflictException when the revision in the Object model is invalid
254     */
255    protected static ODocument toDocument(Map<String, Object> objModel, ODocument docToPopulate, ODatabaseDocumentTx db, String orientDocClass, boolean patch, 
256            boolean topLevel) throws ConflictException {
257        
258        ODocument result = null;
259        if (objModel != null) {
260            if (docToPopulate == null) {
261                result = db.newInstance(orientDocClass);
262            } else {
263                result = docToPopulate;
264                if (!patch) {
265                    // Remove entries from existing doc that don't exist anymore. ODocument.reset resets too much.
266                    List<String> removalList = new ArrayList<String>();
267                    for (java.util.Map.Entry<String, Object> entry : result) {
268                        String key = entry.getKey();
269                        if (!key.equals(ORIENTDB_PRIMARY_KEY) && !objModel.containsKey(key)) {
270                            removalList.add(key);
271                        }
272                    }
273                    for (String entry : removalList) {
274                        logger.trace("Removing entry {} ", entry);
275                        result.removeField(entry);
276                    }
277                }
278            }
279
280            for (Map.Entry<String, Object> entry : objModel.entrySet()) {
281                String key = entry.getKey();
282                Object value = entry.getValue();
283                if (entry.getKey().equals(TAG_ID)) {
284                    // OpenIDM ID mapping
285                    if (topLevel) {
286                        if (!result.containsField(ORIENTDB_PRIMARY_KEY) 
287                                || !result.field(ORIENTDB_PRIMARY_KEY).equals(value)) {
288                            logger.trace("Setting primary key to {}", value);
289                            result.field(ORIENTDB_PRIMARY_KEY, value);
290                        }
291                    }
292                } else if (key.equals(TAG_REV)) {
293                    // OpenIDM revision to document version mapping
294                    if (topLevel) {
295                        String revString = (String) objModel.get(TAG_REV);
296                        if (revString != null) {
297                            int rev = parseVersion(revString);
298                            logger.trace("Setting version to {}", rev);
299                            if (result.getVersion() != rev) {
300                                result.setVersion(rev);
301                            }
302                        }
303                    }
304                } else if (value instanceof Map) {
305                    // TODO: consider if we should replace this with nested maps rather than nested ODocuments
306                    logger.trace("Handling field {} with embedded map {}", key, value);
307                    ODocument existingDoc = null;
308                    if (docToPopulate != null) {
309                        logger.trace("Update existing embedded map entry {}", key);
310                        Object o = docToPopulate.field(entry.getKey());
311                        if (o instanceof ODocument) {
312                            existingDoc = (ODocument) o;
313                        } else {
314                            docToPopulate.removeField(entry.getKey());
315                        }
316                    }
317                    // TODO: below is temporary work-around for OrientDB update not saving embedded ODocument,
318                    // unless it is a new instance
319                    //if (existingDoc == null) {
320                    logger.trace("Instantiate new ODocument to represent embedded map for {}.", key);
321                    existingDoc = new ODocument(); 
322                    //} 
323                    ODocument converted = toDocument((Map<String, Object>) value, existingDoc, db, null, patch, false);
324                    result.field(entry.getKey(), converted, OType.EMBEDDED);
325                } else {
326                    logger.trace("Setting field {} to value {}", key, value);
327                    result.field(entry.getKey(), value);
328                }
329            }
330        }
331
332        return result;
333    }

335    /**
336     * Parse an OpenIDM revision into an OrientDB MVCC version. OrientDB expects these to be ints.
337     * @param revision the revision String with the OrientDB version in it.
338     * @return the OrientDB version
339     * @throws ConflictException if the revision String could not be parsed into the int expected by OrientDB
340     */
341    public static int parseVersion(String revision) throws ConflictException { 
342        int ver = -1;
343        try {
344            ver = Integer.parseInt(revision);
345        } catch (NumberFormatException ex) {
346            throw new ConflictException("OrientDB repository expects revisions as int, " 
347                    + "unable to parse passed revision: " + revision);
348        }
349        return ver;
350    }
