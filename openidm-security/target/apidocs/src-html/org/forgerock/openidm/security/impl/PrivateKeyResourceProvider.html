047/**
048 * A collection resource provider servicing requests on private key entries in a keystore
049 * 
050 * @author ckienle
051 */
052public class PrivateKeyResourceProvider extends EntryResourceProvider {
053
054    private final static Logger logger = LoggerFactory.getLogger(PrivateKeyResourceProvider.class);
055    
056    public PrivateKeyResourceProvider(String resourceName, KeyStoreHandler store, KeyStoreManager manager, RepositoryService repoService) {
057        super(resourceName, store, manager, repoService);
058    }
059
060    @Override
061    protected void storeEntry(JsonValue value, String alias) throws Exception {
062        String type = value.get("type").defaultTo(DEFAULT_CERTIFICATE_TYPE).asString();
063        PrivateKey privateKey = null;
064        String privateKeyPem = value.get("privateKey").asString();
065        if (privateKeyPem == null) {
066            privateKey = getKeyPair(alias).getPrivate();
067        } else {
068            privateKey = ((KeyPair)fromPem(privateKeyPem)).getPrivate();
069        }
070        if (privateKey == null) {
071            throw new NotFoundException("No private key exists for the supplied signed certificate");
072        }
073        List<String> certStringChain = value.get("certs").required().asList(String.class);
074        Certificate [] certChain = readCertificateChain(certStringChain, type);
075        verify(privateKey, certChain[0]);
076        store.getStore().setEntry(alias, new PrivateKeyEntry(privateKey, certChain), 
077                new KeyStore.PasswordProtection(store.getPassword().toCharArray()));
078        store.store();
079    }
080
081    @Override
082    protected JsonValue readEntry(String alias) throws Exception {
083        Key key = store.getStore().getKey(alias, store.getPassword().toCharArray());
084        if (key == null) {
085            throw new NotFoundException("Alias does not correspond to a key entry in " + resourceName);
086        } else {
087            return returnKey(alias, key);
088        }
089    }
090
091    @Override
092    public void createDefaultEntry(String alias) throws Exception {
093        Pair<X509Certificate, PrivateKey> pair = generateCertificate("localhost", 
094                "OpenIDM Self-Signed Certificate", "None", "None", "None", "None",
095                DEFAULT_ALGORITHM, DEFAULT_KEY_SIZE, DEFAULT_SIGNATURE_ALGORITHM, null, null);
096        Certificate cert = pair.getKey();
097        PrivateKey key = pair.getValue();
098        store.getStore().setEntry(alias, new PrivateKeyEntry(key, new Certificate[]{cert}), 
099                new KeyStore.PasswordProtection(store.getPassword().toCharArray()));
100        store.store();
101    }
102}
