025package org.forgerock.openidm.security.impl;
026
027import org.forgerock.json.fluent.JsonValue;
028import org.forgerock.json.resource.ActionRequest;
029import org.forgerock.json.resource.BadRequestException;
030import org.forgerock.json.resource.CollectionResourceProvider;
031import org.forgerock.json.resource.ConflictException;
032import org.forgerock.json.resource.CreateRequest;
033import org.forgerock.json.resource.DeleteRequest;
034import org.forgerock.json.resource.NotFoundException;
035import org.forgerock.json.resource.NotSupportedException;
036import org.forgerock.json.resource.PatchRequest;
037import org.forgerock.json.resource.QueryRequest;
038import org.forgerock.json.resource.QueryResultHandler;
039import org.forgerock.json.resource.ReadRequest;
040import org.forgerock.json.resource.Resource;
041import org.forgerock.json.resource.ResourceException;
042import org.forgerock.json.resource.ResultHandler;
043import org.forgerock.json.resource.ServerContext;
044import org.forgerock.json.resource.UpdateRequest;
045import org.forgerock.openidm.repo.RepositoryService;
046import org.forgerock.openidm.security.KeyStoreHandler;
047import org.forgerock.openidm.security.KeyStoreManager;
048import org.forgerock.openidm.util.ResourceUtil;
049
050/**
051 * A generic collection resource provider servicing requests on entries in a keystore
052 * 
053 * @author ckienle
054 */
055public abstract class EntryResourceProvider extends SecurityResourceProvider implements CollectionResourceProvider {
056
057    public EntryResourceProvider(String resourceName, KeyStoreHandler store, KeyStoreManager manager, RepositoryService repoService) {
058        super(resourceName, store, manager, repoService);
059    }
060
061    @Override
062    public void createInstance(final ServerContext context, final CreateRequest request,
063            final ResultHandler<Resource> handler) {
064        try {
065            if (null != request.getNewResourceId()) {
066                if (store.getStore().containsAlias(request.getNewResourceId())) {
067                    handler.handleError(new ConflictException("The resource with ID '"
068                            + request.getNewResourceId() + "' could not be created because "
069                            + "there is already another resource with the same ID"));
070                } else {
071                    String resourceId = request.getNewResourceId();
072                    storeEntry(request.getContent(), resourceId);
073                    manager.reload();
074                    // Save the store to the repo (if clustered)
075                    saveStore();
076                    handler.handleResult(new Resource(resourceId, null, request.getContent()));
077                }
078            } else {
079                handler.handleError(new BadRequestException(
080                        "A valid resource ID must be specified in the request"));
081            }
082        } catch (Throwable t) {
083            handler.handleError(ResourceUtil.adapt(t));
084        }
085    }
086
087    @Override
088    public void readInstance(final ServerContext context, final String resourceId,
089            final ReadRequest request, final ResultHandler<Resource> handler) {
090        try {
091            if (!store.getStore().containsAlias(resourceId)) {
092                handler.handleError(new NotFoundException());
093            } else {
094                JsonValue result = readEntry(resourceId);
095                handler.handleResult(new Resource(resourceId, null, result));
096            }
097        } catch (Throwable t) {
098            handler.handleError(ResourceUtil.adapt(t));
099        }
100    }
101
102    @Override
103    public void updateInstance(final ServerContext context, final String resourceId,
104            final UpdateRequest request, final ResultHandler<Resource> handler) {
105        try {
106            storeEntry(request.getContent(), resourceId);
107            manager.reload();
108            // Save the store to the repo (if clustered)
109            saveStore();
110            handler.handleResult(new Resource(resourceId, null, request.getContent()));
111        } catch (Throwable t) {
112            handler.handleError(ResourceUtil.adapt(t));
113        }
114    }
115
116    @Override
117    public void deleteInstance(final ServerContext context, final String resourceId,
118            final DeleteRequest request, final ResultHandler<Resource> handler) {
119        try {
120            if (!store.getStore().containsAlias(resourceId)) {
121                handler.handleError(new NotFoundException());
122            } else {
123                store.getStore().deleteEntry(resourceId);
124                store.store();
125                manager.reload();
126                // Save the store to the repo (if clustered)
127                saveStore();
128                handler.handleResult(new Resource(resourceId, null, new JsonValue(null)));
129            }
130        } catch (Throwable t) {
131            handler.handleError(ResourceUtil.adapt(t));
132        }
133    }
134    
135    @Override
136    public void actionCollection(ServerContext context, ActionRequest request, ResultHandler<JsonValue> handler) {
137        final ResourceException e = new NotSupportedException("Action operations are not supported");
138        handler.handleError(e);
139    }
140
141    @Override
142    public void actionInstance(ServerContext context, String resourceId, ActionRequest request,
143            ResultHandler<JsonValue> handler) {
144        final ResourceException e = new NotSupportedException("Action operations are not supported");
145        handler.handleError(e);
146    }
147
148    @Override
149    public void patchInstance(ServerContext context, String resourceId, PatchRequest request, 
150            ResultHandler<Resource> handler) {
151        final ResourceException e = new NotSupportedException("Patch operations are not supported");
152        handler.handleError(e);
153    }
154
155    @Override
156    public void queryCollection(ServerContext context, QueryRequest request, QueryResultHandler handler) {
157        final ResourceException e = new NotSupportedException("Query operations are not supported");
158        handler.handleError(e);
159    }
160    
161    public abstract void createDefaultEntry(String alias) throws Exception;
162    
163    public boolean hasEntry(String alias) throws Exception {
164        return store.getStore().containsAlias(alias);
165    }
166
167    protected abstract void storeEntry(JsonValue value, String alias) throws Exception;
168    
169    protected abstract JsonValue readEntry(String alias) throws Exception;
170}
