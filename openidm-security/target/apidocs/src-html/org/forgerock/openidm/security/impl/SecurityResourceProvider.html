143    /**
144     * Returns a PEM String representation of a object.
145     * 
146     * @param object the object
147     * @return the PEM String representation
148     * @throws Exception
149     */
150    protected String toPem(Object object) throws Exception {
151        StringWriter sw = new StringWriter(); 
152        PEMWriter pw = new PEMWriter(sw); 
153        pw.writeObject(object); 
154        pw.flush(); 
155        return sw.toString();
156    }

158    /**
159     * Returns an object from a PEM String representation
160     * 
161     * @param pem the PEM String representation
162     * @return the object
163     * @throws Exception
164     */
165    protected <T> T fromPem(String pem) throws Exception {
166        StringReader sr = new StringReader(pem);
167        PEMReader pw = new PEMReader(sr);
168        Object object = pw.readObject();
169        return (T)object;
170    }

172    /**
173     * Reads a certificate from a supplied string representation, and a supplied type.
174     * 
175     * @param certString A String representation of a certificate
176     * @param type The type of certificate ("X509").
177     * @return The certificate
178     * @throws Exception
179     */
180    protected Certificate readCertificate(String certString, String type) throws Exception {
181        StringReader sr = new StringReader(certString);
182        PEMReader pw = new PEMReader(sr);
183        Object object = pw.readObject();
184        if (object instanceof X509Certificate) {
185            return (X509Certificate)object;
186        } else {
187            throw ResourceException.getException(ResourceException.BAD_REQUEST, "Unsupported certificate format");
188        }
189    }

191    /**
192     * Reads a certificate chain from a supplied string array representation, and a supplied type.
193     * 
194     * @param certStringChain an array of strings representing a certificate chain
195     * @param type the type of certificates ("X509")
196     * @return the certificate chain
197     * @throws Exception
198     */
199    protected Certificate[] readCertificateChain(List<String> certStringChain, String type) throws Exception {
200        Certificate [] certChain = new Certificate[certStringChain.size()];
201        for (int i=0; i<certChain.length; i++) {
202            certChain[i] = readCertificate(certStringChain.get(i), type);
203        }
204        return certChain;
205    }

275     * Returns a PEM formatted string representation of an object
276     * 
277     * @param object the object to write
278     * @return a PEM formatted string representation of the object
279     * @throws Exception
280     */
281    protected String getCertString(Object object) throws Exception {
282        PEMWriter pemWriter = null;
283        StringWriter sw = null;
284        try {
285            sw = new StringWriter();
286            pemWriter = new PEMWriter(sw);
287            pemWriter.writeObject(object);
288            pemWriter.flush();
289        } finally {
290            pemWriter.close();
291        }
292        return sw.getBuffer().toString();
293    }

315    /**
316     * Generates a self signed certificate using the given properties.
317     *
318     * @param commonName the subject's common name
319     * @param organization the subject's organization name
320     * @param organizationUnit the subject's organization unit name
321     * @param stateOrProvince the subject's state or province
322     * @param country the subject's country code
323     * @param locatity the subject's locality
324     * @param algorithm the algorithm to use
325     * @param keySize the keysize to use
326     * @param signatureAlgorithm the signature algorithm to use
327     * @param validFrom when the certificate is valid from
328     * @param validTo when the certificate is valid until
329     * @return The generated certificate
330     * @throws Exception
331     */
332    protected Pair<X509Certificate, PrivateKey> generateCertificate(String commonName, 
333            String organization, String organizationUnit, String stateOrProvince, 
334            String country, String locatity, String algorithm, int keySize, 
335            String signatureAlgorithm, String validFrom, String validTo) throws Exception {
336        
337        KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance(algorithm); // "RSA","BC"
338        keyPairGenerator.initialize(keySize);
339        KeyPair keyPair = keyPairGenerator.generateKeyPair();
340
341        // Generate self-signed certificate
342        X500NameBuilder builder = new X500NameBuilder(BCStyle.INSTANCE);
343        builder.addRDN(BCStyle.C, country);
344        builder.addRDN(BCStyle.ST, stateOrProvince);
345        builder.addRDN(BCStyle.L, locatity);
346        builder.addRDN(BCStyle.OU, organizationUnit);
347        builder.addRDN(BCStyle.O, organization);
348        builder.addRDN(BCStyle.CN, commonName);
349
350        Date notBefore = null;
351        Date notAfter = null;
352        if (validFrom == null) {
353            notBefore = new Date(System.currentTimeMillis() - 1000L * 60 * 60 * 24 * 30);
354        } else {
355            DateTime notBeforeDateTime = DateUtil.getDateUtil().parseIfDate(validFrom);
356            if (notBeforeDateTime == null) {
357                throw new InternalServerErrorException("Invalid date format for 'validFrom' property");
358            } else {
359                notBefore = notBeforeDateTime.toDate();
360            }
361        }
362        if (validTo == null) {
363            Calendar date = Calendar.getInstance();
364            date.setTime(new Date());
365            date.add(Calendar.YEAR, 10);
366            notAfter = date.getTime();
367        } else {
368            DateTime notAfterDateTime = DateUtil.getDateUtil().parseIfDate(validTo);
369            if (notAfterDateTime == null) {
370                throw new InternalServerErrorException("Invalid date format for 'validTo' property");
371            } else {
372                notAfter = notAfterDateTime.toDate();
373            }
374        }
375
376        BigInteger serial = BigInteger.valueOf(System.currentTimeMillis());
377
378        X509v3CertificateBuilder v3CertGen = new JcaX509v3CertificateBuilder(builder.build(), serial, 
379                notBefore, notAfter, builder.build(), keyPair.getPublic());
380
381        ContentSigner sigGen = new JcaContentSignerBuilder(signatureAlgorithm).setProvider(BC).build(keyPair.getPrivate());
382
383        X509Certificate cert = new JcaX509CertificateConverter().setProvider(BC).getCertificate(v3CertGen.build(sigGen));
384        cert.checkValidity(new Date());
385        cert.verify(cert.getPublicKey());
386
387        return Pair.of(cert, keyPair.getPrivate());
388    }

390    /**
391     * Generates a CSR request.
392     * 
393     * @param alias
394     * @param algorithm
395     * @param signatureAlgorithm
396     * @param keySize
397     * @param params
398     * @return
399     * @throws Exception
400     */
401    protected Pair<PKCS10CertificationRequest, PrivateKey> generateCSR(String alias, String algorithm, String signatureAlgorithm, int keySize, 
402            JsonValue params) throws Exception {
403
404        // Construct the distinguished name
405        StringBuilder sb = new StringBuilder(); 
406        sb.append("CN=").append(params.get("CN").required().asString().replaceAll(",", "\\\\,"));
407        sb.append(", OU=").append(params.get("OU").defaultTo("None").asString().replaceAll(",", "\\\\,"));
408        sb.append(", O=").append(params.get("O").defaultTo("None").asString().replaceAll(",", "\\\\,"));
409        sb.append(", L=").append(params.get("L").defaultTo("None").asString().replaceAll(",", "\\\\,"));
410        sb.append(", ST=").append(params.get("ST").defaultTo("None").asString().replaceAll(",", "\\\\,"));
411        sb.append(", C=").append(params.get("C").defaultTo("None").asString().replaceAll(",", "\\\\,"));
412
413        // Create the principle subject name
414        X509Principal subjectName = new X509Principal(sb.toString());
415        
416        //store.getStore().
417        
418        // Generate the key pair
419        KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance(algorithm);  
420        keyPairGenerator.initialize(keySize); 
421        KeyPair keyPair = keyPairGenerator.generateKeyPair();
422        PublicKey publicKey = keyPair.getPublic();
423        PrivateKey privateKey = keyPair.getPrivate();
424        
425        // Generate the certificate request
426        PKCS10CertificationRequest cr = new PKCS10CertificationRequest(signatureAlgorithm, subjectName, publicKey, null, privateKey);
427        
428        // Store the private key to use when the signed cert is return and updated
429        logger.debug("Storing private key with alias {}", alias);
430        storeKeyPair(alias, keyPair);
431        
432        return Pair.of(cr, privateKey);
433    }

435    /**
436     * Stores a KeyPair (associated with a CSR request on the specified alias) in the repository.
437     * 
438     * @param alias the alias from the CSR
439     * @param keyPair the KeyPair object
440     * @throws JsonResourceException
441     */
442    protected void storeKeyPair(String alias, KeyPair keyPair) throws ResourceException {
443        try {
444            JsonValue keyPairValue = new JsonValue(new HashMap<String, Object>());
445            keyPairValue.put("value" , toPem(keyPair));
446            JsonValue encrypted = getCryptoService().encrypt(keyPairValue, cryptoCipher, cryptoAlias);
447            JsonValue keyMap = new JsonValue(new HashMap<String, Object>());
448            keyMap.put("keyPair", encrypted);
449            storeInRepo(KEYS_CONTAINER, alias, keyMap);
450        } catch (Exception e) {
451            throw ResourceException.getException(ResourceException.INTERNAL_ERROR, e.getMessage(), e);
452        }
453        
454    }

467    /**
468     * Stores an object in the repository
469     * @param id the object's id
470     * @param value the value of the object to store
471     * @throws JsonResourceException
472     */
473    protected void storeInRepo(String container, String id, JsonValue value) throws ResourceException {
474        Resource oldResource;
475        try {
476            oldResource = repoService.read(Requests.newReadRequest(container, id));
477        } catch (NotFoundException e) {
478            logger.debug("creating object " + id);
479            repoService.create(Requests.newCreateRequest(container, id, value));
480            return;
481        }
482        UpdateRequest updateRequest = Requests.newUpdateRequest(container, id, value);
483        updateRequest.setRevision(oldResource.getRevision());
484        repoService.update(updateRequest);
485    }

487    /**
488     * Returns a stored KeyPair (associated with a CSR request on the specified alias) from the repository.
489     * 
490     * @param alias the alias from the CSR
491     * @return the KeyPair
492     * @throws JsonResourceException
493     */
494    protected KeyPair getKeyPair(String alias) throws ResourceException {
495        String id = KEYS_CONTAINER + "/" + alias;
496        Resource keyResource = repoService.read(Requests.newReadRequest(id));
497        if (keyResource.getContent().isNull()) {
498            throw ResourceException.getException(ResourceException.NOT_FOUND, 
499                    "Cannot find stored key for alias " + alias);
500        }
501        try {
502            JsonValue encrypted = keyResource.getContent().get("keyPair");
503            JsonValue keyPairValue = getCryptoService().decrypt(encrypted);
504            return fromPem(keyPairValue.get("value").asString());
505        } catch (Exception e) {
506            throw ResourceException.getException(ResourceException.INTERNAL_ERROR, e.getMessage(), e);
507        }
508    }

510    /**
511     * Verifies that the supplied private key and signed certificate match by signing/verifying some test data.
512     * 
513     * @param privateKey A private key
514     * @param publicKey A public key
515     * @throws JsonResourceException if the verification fails, or an error is encountered.
516     */
517    protected void verify(PrivateKey privateKey, Certificate cert) throws ResourceException {
518        PublicKey publicKey = cert.getPublicKey();
519        byte[] data = { 65, 66, 67, 68, 69, 70, 71, 72, 73, 74 };
520        boolean verified;
521        try {
522            Signature signer = Signature.getInstance(privateKey.getAlgorithm());
523            signer.initSign(privateKey);
524            signer.update(data);
525            byte[] signed = signer.sign();
526            Signature verifier = Signature.getInstance(publicKey.getAlgorithm());
527            verifier.initVerify(publicKey);
528            verifier.update(data);
529            verified = verifier.verify(signed);
530        } catch (Exception e) {
531            throw ResourceException.getException(ResourceException.INTERNAL_ERROR, 
532                    "Error verifying private key and signed certificate", e);
533        }
534        if (!verified) {
535            throw ResourceException.getException(ResourceException.BAD_REQUEST, 
536                    "Private key does not match signed certificate");
537        }
538    }

551    /**
552     * Loads the store from the repository and stores it locally
553     * 
554     * @throws ResourceException
555     */
556    public void loadStoreFromRepo() throws ResourceException {
557        JsonValue keystoreValue = readFromRepo("security/" + resourceName);
558        String keystoreString = keystoreValue.get("storeString").asString();
559        byte [] keystoreBytes = Base64.decode(keystoreString.getBytes());
560        ByteArrayInputStream bais = new ByteArrayInputStream(keystoreBytes);
561        try {
562            KeyStore keystore = null;
563            try {
564                keystore = KeyStore.getInstance(store.getType());     
565                keystore.load(bais, store.getPassword().toCharArray());
566            } finally {
567                bais.close();
568            }
569            store.setStore(keystore);
570        } catch (Exception e) {
571            throw ResourceException.getException(ResourceException.INTERNAL_ERROR, "Error creating keystore from store bytes", e);
572        }
573    }
574    
575    /**
576     * Saves the local store to the repository
577     * 
578     * @throws ResourceException
579     */
580    public void saveStoreToRepo() throws ResourceException {
581        byte [] keystoreBytes = null;
582        FileInputStream fin = null;
583        File file = new File(store.getLocation());
584
585        try {
586            try {
587                fin = new FileInputStream(file);
588                keystoreBytes = new byte[(int) file.length()];
589                fin.read(keystoreBytes);
590            } finally {
591                fin.close();
592            }
593        } catch (Exception e) {
594            throw ResourceException.getException(ResourceException.INTERNAL_ERROR, e.getMessage(), e);
595        }
596        
597        String keystoreString = new String(Base64.encode(keystoreBytes));
598        JsonValue value = new JsonValue(new HashMap<String, Object>());
599        value.add("storeString", keystoreString);
600        storeInRepo("security", resourceName, value);
601    }
