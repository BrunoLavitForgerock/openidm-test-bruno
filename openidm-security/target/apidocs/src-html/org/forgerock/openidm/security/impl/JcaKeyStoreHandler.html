025package org.forgerock.openidm.security.impl;
026
027import org.forgerock.openidm.security.KeyStoreHandler;
028
029import java.io.FileInputStream;
030import java.io.FileOutputStream;
031import java.io.IOException;
032import java.io.InputStream;
033import java.io.OutputStream;
034import java.security.KeyStore;
035import java.security.KeyStoreException;
036import java.security.NoSuchAlgorithmException;
037import java.security.cert.CertificateException;
038
039import javax.net.ssl.SSLContext;
040import javax.net.ssl.TrustManager;
041import javax.net.ssl.TrustManagerFactory;
042
043/**
044 * A NAME does ...
045 *
046 * @author Laszlo Hordos
047 */
048public class JcaKeyStoreHandler implements KeyStoreHandler {
049
050    private String location;
051    private String password;
052    private String type;
053    private KeyStore store;
054
055    public JcaKeyStoreHandler(String type, String location, String password) throws Exception {
056        this.location = location;
057        this.password = password;
058        this.type = type;
059        init();
060    }
061
062    void init() throws IOException, KeyStoreException, CertificateException,
063            NoSuchAlgorithmException {
064        InputStream in = new FileInputStream(location);
065        try {
066            store = KeyStore.getInstance(type);
067            store.load(in, password.toCharArray());
068        } finally {
069            in.close();
070        }
071    }
072
073    @Override
074    public KeyStore getStore() {
075        return store;
076    }
077
078    @Override
079    public void setStore(KeyStore keystore) throws Exception {
080        store = keystore;
081        store();
082    }
083
084    @Override
085    public void store() throws Exception {
086        OutputStream out = new FileOutputStream(location);
087        try {
088            store.store(out, password.toCharArray());
089        } finally {
090            out.close();
091        }
092    }
093
094    @Override
095    public String getPassword() {
096        return password;
097    }
098
099    @Override
100    public String getLocation() {
101        return location;
102    }
103
104    public String getType() {
105        return type;
106    }
107}
