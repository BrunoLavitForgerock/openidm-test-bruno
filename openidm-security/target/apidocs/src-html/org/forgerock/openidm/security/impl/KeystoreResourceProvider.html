058/**
059 * Services requests on a specific keystore
060 *
061 * @author Laszlo Hordos
062 * @author ckienle
063 */
064public class KeystoreResourceProvider extends SecurityResourceProvider implements SingletonResourceProvider {
065
066    /**
067     * Setup logging for the {@link SecurityManager}.
068     */
069    private final static Logger logger = LoggerFactory.getLogger(KeystoreResourceProvider.class);
070
071    public KeystoreResourceProvider(String resourceName, KeyStoreHandler store, KeyStoreManager manager, RepositoryService repoService) {
072        super(resourceName, store, manager, repoService);
073    }
074
075    @Override
076    public void actionInstance(ServerContext context, ActionRequest request, ResultHandler<JsonValue> handler) {
077        try {
078            String alias = request.getContent().get("alias").asString();
079            if (ACTION_GENERATE_CERT.equalsIgnoreCase(request.getAction()) || 
080                    ACTION_GENERATE_CSR.equalsIgnoreCase(request.getAction())) {
081                if (alias == null) {
082                    throw ResourceException.getException(ResourceException.BAD_REQUEST, 
083                            "A valid resource ID must be specified in the request");
084                }
085                String algorithm = request.getContent().get("algorithm").defaultTo(DEFAULT_ALGORITHM).asString();
086                String signatureAlgorithm = request.getContent().get("signatureAlgorithm")
087                        .defaultTo(DEFAULT_SIGNATURE_ALGORITHM).asString();
088                int keySize = request.getContent().get("keySize").defaultTo(DEFAULT_KEY_SIZE).asInteger();
089                JsonValue result = null;
090                if (ACTION_GENERATE_CERT.equalsIgnoreCase(request.getAction())) {
091                    // Generate self-signed certificate
092                    if (store.getStore().containsAlias(alias)) {
093                        handler.handleError(new ConflictException("The resource with ID '" + alias 
094                                + "' could not be created because there is already another resource with the same ID"));
095                    } else {
096                        logger.info("Generating a new self-signed certificate with the alias {}", alias);
097                        String domainName = request.getContent().get("domainName").required().asString();
098                        String validFrom = request.getContent().get("validFrom").asString();
099                        String validTo = request.getContent().get("validTo").asString();
100
101                        // Generate the cert
102                        Pair<X509Certificate, PrivateKey> pair = generateCertificate(domainName, algorithm, 
103                                keySize, signatureAlgorithm, validFrom, validTo);
104                        Certificate cert = pair.getKey();
105                        PrivateKey key = pair.getValue();
106
107                        String password = request.getContent().get("password").defaultTo(
108                                Param.getKeystoreKeyPassword()).asString();
109
110                        // Add it to the store and reload
111                        logger.debug("Adding certificate entry under the alias {}", alias);
112                        store.getStore().setCertificateEntry(alias, cert);
113                        store.store();
114                        manager.reload();
115                        // Save the store to the repo (if clustered)
116                        saveStore();
117
118                        result = returnCertificate(alias, cert);
119                    }
120                } else {
121                    // Generate CSR
122                    Pair<PKCS10CertificationRequest, PrivateKey> csr = generateCSR(alias, algorithm, 
123                            signatureAlgorithm, keySize, request.getContent());
124                    result = returnCertificateRequest(alias, csr.getKey());
125                    if (request.getContent().get("returnPrivateKey").defaultTo(false).asBoolean()) {
126                        result.put("privateKey", getKeyMap(csr.getRight()));
127                    }
128                }
129                handler.handleResult(result);
130            } else {
131                handler.handleError(new BadRequestException("Unsupported action " + request.getAction()));
132            }
133            
134        } catch (Throwable t) {
135            handler.handleError(ResourceUtil.adapt(t));
136        }
137    }
138
139    @Override
140    public void patchInstance(ServerContext context, PatchRequest request,
141            ResultHandler<Resource> handler) {
142        final ResourceException e = new NotSupportedException("Patch operations are not supported");
143        handler.handleError(e);
144    }
145
146    @Override
147    public void readInstance(final ServerContext context, final ReadRequest request,
148            final ResultHandler<Resource> handler) {
149        try {
150            JsonValue content = new JsonValue(new LinkedHashMap<String, Object>(5));
151            content.put("type", store.getStore().getType());
152            content.put("provider", store.getStore().getProvider());
153            Enumeration<String> aliases = store.getStore().aliases();
154            List<String> aliasList = new ArrayList<String>();
155            while (aliases.hasMoreElements()) {
156                aliasList.add(aliases.nextElement());
157            }
158            content.put("aliases", aliasList);
159            handler.handleResult(new Resource(resourceName, null, content));
160        } catch (Throwable t) {
161            handler.handleError(ResourceUtil.adapt(t));
162        }
163    }
164
165    @Override
166    public void updateInstance(ServerContext context, UpdateRequest request,
167            ResultHandler<Resource> handler) {
168        final ResourceException e = new NotSupportedException("Update operations are not supported");
169        handler.handleError(e);
170    }
171}
