075/**
076 * A Security Manager Service which handles operations on the java security
077 * keystore and truststore files.
078 *
079 * @author ckienle
080 */
081@Component(name = SecurityManager.PID, policy = ConfigurationPolicy.IGNORE, metatype = true, 
082        description = "OpenIDM Security Management Service", immediate = true)
083@Service
084@Properties({
085    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
086    @Property(name = Constants.SERVICE_DESCRIPTION, value = "Security Management Service"),
087    @Property(name = ServerConstants.ROUTER_PREFIX, value = "/security/*") })
088public class SecurityManager implements RequestHandler, KeyStoreManager {
089
090    public static final String PID = "org.forgerock.openidm.security";
091
092    /**
093     * Setup logging for the {@link SecurityManager}.
094     */
095    private final static Logger logger = LoggerFactory.getLogger(SecurityManager.class);
096    
097    @Reference
098    protected RepositoryService repoService;
099
100    @Reference
101    private CryptoUpdateService cryptoUpdateService;
102
103    private final Router router = new Router();
104    
105    private KeyStoreHandler trustStoreHandler = null;
106    private KeyStoreHandler keyStoreHandler = null;
107
108    @Activate
109    void activate(ComponentContext compContext) throws Exception {
110        logger.debug("Activating Security Management Service {}", compContext);
111        // Add the Bouncy Castle provider
112        Security.addProvider(new BouncyCastleProvider());
113        
114        String keyStoreType = Param.getKeystoreType();
115        String keyStoreLocation = Param.getKeystoreLocation();
116        String keyStorePassword = Param.getKeystorePassword(false); 
117        
118        String trustStoreType = Param.getTruststoreType();
119        String trustStoreLocation = Param.getTruststoreLocation();
120        String trustStorePassword = Param.getTruststorePassword(false);
121
122        // Set System properties
123        if (System.getProperty("javax.net.ssl.keyStore") == null) {
124            System.setProperty("javax.net.ssl.keyStore", keyStoreLocation);
125            System.setProperty("javax.net.ssl.keyStorePassword", keyStorePassword);
126            System.setProperty("javax.net.ssl.keyStoreType", keyStoreType);
127        }
128        if (System.getProperty("javax.net.ssl.trustStore") == null) {
129            System.setProperty("javax.net.ssl.trustStore", trustStoreLocation);
130            System.setProperty("javax.net.ssl.trustStorePassword", trustStorePassword);
131            System.setProperty("javax.net.ssl.trustStoreType", trustStoreType);
132        }
133        
134        keyStoreHandler = new JcaKeyStoreHandler(keyStoreType, keyStoreLocation, keyStorePassword);
135        KeystoreResourceProvider keystoreProvider = new KeystoreResourceProvider("keystore", keyStoreHandler, this, repoService);
136        EntryResourceProvider keystoreCertProvider = new CertificateResourceProvider("keystore", keyStoreHandler, this, repoService);
137        EntryResourceProvider privateKeyProvider = new PrivateKeyResourceProvider("keystore", keyStoreHandler, this, repoService);
138
139        router.addRoute("/keystore", keystoreProvider);
140        router.addRoute("/keystore/cert", keystoreCertProvider);
141        router.addRoute("/keystore/privatekey", privateKeyProvider);
142
143        trustStoreHandler = new JcaKeyStoreHandler(trustStoreType, trustStoreLocation, trustStorePassword);
144        KeystoreResourceProvider truststoreProvider = new KeystoreResourceProvider("truststore", trustStoreHandler, this, repoService);
145        EntryResourceProvider truststoreCertProvider = new CertificateResourceProvider("truststore", trustStoreHandler, this, repoService);
146
147        router.addRoute("/truststore", truststoreProvider);
148        router.addRoute("/truststore/cert", truststoreCertProvider);
149        
150        String instanceType = IdentityServer.getInstance().getProperty("openidm.instance.type", ClusterUtils.TYPE_STANDALONE);
151        
152        String privateKeyAlias = Param.getProperty("openidm.https.keystore.cert.alias");
153        try {
154            if (instanceType.equals(ClusterUtils.TYPE_CLUSTERED_ADDITIONAL)) {
155                // Load keystore and truststore from the repository
156                keystoreProvider.loadStoreFromRepo();
157                truststoreProvider.loadStoreFromRepo();
158                // Reload the SSL context
159                reload();
160                // Update CryptoService
161                cryptoUpdateService.updateKeySelector(keyStoreHandler.getStore(), keyStorePassword);
162            } else {
163                // Check if the default private key alias exists
164                if (!privateKeyProvider.hasEntry(privateKeyAlias)) {
165                    // Create the default private key
166                    privateKeyProvider.createDefaultEntry(privateKeyAlias);
167                    // Reload the SSL context
168                    reload();
169
170                    try {
171                        Config.updateConfig(null);
172                    } catch (NullPointerException e) {
173                        // TODO Auto-generated catch block
174                        e.printStackTrace();
175                    }
176                }
177                // If this is the first/primary node in a cluster, then save the keystore and truststore to the repository
178                if (instanceType.equals(ClusterUtils.TYPE_CLUSTERED_FIRST)) {
179                    keystoreProvider.saveStoreToRepo();
180                    truststoreProvider.saveStoreToRepo();
181                }
182            }
183        } catch (Exception e) {
184            logger.warn("Error initializing keys", e);
185        }
186    }
187
188    @Deactivate
189    void deactivate(ComponentContext compContext) {
190        logger.debug("Deactivating Security Management Service {}", compContext);
191        router.removeAllRoutes();
192    }
193    
194    // ----- Implementation of KeyStoreManager interface
195    
196    public void reload() throws Exception {
197        TrustManagerFactory tmf = TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
198        tmf.init(trustStoreHandler.getStore());
199        TrustManager [] trustManagers = tmf.getTrustManagers();
200
201        KeyManagerFactory kmf = KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
202        kmf.init(keyStoreHandler.getStore(), keyStoreHandler.getPassword().toCharArray());
203        KeyManager [] keyManagers = kmf.getKeyManagers();
204
205        
206        SSLContext context = SSLContext.getInstance("SSL");
207        context.init(keyManagers, trustManagers, null);
208        SSLContext.setDefault(context);
209    }
