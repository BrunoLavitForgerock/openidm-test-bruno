024package org.forgerock.openidm.metadata;
025
026import java.util.List;
027
028import org.forgerock.json.fluent.JsonPointer;
029import org.forgerock.json.fluent.JsonValue;
030
031/**
032 * Meta data provider interface to describe configuration requirements of a
033 * bundle. Use a meta-data.json file to declare a meta data provider for a
034 * bundle*
035 *
036 * @author aegloff
037 * @author ckienle
038 */
039public interface MetaDataProvider {
040
041    /**
042     * Meta-data describing which configuration properties need to be encrypted
043     * for a given configuration.
044     *
045     * @param pidOrFactory
046     *            the PID of either the managed service; or for factory
047     *            configuration the PID of the Managed Service Factory
048     * @param instanceAlias
049     *            null for plain managed service, or the subname (alias) for the
050     *            managed factory configuration newBuilder
051     * @param config
052     *            the new configuration that is being set. May or may not
053     *            already have encrypted values.
054     *
055     * @return a list of configuration properties (identified by JSON pointers)
056     *         that need to be encrypted if this MetaDataProvider is responsible
057     *         for this configuration. Empty list if none should be encrypted.
058     *         Null if this provider is not responsible for this configuration.
059     * @throws WaitForMetaData
060     *             thrown if this provider knows that the given configuration
061     *             has associated meta-data, but the meta-data is not yet
062     *             available.
063     * @throws NotConfiguration
064     *             throws when the config parameter is not represent a real
065     *             configuration.
066     */
067    List<JsonPointer> getPropertiesToEncrypt(String pidOrFactory, String instanceAlias,
068            JsonValue config) throws WaitForMetaData, NotConfiguration;
069
070    /**
071     * Sets a callback to be used to refresh/update the configuration
072     * requirements/properties.
073     *
074     * @param callback
075     *            a MetaDataProviderCallback implementation
076     */
077    public void setCallback(MetaDataProviderCallback callback);
078}
