090    public static boolean jsonIsNull(JsonValue value) {
091        return (value == null || value.isNull());
092    }
093
094    public static boolean isEncrypted(String value) {
095        boolean encrypted = false;
096        // TODO: delegate the sanity check if String is a candidate for parsing
097        // to the crypto lib
098        boolean candidate =
099                value != null && value.startsWith("{\"$crypto\":{") && value.endsWith("}}");
100        if (candidate) {
101            try {
102                JsonValue jsonValue = parseStringified(value);
103                encrypted = JsonCrypto.isJsonCrypto(jsonValue);
104            } catch (JsonException ex) {
105                encrypted = false; // IF we can't parse the string assume it's
106                // not in an encrypted format we support
107            }
108        }
109        return encrypted;
110    }

131    public static JsonValue parseStringified(String content) {
132        JsonValue jsonValue = null;
133        try {
134            Object parsedValue = OBJECT_MAPPER.readValue(content, Object.class);
135            jsonValue = new JsonValue(parsedValue);
136        } catch (IOException ex) {
137            throw new JsonException("String passed into parsing is not valid JSON", ex);
138        }

158    public static JsonTransformer getPropertyJsonTransformer(final JsonValue properties,
159            boolean allowUnresolved) {
160        if (jsonIsNull(properties)) {
161            return null;
162        }
163        return new PropertyTransformer(properties, allowUnresolved);
164    }
165
166    public static class PropertyTransformer implements JsonTransformer {
167
168        private final PropertyAccessor properties;
169        private final boolean eager;
170
171        PropertyTransformer(final JsonValue properties, boolean allowUnresolved) {
172            this.eager = !allowUnresolved;
173            this.properties = new PropertyAccessor() {
174                @Override
175                @SuppressWarnings("unchecked")
176                public <T> T getProperty(String key, T defaultValue, Class<T> expected) {
177                    JsonPointer pointer = new JsonPointer(key.split("\\."));
178                    try {
179                        JsonValue newValue = properties.get(pointer);
180                        if (null != newValue) {
181                            return (T) newValue.required().expect(expected).getObject();
182                        }
183                    } catch (JsonValueException e) {
184                        logger.trace("Failed to substitute variable {}", key, e);
185                        /*
186                         * Expected if the value is null or the type does not
187                         * match
188                         */
189                    } catch (Throwable t) {
190                        logger.debug("Failed to substitute variable with unexpected error {}", key,
191                                t);
192                    }
193                    if (eager && null == defaultValue) {
194                        StringBuilder sb =
195                                new StringBuilder("Failed to resolve mandatory property: ")
196                                        .append(key);
197                        if (null != expected && !Object.class.equals(expected)) {
198                            sb.append(" expecting ").append(expected.getSimpleName()).append(
199                                    " class");
200                        }
201                        throw new JsonValueException(null, sb.toString());
202                    }
203                    return defaultValue;
204                }
205            };
206        }
207
208        /**
209         * {@inheritDoc}
210         */
211        @Override
212        public void transform(JsonValue value) {
213            if (null != value && value.isString()) {
214                try {
215                    value.setObject(PropertyUtil.substVars(value.asString(), properties,
216                            PropertyUtil.Delimiter.DOLLAR, false));
217                } catch (JsonValueException e) {
218                    throw new JsonValueException(value, e.getMessage());
219                }
220            }
221        }
222    }

224    /**
225     * Indenter, part of formatting Jackson output in pretty print Makes the
226     * number of spaces to use per indent configurable.
227     *
228     */
229    private static class PrettyIndenter implements Indenter {
230        // Default to 4 spaces per level
231        int noOfSpaces = 4;
232
233        final static String SYSTEM_LINE_SEPARATOR;
234        static {
235            String lf = null;
236            try {
237                lf = System.getProperty("line.separator");
238            } catch (Throwable t) {
239            } // access exception?
240            SYSTEM_LINE_SEPARATOR = (lf == null) ? "\n" : lf;
241        }
242
243        final static int SPACE_COUNT = 64;
244        final static char[] SPACES = new char[SPACE_COUNT];
245        static {
246            Arrays.fill(SPACES, ' ');
247        }
248
249        /**
250         * Configure how many spaces to use per indent. Default is 4 spaces.
251         *
252         * @param noOfSpaces
253         */
254        public void setIndentSpaces(int noOfSpaces) {
255            this.noOfSpaces = noOfSpaces;
256        }
257
258        public boolean isInline() {
259            return false;
260        }
261
262        @Override
263        public void writeIndentation(JsonGenerator jg, int level) throws IOException,
264                JsonGenerationException {
265            jg.writeRaw(SYSTEM_LINE_SEPARATOR);
266            level = level * noOfSpaces;
267            while (level > SPACE_COUNT) { // should never happen but...
268                jg.writeRaw(SPACES, 0, SPACE_COUNT);
269                level -= SPACES.length;
270            }
271            jg.writeRaw(SPACES, 0, level);
272        }
273    }
