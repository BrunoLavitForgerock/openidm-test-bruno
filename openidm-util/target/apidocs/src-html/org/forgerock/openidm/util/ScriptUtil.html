045    public static Map<String, Object> getRequestMap(Request request, ServerContext context) {
046        Map<String, Object> requestMap = new HashMap<String, Object>();
047        JsonValue value = new JsonValue(null);
048        String id = request.getResourceName();
049        if (request instanceof ActionRequest) {
050            value = ((ActionRequest)request).getContent();
051            requestMap.put("params", ((ActionRequest)request).getAdditionalParameters());
052            requestMap.put("method", "action");
053            requestMap.put("action", ((ActionRequest)request).getAction());
054        } else if (request instanceof CreateRequest) {
055            CreateRequest createRequest = (CreateRequest)request;
056            value = createRequest.getContent();
057            requestMap.put("method", "create");
058            if (createRequest.getNewResourceId() != null) {
059                id = createRequest.getResourceName() + "/" + createRequest.getNewResourceId();
060            }
061        } else if (request instanceof ReadRequest) {
062            requestMap.put("method", "read");
063        } else if (request instanceof UpdateRequest) {
064            value = ((UpdateRequest)request).getContent();
065            requestMap.put("method", "update");
066        } else if (request instanceof DeleteRequest) {
067            requestMap.put("method", "delete");
068        } else if (request instanceof PatchRequest) {
069            requestMap.put("method", "patch");
070            List<PatchOperation> ops = ((PatchRequest)request).getPatchOperations();
071            JsonValue opsValue = new JsonValue(new ArrayList<Object>());
072            for (PatchOperation op : ops) {
073                opsValue.add(op.toJsonValue().getObject());
074            }
075            value = opsValue;
076        } else if (request instanceof QueryRequest) {
077            final QueryRequest queryRequest = (QueryRequest) request;
078            final Map<String, String> params = queryRequest.getAdditionalParameters();
079            params.put("_queryId", queryRequest.getQueryId());
080            params.put("_queryExpression", queryRequest.getQueryExpression());
081            params.put("_queryFilter", queryRequest.getQueryFilter() != null ? queryRequest.getQueryFilter().toString() : null);
082            requestMap.put("params", params);
083            requestMap.put("method", "query");
084        } else {
085            requestMap.put("method", null);
086        }
087        if (context.containsContext(SecurityContext.class)) {
088            SecurityContext securityContext = context.asContext(SecurityContext.class);
089            Map<String, Object> security = new HashMap<String, Object>();
090            security.putAll(securityContext.getAuthorizationId());
091            security.put("username", securityContext.getAuthenticationId());
092            requestMap.put("security", security);
093        }
094        
095        if (isFromHttp(context)) {
096            HttpContext httpContext = context.asContext(HttpContext.class);
097            requestMap.put("headers", httpContext.getHeaders());
098            requestMap.put("fromHttp", "true");
099            requestMap.put("params", httpContext.getParameters());
100        } else {
101            requestMap.put("fromHttp", "false");
102        }
103        requestMap.put("external", String.valueOf(ContextUtil.isExternal(context)));
104        requestMap.put("type", request.getRequestType().toString());
105        requestMap.put("value", value.getObject());
106        requestMap.put("id", id);
107        
108        return requestMap;
109    }
110
111    private static boolean isFromHttp(ServerContext context) {
112        Context c = context.getParent();
113        if (c != null && c.getParent() != null && HttpContext.class.isAssignableFrom(c.getParent().getClass())) {
114            return true;
115        }
116        return false;
117    }
118
119    public static  LazyMap<String, Object> getLazyContext(final ServerContext context) {
120        return new LazyMap<String, Object>(new Factory<Map<String, Object>>() {
121            @Override
122            public Map<String, Object> newInstance() {
123                if (null != context) {
124                    return context.toJsonValue().required().asMap();
125                }
126                else {
127                    return Collections.emptyMap();
128                }
129            }
130        });
131    }
