056    /**
057     * Expands any interpolation contained within the JsonValue object in-place.
058     *
059     * @param json
060     *            JsonValue to parse for macros
061     */
062    public static void expand(JsonValue json) {
063        Iterator<String> iter = json.keys().iterator();
064        while (iter.hasNext()) {
065            String key = iter.next();
066
067            String expanded = parse(json.get(key));
068            if (expanded != null) {
069                json.put(key, expanded);
070            }
071        }
072    }
073
074    /**
075     * Start the string parsing. If base is not a string, will return null.
076     *
077     * @param base
078     *            base JsonValue object to begin parsing from
079     * @return a string with any interpolation expanded or null if base is not a
080     *         string
081     */
082    private static String parse(JsonValue base) {
083        if (!base.isString()) {
084            return null;
085        }
086
087        return buildString(base.asString());
088    }
089
090    /**
091     * Begins building the string from interpolation and normal string contents.
092     *
093     * @param str
094     *            string to interpolate from
095     * @return a string after interpolation
096     */
097    public static String buildString(String str) {
098        StringBuilder builder = new StringBuilder();
099
100        List<Integer> possibleLocations = possibleLocations(str);
101        if (possibleLocations.isEmpty()) {
102            return null;
103        }
104
105        List<Integer[]> confirmedLocations = confirmedLocations(str, possibleLocations);
106        if (confirmedLocations.isEmpty()) {
107            return null;
108        }
109
110        int lastEnd = 0;
111        for (Integer[] pair : confirmedLocations) {
112            int start = pair[0];
113            int length = pair[1];
114            int end = start + length;
115            builder.append(str.substring(lastEnd, start));
116            builder.append(interpolate(str.substring(start, end)));
117            lastEnd = pair[0] + pair[1];
118        }
119
120        return builder.toString();
121    }
122
123    /**
124     * Identifies any possible interpolation locations (begins by looking for
125     * "${").
126     *
127     * @param str
128     *            string to look through for interpolation sites
129     * @return list of indices where the interpolation sites begin
130     */
131    private static List<Integer> possibleLocations(String str) {
132        List<Integer> possibleLocations = new ArrayList<Integer>();
133
134        int lastIndex = -1;
135        int index;
136        while ((index = str.indexOf("${", lastIndex + 1)) >= 0) {
137            if (lastIndex == index) {
138                break;
139            }
140            possibleLocations.add(index);
141            lastIndex = index;
142        }
143
144        return possibleLocations;
145    }
146
147    /**
148     * Confirm interpolation sites by looking for a closing brace.
149     *
150     * @param str
151     *            string to confirm interpolation sites from
152     * @param possibleLocations
153     *            list of string indicies indicating possible interpolation
154     *            beginnings
155     * @return list paired integers containing (starting location, length of
156     *         interpolation string)
157     */
158    private static List<Integer[]> confirmedLocations(String str, List<Integer> possibleLocations) {
159        List<Integer[]> confirmedLocations = new ArrayList<Integer[]>();
160
161        Integer[] lastPair = { -1, -1 };
162        for (Integer start : possibleLocations) {
163            int length = 0;
164
165            // Ignore any escaped \${}
166            if (start != 0 && str.charAt(start - 1) == '\\') {
167                continue;
168            }
169
170            // Determine the length and existence of a ${} block
171            boolean found = false;
172            for (int i = start; i < str.length(); i++) {
173                length += 1;
174                if (str.charAt(i) == '}') {
175                    found = true;
176                    break;
177                }
178            }
179
180            // Don't add overlapping pairs -- this will keep "${ ${hi} }" from
181            // being interpolated
182            // technically it will wind up "${ ${hi}"
183            if ((lastPair[0] + lastPair[1] < start) && found) {
184                Integer[] pair = { start, length };
185                confirmedLocations.add(pair);
186            }
187        }
188
189        return confirmedLocations;
190    }

192    /**
193     * Interpolates the macros contained in the interpolation braces.
194     *
195     * <b>NOTE:</b> for ease of tokenization, this expects each token to have a
196     * space between each component <b><i>e.g.</i></b> "Time.now + 1d" rather
197     * than "Time.now+1d" <br>
198     * <br>
199     * <b>TODO:</b> Proper tokenizing
200     *
201     * @param str
202     *            interpolation string
203     * @return interpolated string
204     */
205    private static String interpolate(String str) {
206        String toInterpolate = str.substring(2, str.length() - 1); // Strip ${
207                                                                   // and }
208        List<String> tokens = Arrays.asList(toInterpolate.split(" "));
209
210        StringBuilder builder = new StringBuilder();
211        Iterator<String> iter = tokens.iterator();
212        while (iter.hasNext()) {
213            String token = iter.next();
214
215            if (token.equals("Time.now")) {
216                builder.append(handleTime(tokens, iter));
217            } else {
218                logger.warn("Unrecognized token: {}", token);
219                builder.append(token);
220            }
221        }
222
223        return builder.toString();
224    }
225
226    /**
227     * Handles the Time.now macro
228     *
229     * @param tokens
230     *            list of tokens
231     * @param iter
232     *            iterator used to iterate over the list of tokens
233     * @return string containing the interpolated time token
234     */
235    private static String handleTime(List<String> tokens, Iterator<String> iter) {
236        DateTime dt = new DateTime();
237
238        // Add some amount
239        if (iter.hasNext()) {
240            String operationToken = iter.next();
241            if (operationToken.equals("+") || operationToken.equals("-")) {
242                if (iter.hasNext()) {
243                    String quantityToken = iter.next(); // Get the magnitude to
244                                                        // add or subtract
245
246                    ReadablePeriod period = getTimePeriod(quantityToken);
247
248                    if (operationToken.equals("-")) {
249                        dt = dt.minus(period);
250                    } else {
251                        dt = dt.plus(period);
252                    }
253                } else {
254                    logger.warn("Token '{}' not followed by a quantity", operationToken);
255                }
256            } else {
257                logger.warn("Invalid token '{}', must be operator '+' or '-'", operationToken);
258            }
259        }
260
261        return DATE_UTIL.formatDateTime(dt);
262    }
263
264    /**
265     * Defines the magnitudes that can be added to the timestamp
266     *
267     * @param token
268     *            token of form "[number][magnitude]" (ex. "1d")
269     * @return integer indicating the magnitude of the date for the calendar
270     *         system
271     */
272    public static ReadablePeriod getTimePeriod(String token) {
273        String valString = token.substring(0, token.length() - 1);
274        int value = Integer.parseInt(valString);
275        char mag = token.charAt(token.length() - 1);
276
277        ReadablePeriod period;
278
279        switch (mag) {
280        case 's':
281            period = Seconds.seconds(value);
282            break;
283        case 'm':
284            period = Minutes.minutes(value);
285            break;
286        case 'h':
287            period = Hours.hours(value);
288            break;
289        case 'd':
290            period = Days.days(value);
291            break;
292        case 'M':
293            period = Months.months(value);
294            break;
295        case 'y':
296            period = Years.years(value);
297            break;
298        default:
299            logger.warn("Invalid date magnitude: {}. Defaulting to seconds.", mag);
300            period = Seconds.seconds(value);
301            break;
302        }
303
304        return period;
305    }
