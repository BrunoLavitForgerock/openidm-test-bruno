064    /**
065     * Retrieve the {@code UriTemplateVariables} from the context.
066     * <p/>
067     *
068     * @param context
069     *
070     * @return an unmodifiableMap or null if the {@code context} does not
071     *         contains {@link RouterContext}
072     */
073    public static Map<String, String> getUriTemplateVariables(Context context) {
074        RouterContext routerContext =
075                context.containsContext(RouterContext.class) ? context
076                        .asContext(RouterContext.class) : null;
077        if (null != routerContext) {
078            return Collections.unmodifiableMap(routerContext.getUriTemplateVariables());
079        }
080        return null;
081    }

083    public static boolean applyPatchOperations(final List<PatchOperation> operations,
084            final JsonValue newContent) throws ResourceException {
085        boolean isModified = false;
086        if (null != operations) {
087            for (final PatchOperation operation : operations) {
088                try {
089                    if (operation.isAdd()) {
090                        newContent.putPermissive(operation.getField(), operation.getValue()
091                                .getObject());
092                    } else if (operation.isRemove()) {
093                        if (operation.getValue().isNull()) {
094                            // Remove entire value.
095                            newContent.remove(operation.getField());
096                        } else {
097                            // Find matching value(s) and remove (assumes
098                            // reference to array).
099                            final JsonValue value = newContent.get(operation.getField());
100                            if (value != null) {
101                                if (value.isList()) {
102                                    final Object valueToBeRemoved =
103                                            operation.getValue().getObject();
104                                    final Iterator<Object> iterator = value.asList().iterator();
105                                    while (iterator.hasNext()) {
106                                        if (valueToBeRemoved.equals(iterator.next())) {
107                                            iterator.remove();
108                                        }
109                                    }
110                                } else {
111                                    // Single valued field.
112                                    final Object valueToBeRemoved =
113                                            operation.getValue().getObject();
114                                    if (valueToBeRemoved.equals(value.getObject())) {
115                                        newContent.remove(operation.getField());
116                                    }
117                                }
118                            }
119                        }
120                    } else if (operation.isReplace()) {
121                        newContent.remove(operation.getField());
122                        if (!operation.getValue().isNull()) {
123                            newContent.putPermissive(operation.getField(), operation.getValue()
124                                    .getObject());
125                        }
126                    } else if (operation.isIncrement()) {
127                        final JsonValue value = newContent.get(operation.getField());
128                        final Number amount = operation.getValue().asNumber();
129                        if (value == null) {
130                            throw new BadRequestException("The field '" + operation.getField()
131                                    + "' does not exist");
132                        } else if (value.isList()) {
133                            final List<Object> elements = value.asList();
134                            for (int i = 0; i < elements.size(); i++) {
135                                elements.set(i, increment(operation, elements.get(i), amount));
136                            }
137                        } else {
138                            newContent.put(operation.getField(), increment(operation, value
139                                    .getObject(), amount));
140                        }
141                    }
142                    isModified = true;
143                } catch (final JsonValueException e) {
144                    throw new ConflictException("The field '" + operation.getField()
145                            + "' does not exist");
146                }
147            }
148        }
149        return isModified;
150    }
151
152    private static Object increment(final PatchOperation operation, final Object object,
153            final Number amount) throws BadRequestException {
154        if (object instanceof Long) {
155            return ((Long) object) + amount.longValue();
156        } else if (object instanceof Integer) {
157            return ((Integer) object) + amount.intValue();
158        } else if (object instanceof Float) {
159            return ((Float) object) + amount.floatValue();
160        } else if (object instanceof Double) {
161            return ((Double) object) + amount.doubleValue();
162        } else {
163            throw new BadRequestException("The field '" + operation.getField()
164                    + "' is not a number");
165        }
166    }

168    /**
169     * Adapts a {@code Throwable} to a {@code ResourceException}. If the
170     * {@code Throwable} is an JSON {@code JsonValueException} then an
171     * appropriate {@code ResourceException} is returned, otherwise an
172     * {@code InternalServerErrorException} is returned.
173     *
174     * @param t
175     *            The {@code Throwable} to be converted.
176     * @return The equivalent resource exception.
177     */
178    public static ResourceException adapt(final Throwable t) {
179        int resourceResultCode;
180        try {
181            throw t;
182        } catch (final ResourceException e) {
183            return e;
184        } catch (final JsonValueException e) {
185            resourceResultCode = ResourceException.BAD_REQUEST;
186        } catch (final Throwable tmp) {
187            resourceResultCode = ResourceException.INTERNAL_ERROR;
188        }
189        return ResourceException.getException(resourceResultCode, t.getMessage(), t);
190    }

192    public static ResourceException notSupported(final Request request) {
193        return new NotSupportedException(ResourceMessages.ERR_OPERATION_NOT_SUPPORTED_EXPECTATION
194                .get(request.getRequestType().name()).toString());
195    }
