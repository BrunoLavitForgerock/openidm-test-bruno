016package org.forgerock.openidm.util;
017
018import java.util.Date;
019
020import org.forgerock.openidm.core.ServerConstants;
021import org.joda.time.Chronology;
022import org.joda.time.DateTime;
023import org.joda.time.DateTimeZone;
024import org.joda.time.chrono.ISOChronology;
025import org.joda.time.format.DateTimeFormatter;
026import org.joda.time.format.ISODateTimeFormat;
027
028/**
029 * Manages timestamp strings in ISO8601 format
030 * <p>
031 * <b>Example:</b> <br>
032 * 2011-09-09T14:58:17.6+02:00 <br>
033 * 2011-09-09T14:58:17.65+02:00 <br>
034 * 2011-09-09T14:58:17.654+02:00 <br>
035 * 2011-09-09T14:58:17.654Z
036 *
037 * @author $author$
038 * @version $Revision$ $Date$
039 */
040public final class DateUtil {
041    private Chronology chrono;
042
043    /**
044     * Fetches a DateUtil that is set in the default timezone.
045     *
046     * @return DateUtil set in the default timezone
047     */
048    public static DateUtil getDateUtil() {
049        return new DateUtil();
050    }
051
052    /**
053     * Returns a DateUtil using a specified timezone.
054     *
055     * @param zone
056     *            string representation of a timezone i.e. "UTC" or "Asia/Tokyo"
057     * @return DateUtil set with the supplied timezone
058     */
059    public static DateUtil getDateUtil(String zone) {
060        return new DateUtil(zone);
061    }
062
063    /**
064     * Creates a DateUtil using a specified timezone.
065     *
066     * @param zone
067     *            DateTimeZone object
068     * @return DateUtil set with the supplied timezone
069     */
070    public static DateUtil getDateUtil(DateTimeZone zone) {
071        return new DateUtil(zone);
072    }
073
074    /**
075     * Creates a DateUtil using the default timezone and generates ISO8601
076     * timestamps.
077     */
078    private DateUtil() {
079        this(DateTimeZone.getDefault());
080    }
081
082    /**
083     * Creates a DateUtil using a specified timezone and generates ISO8601
084     * timestamps.
085     *
086     * @param zone
087     *            string representation of a timezone. i.e. "UTC" or
088     *            "Asia/Tokyo"
089     */
090    private DateUtil(String zone) {
091        this(DateTimeZone.forID(zone));
092    }
093
094    /**
095     * Creates a DateUtil using a specified timezone and generates ISO8601
096     * timestamps.
097     *
098     * @param zone
099     *            timezone object
100     */
101    private DateUtil(DateTimeZone zone) {
102        chrono = ISOChronology.getInstance(zone);
103    }
104
105    /**
106     * Generate a formatted timestamp for the current time.
107     *
108     * @return String containing a timestamp
109     */
110    public String now() {
111        return new DateTime(chrono).toString();
112    }
113
114    public DateTime currentDateTime() {
115        return new DateTime(chrono);
116    }
117
118    /**
119     * Formats a given DateTime into a timestamp.
120     *
121     * @param date
122     *            DateTime object to convert
123     * @return String containing the formatted timestamp
124     */
125    public String formatDateTime(DateTime date) {
126        return date.withChronology(chrono).toString();
127    }
128
129    /**
130     * Formats a given date into a timestamp.
131     *
132     * @param date
133     *            date object to convert
134     * @return String containing the formatted timestamp
135     */
136    public String formatDateTime(Date date) {
137        DateTime dt = new DateTime(date, chrono);
138        return dt.toString();
139    }
140
141    /**
142     * Parses an ISO8601 compliant timestamp into a DateTime object.
143     *
144     * @param timestamp
145     *            timestamp to parse
146     * @return DateTime using the zone and chronology indicated by the timestamp
147     */
148    public DateTime parseTimestamp(String timestamp) {
149        DateTimeFormatter parser = ISODateTimeFormat.dateTime();
150        return parser.withOffsetParsed().parseDateTime(timestamp);
151    }
152
153    /**
154     * Parses an ISO8601 compliant timestamp into a DateTime object.
155     *
156     * Checks the length of the timestamp and returns null if the string is not
157     * ISO8601 compliant Accepted formats:
158     *
159     * <pre>
160     * yyyy-MM-ddTHH:mm:ss.SSSZ,
161     * yyyy-MM-ddTHH:mm:ss.SSS+00,
162     * yyyy-MM-ddTHH:mm:ss.SSS+00:00
163     * </pre>
164     *
165     * @param timestamp
166     *            timestamp to parse
167     * @return
168     */
169    public DateTime parseIfDate(String timestamp) {
170        DateTime d = null;
171        if (timestamp.length() > 23 && timestamp.length() < 30) {
172            try {
173                d = parseTimestamp(timestamp);
174            } catch (IllegalArgumentException e) {
175                /* ignore */
176            }
177        }
178        return d;
179    }
180
181    /**
182     * return the number of days between the two dates.
183     *
184     * @param start
185     *            Start date
186     * @param end
187     *            End date
188     * @param includeDay
189     *            include both Days (increase the result with one)
190     * @return number of days
191     */
192    public static int getDateDifferenceInDays(Date start, Date end, Boolean includeDay) {
193        Integer result = null;
194        if (start != null && end != null) {
195            Long l = 86400000L;
196            Long r = (end.getTime() - start.getTime()) / l;
197            result = r.intValue();
198            if (includeDay) {
199                result++;
200            }
201        }
202        return result;
203    }
204
205    public static void main(String[] args) {
206        System.out.println(new DateUtil(ServerConstants.TIME_ZONE_UTC).now());
207    }
208}
