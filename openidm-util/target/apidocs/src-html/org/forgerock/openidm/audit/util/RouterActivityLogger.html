025package org.forgerock.openidm.audit.util;
026
027import java.util.HashMap;
028import java.util.Map;
029
030import org.forgerock.json.fluent.JsonValue;
031import org.forgerock.json.resource.ConnectionFactory;
032import org.forgerock.json.resource.Context;
033import org.forgerock.json.resource.RequestType;
034import org.forgerock.json.resource.Requests;
035import org.forgerock.json.resource.Resource;
036import org.forgerock.json.resource.ResourceException;
037import org.forgerock.json.resource.RootContext;
038import org.forgerock.json.resource.SecurityContext;
039import org.forgerock.json.resource.ServerContext;
040import org.forgerock.openidm.core.IdentityServer;
041import org.forgerock.openidm.util.DateUtil;
042import org.forgerock.openidm.util.JsonUtil;
043import org.slf4j.Logger;
044import org.slf4j.LoggerFactory;
045
046/**
047 * Creates an audit activity log message using the router.
048 */
049public class RouterActivityLogger implements ActivityLogger {
050
051    /**
052     * Setup logging for the {@link ActivityLogger}.
053     */
054    final static Logger logger = LoggerFactory.getLogger(RouterActivityLogger.class);
055
056    private final ConnectionFactory connectionFactory;
057    private final boolean suspendException;
058    private final DateUtil dateUtil;
059
060    /**
061     * Creates an AuditLogger to create activity messages on the router.
062     *
063     * @param connectionFactory The {@link ConnectionFactory} to use.
064     */
065    public RouterActivityLogger(ConnectionFactory connectionFactory) {
066        this(connectionFactory,
067                "suspend".equals(IdentityServer.getInstance().getProperty(ActivityLogger.class.getName().toLowerCase())));
068    }
069
070    /**
071     * Creates an AuditLogger to create activity messages on the router.
072     *
073     * @param connectionFactory The {@link ConnectionFactory} to use.
074     * @param suspendException whether to throw Exceptions on failure to log or not
075     */
076    public RouterActivityLogger(ConnectionFactory connectionFactory, boolean suspendException) {
077        this.connectionFactory = connectionFactory;
078        this.suspendException = suspendException;
079        // TODO Allow for configured dateUtil
080        dateUtil = DateUtil.getDateUtil("UTC");
081    }
082
083    private String getRequester(Context context) {
084        SecurityContext securityContext = context.asContext(SecurityContext.class);
085        return securityContext != null
086                ? securityContext.getAuthenticationId()
087                : null;
088    }
089
090    /**
091     * {@inheritDoc}
092     */
093    @Override
094    public void log(ServerContext context, RequestType requestType, String message, String objectId,
095                    JsonValue before, JsonValue after, Status status) throws ResourceException {
096        if (requestType == null) {
097            throw new NullPointerException("Request can not be null when audit.");
098        }
099        // TODO: convert to flyweight?
100        try {
101            Map<String, Object> activity = buildLog(context, requestType, message, objectId, before, after, status);
102            connectionFactory.getConnection().create(new ServerContext(context),
103                    Requests.newCreateRequest("audit/activity", new JsonValue(activity)));
104        } catch (ResourceException ex) {
105            // TODO: should this stop the activity itself?
106            if (suspendException) {
107                // log on exception if we're suspending the exception-propagation
108                logger.warn("Failed to write activity log {}", ex);
109            } else {
110                // let the caller handle/log the exception
111                throw ex;
112            }
113        }
114    }
115
116    private Map<String, Object> buildLog(Context context, RequestType requestType, String message,
117            String objectId, JsonValue before, JsonValue after, Status status) {
118        String rev = null;
119        if (after != null && after.get(Resource.FIELD_CONTENT_REVISION).isString()) {
120            rev = after.get(Resource.FIELD_CONTENT_REVISION).asString();
121        } else if (before != null && before.get(Resource.FIELD_CONTENT_REVISION).isString()) {
122            rev = before.get(Resource.FIELD_CONTENT_REVISION).asString();
123        }
124
125        String method = requestType.name().toLowerCase();
126
127        // TODO: make configurable
128        if (method != null
129                && (RequestType.READ.equals(requestType) || RequestType.QUERY.equals(requestType))) {
130            before = null;
131            after = null;
132        }
133
134        Context root = context.asContext(RootContext.class);
135        Context parent = context.getParent();
136
137        Map<String, Object> activity = new HashMap<String, Object>();
138        activity.put(TIMESTAMP, dateUtil.now());
139        activity.put(ACTION, method);
140        activity.put(MESSAGE, message);
141        activity.put(OBJECT_ID, objectId);
142        activity.put(REVISION, rev);
143        activity.put(ACTIVITY_ID, context.getId());
144        activity.put(ROOT_ACTION_ID, root.getId());
145        activity.put(PARENT_ACTION_ID, parent.getId());
146        activity.put(REQUESTER, getRequester(context));
147        activity.put(BEFORE, JsonUtil.jsonIsNull(before) ? null : before.getWrappedObject());
148        activity.put(AFTER, JsonUtil.jsonIsNull(after) ? null : after.getWrappedObject());
149        activity.put(STATUS, status == null ? null : status.toString());
150
151        return activity;
152    }
153}
