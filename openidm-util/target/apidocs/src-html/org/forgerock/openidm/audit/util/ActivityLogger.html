025package org.forgerock.openidm.audit.util;
026
027import org.forgerock.json.fluent.JsonValue;
028import org.forgerock.json.resource.RequestType;
029import org.forgerock.json.resource.ResourceException;
030import org.forgerock.json.resource.ServerContext;
031
032/**
033 * Create an audit activity log for a request result.
034 *
035 * @author brmiller
036 */
037public interface ActivityLogger {
038
039    /** the activity log timestamp field */
040    public static final String TIMESTAMP = "timestamp";
041    /** the activity log action field */
042    public static final String ACTION = "action";
043    /** the activity log message field */
044    public static final String MESSAGE = "message";
045    /** the activity log object id field */
046    public static final String OBJECT_ID = "objectId";
047    /** the activity log revision field */
048    public static final String REVISION = "rev";
049    /** the activity log activity id field */
050    public static final String ACTIVITY_ID = "activityId";
051    /** the activity log root action id field */
052    public static final String ROOT_ACTION_ID = "rootActionId";
053    /** the activity log parent action id field */
054    public static final String PARENT_ACTION_ID = "parentActionId";
055    /** the activity log requester field */
056    public static final String REQUESTER = "requester";
057    /** the activity log object before field */
058    public static final String BEFORE = "before";
059    /** the activity log object after field */
060    public static final String AFTER = "after";
061    /** the activity log status field */
062    public static final String STATUS = "status";
063    /** the activity log changed fields field */
064    public static final String CHANGED_FIELDS = "changedFields";
065    /** the activity log password changed field */
066    public static final String PASSWORD_CHANGED = "passwordChanged";
067
068    /**
069     * Write an activity audit log.
070     *
071     * @param context the context of the request to log
072     * @param requestType the type of request to log
073     * @param message
074     * @param objectId the resourceId being operated on
075     * @param before the object value "before" the request
076     * @param after the object value "after" the request
077     * @param status the status of the operation
078     * @throws ResourceException on failure to write to audit log unless suppressed.
079     */
080    void log(ServerContext context, RequestType requestType, String message, String objectId,
081             JsonValue before, JsonValue after, Status status) throws ResourceException;
082}
