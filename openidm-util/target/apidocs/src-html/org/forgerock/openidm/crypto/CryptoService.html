025package org.forgerock.openidm.crypto;
026
027import java.util.List;
028
029import org.forgerock.json.crypto.JsonCryptoException;
030import org.forgerock.json.crypto.JsonEncryptor;
031import org.forgerock.json.fluent.JsonException;
032import org.forgerock.json.fluent.JsonTransformer;
033import org.forgerock.json.fluent.JsonValue;
034
035/**
036 * Provides encryption and decryption services to OpenIDM components.
037 *
038 * @author Paul C. Bryan
039 * @author aegloff
040 */
041public interface CryptoService {
042
043    /**
044     * TODO: Description.
045     *
046     * @param cipher
047     *            the cipher with which to encrypt the value.
048     * @param alias
049     *            the key alias in the key store with which to encrypt the
050     *            value.
051     * @return TODO.
052     * @throws JsonCryptoException
053     *             TODO.
054     */
055    JsonEncryptor getEncryptor(String cipher, String alias) throws JsonCryptoException;
056
057    /**
058     * Returns a list of decryption transformers that can decrypt JSON values.
059     *
060     * @return a list of transformers that can decrypt JSON values.
061     */
062    List<JsonTransformer> getDecryptionTransformers();
063
064    /**
065     * Encrypts a JSON value.
066     *
067     * @param value
068     *            the JSON value to be encrypted.
069     * @param cipher
070     *            the cipher with which to encrypt the value.
071     * @param alias
072     *            the key alias in the key store with which to encrypt the
073     *            value.
074     * @return a copy of the value, encrypted with the specified cipher and key.
075     * @throws JsonCryptoException
076     *             if and invalid cipher and/or alias is provided.
077     * @throws JsonException
078     *             if an exception occurred encrypting the value.
079     */
080    JsonValue encrypt(JsonValue value, String cipher, String alias) throws JsonCryptoException,
081            JsonException;
082
083    /**
084     * Decrypts a JSON value and all of its children.
085     *
086     * @param value
087     *            the JSON value to be decrypted.
088     * @return a deep copy of the value, with all values decrypted.
089     * @throws JsonException
090     *             if an exception occurred decrypting the value.
091     */
092    JsonValue decrypt(JsonValue value);
093
094    /**
095     * Decrypts a String if in a format supported by this service.
096     *
097     * @param value
098     *            the Stringified value to be decrypted.
099     * @return The decrypted structure value, with all values decrypted.
100     * @throws JsonException
101     *             if an exception occurred decrypting the value.
102     */
103    JsonValue decrypt(String value);
104
105    /**
106     * Decrypts a JSON value and all of its children if necessary. If not,
107     * returns the original object.
108     *
109     * <p>
110     * Note that if the argument is null, this return a JsonValue wrapping null.
111     *
112     * @param value
113     *            the JSON value to be decrypted.
114     * @return The structure value, decrypted if necessary.
115     * @throws JsonException
116     *             if an exception occurred decrypting the value.
117     */
118    JsonValue decryptIfNecessary(JsonValue value);
119
120    /**
121     * Decrypts a String if in a format supported by this service and is
122     * decryption is necessary. If decryption is unnecessary, it returns the
123     * structure value.
124     *
125     * <p>
126     * Note that if the argument is null, this will return a JsonValue wrapping
127     * null.
128     *
129     * @param value
130     *            the Stringified value to be decrypted.
131     * @return The structure value, decrypted if necessary.
132     * @throws JsonException
133     *             if an exception occurred decrypting the value.
134     */
135    JsonValue decryptIfNecessary(String value);
136
137    /**
138     * Detects if a JSON value is encrypted in a format supported by this
139     * service.
140     *
141     * @param value
142     *            the JSON value to check.
143     * @return true if encrypted, false otherwise.
144     */
145    boolean isEncrypted(JsonValue value);
146
147    /**
148     * Detects if a String is encrypted in a format supported by this service.
149     *
150     * @param value
151     *            the JSON value to check.
152     * @return true if encrypted, false otherwise.
153     */
154    boolean isEncrypted(String value);
155}
