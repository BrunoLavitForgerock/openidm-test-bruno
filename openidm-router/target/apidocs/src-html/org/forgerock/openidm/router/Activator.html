025package org.forgerock.openidm.router;
026
027import java.util.Dictionary;
028import java.util.Hashtable;
029
030import org.forgerock.json.resource.Connection;
031import org.forgerock.json.resource.ConnectionProvider;
032import org.forgerock.json.resource.PersistenceConfig;
033import org.forgerock.json.resource.RequestHandler;
034import org.forgerock.json.resource.ResourceException;
035import org.forgerock.json.resource.Resources;
036import org.forgerock.openidm.core.ServerConstants;
037import org.osgi.framework.BundleActivator;
038import org.osgi.framework.BundleContext;
039import org.osgi.framework.Constants;
040import org.osgi.framework.ServiceRegistration;
041
042/**
043 * A NAME does ...
044 *
045 * @author Laszlo Hordos
046 */
047// TODO Delete this when openidm-core is migrated
048public class Activator implements BundleActivator {
049
050    private RouterRegistryImpl routerRegistry;
051
052    private ServiceRegistration factoryServiceRegistration = null;
053    private ServiceRegistration<RequestHandler> routerServiceRegistration = null;
054    private ServiceRegistration<PersistenceConfig> persistenceConfigRegistration = null;
055
056    @Override
057    public void start(BundleContext context) throws Exception {
058        routerRegistry = new RouterRegistryImpl(context);
059        Dictionary<String, Object> properties = new Hashtable<String, Object>(5);
060        properties.put(Constants.SERVICE_DESCRIPTION, "Router route group service");
061        properties.put(Constants.SERVICE_VENDOR, ServerConstants.SERVER_VENDOR_NAME);
062        properties.put(Constants.SERVICE_PID, RouterRegistry.class.getName());
063
064        factoryServiceRegistration =
065                context.registerService(RouterRegistry.class.getName(), routerRegistry, properties);
066
067        properties = new Hashtable<String, Object>(5);
068        properties.put(Constants.SERVICE_DESCRIPTION, "Router route group service");
069        properties.put(Constants.SERVICE_VENDOR, ServerConstants.SERVER_VENDOR_NAME);
070        properties.put("org.forgerock.openidm.router", "true");
071
072        routerServiceRegistration =
073                context.registerService(RequestHandler.class, routerRegistry.getInternalRouter(), properties);
074
075        final Connection connection = Resources.newInternalConnection(routerRegistry.getInternalRouter());
076
077        persistenceConfigRegistration =
078                context.registerService(PersistenceConfig.class, PersistenceConfig.builder()
079                        .connectionProvider(new ConnectionProvider() {
080                            @Override
081                            public Connection getConnection(String connectionId)
082                                    throws ResourceException {
083                                return connection;
084                            }
085
086                            @Override
087                            public String getConnectionId(Connection connection)
088                                    throws ResourceException {
089                                return "DEFAULT";
090                            }
091                        }).classLoader(this.getClass().getClassLoader()).build(), properties);
092    }
093
094    @Override
095    public void stop(BundleContext context) throws Exception {
096        if (null != routerRegistry) {
097            routerRegistry.deactivate();
098        }
099        if (null != persistenceConfigRegistration) {
100            persistenceConfigRegistration.unregister();
101            persistenceConfigRegistration = null;
102        }
103        if (null != factoryServiceRegistration) {
104            factoryServiceRegistration.unregister();
105            factoryServiceRegistration = null;
106        }
107        if (null != routerServiceRegistration) {
108            routerServiceRegistration.unregister();
109            routerServiceRegistration = null;
110        }
111    }
112}
