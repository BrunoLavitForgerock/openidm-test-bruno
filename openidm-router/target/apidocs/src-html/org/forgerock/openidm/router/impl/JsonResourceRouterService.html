040import org.forgerock.json.resource.FutureResult;
041import org.forgerock.json.resource.InternalServerContext;

052/**
053 * Provides internal routing for a top-level object set.
054 *
055 * @author Paul C. Bryan
056 * @author aegloff
057 * @author brmiller
058 */
059@Component(name = JsonResourceRouterService.PID, policy = ConfigurationPolicy.OPTIONAL,
060        metatype = true, configurationFactory = false, immediate = true)
061@Service
062@Properties({
063    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
064    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM internal JSON resource router"),
065    @Property(name = ServerConstants.ROUTER_PREFIX, value = "/") })
066public class JsonResourceRouterService implements ConnectionFactory {
067
068    // Public Constants
069    public static final String PID = "org.forgerock.openidm.internal";
070
071    /** Setup logging for the {@link JsonResourceRouterService}. */
072    private final static Logger logger = LoggerFactory.getLogger(JsonResourceRouterService.class);
073
074    /** Event name prefix for monitoring the router */
075    public final static String EVENT_ROUTER_PREFIX = "openidm/internal/router/";
076
077    @Reference(target = "(service.pid=org.forgerock.openidm.router)")
078    private ConnectionFactory connectionFactory = null;
079
080    /** internal connection factory for internal routing */
081    private ConnectionFactory internal = null;
082
083    @Activate
084    void activate(ComponentContext context) {
085        internal = newInternalConnectionFactory(connectionFactory);
086        logger.info("Reconciliation service activated.");
087    }
088
089    @Modified
090    void modified(ComponentContext context) {
091        activate(context);
092        logger.info("Reconciliation service modified.");
093    }
094
095    @Deactivate
096    void deactivate(ComponentContext context) {
097        logger.info("Reconciliation service deactivated.");
098    }
099
100    // ----- Implementation of ConnectionFactory - delegate to this object's internal ConnectionFactory
101    // which decorates the router ConnectionFactory
102
103    @Override
104    public Connection getConnection() throws ResourceException {
105        return internal.getConnection();
106    }
107
108    @Override
109    public FutureResult<Connection> getConnectionAsync(ResultHandler<? super Connection> handler) {
110        return internal.getConnectionAsync(handler);
111    }
112    @Override
113    public void close() {
114        internal.close();
115    }
116
117    private ConnectionFactory newInternalConnectionFactory(final ConnectionFactory connectionFactory) {
118        return new ConnectionFactory() {
119            @Override
120            public void close() {
121                connectionFactory.close();
122            }
123
124            @Override
125            public Connection getConnection() throws ResourceException {
126                return new AbstractConnectionWrapper<Connection>(connectionFactory.getConnection()) {
127                    @Override
128                    protected Context transform(Context context) {
129                        return new InternalServerContext(context);
130                    }
131                };
132            }
133
134            @Override
135            public FutureResult<Connection> getConnectionAsync(final ResultHandler<? super Connection> handler) {
136                try {
137                    final Connection connection = getConnection();
138                    final FutureResult<Connection> future = Resources.newCompletedFutureResult(connection);
139                    if (handler != null) {
140                        handler.handleResult(connection);
141                    }
142                    return future;
143                } catch (ResourceException e) {
144                    throw new RuntimeException("Can't obtain connection", e);
145                }
146            }
147        };
148    }
