025package org.forgerock.openidm.router;
026
027import java.util.Collections;
028import java.util.Dictionary;
029import java.util.HashSet;
030import java.util.Hashtable;
031import java.util.Set;
032
033import org.apache.commons.lang3.StringUtils;
034import org.forgerock.json.resource.CollectionResourceProvider;
035import org.forgerock.json.resource.RequestHandler;
036import org.forgerock.json.resource.Route;
037import org.forgerock.json.resource.Router;
038import org.forgerock.json.resource.RoutingMode;
039import org.forgerock.json.resource.SingletonResourceProvider;
040import org.forgerock.openidm.core.ServerConstants;
041import org.osgi.framework.Constants;
042
043/**
044 * A RouteBuilder does ...
045 *
046 * @author Laszlo Hordos
047 */
048public final class RouteBuilder {
049
050    private final Set<RouteItem> routes;
051
052    private CollectionResourceProvider collection;
053    private SingletonResourceProvider singleton;
054    private RequestHandler handler;
055    private RoutingMode mode = RoutingMode.EQUALS;
056    private String uriTemplate;
057
058    private RouteBuilder() {
059        this.routes = new HashSet<RouteItem>();
060    }
061
062    private RouteBuilder(Set<RouteItem> initial) {
063        this.routes = initial;
064    }
065
066    public static RouteBuilder newBuilder() {
067        return new RouteBuilder();
068    }
069
070    public RouteBuilder withService(Object service) {
071        if (service instanceof CollectionResourceProvider) {
072            return withCollectionResourceProvider((CollectionResourceProvider) service);
073        } else if (service instanceof SingletonResourceProvider) {
074            return withSingletonResourceProvider((SingletonResourceProvider) service);
075        } else if (service instanceof RequestHandler) {
076            return withRequestHandler((RequestHandler) service);
077        }
078        return this;
079    }
080
081    public RouteBuilder withServiceAndTemplate(Object service, String uriTemplate) {
082        return bind(service, uriTemplate);
083    }
084
085    public RouteBuilder withServiceAndTemplate(Object service, String[] uriTemplate) {
086        return bind(service, uriTemplate);
087    }
088
089    public RouteBuilder bind(Object service, Object routerPrefix) {
090        if (null == routerPrefix || null == service) {
091            return this;
092        }
093        if (routerPrefix.getClass().isArray()) {
094            if (routerPrefix.getClass().getComponentType() == String.class) {
095                for (String pattern : (String[]) routerPrefix) {
096                    withService(service).withTemplate(pattern).buildNext();
097                }
098            }
099        } else if (routerPrefix instanceof String) {
100            withService(service).withTemplate((String) routerPrefix).buildNext();
101        }
102        return this;
103    }
104
105    public RouteBuilder withTemplate(String uriTemplate) {
106        if (StringUtils.isNotBlank(uriTemplate)) {
107            if ('*' == uriTemplate.charAt(uriTemplate.length() - 1)) {
108                withModeStartsWith().withTemplate(
109                        uriTemplate.substring(0, uriTemplate.length() - 1));
110            } else {
111                this.uriTemplate = uriTemplate;
112            }
113        }
114        return this;
115    }
116
117    public RouteBuilder verify() {
118        if ((null == collection) && (null == singleton) && (null == handler)) {
119            throw new NullPointerException("Failed because the service is not set");
120        }
121        if (StringUtils.isBlank(uriTemplate)) {
122            throw new NullPointerException("Failed because the uriTemplate is not set");
123        }
124        return this;
125    }
126
127    public RouteBuilder buildNext() {
128        try {
129            verify();
130            routes.add(new RouteItem(collection, singleton, handler, mode, uriTemplate));
131        } catch (NullPointerException e) {
132            /* ignore */
133        }
134        this.collection = null;
135        this.singleton = null;
136        this.handler = null;
137        this.uriTemplate = null;
138        return withModeEquals();
139    }
140
141    public boolean isNotEmpty() {
142        return !routes.isEmpty();
143    }
144
145    public RouteBuilder withModeEquals() {
146        this.mode = RoutingMode.EQUALS;
147        return this;
148    }
149
150    public RouteBuilder withModeStartsWith() {
151        this.mode = RoutingMode.STARTS_WITH;
152        return this;
153    }
154
155    public RouteBuilder seal() {
156        return new RouteBuilder(Collections.unmodifiableSet(buildNext().routes));
157    }
158
159    public Dictionary<String, Object> buildServiceProperties() {
160        if (routes.isEmpty()) {
161            return null;
162        }
163        StringBuilder sb = new StringBuilder("Route group of {");
164        Dictionary<String, Object> properties = new Hashtable<String, Object>(5);
165        properties.put(Constants.SERVICE_VENDOR, ServerConstants.SERVER_VENDOR_NAME);
166        if (routes.size() == 1) {
167            String template = routes.iterator().next().uriTemplate;
168            sb.append(template);
169            properties.put(ServerConstants.ROUTER_PREFIX, template);
170        } else if (routes.size() > 1) {
171            Object[] params = routes.toArray();
172            for (int i = 0; i < params.length; i++) {
173                params[i] = ((RouteItem) params[i]).uriTemplate;
174                if (i > 0) {
175                    sb.append(", ");
176                }
177                sb.append(params[i]);
178            }
179            properties.put(ServerConstants.ROUTER_PREFIX, params);
180        }
181        properties.put(Constants.SERVICE_DESCRIPTION, sb.append("}").toString());
182        return properties;
183    }
184
185    public RouteBuilder withCollectionResourceProvider(CollectionResourceProvider service) {
186        this.collection = service;
187        this.singleton = null;
188        this.handler = null;
189        return this;
190    }
191
192    public RouteBuilder withSingletonResourceProvider(SingletonResourceProvider service) {
193        this.collection = null;
194        this.singleton = service;
195        this.handler = null;
196        return this;
197    }
198
199    public RouteBuilder withRequestHandler(RequestHandler service) {
200        this.collection = null;
201        this.singleton = null;
202        this.handler = service;
203        return this;
204    }
205
206    public Route[] register(Router router) {
207        if (routes.isEmpty()) {
208            return new Route[0];
209        }
210        Set<Route> registered = new HashSet<Route>();
211        for (RouteItem r : routes) {
212            Route o = r.register(router);
213            if (null != o) {
214                registered.add(o);
215            }
216        }
217        return registered.toArray(new Route[registered.size()]);
218    }
219
220    private static class RouteItem {
221        private final CollectionResourceProvider collection;
222        private final SingletonResourceProvider singleton;
223        private final RequestHandler handler;
224        private final RoutingMode mode;
225        private final String uriTemplate;
226
227        private RouteItem(CollectionResourceProvider collection,
228                SingletonResourceProvider singleton, RequestHandler handler, RoutingMode mode,
229                String uriTemplate) {
230            this.collection = collection;
231            this.singleton = singleton;
232            this.handler = handler;
233            this.mode = mode;
234            this.uriTemplate = uriTemplate;
235        }
236
237        private Route register(Router router) {
238            if (null != router) {
239                if (null != collection) {
240                    return router.addRoute(uriTemplate, collection);
241                } else if (null != singleton) {
242                    return router.addRoute(uriTemplate, singleton);
243                } else if (null != handler) {
244                    return router.addRoute(mode, uriTemplate, handler);
245                }
246            }
247            return null;
248        }
249
250        @Override
251        public boolean equals(Object o) {
252            if (this == o) {
253                return true;
254            }
255            if (o == null || getClass() != o.getClass()) {
256                return false;
257            }
258
259            RouteItem routeItem = (RouteItem) o;
260
261            if (collection != null ? !collection.equals(routeItem.collection)
262                    : routeItem.collection != null) {
263                return false;
264            }
265            if (handler != null ? !handler.equals(routeItem.handler) : routeItem.handler != null) {
266                return false;
267            }
268            if (mode != routeItem.mode) {
269                return false;
270            }
271            if (singleton != null ? !singleton.equals(routeItem.singleton)
272                    : routeItem.singleton != null) {
273                return false;
274            }
275            if (!uriTemplate.equals(routeItem.uriTemplate)) {
276                return false;
277            }
278
279            return true;
280        }
281
282        @Override
283        public int hashCode() {
284            int result = collection != null ? collection.hashCode() : 0;
285            result = 31 * result + (singleton != null ? singleton.hashCode() : 0);
286            result = 31 * result + (handler != null ? handler.hashCode() : 0);
287            result = 31 * result + mode.hashCode();
288            result = 31 * result + uriTemplate.hashCode();
289            return result;
290        }
291    }
292}
