024package org.forgerock.openidm.http;
025
026import java.io.IOException;
027import java.io.InputStream;
028import java.net.URL;
029import java.util.ArrayList;
030import java.util.Dictionary;
031import java.util.Enumeration;
032import java.util.Hashtable;
033import java.util.List;
034
035import org.apache.felix.scr.annotations.Activate;
036import org.apache.felix.scr.annotations.Component;
037import org.apache.felix.scr.annotations.ConfigurationPolicy;
038import org.apache.felix.scr.annotations.Deactivate;
039import org.apache.felix.scr.annotations.Reference;
040import org.ops4j.pax.web.service.WebContainer;
041import org.osgi.framework.BundleContext;
042import org.osgi.service.component.ComponentContext;
043import org.osgi.service.http.HttpContext;
044import org.slf4j.Logger;
045import org.slf4j.LoggerFactory;
046
047/**
048 * Http context to share amongst OpenIDM servlets to allow for applying uniform
049 * security handling
050 * 
051 * @author aegloff
052 * @author ckienle
053 */
054@Component(name = "org.forgerock.openidm.http.context", immediate = true,
055        policy = ConfigurationPolicy.IGNORE)
056public final class ContextRegistrator {
057    final static Logger logger = LoggerFactory.getLogger(ContextRegistrator.class);
058    public static final String OPENIDM = "openidm";
059
060    @Reference
061    private WebContainer httpService;
062
063    HttpContext httpContext;
064
065    static ComponentContext context;
066
067    List<SecurityConfigurator> securityConfigurators = new ArrayList<SecurityConfigurator>();
068
069    /**
070     * Allow access for the fragments
071     * 
072     * @return bundle context if activated, null otherwise
073     */
074    public static BundleContext getBundleContext() {
075        if (context != null) {
076            return context.getBundleContext();
077        } else {
078            return null;
079        }
080    }
081
082    @Activate
083    protected void activate(ComponentContext context) throws Exception {
084        ContextRegistrator.context = context;
085        httpContext = httpService.createDefaultHttpContext();
086        Dictionary<String, Object> contextProps = new Hashtable<String, Object>();
087        contextProps.put("openidm.contextid", "shared");
088        // TODO: Consider the HttpContextMapping it allows to configure the path
089        context.getBundleContext().registerService(HttpContext.class.getName(), httpContext,
090                contextProps);
091        logger.debug("Registered OpenIDM shared http context");
092
093        // Apply the pluggable security configurations on the httpContext
094        initSecurityConfigurators();
095        activateSecurityConfigurators(context, httpContext);
096    }
097
098    @Deactivate
099    protected void deactivate(ComponentContext context) {
100        deactivateSecurityConfigurators(context, httpContext);
101    }
102
103    /**
104     * Loads and instantiates the pluggable Security Configurators
105     * 
106     * To allow for pluggability with fragments a simple convention is used to
107     * find security configurator(s); 1. A properties file contains a property
108     * security.configurator.class of a class in the bundle fragment that
109     * implements SecurityConfigurator and has a no-arg constructor 2. The
110     * properties file is name <prefix>securityconfigurator.properties and
111     * placed somewhere in the bundle or attached fragments
112     */
113    private void initSecurityConfigurators() {
114
115        Enumeration<URL> entries =
116                context.getBundleContext().getBundle().findEntries("/",
117                        "*securityconfigurator.properties", true);
118        while (entries != null && entries.hasMoreElements()) {
119            URL entry = entries.nextElement();
120            logger.trace("Handle properties file at {}", entry.getPath());
121
122            InputStream is = null;
123            java.util.Properties props = new java.util.Properties();
124            try {
125                is = entry.openStream();
126                props.load(is);
127            } catch (IOException ex) {
128                logger.warn("Failed to load security extension properties file", ex);
129                try {
130                    is.close();
131                } catch (Exception cex) {
132                    logger.warn("Failure during close of properties file.", cex);
133                }
134            }
135            logger.trace("Loaded {}: {}", entry.getPath(), props);
136            if (props != null) {
137                String clazzName = (String) props.get("security.configurator.class");
138                logger.debug("Initiating security configurator for class: {}", clazzName);
139                SecurityConfigurator configurator = instantiateSecurityConfigurator(clazzName);
140                if (configurator != null) {
141                    securityConfigurators.add(configurator);
142                }
143            }
144        }
145    }
146
147    /**
148     * @param clazzName
149     *            name of SecurityConfigurator to load and instantiate
150     * @return the security configurator instance if it was successfully
151     *         instantiated, null if not. Logs any failures
152     */
153    SecurityConfigurator instantiateSecurityConfigurator(String clazzName) {
154        SecurityConfigurator configurator = null;
155        Class configuratorClazz = null;
156        try {
157            configuratorClazz = context.getBundleContext().getBundle().loadClass(clazzName);
158            logger.debug("Loaded configurator class {}", clazzName);
159        } catch (ClassNotFoundException ex) {
160            logger.debug("Security configurator not present: {}", clazzName);
161        }
162        try {
163            if (configuratorClazz != null) {
164                Object instance = configuratorClazz.newInstance();
165                logger.debug("Instantiated configurator {}", instance);
166                configurator = (SecurityConfigurator) instance;
167            }
168        } catch (Exception ex) {
169            logger.warn("Failed to load security configurator class {}", clazzName, ex);
170        }
171        return configurator;
172    }
173
174    /**
175     * Activate security configurators if present to enable security
176     * 
177     * @param context
178     *            the component context of the main bundle
179     * @param httpContext
180     *            the shared http context to configure
181     */
182    private void activateSecurityConfigurators(ComponentContext context, HttpContext httpContext) {
183        for (SecurityConfigurator configurator : securityConfigurators) {
184            configurator.activate(httpService, httpContext, context);
185            logger.info("Activated security configurator {}", configurator.getClass().getName());
186        }
187    }
188
189    /**
190     * Deactivate security configurators if present to cleanup
191     * 
192     * @param context
193     *            the component context of the main bundle
194     * @param httpContext
195     *            the shared http context to configure
196     */
197    private void deactivateSecurityConfigurators(ComponentContext context, HttpContext httpContext) {
198        for (SecurityConfigurator configurator : securityConfigurators) {
199            configurator.deactivate(httpService, httpContext, context);
200            logger.debug("Deactivated security configurator {}", configurator.getClass().getName());
201        }
202    }
203
204    public String getHttpContextId() {
205        return OPENIDM;
206    }
207}
