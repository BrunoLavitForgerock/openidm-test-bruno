047/**
048 * @author $author$
049 * @version $Revision$ $Date$
050 */
051public class InteractiveObjectSetService implements /* RequestHandler, */ServiceListener {
052
053    private final static Logger TRACE = LoggerFactory.getLogger(InteractiveObjectSetService.class);
054
055    public static final String ROUTER_SERVICE_FILTER = "(service.pid=org.forgerock.openidm.router)";
056
057    private ConnectionProvider router = null;
058
059    private CommandSession session;
060    private final ObjectMapper mapper = new ObjectMapper();
061
062    private final BundleContext context;
063    private final Thread parent;
064
065    public InteractiveObjectSetService(final Thread parent, BundleContext context,
066            CommandSession session) {
067        this.parent = parent;
068        this.context = context;
069        this.session = session;
070        try {
071            ServiceReference[] ref =
072                    context.getServiceReferences(ConnectionProvider.class.getName(),
073                            ROUTER_SERVICE_FILTER);
074            if (null != ref && ref.length > 0) {
075                router = (ConnectionProvider) context.getService(ref[0]);
076            }
077        } catch (InvalidSyntaxException e) {
078        }
079
080    }
081
082    /**
083     * {@inheritDoc}
084     */
085    public JsonValue handle(JsonValue request) throws ResourceException {
086        synchronized (this) {
087            session.getConsole().println("Incoming request:");
088            if (null == request) {
089                session.getConsole().println("null");
090            } else {
091                try {
092                    StringWriter wr = new StringWriter();
093                    mapper.writerWithDefaultPrettyPrinter().writeValue(wr, request.getObject());
094                    session.getConsole().println(wr.toString());
095                } catch (IOException e) {
096                    session.getConsole().append("Input object serialization exception: ").println(
097                            e.getMessage());
098                }
099            }
100            JsonValue response = null;
101            Scanner input = new Scanner(session.getKeyboard());
102            do {
103                switch (printInputHelp(input)) {
104                case 1: {
105                    if (null != router) {
106                        return redirect(request, input);
107                    } else {
108                        session.getConsole().println(
109                                "Router is not available, please select something else.");
110                    }
111                    break;
112                }
113                case 2: {
114                    break;
115                }
116                case 3: {
117                    response = loadFromConsole(input);
118                    break;
119                }
120                case 4: {
121                    return null;
122                }
123                case 5: {
124                    printExceptionHelp(input);
125                }
126                case 6: {
127                    parent.interrupt();
128                    return new JsonValue(new HashMap<String, Object>());
129                }
130                default: {
131                    session.getConsole().println("Your should select something [1..6]");
132                }
133                }
134            } while (null == response);
135            return response;
136        }
137    }
138
139    private ConnectionProvider getRouter() {
140        return router;
141    }
142
143    private int printInputHelp(Scanner input) {
144        session.getConsole().println("Chose one from the following option and type in the number:");
145        session.getConsole().println("1: Redirect to a router call");
146        session.getConsole().println("2: Read response from file");
147        session.getConsole().println("3: Read response from console");
148        session.getConsole().println("4: Return null");
149        session.getConsole().println("5: Throw JsonResourceException");
150        session.getConsole().println("6: Exit and shutdown the service");
151        return input.nextInt();
152    }
153
154    private JsonValue redirect(JsonValue request, Scanner input) throws ResourceException {
155        session.getConsole().append("Current id: ").println(request.get("id").asString());
156        session.getConsole().print("Type in the new id: ");
157        String id = input.next();
158        request.put("id", id);
159        return null;// getRouter().handle(request);
160    }
161
162    private JsonValue readFile(File inputFile) throws Exception {
163        try {
164            if (inputFile.exists()) {
165                Object map = mapper.readValue(inputFile, Map.class);
166                return new JsonValue(map);
167            }
168        } catch (Exception e) {
169            session.getConsole().format("Error reading file: %s. Exception: %s",
170                    inputFile.getAbsolutePath(), e.getMessage());
171        }
172        return new JsonValue(new HashMap());
173    }
174
175    private JsonValue loadFromConsole(Scanner scanner) {
176        session.getConsole().println();
177        session.getConsole().println("> Press ctrl-D to finish input");
178        session.getConsole().println("Start data input:");
179        String input = null;
180        StringBuilder stringBuilder = new StringBuilder();
181        while (scanner.hasNext()) {
182            input = scanner.next();
183            if (null == input) {
184                // control-D pressed
185                break;
186            } else {
187                stringBuilder.append(input);
188            }
189        }
190        try {
191            Object map = mapper.readValue(stringBuilder.toString(), Map.class);
192            return new JsonValue(map);
193        } catch (IOException e) {
194            session.getConsole().append("[Exception] Failed to read input from console. Reason: ")
195                    .println(e);
196            return null;
197        }
198    }
199
200    private void printExceptionHelp(Scanner scanner) throws ResourceException {
201        session.getConsole().println("Throw a JsonResourceException");
202        session.getConsole().print("Type in the numeric code of the exception: ");
203        throw ResourceException.getException(scanner.nextInt(), "Test exception");
204    }
205
206    @Override
207    public void serviceChanged(ServiceEvent event) {
208        ServiceReference sr = event.getServiceReference();
209        switch (event.getType()) {
210        case ServiceEvent.REGISTERED: {
211            router = (ConnectionProvider) context.getService(sr);
212            break;
213        }
214        case ServiceEvent.MODIFIED: {
215            router = (ConnectionProvider) context.getService(sr);
216            break;
217        }
218        case ServiceEvent.UNREGISTERING: {
219            router = null;
220            break;
221        }
222        default:
223            break;
224        }
225    }
