025package org.forgerock.openidm.shell.felixgogo.debug;
026
027import org.apache.felix.service.command.CommandSession;
028import org.apache.felix.service.command.Descriptor;
029import org.forgerock.json.resource.RequestHandler;
030import org.forgerock.openidm.core.ServerConstants;
031import org.osgi.framework.BundleContext;
032import org.osgi.framework.InvalidSyntaxException;
033import org.osgi.framework.ServiceRegistration;
034
035import java.util.Dictionary;
036import java.util.Hashtable;
037
038/**
039 * @author $author$
040 * @version $Revision$ $Date$
041 */
042public class DebugCommands {
043
044    public static final String[] functions = {"mockservice", "help"};
045
046    /**
047     * Bundle Context
048     */
049    private BundleContext context;
050
051
052    public DebugCommands(BundleContext context) {
053        this.context = context;
054    }
055
056
057    @Descriptor("Start a new object service and register into the router service")
058    public void mockservice(CommandSession session, @Descriptor("URL Prefix") String[] params) {
059        try {
060            InteractiveObjectSetService debugRouter = new InteractiveObjectSetService(Thread.currentThread(), context, session);
061            context.addServiceListener(debugRouter, InteractiveObjectSetService.ROUTER_SERVICE_FILTER);
062            Dictionary<String, Object> props = new Hashtable<String, Object>();
063            props.put(ServerConstants.ROUTER_PREFIX, params.length > 0 ? params[0] : "debugrouter");
064            ServiceRegistration srv = context.registerService(RequestHandler.class.getName(), debugRouter, props);
065            boolean run = true;
066            while (run) {
067                try {
068                    Thread.sleep(1000000000);
069                } catch (InterruptedException e) {
070                    run = false;
071                    session.getConsole().println("Interrupt debug server.");
072                }
073            }
074            context.removeServiceListener(debugRouter);
075            srv.unregister();
076            session.getConsole().println("Unregister debug server.");
077        } catch (InvalidSyntaxException e) {
078            //This should never happen
079        }
080    }
081
082    @Descriptor("Display help messages")
083    public String help() {
084        return "mockservice\t Start new service";
085    }
086}
