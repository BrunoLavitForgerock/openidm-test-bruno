082    @Descriptor("Export or import a SecretKeyEntry. The Java Keytool does not allow for exporting or importing SecretKeyEntries.")
083    public void keytool(CommandSession session, @Parameter(names = { "-i", "--import" },
084            presentValue = "true", absentValue = "false") boolean doImport, @Parameter(names = {
085        "-e", "--export" }, presentValue = "true", absentValue = "false") boolean doExport,
086            @Descriptor("key alias") String alias) {
087        if (doImport ^ doExport) {
088            String type = IdentityServer.getInstance().getProperty("openidm.keystore.type", KeyStore.getDefaultType());
089            String provider = IdentityServer.getInstance().getProperty("openidm.keystore.provider");
090            String location = IdentityServer.getInstance().getProperty("openidm.keystore.location");
091
092            try {
093                KeyStore ks = (provider == null || provider.trim().length() == 0 ? KeyStore.getInstance(type) : KeyStore.getInstance(type, provider));
094                if (location != null) {
095                    File configFile = IdentityServer.getFileForInstallPath(location);
096                    if (configFile.exists()) {
097                        FileInputStream in = null;
098                        try {
099                            in = new FileInputStream(configFile);
100                            if (null != in) {
101                                session.getConsole().append("Use KeyStore from: ").println(configFile.getAbsolutePath());
102                                // TODO Don't use the System in OSGi.
103                                char[] passwordArray = System.console().readPassword("Please enter the password: ");
104                                char[] passwordCopy = Arrays.copyOf(passwordArray, passwordArray.length);
105                                Arrays.fill(passwordArray, ' ');
106                                ks.load(in, passwordCopy);
107                                if (null != in) {
108                                    in.close();
109                                    in = null;
110                                }
111                                if (doExport) {
112                                    KeyStore.Entry key = ks.getEntry(alias, new KeyStore.PasswordProtection(passwordCopy));
113                                    if (key instanceof KeyStore.SecretKeyEntry) {
114                                        KeyStore.SecretKeyEntry secretKeyEntry = (KeyStore.SecretKeyEntry) key;
115                                        session.getConsole().append("[OK] ").println(secretKeyEntry);
116                                        StringBuilder sb = new StringBuilder(secretKeyEntry.getSecretKey().getAlgorithm());
117                                        sb.append(":").append(new BigInteger(1, secretKeyEntry.getSecretKey().getEncoded()).toString(16));
118                                        session.getConsole().println(sb);
119                                    } else {
120                                        session.getConsole().println("SecretKeyEntry with this alias is not in KeyStore");
121                                    }
122                                } else if (doImport) {
123                                    if (ks.containsAlias(alias)) {
124                                        session.getConsole().println("KeyStore contains a key with this alias");
125                                    } else {
126                                        session.getConsole().println("Enter the key: ");
127                                        Scanner scanner = new Scanner(session.getKeyboard());
128                                        String[] tokens = scanner.nextLine().split(":");
129                                        if (tokens.length == 2) {
130                                            byte[] encoded = new BigInteger(tokens[1], 16).toByteArray();
131                                            javax.crypto.SecretKey mySecretKey = new SecretKeySpec(encoded, tokens[0]);
132                                            KeyStore.SecretKeyEntry skEntry = new KeyStore.SecretKeyEntry(mySecretKey);
133                                            ks.setEntry(alias, skEntry, new KeyStore.PasswordProtection(passwordCopy));
134                                            FileOutputStream fos = null;
135                                            try {
136                                                fos = new FileOutputStream(configFile);
137                                                ks.store(fos, passwordCopy);
138                                            } finally {
139                                                if (fos != null) {
140                                                    fos.close();
141                                                }
142                                            }
143                                        } else {
144                                            session.getConsole().println("Invalid key input");
145                                        }
146                                    }
147                                }
148                            }
149                        } catch (IOException ioe) {
150                            if (null != in) {
151                                in.close();
152                            }
153                        }
154                    } else {
155                        session.getConsole().append("KeyStore file: ").append(configFile.getAbsolutePath()).println(" does not exists.");
156                    }
157                }
158            } catch (Exception e) {
159                session.getConsole().println(e.getMessage());
160            }
161        } else {
162            session.getConsole().println("Import or Export have to be exclusively defined.");
163        }
164    }

166    @Descriptor("Validates all json configuration files in the configuration (default: /conf) folder.")
167    public void validate(CommandSession session) {
168        File file = IdentityServer.getFileForProjectPath("conf");
169        session.getConsole().println(
170                "...................................................................");
171        if (file.isDirectory()) {
172            session.getConsole().println("[Validating] Load JSON configuration files from:");
173            session.getConsole().append("[Validating] \t").println(file.getAbsolutePath());
174            FileFilter filter = new FileFilter() {
175                public boolean accept(File f) {
176                    return (f.isDirectory()) || (f.getName().endsWith(".json"));
177                }
178            };
179            ObjectMapper mapper = new ObjectMapper();
180            File[] files = file.listFiles(filter);
181            for (File subFile : files) {
182                if (subFile.isDirectory())
183                    continue;
184                // TODO pretty print
185                try {
186                    mapper.readValue(subFile, Object.class);
187                    prettyPrint(session.getConsole(), subFile.getName(), null);
188                } catch (Exception e) {
189                    prettyPrint(session.getConsole(), subFile.getName(), e);
190                }
191            }
192        } else {
193            session.getConsole().append("[Validating] ").append(
194                    "Configuration directory not found at: ").println(file.getAbsolutePath());
195        }
196    }

198    @Descriptor("Encrypt the input string.")
199    public void encrypt(CommandSession session, @Parameter(names = { "-j", "--json" },
200            presentValue = "false", absentValue = "true") boolean isString) {
201        session.getConsole().append("Enter the ").append(isString ? "String" : "Json").println(
202                " value");
203        encrypt(session, isString, loadFromConsole(session));
204    }
205
206    @Descriptor("Encrypt the input string.")
207    public void encrypt(CommandSession session, @Parameter(names = { "-j", "--json" },
208            presentValue = "false", absentValue = "true") boolean isString,
209            @Descriptor("source string to encrypt") String name) {
210        try {
211            CryptoServiceImpl cryptoSvc = (CryptoServiceImpl) CryptoServiceFactory.getInstance();
212            cryptoSvc.activate(null);
213            
214            JsonValue value = new JsonValue(isString ? name : mapper.readValue(name, Object.class));
215            String cipher = ServerConstants.SECURITY_CRYPTOGRAPHY_DEFAULT_CIPHER;
216            String alias = IdentityServer.getInstance().getProperty("openidm.config.crypto.alias", "openidm-config-default");
217            JsonValue secure = cryptoSvc.encrypt(value, cipher, alias);
218                            
219            StringWriter wr = new StringWriter();
220            mapper.writerWithDefaultPrettyPrinter().writeValue(wr, secure.getObject());
221            session.getConsole().println("-----BEGIN ENCRYPTED VALUE-----");
222            session.getConsole().println(wr.toString());
223            session.getConsole().println("------END ENCRYPTED VALUE------");
224        } catch (JsonCryptoException e) {
225            e.printStackTrace();
226        } catch (IOException e) {
227            e.printStackTrace();
228        }
229    }
230
231    private void prettyPrint(PrintStream out, String name, Exception reason) {
232        out.append("[Validating] ").append(name).append(" ").append(
233                DOTTED_PLACEHOLDER.substring(Math.min(name.length(), DOTTED_PLACEHOLDER.length())));
234        if (null == reason) {
235            out.println(" SUCCESS");
236        } else {
237            out.println(" FAILED");
238            out.append("\t[").append(reason.getMessage()).println("]");
239        }
240    }
241
242    private String loadFromConsole(CommandSession session) {
243        session.getConsole().println();
244        session.getConsole().println("> Press ctrl-D to finish input");
245        session.getConsole().println("Start data input:");
246        String input = null;
247        StringBuilder stringBuilder = new StringBuilder();
248        Scanner scanner = new Scanner(session.getKeyboard());
249        while (scanner.hasNext()) {
250            input = scanner.next();
251            if (null == input) {
252                // control-D pressed
253                break;
254            } else {
255                stringBuilder.append(input);
256            }
257        }
258        session.getConsole().println();
259        return stringBuilder.toString();
260    }
