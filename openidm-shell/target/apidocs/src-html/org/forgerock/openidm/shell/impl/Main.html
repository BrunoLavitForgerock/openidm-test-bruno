025package org.forgerock.openidm.shell.impl;
026
027import org.apache.felix.gogo.runtime.CommandProcessorImpl;
028import org.apache.felix.gogo.runtime.threadio.ThreadIOImpl;
029import org.apache.felix.gogo.shell.Console;
030import org.apache.felix.service.command.CommandSession;
031import org.apache.felix.service.command.Converter;
032import org.apache.felix.service.command.Function;
033import org.forgerock.openidm.core.IdentityServer;
034import org.forgerock.openidm.shell.CustomCommandScope;
035
036import java.io.InputStream;
037import java.io.PrintStream;
038import java.lang.reflect.Field;
039import java.lang.reflect.Method;
040import java.util.Map;
041import java.util.ServiceLoader;
042
043/**
044 * @author $author$
045 * @version $Revision$ $Date$
046 */
047public class Main {
048    protected static CommandProcessorImpl processor;
049
050    public static void main(String[] args) throws Exception {
051        initProcessor();
052        CommandSession session = processor.createSession(System.in, System.out, System.err);
053        session.put("prompt", "openidm# ");
054        session.put("_cwd", IdentityServer.getFileForPath("."));
055        // start shell
056        if (args.length == 0) {
057            Thread thread = new Thread(new Console(session), "OpenIDM shell");
058            thread.start();
059        } else {
060            processor.eval(session, args);
061        }
062    }
063
064    private static void initProcessor() {
065        processor = new CommandProcessorImpl(new ThreadIOImpl());
066
067
068        // Setup the variables and commands exposed in an OSGi environment.
069        //processor.addConstant(CONTEXT, context);
070        processor.addCommand("osgi", processor, "addCommand");
071        processor.addCommand("osgi", processor, "removeCommand");
072        processor.addCommand("osgi", processor, "eval");
073        ServiceLoader<CustomCommandScope> ldr = ServiceLoader.load(CustomCommandScope.class);
074        for (CustomCommandScope cmdScope : ldr) {
075            if (null != cmdScope.getScope() && null != cmdScope.getFunctionMap()) {
076                for (Map.Entry<String, String> entry : cmdScope.getFunctionMap().entrySet()) {
077                    Function target = new CommandProxy(cmdScope,
078                            entry.getKey());
079                    processor.addCommand(cmdScope.getScope(), target, entry.getKey());
080                }
081            }
082        }
083
084        ServiceLoader<Converter> cldr = ServiceLoader.load(Converter.class);
085        for (Converter converter : cldr) {
086            processor.addConverter(converter);
087        }
088    }
089
090    public static Method findCommandMethod(Class<? extends CustomCommandScope> commandsProvider, String methodName) {
091        try {
092            return commandsProvider.getMethod(methodName, InputStream.class, PrintStream.class, String[].class);
093        } catch (NoSuchMethodException e) {
094            return null;
095        }
096    }
097}
