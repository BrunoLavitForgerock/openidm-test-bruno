037/**

064    /**
065     * Generate CommandProvider class and newBuilder for this class based on parameters
066     *
067     * @param service  commands service
068     * @param commands commands map (name=help)
069     * @param suffix   unique class suffix
070     * @return generated CommandProvider newBuilder
071     * @throws Exception if something went wrong
072     */
073    public static Object generate(CustomCommandScope service, Map<String, String> commands, String suffix) throws Exception {
074        // generate class with unique name
075        //javassist.CtClass ctClass = POOL.makeClass(AbstractFelixCommandsService.class.getName() + suffix);
076
077        try {
078            ClassWriter cw = new ClassWriter(0);
079            MethodVisitor mv;
080            AnnotationVisitor av0;
081            String className = AbstractFelixCommandsService.class.getName().replace('.', '/') + suffix;
082            cw.visit(V1_5, ACC_PUBLIC + ACC_SUPER, className, null, AbstractFelixCommandsService.class.getName().replace('.', '/'), null);
083
084            //cw.visitSource("AbstractFelixCommandsServiceSample.java", null);
085
086            {
087                mv = cw.visitMethod(ACC_PUBLIC, "<init>", "(Ljava/lang/Object;)V", null, null);
088                mv.visitCode();
089                Label l0 = new Label();
090                mv.visitLabel(l0);
091                mv.visitLineNumber(10, l0);
092                mv.visitVarInsn(ALOAD, 0);
093                mv.visitVarInsn(ALOAD, 1);
094                mv.visitMethodInsn(INVOKESPECIAL, AbstractFelixCommandsService.class.getName().replace('.', '/'), "<init>", "(Ljava/lang/Object;)V");
095                Label l1 = new Label();
096                mv.visitLabel(l1);
097                mv.visitLineNumber(11, l1);
098                mv.visitInsn(RETURN);
099                Label l2 = new Label();
100                mv.visitLabel(l2);
101                mv.visitLocalVariable("this", "L" + className + ";", null, l0, l2, 0);
102                mv.visitLocalVariable("service", "Ljava/lang/Object;", null, l0, l2, 1);
103                mv.visitMaxs(2, 2);
104                mv.visitEnd();
105            }
106
107            /*javassist.bytecode.ClassFile ccFile = ctClass.getClassFile();
108            ccFile.setVersionToJava5();
109            javassist.bytecode.ConstPool constPool = ccFile.getConstPool();
110
111            // set superclass
112            javassist.CtClass abstractCtClass = POOL.getCtClass(AbstractFelixCommandsService.class.getName());
113            ctClass.setSuperclass(abstractCtClass);
114
115            // add constructor
116            javassist.CtClass serviceCtClass = POOL.getCtClass(Object.class.getName());
117            javassist.CtConstructor ctConstructor = new javassist.CtConstructor(new javassist.CtClass[]{serviceCtClass}, ctClass);
118            ctConstructor.setModifiers(javassist.Modifier.PUBLIC);
119            ctConstructor.setBody("super($1);");
120            ctClass.addConstructor(ctConstructor);
121
122            // add method for each command
123            javassist.CtClass sessionCtClass = POOL.getCtClass(CommandSession.class.getName());
124            javassist.CtClass stringArrayCtClass = POOL.getCtClass(String[].class.getName());*/
125            Set<String> names = commands.keySet();
126            for (String name : names) {
127                if (isMethodAvailable(service, name)) {
128                    mv = cw.visitMethod(ACC_PUBLIC, name, "(Lorg/apache/felix/service/command/CommandSession;[Ljava/lang/String;)V", null, null);
129                    {
130                        av0 = mv.visitAnnotation("Lorg/apache/felix/service/command/Descriptor;", true);
131                        av0.visit("value", commands.get(name));
132                        av0.visitEnd();
133                    }
134                    mv.visitCode();
135                    Label l0 = new Label();
136                    mv.visitLabel(l0);
137                    mv.visitLineNumber(15, l0);
138                    mv.visitVarInsn(ALOAD, 0);
139                    mv.visitLdcInsn(name);
140                    mv.visitVarInsn(ALOAD, 1);
141                    mv.visitVarInsn(ALOAD, 2);
142                    mv.visitMethodInsn(INVOKEVIRTUAL, className, "runCommand", "(Ljava/lang/String;Lorg/apache/felix/service/command/CommandSession;[Ljava/lang/String;)V");
143                    Label l1 = new Label();
144                    mv.visitLabel(l1);
145                    mv.visitLineNumber(16, l1);
146                    mv.visitInsn(RETURN);
147                    Label l2 = new Label();
148                    mv.visitLabel(l2);
149                    mv.visitLocalVariable("this", "L" + className + ";", null, l0, l2, 0);
150                    mv.visitLocalVariable("session", "Lorg/apache/felix/service/command/CommandSession;", null, l0, l2, 1);
151                    mv.visitLocalVariable("args", "[Ljava/lang/String;", null, l0, l2, 2);
152                    mv.visitMaxs(4, 3);
153                    mv.visitEnd();
154
155                    /*javassist.CtMethod ctMethod = new javassist.CtMethod(javassist.CtClass.voidType, name, new javassist.CtClass[]{
156                            sessionCtClass, stringArrayCtClass
157                    }, ctClass);
158                    ctMethod.setModifiers(javassist.Modifier.PUBLIC);
159                    ctMethod.setBody("runCommand(\"" + name + "\", $1, $2);");
160                    ctClass.addMethod(ctMethod);
161
162                    // add GoGo descriptor for this shell command
163                    javassist.bytecode.AnnotationsAttribute annotationsAttribute = new javassist.bytecode.AnnotationsAttribute(constPool, javassist.bytecode.AnnotationsAttribute.visibleTag);
164                    javassist.bytecode.annotation.Annotation annotation = new javassist.bytecode.annotation.Annotation(Descriptor.class.getName(), constPool);
165                    annotation.addMemberValue("value", new javassist.bytecode.annotation.StringMemberValue(commands.get(name), constPool));
166                    annotationsAttribute.addAnnotation(annotation);
167                    ctMethod.getMethodInfo().addAttribute(annotationsAttribute);*/
168                }
169            }
170
171            cw.visitEnd();
172            // create new newBuilder
173            /*Class<?> aClass = ctClass.toClass(FelixGogoCommandsServiceGenerator.class.getClassLoader(), null);
174            */
175            Class<?> aClass = classLoader.defineClass(className, cw.toByteArray());
176            Constructor<?> constructor = aClass.getConstructor(Object.class);
177            return constructor.newInstance(service);
178        } catch (Exception e) {
179            //ctClass.detach();
180            throw e;
181        }
182    }
183
184    private static boolean isMethodAvailable(CustomCommandScope commandsProvider, String methodName) {
185        return null != Main.findCommandMethod(commandsProvider.getClass(), methodName);
186    }
