019package org.forgerock.openidm.shell.impl;
020
021import org.apache.commons.lang3.StringUtils;
022import org.apache.felix.service.command.CommandSession;
023import org.apache.felix.service.command.Descriptor;
024import org.forgerock.openidm.shell.CustomCommandScope;
025
026import java.io.PrintStream;
027import java.lang.reflect.Method;
028import java.util.*;
029
030/**
031 * @author $author$
032 * @version $Revision$ $Date$
033 */
034public class BasicCommandScope extends CustomCommandScope {
035    /**
036     * {@inheritDoc}
037     */
038    public Map<String, String> getFunctionMap() {
039        Map<String, String> help = new HashMap<String, String>();
040        help.put("help", getShortHeader("help"));
041        help.put("exit", getShortHeader("exit"));
042        return help;
043
044    }
045
046    /**
047     * {@inheritDoc}
048     */
049    public String getScope() {
050        return "basic";
051    }
052
053
054    @Descriptor("Displays available commands.")
055    public void help(CommandSession session) {
056        ServiceLoader<CustomCommandScope> ldr = ServiceLoader.load(CustomCommandScope.class);
057        PrintStream console = session.getConsole();
058
059        for (CustomCommandScope cmdScope : ldr) {
060            String scope = cmdScope.getScope();
061            Map<String, String> functionMap = cmdScope.getFunctionMap();
062
063            if (StringUtils.isNotEmpty(scope) && functionMap != null) {
064                int maxEntryLen = 0;
065
066                for (Map.Entry<String, String> entry : functionMap.entrySet()) {
067                    int len = scope.length() + entry.getKey().length() + 4; // 4 for ':' +3 space
068                    maxEntryLen = len > maxEntryLen ? len : maxEntryLen;
069                }
070
071                StringBuilder spaceBuilder = new StringBuilder();
072
073                for (int i = 0; i < maxEntryLen; i++) {
074                    spaceBuilder.append(' ');
075                }
076
077                String spacer = spaceBuilder.toString();
078
079                for (Map.Entry<String, String> entry : functionMap.entrySet()) {
080                    String name = scope + ":" + entry.getKey();
081                    String desc = entry.getValue();
082
083                    console.append(LEAD_OPTION_SPACE).append(name)
084                            .append(spacer.substring(Math.min(name.length(), spacer.length())))
085                            .println(desc);
086                }
087            }
088        }
089    }
090
091    @Descriptor("Exit from the console.")
092    public void exit() {
093        System.exit(0);
094    }
095
096    @Descriptor("Displays information about a specific command.")
097    public void help(CommandSession session, @Descriptor("target command") String name) {
098        Map<String, List<Method>> commands = getCommands();
099
100        List<Method> methods = null;
101
102        // If the specified command doesn't have a scope, then
103        // search for matching methods by ignoring the scope.
104        int scopeIdx = name.indexOf(':');
105        if (scopeIdx < 0) {
106            for (Map.Entry<String, List<Method>> entry : commands.entrySet()) {
107                String k = entry.getKey().substring(entry.getKey().indexOf(':') + 1);
108                if (name.equals(k)) {
109                    name = entry.getKey();
110                    methods = entry.getValue();
111                    break;
112                }
113            }
114        }
115        // Otherwise directly look up matching methods.
116        else {
117            methods = commands.get(name);
118        }
119
120        if ((methods != null) && (methods.size() > 0)) {
121            StringBuilder help = new StringBuilder();
122            for (int i = 0; i < methods.size(); i++) {
123                Method m = methods.get(i);
124                if (i > 0) {
125                    help.append("\n");
126                }
127                help.append("Info: ").append(getHeader(m)).append("\n");
128                help.append(getUsage(m)).append("\n");
129            }
130            session.getConsole().print(help.toString());
131        }
132    }
133
134    private Map<String, List<Method>> getCommands() {
135        Map<String, List<Method>> commands = new TreeMap();
136
137        ServiceLoader<CustomCommandScope> ldr = ServiceLoader.load(CustomCommandScope.class);
138        for (CustomCommandScope cmdScope : ldr) {
139            if (null != cmdScope.getScope() && null != cmdScope.getFunctionMap()) {
140
141                for (String func : cmdScope.getFunctionMap().keySet()) {
142                    commands.put(cmdScope.getScope() + ":" + func, new ArrayList());
143                }
144
145                if (!commands.isEmpty()) {
146                    Method[] methods = cmdScope.getClass().getMethods();
147                    for (Method method : methods) {
148                        List<Method> commandMethods = commands.get(cmdScope.getScope() + ":"
149                                + method.getName());
150                        if (commandMethods != null) {
151                            commandMethods.add(method);
152                        }
153                    }
154                }
155
156                // Remove any missing commands.
157                Iterator<Map.Entry<String, List<Method>>> it = commands.entrySet().iterator();
158                while (it.hasNext()) {
159                    if (it.next().getValue().size() == 0) {
160                        it.remove();
161                    }
162                }
163            }
164        }
165
166        return commands;
167    }
168}
