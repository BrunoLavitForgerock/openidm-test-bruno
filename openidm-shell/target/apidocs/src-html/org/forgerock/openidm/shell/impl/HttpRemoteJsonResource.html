228    public ClientResource getClientResource(Reference ref) {
229        ClientResource clientResource = new ClientResource(new Context(), new Reference(baseReference, ref));
230
231        List<Preference<MediaType>> acceptedMediaTypes = new ArrayList<Preference<MediaType>>(1);
232        acceptedMediaTypes.add(new Preference<MediaType>(MediaType.APPLICATION_JSON));
233        clientResource.getClientInfo().setAcceptedMediaTypes(acceptedMediaTypes);
234        clientResource.getLogger().setLevel(Level.WARNING);
235        
236        ChallengeResponse rc = new ChallengeResponse(ChallengeScheme.HTTP_BASIC, username, password);
237        clientResource.setChallengeResponse(rc);
238
239        return clientResource;
240    }

242    @SuppressWarnings({ "rawtypes", "unchecked" })
243    public JsonValue getResponse(ClientResource clientResource, Representation response) 
244            throws org.forgerock.json.resource.ResourceException {
245        // Check if the request failed
246        if (!clientResource.getStatus().isSuccess()) {
247            throw org.forgerock.json.resource.ResourceException.getException(clientResource
248                    .getStatus().getCode(), clientResource.getStatus().getDescription(),
249                    clientResource.getStatus().getThrowable());
250        }
251
252        JsonValue result = null;
253        if (null != response && response instanceof EmptyRepresentation == false) {
254            try {
255                // Parse the response
256                result = new JsonValue(new JacksonRepresentation(response, Map.class).getObject());
257            } catch (IOException e) {
258                throw new InternalServerErrorException(e);
259            }
260        } else {
261            result = new JsonValue(null);
262        }
263        return result;
264    }

266    public JsonValue handle(Request request, String id, JsonValue params)
267            throws org.forgerock.json.resource.ResourceException {
268        Representation response = null;
269        ClientResource clientResource = null;
270        try {
271            Reference remoteRef = new Reference(id);
272            
273            // Get the client resource corresponding to this request's resource name
274            clientResource = getClientResource(remoteRef);
275
276            // Prepare query params
277            if (params != null && !params.isNull()) {
278                for (Map.Entry<String, Object> entry : params.expect(Map.class).asMap().entrySet()) {
279                    if (entry.getValue() instanceof String) {
280                        clientResource.addQueryParameter(entry.getKey(), (String) entry.getValue());
281                    }
282                }
283            }
284
285            // Payload
286            Representation representation = null;
287            JsonValue value = getRequestValue(request);
288            if (!value.isNull()) {
289                representation = new JacksonRepresentation<Map>(value.expect(Map.class).asMap());
290            }
291
292            // ETag
293            Conditions conditions = new Conditions();
294
295            switch (request.getRequestType()) {
296            case CREATE:
297                conditions.setNoneMatch(Arrays.asList(Tag.ALL));
298                clientResource.getRequest().setConditions(conditions);
299                response = clientResource.put(representation);
300                break;
301            case READ:
302                response = clientResource.get();
303                break;
304            case UPDATE:
305                conditions.setMatch(getTag(((UpdateRequest)request).getRevision()));
306                clientResource.getRequest().setConditions(conditions);
307                response = clientResource.put(representation);
308                break;
309            case DELETE:
310                conditions.setMatch(Arrays.asList(Tag.ALL));
311                clientResource.getRequest().setConditions(conditions);
312                response = clientResource.delete();
313                break;
314            case PATCH:
315                conditions.setMatch(getTag(((PatchRequest)request).getRevision()));
316                clientResource.getRequest().setConditions(conditions);
317                clientResource.setMethod(PATCH);
318                clientResource.getRequest().setEntity(representation);
319                response = clientResource.handle();
320                break;
321            case QUERY:
322                response = clientResource.get();
323                break;
324            case ACTION:
325                //clientResource.getRequest().setEntity(representation);
326                response = clientResource.post(representation);
327                break;
328            default:
329                throw new BadRequestException();
330            }
331
332            if (!clientResource.getStatus().isSuccess()) {
333                throw org.forgerock.json.resource.ResourceException.getException(clientResource
334                        .getStatus().getCode(), clientResource.getStatus().getDescription(),
335                        clientResource.getStatus().getThrowable());
336            }
337
338            JsonValue result = null;
339
340            if (null != response && response instanceof EmptyRepresentation == false) {
341                result = new JsonValue(new JacksonRepresentation(response, Map.class).getObject());
342            } else {
343                result = new JsonValue(null);
344            }
345            return result;
346        } catch (JsonValueException jve) {
347            throw new BadRequestException(jve);
348        } catch (ResourceException e) {
349            StringBuilder sb = new StringBuilder(e.getStatus().getDescription());
350            if (null != clientResource) {
351                try {
352                    sb.append(" ").append(clientResource.getResponse().getEntity().getText());
353                } catch (IOException e1) {
354                }
355            }
356            throw org.forgerock.json.resource.ResourceException.getException(e.getStatus()
357                    .getCode(), sb.toString(), e.getCause());
358        } catch (Exception e) {
359            throw new InternalServerErrorException(e);
360        } finally {
361            if (null != response) {
362                response.release();
363            }
364        }
365    }

367    private JsonValue getRequestValue(Request request) throws Exception {
368        switch (request.getRequestType()) {
369        case CREATE:
370            return ((CreateRequest)request).getContent();
371        case UPDATE:
372            return new JsonValue(((UpdateRequest)request).getContent());
373        case PATCH:
374            ObjectMapper mapper = new ObjectMapper();
375            List<PatchOperation> ops = ((PatchRequest)request).getPatchOperations();
376            JsonValue value = new JsonValue(new ArrayList<Object>());
377            for (PatchOperation op : ops) {
378               value.add(new JsonValue(mapper.readValue(op.toString(), Object.class)));
379            }
380            return value;
381        case ACTION:
382            JsonValue content = ((ActionRequest)request).getContent();
383            if (content != null && !content.isNull()) {
384                return content;
385            } else {
386                return new JsonValue(new HashMap<String, Object>());
387            }
388        }
389        return new JsonValue(null);
390    }

392    private List<Tag> getTag(String tag) {
393        List<Tag> result = new ArrayList<Tag>(1);
394        if (null != tag && tag.trim().length() > 0) {
395            result.add(Tag.parse(tag));
396        }
397        return result;
398    }
