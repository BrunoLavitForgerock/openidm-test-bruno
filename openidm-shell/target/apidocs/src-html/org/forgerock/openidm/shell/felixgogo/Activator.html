019package org.forgerock.openidm.shell.felixgogo;
020
021import org.apache.felix.service.command.CommandProcessor;
022import org.forgerock.openidm.shell.CustomCommandScope;
023import org.forgerock.openidm.shell.felixgogo.debug.DebugCommands;
024import org.osgi.framework.*;
025import org.osgi.util.tracker.ServiceTracker;
026import org.osgi.util.tracker.ServiceTrackerCustomizer;
027
028import java.io.InputStream;
029import java.io.PrintStream;
030import java.util.Dictionary;
031import java.util.HashMap;
032import java.util.Hashtable;
033import java.util.Map;
034import java.util.logging.Level;
035import java.util.logging.Logger;
036
037/**
038 * Felix GoGo shell adapter activator
039 *
040 * Based on Apache License 2.0 licensed osgilab org.knowhowlab.osgi.shell.felixgogo
041 */
042public class Activator implements BundleActivator {
043    private static final Logger LOG = Logger.getLogger(Activator.class.getName());
044
045    private static final String COMMANDS_DESCRIPTION_PROPERTY = "openidm.osgi.shell.commands";
046    private static final String GROUP_ID_PROPERTY = "openidm.osgi.shell.group.id";
047
048    /**
049     * Felix GoGo shell API supports groups.
050     * Filter requires shell commands description and group id
051     */
052    private static final String SHELL_COMMANDS_SERVICE_FILTER = "(&" +
053            "(" + COMMANDS_DESCRIPTION_PROPERTY + "=*)" +
054            "(" + GROUP_ID_PROPERTY + "=*)" +
055            ")";
056
057
058    /**
059     * Bundle Context
060     */
061    private BundleContext bc;
062
063    /**
064     * Command provides service tracker
065     */
066    private ServiceTracker shellCommandsTracker;
067
068    private Map<ServiceReference, ServiceRegistration> commandRegistrations = new HashMap<ServiceReference, ServiceRegistration>();
069
070    public void start(BundleContext context) throws Exception {
071        bc = context;
072        shellCommandsTracker = new ServiceTracker(bc, bc.createFilter("(objectClass=" + CustomCommandScope.class.getName() + ")"),
073                new ShellCommandsCustomizer());
074        shellCommandsTracker.open();
075
076        Dictionary<String, Object> props = new Hashtable<String, Object>();
077        props.put(CommandProcessor.COMMAND_SCOPE, "debug");
078        props.put(CommandProcessor.COMMAND_FUNCTION, DebugCommands.functions);
079        bc.registerService(DebugCommands.class.getName(), new DebugCommands(bc), props);
080    }
081
082    public void stop(BundleContext bundleContext) throws Exception {
083        shellCommandsTracker.close();
084        shellCommandsTracker = null;
085
086        bc = null;
087    }
088
089    /**
090     * Validate Command method
091     *
092     * @param service     service newBuilder
093     * @param commandName command method name
094     * @return <code>true</code> if method is peresent in service, <code>public</code> and
095     *         has params <code>PrintStream</code> and <code>String[]</code>, otherwise - <code>false</code>
096     */
097    private boolean isValidCommandMethod(Object service, String commandName) {
098        try {
099            service.getClass().getMethod(commandName, InputStream.class, PrintStream.class, String[].class);
100            return true;
101        } catch (NoSuchMethodException e) {
102            return false;
103        }
104    }
105
106    /**
107     * Command provides service tracker customizer
108     */
109    private class ShellCommandsCustomizer implements ServiceTrackerCustomizer {
110
111        public Object addingService(ServiceReference reference) {
112            CustomCommandScope service = (CustomCommandScope) bc.getService(reference);
113            Object groupId = service.getScope();
114            // if property value null or not String - ignore service
115            if (groupId == null || !(groupId instanceof String)) {
116                LOG.warning(GROUP_ID_PROPERTY + " property is null or invalid. Ignore service");
117                return null;
118            }
119            // get service ranking propety. if not null - use it on Command services registration
120            Map<String, String> commandMap = service.getFunctionMap();
121            if (!commandMap.isEmpty()) {
122                Dictionary<String, Object> props = new Hashtable<String, Object>();
123                Integer ranking = (Integer) reference.getProperty(Constants.SERVICE_RANKING);
124                Long serviceId = (Long) reference.getProperty(Constants.SERVICE_ID);
125                if (ranking != null) {
126                    props.put(Constants.SERVICE_RANKING, ranking);
127                }
128                props.put(CommandProcessor.COMMAND_SCOPE, groupId);
129                props.put(CommandProcessor.COMMAND_FUNCTION, commandMap.keySet().toArray(new String[commandMap.size()]));
130                try {
131                    // generate class
132                    Object commandsProvider = FelixGogoCommandsServiceGenerator.generate(service, commandMap, serviceId.toString());
133                    commandRegistrations.put(reference,
134                            bc.registerService(commandsProvider.getClass().getName(), commandsProvider, props));
135                } catch (Exception e) {
136                    LOG.log(Level.WARNING, "Unable to initialize group: " + groupId, e);
137                }
138                return service;
139            } else {
140                return null;
141            }
142        }
143
144        public void modifiedService(ServiceReference reference, Object service) {
145            // ignore
146        }
147
148        public void removedService(ServiceReference reference, Object service) {
149            // unregister CommandGroup services that belongs to this service registration
150            Long serviceId = (Long) reference.getProperty(Constants.SERVICE_ID);
151            // detach class
152            FelixGogoCommandsServiceGenerator.clean(serviceId.toString());
153            ServiceRegistration registration = commandRegistrations.remove(reference);
154            if (registration != null) {
155                registration.unregister();
156            }
157            bc.ungetService(reference);
158        }
159    }
160}
