025package org.forgerock.openidm.shell;
026
027import java.util.ArrayList;
028import java.util.Collections;
029import java.util.Comparator;
030import java.util.List;
031import java.util.Map;
032import java.lang.annotation.Annotation;
033import java.lang.reflect.Method;
034
035import org.apache.commons.lang3.StringUtils;
036import org.apache.felix.service.command.Descriptor;
037import org.apache.felix.service.command.Parameter;
038import org.forgerock.openidm.shell.felixgogo.MetaVar;
039
040/**
041 *
042 * @author $author$
043 * @version $Revision$ $Date$
044 */
045public abstract class CustomCommandScope {
046    protected static final String LEAD_OPTION_SPACE = "  ";
047    protected static final String OPTIONS_SPACE = "                                  ";
048
049    /**
050     * Get the {@link org.apache.felix.service.command.CommandProcessor#COMMAND_FUNCTION} value.
051     * <p/>
052     * TODO add description
053     *
054     * @return retrun a new map where the key is the command name and the value is the description.
055     */
056    public abstract Map<String, String> getFunctionMap();
057
058    /**
059     * Get the {@link org.apache.felix.service.command.CommandProcessor#COMMAND_SCOPE} value.
060     * <p/>
061     * TODO add description
062     *
063     * @return
064     */
065    public abstract String getScope();
066
067    /**
068     * Gets usage information for method by name. Fetches the method with the longest list of arguments.
069     * @param name Name of the method to search for
070     * @return String containing usage information
071     * @throws NoSuchMethodException if no such method can be found
072     */
073    public String getUsage(String name) throws NoSuchMethodException {
074        Method m = getLongestMethodByName(name);
075        return getUsage(m);
076    }
077
078    /**
079     * Gets usage information for a specified method.
080     * @param method Method object to pull information from.
081     * @return String containing usage information.
082     */
083    public String getUsage(Method method) {
084        StringBuilder usage = new StringBuilder("Usage: ").append(method.getName());
085        List<String> args = getArguments(method);
086        List<String> opts = getParameters(method);
087
088        if (opts.size() > 0) {
089            usage.append(" [options]");
090        }
091
092        for (String arg : args) {
093            usage.append(" <"+arg+">");
094        }
095
096        usage.append("\nScope: ").append(getScope());
097
098        if (!opts.isEmpty()) {
099            usage.append("\nOptions:");
100        }
101
102        for (String opt : opts) {
103            usage.append("\n").append(opt);
104        }
105
106        return usage.toString();
107    }
108
109    /**
110     * Fetches the header for the method with the longest argument list
111     * @param name Name of the method to search for
112     * @return String containing header information for the method
113     */
114    protected String getLongHeader(String name) {
115        String header;
116        try {
117            Method method = getLongestMethodByName(name);
118            header = getHeader(method);
119        } catch (NoSuchMethodException e) {
120            header = "[WARNING] No such method exists.";
121        }
122        return header;
123    }
124
125    /**
126     * Fetches the header for the method with the shortest argument list
127     * @param name Name of the method to search for
128     * @return String containing header information for the method
129     */
130    protected String getShortHeader(String name) {
131        String header;
132        try {
133            Method method = getShortestMethodByName(name);
134            header = getHeader(method);
135        } catch (NoSuchMethodException e) {
136            header = "[WARNING] No such method exists.";
137        }
138        return header;
139    }
140
141    /**
142     * Fetches the header from a specified method
143     * @param method method to pull a header from
144     * @return String containing the header for the specified method
145     */
146    protected String getHeader(Method method) {
147        Descriptor desc = method.getAnnotation(Descriptor.class);
148        return desc.value();
149    }
150
151    /**
152     * Fetches the list of arguments from a specified method.
153     * Arguments are defined by annotated method arguments with only a @Descriptor.
154     * @param method method to fetch the list of arguments from
155     * @return a list of Strings containing argument descriptions
156     */
157    protected List<String> getArguments(Method method) {
158        List<String> args = new ArrayList<String>();
159        Annotation[][] parameterAnnotations = method.getParameterAnnotations();
160
161        for (Annotation[] annotations : parameterAnnotations) {
162            String desc = null;
163            boolean foundParam = false;
164
165            for (Annotation a : annotations) {
166                if (a instanceof Parameter) {
167                    foundParam = true;
168                } else if (a instanceof Descriptor) {
169                    Descriptor d = (Descriptor) a;
170                    desc = d.value();
171                }
172            }
173
174            if (desc != null && !foundParam) {
175                args.add(desc);
176            }
177        }
178        return args;
179    }
180
181    /**
182     * Fetches the list of parameters from a specified method.
183     * Parameters are defined as by method arguments annotated with @Parameter
184     * @param method method to fetch the list of parameters from
185     * @return a list of Strings containing parameter descriptions
186     */
187    protected List<String> getParameters(Method method) {
188        List<String> opts = new ArrayList<String>();
189        Annotation[][] parameterAnnotations = method.getParameterAnnotations();
190
191        for (Annotation[] annotations : parameterAnnotations) {
192            String names = null;
193            String desc = "";
194            String metaVar = null;
195
196            for (Annotation a : annotations) {
197                if (a instanceof Parameter) {
198                    Parameter param = (Parameter) a;
199                    names = StringUtils.join(param.names(), ", ");
200                } else if (a instanceof MetaVar) {
201                    MetaVar m = (MetaVar) a;
202                    metaVar = m.value();
203                } else if (a instanceof Descriptor) {
204                    Descriptor d = (Descriptor) a;
205                    desc = d.value();
206                }
207            }
208
209            String namesWithMeta = StringUtils.isBlank(metaVar) ? names : names + " " + metaVar;
210
211            if (names != null) {
212                String str = LEAD_OPTION_SPACE + namesWithMeta +
213                             OPTIONS_SPACE.substring(Math.min(namesWithMeta.length(), OPTIONS_SPACE.length())) +
214                             desc;
215                opts.add(str);
216            }
217        }
218        return opts;
219    }
220
221    /**
222     * Fetches a sorted list of methods all sharing a specified name.
223     * This list is returned in increasing order of parameters.
224     * @param name the name of the method list to retrieve
225     * @return a sorted list of methods
226     */
227    protected List<Method> getAllMethodsByName(String name) {
228        Method[] allMethods = this.getClass().getDeclaredMethods();
229        List<Method> allNamedMethods = new ArrayList<Method>();
230        for (Method m : allMethods) {
231            if (m.getName().equals(name)) {
232                allNamedMethods.add(m);
233            }
234        }
235
236        Collections.sort(allNamedMethods, new Comparator<Method>() {
237            public int compare(Method o1, Method o2) {
238                Integer l1 = o1.getParameterTypes().length;
239                Integer l2 = o2.getParameterTypes().length;
240                return l1.compareTo(l2);
241            }
242        });
243
244        return allNamedMethods;
245    }
246
247    /**
248     * Fetch a method specified by name (gets the method with the largest number of parameters).
249     * @param name of the method to fetch
250     * @return the method with the largest number of parameters by name
251     * @throws NoSuchMethodException if no such method exists
252     */
253    protected Method getLongestMethodByName(String name) throws NoSuchMethodException {
254        List<Method> methods = getAllMethodsByName(name);
255        if (methods.isEmpty()) {
256            throw new NoSuchMethodException();
257        }
258        return methods.get(methods.size() - 1);
259    }
260
261    /**
262     * Fetch a method specified by name (gets the method with the shortest number of parameters).
263     * @param name of the method to fetch
264     * @return the method with the largest number of parameters by name
265     * @throws NoSuchMethodException if no such method exists
266     */
267    protected Method getShortestMethodByName(String name) throws NoSuchMethodException {
268        List<Method> methods = getAllMethodsByName(name);
269        if (methods.isEmpty()) {
270            throw new NoSuchMethodException();
271        }
272        return methods.get(0);
273    }
274}
