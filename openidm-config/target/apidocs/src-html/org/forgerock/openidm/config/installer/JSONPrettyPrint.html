017package org.forgerock.openidm.config.installer;
018
019import org.codehaus.jackson.util.DefaultPrettyPrinter;
020import org.codehaus.jackson.impl.Indenter;
021import org.codehaus.jackson.map.ObjectWriter;
022import org.codehaus.jackson.map.ObjectMapper;
023import org.codehaus.jackson.PrettyPrinter;
024
025/**
026 * Helper to generate customer formatted (pretty print) Jackson JSON output
027 * 
028 * @author aegloff
029 *
030 */
031public class JSONPrettyPrint {
032    
033    ObjectWriter writer;
034    
035    public JSONPrettyPrint() {
036        ObjectMapper mapper = new ObjectMapper();
037        Indenter indenter = new PrettyIndenter();
038        DefaultPrettyPrinter prettyPrinter = new DefaultPrettyPrinter();
039        prettyPrinter.indentObjectsWith(indenter);
040        prettyPrinter.indentArraysWith(indenter);
041        writer = mapper.prettyPrintingWriter(prettyPrinter);
042    }
043    
044    public ObjectWriter getWriter() {
045        return writer;
046    }
047}
