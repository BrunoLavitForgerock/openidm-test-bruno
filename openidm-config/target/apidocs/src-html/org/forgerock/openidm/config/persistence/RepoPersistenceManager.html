100    /**
101     * Handle the system notifying that it's ready to install configs.
102     */
103    public void checkReady() throws BootstrapFailure {
104        if (requireRepository) {
105            ServiceTracker repoTracker = null;
106            try {
107                if (repo == null) {
108                    Filter filter = ctx.createFilter("(" + Constants.OBJECTCLASS + "=" + RepoBootService.class.getName() + ")");
109                    repoTracker = new ServiceTracker(ctx, filter, null);
110                    repoTracker.open();
111                    logger.debug("Bootstrapping repository");
112                    RepoBootService rawRepo = (RepoBootService) repoTracker.waitForService(5000);
113                    if (rawRepo != null) {
114                        logger.debug("Bootstrap obtained repository");   
115                        repo = rawRepo;
116                    } else {
117                        logger.info("Failed to bootstrap repo, returned null");
118                    }
119                }
120            } catch (InterruptedException ex) {
121                logger.warn("Failed to bootstrap repo " + ex.getMessage(), ex);
122            } catch (InvalidSyntaxException ex) {
123                logger.warn("Failed to bootstrap repo " + ex.getMessage(), ex);
124            } finally {
125                if (repoTracker != null) {
126                    repoTracker.close();
127                }
128            }                
129            if (repo == null) {
130                throw new BootstrapFailure("Failed to acquire the bootstrap repository service to access configuration persistence.");
131            }
132        }
133    }
134
135    private boolean isReady(int retries) {
136        try {
137            checkReady();
138        } catch (BootstrapFailure e) {
139            if (retries > 0) isReady(--retries);
140        }
141        return requireRepository ? repo != null : true;
142    }

144    /**
145     * Returns <code>true</code> if a persisted <code>Dictionary</code> exists
146     * for the given <code>pid</code>. 
147     * 
148     * @param pid The identifier for the dictionary to test.
149     */
150    public boolean exists(String pid) {
151        logger.debug("Config exists call for {}", pid);
152        
153        boolean exists = false;
154
155        if (isReady(0) && requireRepository) {
156            String id = pidToId(pid);
157            try {
158                ReadRequest readRequest = Requests.newReadRequest(id);
159                Resource existing = repo.read(readRequest);
160                exists = (existing != null);
161            } catch (NotFoundException ex) {
162                exists = false;
163            } catch (ResourceException ex) {
164                throw new RuntimeException("Failed to check if configuration exists in repository: " + ex.getMessage(), ex);
165            }
166        } 
167        if (!exists) {
168            exists = tempStore.containsKey(pid);
169            if (exists) {
170                logger.debug("Entry exists in temporary store for '{}'", pid);
171            }
172        } else {
173            logger.debug("Entry exists for '{}'", pid);
174        }
175       
176        if (!exists) {
177            logger.debug("Entry does not exist for '{}'", pid);
178        }
179        return exists;
180    }

182    /**
183     * Returns the <code>Dictionary</code> for the given <code>pid</code>.
184     * 
185     * @param pid The identifier for the dictionary to load.
186     * 
187     * @return The dictionary for the identifier. This must not be
188     *      <code>null</code> but may be empty.
189     *      
190     * @throws IOException If an error occurrs loading the dictionary. An
191     *      <code>IOException</code> must also be thrown if no dictionary
192     *      exists for the given identifier. 
193     */
194    public Dictionary load(String pid) throws IOException {
195        logger.debug("Config load call for {}", pid);
196        Dictionary result = null;
197        
198        try {
199            if (isReady(0) && requireRepository) {
200                String id = pidToId(pid);
201                ReadRequest readRequest = Requests.newReadRequest(id);
202                Resource existing = repo.read(readRequest);
203                logger.debug("Config loaded {} {}", pid, existing);
204                result = mapToDict(existing.getContent().asMap());
205            } else if (!requireRepository) {
206                result = tempStore.get(pid);
207                if (result == null) {
208                    throw new IOException("No entry for " + pid + " exists.");
209                }
210
211                logger.debug("Config loaded from temporary store {} {}", pid, result);
212            }
213        } catch (NotFoundException ex) {
214            result = tempStore.get(pid);
215            if (result == null) {
216                throw new IOException("No entry for " + pid + " exists.");
217            }
218            
219            logger.debug("Config loaded from temporary store {} {}", pid, result);
220        } catch (ResourceException ex) {
221            throw new IOException("Failed to load configuration in repository: " + ex.getMessage(), ex);
222        }
223        return result;
224    }

226    /**
227     * Returns an enumeration of all <code>Dictionary</code> objects known to 
228     * this persistence manager.Dictionary
229     * <p>
230     * Implementations of this method are allowed to return lazy enumerations.
231     * That is, it is allowable for the enumeration to not return a dictionary
232     * if loading it results in an error.
233     * 
234     * @return A possibly empty Enumeration of all dictionaries.
235     * 
236     * @throws IOException If an error occurrs getting the dictionaries.
237     */
238    public Enumeration getDictionaries() throws IOException {
239        if (isReady(5)) {
240            logger.debug("Config getDictionaries call from repository");
241        } else {
242            logger.debug("Config getDictionaries call from temporary store");
243        }
244        return new Enumeration() {
245            java.util.Iterator memIter = tempStore.values().iterator();
246            java.util.Iterator dbIter = null;
247            List<String[]> returnedIds = new ArrayList<String[]>();
248            
249            @Override
250            public boolean hasMoreElements() {
251                boolean hasMore = false;
252                try {
253                    hasMore = memIter.hasNext();
254                    
255                    if (!hasMore) {
256                        if (requireRepository && repo != null && dbIter == null) {
257                            QueryRequest r = Requests.newQueryRequest("/config");
258                            r.setQueryId("query-all-ids");
259                            logger.debug("Attempt query query-all-ids");
260                            final List<Map<String, Object>> queryResult = new ArrayList<Map<String, Object>>();
261                            List<Resource> results = repo.query(r);
262                            for (Resource resource : results) {
263                                queryResult.add(resource.getContent().asMap());
264                            }
265                            dbIter = queryResult.iterator();
266                        }
267                        if (dbIter != null) {
268                            hasMore = dbIter.hasNext();
269                        }
270                    }
271                } catch (RuntimeException ex) {
272                    logger.warn("Failure getting configuration dictionaries for hasMoreElements " + ex.getMessage(), ex);
273                    throw ex;
274                } catch (Exception ex) {
275                    logger.warn("Failure getting configuration dictionaries for hasMoreElements " + ex.getMessage(), ex);
276                    throw new RuntimeException(ex);
277                }
278                
279                return hasMore;
280            }
281            
282            public Object nextElement() {
283                try {
284                    if (memIter.hasNext()) {
285                        return memIter.next();
286                    } else {
287                        Map entry = (Map) dbIter.next();
288                        String entryId = (String) entry.get("_id");
289                        return load(entryId);
290                    }
291                } catch (RuntimeException ex) {
292                    logger.warn("Failure getting configuration dictionaries for nextElement " + ex.getMessage(), ex);
293                    throw ex;
294                } catch (Exception ex) {
295                    logger.warn("Failure getting configuration dictionaries for nextElement " + ex.getMessage(), ex);
296                    throw new RuntimeException(ex);
297                }
298            }
299        };
300    }

302    /**
303     * Stores the <code>Dictionary</code> under the given <code>pid</code>.
304     * 
305     * @param pid The identifier of the dictionary.
306     * @param properties The <code>Dictionary</code> to store.
307     * 
308     * @throws IOException If an error occurrs storing the dictionary. If this
309     *      exception is thrown, it is expected, that
310     *      {@link #exists(String) exists(pid} returns <code>false</code>.
311     */
312    public void store(String pid, Dictionary properties) throws IOException {
313        logger.debug("Store call for {} {}", pid, properties);
314        
315        // Store config handling settings in memory
316        if (pid.startsWith("org.apache.felix.fileinstall")) {
317            tempStore.put(pid, properties);
318            return;
319        }
320        
321        try {
322            if (isReady(0) && requireRepository) {
323                String id = pidToId(pid);
324                Map<String,Object> obj = dictToMap(properties);
325                Map<String,Object> existing = null;
326                try {
327                    ReadRequest readRequest = Requests.newReadRequest(id);
328                    existing = repo.read(readRequest).getContent().asMap();
329                } catch (NotFoundException ex) {
330                    // Just detect that it doesn't exist
331                }
332                if (existing != null) {
333                    String rev = (String) existing.get("_rev");
334                    
335                    existing.remove("_rev");
336                    existing.remove("_id");
337                    obj.remove("_rev"); // beware, this means _id and _rev should not be in config file
338                    obj.remove("_id"); // beware, this means _id and _rev should not be in config file
339                    obj.remove(RepoPersistenceManager.BUNDLE_LOCATION);
340                    obj.remove(RepoPersistenceManager.FELIX_FILEINSTALL_FILENAME);
341                    if(!existing.equals(obj)) {
342                        logger.trace("Not matching {} {}", existing, obj);
343                        boolean retry;
344                        do {
345                            retry = false;
346                            try {
347                                UpdateRequest r = Requests.newUpdateRequest(id, new JsonValue(obj));
348                                r.setRevision(rev);
349                                repo.update(r);
350                            } catch (PreconditionFailedException ex) {
351                                logger.debug("Concurrent change during update, retrying {} {}", pid, rev);
352                                ReadRequest readRequest = Requests.newReadRequest(id);
353                                existing = repo.read(readRequest).getContent().asMap();
354                                retry = true;
355                            }
356                        } while (retry);
357                        logger.debug("Updated existing config {} {} {}", new Object[] {pid, rev, obj});
358                    } else {
359                        logger.debug("Existing config same as store request, ignoring {} {} {}", new Object[] {pid, rev, obj});
360                    }
361                } else {
362                    logger.trace("Creating: {} {} ", id, obj);
363                    // This may create a new (empty) configuration, which felix marks with _felix___cm__newConfiguration=true
364                    String newResourceId = id.substring(CONFIG_CONTEXT_PREFIX.length());
365                    CreateRequest createRequest = Requests.newCreateRequest(CONFIG_CONTEXT_PREFIX, new JsonValue(obj));
366                    createRequest.setNewResourceId(newResourceId);
367                    obj = repo.create(createRequest).getContent().asMap();
368                    logger.debug("Stored new config in repository {} {}", pid, obj);
369                }
370            } else {
371                tempStore.put(pid, properties);
372                logger.debug("Stored in memory {} {}", pid, properties);
373            }
374        } catch (ResourceException ex) {
375            throw new IOException("Failed to store configuration in repository: " + ex.getMessage(), ex);
376        }
377    }

379    /**
380     * Removes the <code>Dictionary</code> for the given <code>pid</code>. If
381     * such a dictionary does not exist, this method has no effect.
382     * 
383     * @param pid The identifier of the dictionary to delet.
384     * 
385     * @throws IOException If an error occurrs deleting the dictionary. This
386     *      exception must not be thrown if no dictionary with the given
387     *      identifier exists.
388     */
389    public void delete(String pid) throws IOException {
390        logger.debug("delete call for {}", pid);
391        Object removed = tempStore.remove(pid);
392        if (removed != null) {
393            logger.debug("Deleted {} from temporary store", pid);
394        }
395        try {
396            if (isReady(0) && requireRepository) {
397                String id = pidToId(pid);
398                boolean retry;
399                String rev = null;
400                do {
401                    retry = false;
402                    try {
403
404                        ReadRequest readRequest = Requests.newReadRequest(id);
405                        Map<String, Object> existing = repo.read(readRequest).getContent().asMap();
406                        if (existing != null) {
407                            rev = (String) existing.get("_rev");
408                            DeleteRequest r = Requests.newDeleteRequest(id);
409                            r.setRevision(rev);
410                            repo.delete(r);
411                            logger.debug("Deleted {}", pid);
412                        }
413                    } catch (PreconditionFailedException ex) {
414                        logger.debug("Concurrent change during delete, retrying {} {}", pid, rev);
415                        retry = true;
416                    } catch (NotFoundException ex) {
417                        // If it doesn't exists (anymore) that's fine
418                    }
419                } while (retry);
420
421            }
422        } catch (ResourceException ex) {
423            throw new IOException("Failed to delete configuration + " + pid + " in repository: " + ex.getMessage(), ex);
424        }
425    }

427    /**
428     * Convert OSGi pid to an ID suitable for addressing the repository
429     * @param pid the OSGi pid
430     * @return the qualified objectset id
431     */
432    String pidToId(String pid) {
433        if (pid.indexOf("|") > -1) {
434            return CONFIG_CONTEXT_PREFIX + pid.substring(0, pid.indexOf("|"));
435        }
436        return CONFIG_CONTEXT_PREFIX + pid;
437    }

439    /**
440     * Convert dictionary and contents to a Map, compatible with the JSON model.
441     * Arrays are encoded with special meta-data entries marking their origin as array.
442     * 
443     * @param dict the (OSGi configuration) dictionary
444     * @return the converted map
445     */
446    Map<String,Object> dictToMap(Dictionary dict) {
447        Map<String,Object> convert = new HashMap<String,Object>();
448        Enumeration keysEnum = dict.keys();
449        while(keysEnum.hasMoreElements()) {
450            String key = (String) keysEnum.nextElement();
451            Object rawValue = dict.get(key);
452            // Deal with outdated collection classes
453            Object putValue = rawValue;
454            if (rawValue instanceof Dictionary) {
455                putValue = dictToMap((Dictionary) rawValue); // Beware, does not support recursive dictionaries
456            } else if (rawValue instanceof Vector) {
457                putValue = new ArrayList((Vector) rawValue);
458            } else if (rawValue instanceof Object[]) {
459                List convList = new ArrayList(Arrays.asList((Object[]) rawValue));
460                
461                // Add marker entries to mark this as coming from an array
462                convList.add(OPENIDM_ORIG_ARRAY); 
463                convList.add(OPENIDM_ORIG_ARRAY_TYPE + rawValue.getClass().getComponentType().getName());
464
465                putValue = convList;
466            }
467            convert.put(toEscapedKey(key), putValue);
468        }
469        return convert;
470    }

472    /**
473     * Convert the JSON model map and contents to types and contents 
474     * compatible with OSGi configuration Dictionary
475     * e.g. using legacy types (Hashtables, Vectors), 
476     * Encoded property names decoded.
477     * 
478     * @param properties map to convert
479     * @return converted to legacy types
480     * @throws IOException if the conversion failed 
481     */
482    Dictionary mapToDict(Map properties) throws IOException {
483        if (properties instanceof Dictionary) {
484            return (Dictionary) properties;
485        } else {
486            Hashtable converted = new Hashtable();
487            for(Object entry : properties.keySet()) {
488                Object key = entry;
489                Object value = properties.get(entry);
490                if (entry instanceof String) {
491                    key = fromEscapedKey((String) entry);
492                }
493                if (value instanceof List) {
494                    List listToInspect = (List) value;
495                    if (listToInspect.contains(OPENIDM_ORIG_ARRAY)) {
496                        listToInspect.remove(OPENIDM_ORIG_ARRAY); // remove this marker entry
497                        String typeListItem = null;
498                        for (Object listItem : listToInspect) {
499                            if (listItem instanceof String && ((String)listItem).startsWith(OPENIDM_ORIG_ARRAY_TYPE)) {
500                                
501                                typeListItem = (String) listItem;
502                            }
503                        }
504                        String origType = typeListItem.substring(OPENIDM_ORIG_ARRAY_TYPE.length());
505                        listToInspect.remove(typeListItem); // remove this marker/meta-data entry
506                        
507                        try {
508                            Class origArrayClazz = null; 
509                            origArrayClazz = ctx.getBundle().loadClass(origType);
510                            value = listToInspect.toArray((Object[])Array.newInstance(origArrayClazz, 0));
511                        } catch (Exception ex) {
512                            logger.warn("Failed to convert back to original array type " + origType + " " + ex.getMessage(), ex);
513                            throw new IOException("Failed to convert back to original array type " + origType + " " + ex.getMessage(), ex);
514                        }
515                    } else {
516                        value = new java.util.Vector((List)value);
517                    }
518                }
519                converted.put(key, value);
520            }
521            return converted;
522        }
523    }

525    /*
526     * Convert to a key (property name) compatible with storing in the repository
527     * 
528     * Note: does not support keys that contain a sequence of "_.", which should not occur in our configuration files
529     */
530    String toEscapedKey(String rawKey) {
531        return rawKey.replaceAll("\\.", "__");
532    }
533    
534    /*
535     * Convert from encoded version for repository storage to the original key
536     */
537    String fromEscapedKey(String escapedKey) {
538        return escapedKey.replaceAll("__", ".");
539    }
