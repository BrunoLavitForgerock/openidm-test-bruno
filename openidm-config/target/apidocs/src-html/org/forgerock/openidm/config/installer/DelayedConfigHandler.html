037/**
038 * Handles configuration that needs to wait for related meta-data to become available
039 * (e.g. for encryption) before it can be provided to OSGi
040 * 
041 * @author aegloff
042 */
043public class DelayedConfigHandler implements ProviderListener {
044    final static Logger logger = LoggerFactory.getLogger(DelayedConfigHandler.class);
045    
046    ConfigCrypto configCrypto;
047    List<DelayedConfig> delayedConfigs = new ArrayList<DelayedConfig>();
048    
049    /**
050     * {@inheritDoc}
051     */
052    public void init(ConfigCrypto configCrypto) {
053        this.configCrypto = configCrypto;
054    }
055    
056    /**
057     * Add a configuration to the delayed pool
058     * 
059     * @param config the configuration to handle when new meta-data becomes available
060     */
061    public void addConfig(DelayedConfig config) {
062        delayedConfigs.add(config);
063    }
064    
065    /**
066     * Detects changed meta data providers
067     */
068    public void addedProvider(Object originId, MetaDataProvider provider) {
069        for (DelayedConfig config : delayedConfigs) {
070            if (configCrypto != null) {
071                List<JsonPointer> props = null;
072                try {
073                    props = config.configCrypto.getPropertiesToEncrypt(config.pidOrFactory, config.factoryAlias, config.parsedConfig);
074                    try {
075                        // Meta data now found, handle it
076                        config.configInstaller.setConfig(config.newConfig, new String[] {config.pidOrFactory, config.factoryAlias}, config.file);
077                    } catch (Exception ex) {
078                        logger.warn("Setting delayed configuration failed for {} {}", new Object[] {config.pidOrFactory, config.factoryAlias, ex});
079                    }
080                } catch (WaitForMetaData ex) {
081                    // Still not available for this config, leave it in delayed config
082                    logger.trace("Still no meta data provider for {}-{}", config.pidOrFactory, config.factoryAlias );
083                }
084            } else {
085                logger.warn("Provider change received before DelayedConfigHandler properly initialized");
086            }
087        }
088    }
089}
