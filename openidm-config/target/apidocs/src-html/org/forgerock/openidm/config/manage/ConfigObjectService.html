126    /**
127     * Gets an object from the object set by identifier.
128     * <p/>
129     * The object may contain metadata properties, including object identifier {@code _id},
130     * and object version {@code _rev} to enable optimistic concurrency supported by OpenIDM.
131     *
132     * @param resourceName the identifier of the resource to retrieve from the object set.
133     * @return the requested object.
134     * @throws NotFoundException   if the specified object could not be found.
135     * @throws ForbiddenException  if access to the object is forbidden.
136     * @throws BadRequestException if the passed identifier is invalid
137     */
138    public Map<String, Object> read(ResourceName resourceName) throws ResourceException {
139        logger.debug("Invoking read {}", resourceName.toString());
140        Map<String, Object> result = null;
141
142        try {
143
144            if (resourceName.isEmpty()) {
145                // List all configurations
146                result = new HashMap<String, Object>();
147                Configuration[] rawConfigs = configAdmin.listConfigurations(null);
148                List configList = new ArrayList();
149                if (null != rawConfigs) {
150                    for (Configuration conf : rawConfigs) {
151                        Map<String, Object> configEntry = new LinkedHashMap<String, Object>();
152
153                        String alias = null;
154                        Dictionary properties = conf.getProperties();
155                        if (properties != null) {
156                            alias = (String) properties.get(JSONConfigInstaller.SERVICE_FACTORY_PID_ALIAS);
157                        }
158                        String pid = ConfigBootstrapHelper.unqualifyPid(conf.getPid());
159                        String factoryPid = ConfigBootstrapHelper.unqualifyPid(conf.getFactoryPid());
160                        // If there is an alias for factory config is available, make a nicer ID then the internal PID
161                        String id = factoryPid != null && alias != null
162                                ? factoryPid + "/" + alias
163                                : pid;
164
165                        configEntry.put("_id", id);
166                        configEntry.put("pid", pid);
167                        configEntry.put("factoryPid", factoryPid);
168                        configList.add(configEntry);
169                    }
170                }
171                result.put("configurations", configList);
172                logger.debug("Read list of configurations with {} entries", configList.size());
173            } else {
174                Configuration config = findExistingConfiguration(new ParsedId(resourceName));
175                if (config == null) {
176                    throw new NotFoundException("No configuration exists for id " + resourceName.toString());
177                }
178                Dictionary props = config.getProperties();
179                JSONEnhancedConfig enhancedConfig = new JSONEnhancedConfig();
180                JsonValue value = enhancedConfig.getConfiguration(props, resourceName.toString(), false);
181                result = value.asMap();
182                logger.debug("Read configuration for service {}", resourceName);
183            }
184        } catch (ResourceException ex) {
185            throw ex;
186        } catch (Exception ex) {
187            logger.warn("Failure to load configuration for {}", resourceName, ex);
188            throw new InternalServerErrorException("Failure to load configuration for " + resourceName + ": " + ex.getMessage(), ex);
189        }
190        return result;
191    }

208    /**
209     * Creates a new object in the object set.
210     * <p/>
211     * This method sets the {@code _id} property to the assigned identifier for the object,
212     * and the {@code _rev} property to the revised object version (For optimistic concurrency)
213     *
214     * @param resourceName for multi-instance config, the factory pid to use
215     * @param id the client-generated identifier to use, or {@code null} if server-generated identifier is requested.
216     * @param obj    the contents of the object to create in the object set.
217     * @throws NotFoundException           if the specified id could not be resolved.
218     * @throws ForbiddenException          if access to the object or object set is forbidden.
219     * @throws PreconditionFailedException if an object with the same ID already exists.
220     * @throws BadRequestException         if the passed identifier is invalid
221     */
222    public void create(ResourceName resourceName, String id, Map<String, Object> obj) throws ResourceException {
223        logger.debug("Invoking create configuration {} {} {}", new Object[] { resourceName.toString(), id, obj });
224        if (id == null || id.length() == 0) {
225            throw new BadRequestException("The passed identifier to create is null");
226        }
227        ParsedId parsedId = new ParsedId(resourceName, id);
228        try {
229            Configuration config = null;
230            if (parsedId.isFactoryConfig()) {
231                String qualifiedFactoryPid = ConfigBootstrapHelper.qualifyPid(parsedId.factoryPid);
232                if ("org.forgerock.openidm.router".equalsIgnoreCase(qualifiedFactoryPid)) {
233                    throw new BadRequestException("router config can not be factory config");
234                }
235                config = configAdmin.createFactoryConfiguration(qualifiedFactoryPid, null);
236            } else {
237                String qualifiedPid = ConfigBootstrapHelper.qualifyPid(parsedId.pid);
238                config = configAdmin.getConfiguration(qualifiedPid, null);
239            }
240            if (config.getProperties() != null) {
241                throw new PreconditionFailedException("Can not create a new configuration with ID "
242                        + parsedId + ", configuration for this ID already exists.");
243            }
244
245            Dictionary dict = configCrypto.encrypt(parsedId.getPidOrFactoryPid(), parsedId.instanceAlias, null, new JsonValue(obj));
246            if (parsedId.isFactoryConfig()) {
247                dict.put(JSONConfigInstaller.SERVICE_FACTORY_PID_ALIAS, parsedId.instanceAlias); // The alias for the PID as understood by fileinstall
248            }
249
250            config.update(dict);
251            logger.debug("Created new configuration for {} with {}", parsedId.toString(), dict);
252        } catch (ResourceException ex) {
253            throw ex;
254        } catch (WaitForMetaData ex) {
255            logger.info("No meta-data provider available yet to create and encrypt configuration for {}, retry later.", parsedId.toString(), ex);
256            throw new InternalServerErrorException("No meta-data provider available yet to create and encrypt configuration for "
257                    + parsedId.toString() + ", retry later.", ex);
258        } catch (Exception ex) {
259            logger.warn("Failure to create configuration for {}", parsedId.toString(), ex);
260            throw new InternalServerErrorException("Failure to create configuration for " + parsedId.toString() + ": " + ex.getMessage(), ex);
261        }
262    }

264    @Override
265    public void handleUpdate(ServerContext context, UpdateRequest request,
266            ResultHandler<Resource> handler) {
267        try {
268            String rev = request.getRevision();
269            JsonValue content = request.getContent();
270            update(request.getResourceNameObject(), rev, content.asMap());
271            Resource resource = new Resource(request.getResourceName(), null, content);
272            handler.handleResult(resource);
273        } catch (ResourceException e) {
274            handler.handleError(e);
275        } catch (Exception e) {
276            handler.handleError(new InternalServerErrorException(e.getMessage(), e));
277        }
278    }

280    /**
281     * Updates the specified object in the object set.
282     * <p/>
283     * This implementation requires MVCC and hence enforces that clients state what revision they expect
284     * to be updating
285     * <p/>
286     * If successful, this method updates metadata properties within the passed object,
287     * including: a new {@code _rev} value for the revised object's version
288     *
289     * @param resourceName the identifier of the resource to be updated
290     * @param rev    the version of the object to update; or {@code null} if not provided.
291     * @param obj    the contents of the object to put in the object set.
292     * @throws ConflictException           if version is required but is {@code null}.
293     * @throws ForbiddenException          if access to the object is forbidden.
294     * @throws NotFoundException           if the specified object could not be found.
295     * @throws PreconditionFailedException if version did not match the existing object in the set.
296     * @throws BadRequestException         if the passed identifier is invalid
297     */
298    public void update(ResourceName resourceName, String rev, Map<String, Object> obj) throws ResourceException {
299        logger.debug("Invoking update configuration {} {}", resourceName.toString(), rev);
300        if (resourceName.isEmpty()) {
301            throw new BadRequestException("The passed identifier to update is empty");
302        }
303        else if (resourceName.size() > 2) {
304            throw new BadRequestException("The passed identifier to update has more than two parts");
305        }
306
307        try {
308            ParsedId parsedId = new ParsedId(resourceName);
309            Configuration config = findExistingConfiguration(parsedId);
310
311            Dictionary existingConfig = (config == null ? null : config.getProperties());
312            if (existingConfig == null) {
313                throw new NotFoundException("No existing configuration found for " + resourceName.toString() + ", can not update the configuration.");
314            }
315            existingConfig = configCrypto.encrypt(parsedId.getPidOrFactoryPid(), parsedId.instanceAlias, existingConfig, new JsonValue(obj));
316            config.update(existingConfig);
317            logger.debug("Updated existing configuration for {} with {}", resourceName.toString(), existingConfig);
318        } catch (ResourceException ex) {
319            throw ex;
320        } catch (Exception ex) {
321            logger.warn("Failure to update configuration for {}", resourceName.toString(), ex);
322            throw new InternalServerErrorException("Failure to update configuration for " + resourceName.toString() + ": " + ex.getMessage(), ex);
323        }
324    }

341    /**
342     * Deletes the specified object from the object set.
343     *
344     * @param resourceName the identifier of the resource to be deleted.
345     * @param rev    the version of the object to delete or {@code null} if not provided.
346     * @throws NotFoundException           if the specified object could not be found.
347     * @throws ForbiddenException          if access to the object is forbidden.
348     * @throws ConflictException           if version is required but is {@code null}.
349     * @throws PreconditionFailedException if version did not match the existing object in the set.
350     */
351    public void delete(ResourceName resourceName, String rev) throws ResourceException {
352        logger.debug("Invoking delete configuration {} {}", resourceName.toString(), rev);
353        if (resourceName.isEmpty()) {
354            throw new BadRequestException("The passed identifier to delete is null");
355        }
356        try {
357            Configuration config = findExistingConfiguration(new ParsedId(resourceName));
358
359            Dictionary existingConfig = config.getProperties();
360            if (existingConfig == null) {
361                throw new NotFoundException("No existing configuration found for " + resourceName.toString() + ", can not delete the configuration.");
362            }
363            config.delete();
364            logger.debug("Deleted configuration for {}", resourceName.toString());
365        } catch (ResourceException ex) {
366            throw ex;
367        } catch (Exception ex) {
368            logger.warn("Failure to delete configuration for {}", resourceName.toString(), ex);
369            throw new InternalServerErrorException("Failure to delete configuration for " + resourceName.toString() + ": " + ex.getMessage(), ex);
370        }
371    }

393    /**
394     * Locate an existing configuration based on its id, which can be
395     * a pid or for factory configurations the <factory pid>/<alias>
396     * pids can be qualified or if they use the default openidm prefix unqualified
397     *
398     * @param fullId the id
399     * @return the configuration if found, null if not
400     * @throws IOException
401     * @throws InvalidSyntaxException
402     */
403    Configuration findExistingConfiguration(ParsedId parsedId) throws IOException, InvalidSyntaxException, BadRequestException {
404
405        String filter = null;
406        if (parsedId.isFactoryConfig()) {
407            String factoryPid = ConfigBootstrapHelper.qualifyPid(parsedId.factoryPid);
408            filter = "(&(" + ConfigurationAdmin.SERVICE_FACTORYPID + "=" + factoryPid + ")(" + JSONConfigInstaller.SERVICE_FACTORY_PID_ALIAS + "=" + parsedId.instanceAlias + "))";
409        } else {
410            String pid = ConfigBootstrapHelper.qualifyPid(parsedId.pid);
411            filter = "(" + Constants.SERVICE_PID + "=" + pid + ")";
412        }
413        logger.trace("List configurations with filter: {}", filter);
414        Configuration[] configurations = configAdmin.listConfigurations(filter);
415        logger.debug("Configs found: {}", configurations);
416        if (configurations != null && configurations.length > 0) {
417            return configurations[0];
418        } else {
419            return null;
420        }
421    }

441class ParsedId {
442    final static Logger logger = LoggerFactory.getLogger(ParsedId.class);
443
444    public String pid;
445    public String factoryPid;
446    public String instanceAlias;
447
448    public ParsedId(ResourceName resourceName) throws BadRequestException {
449        // OSGi pid with spaces is disallowed; replace any spaces we get to be kind
450        ResourceName stripped = resourceName.toString().contains(" ")
451                ? ResourceName.valueOf(resourceName.toString().replaceAll(" ", "_"))
452                : resourceName;
453
454        switch (stripped.size()) {
455            case 2:
456                factoryPid = stripped.parent().toString();
457                instanceAlias = stripped.leaf();
458                break;
459
460            case 1:
461                pid = stripped.toString();
462                break;
463
464            default:
465                throw new BadRequestException("The passed resourceName has more than two parts");
466        }
467
468        if (null != factoryPid) {
469            logger.trace("Factory configuration pid: {} instance alias: {}", factoryPid, instanceAlias);
470        } else {
471            logger.trace("Managed service configuration pid: {}", pid);
472        }
473    }
474
475    public ParsedId(ResourceName resourceName, String id) throws BadRequestException {
476        if (resourceName.isEmpty()) {
477            // single-instance config
478            pid = id;
479        } else {
480            // multi-instance config
481            factoryPid = resourceName.toString();
482            instanceAlias = id;
483        }
484    }
485
486    /**
487     * @return is this ID represents a managed factory configuration, or false if it is a managed service configuraiton
488     */
489    public boolean isFactoryConfig() {
490        return (instanceAlias != null);
491    }
492
493    /**
494     * Get the qualified pid of the managed service or managed factory depending on the configuration represented
495     * Some APIs do not distinguish between single managed service PID and managed factory PID
496     *
497     * @return the qualified pid if this ID represents a managed service configuration, or the managed factory PID
498     *         if it represents a managed factory configuration
499     */
500    public String getPidOrFactoryPid() {
501        return isFactoryConfig()
502                ? ConfigBootstrapHelper.qualifyPid(factoryPid)
503                : ConfigBootstrapHelper.qualifyPid(pid);
504    }
505
506    public String toString() {
507        return isFactoryConfig()
508                ? (factoryPid + "-" + instanceAlias)
509                : pid;
510    }
