065    BundleContext context;

075    private ConfigCrypto(BundleContext context, ProviderListener delayedHandler) {
076        this.context = context;
077        this.delayedHandler = delayedHandler;
078        this.delayedHandler.init(this);
079        alias = IdentityServer.getInstance().getProperty("openidm.config.crypto.alias", "openidm-config-default");
080        logger.info("Using keystore alias {} to handle config encryption", alias);
081
082        providerTracker = new ProviderTracker(context, delayedHandler, false);
083        
084        // TODO: add bundle listeners to track new installs and remove uninstalls
085    }
086    
087    public synchronized static ConfigCrypto getInstance(BundleContext context, ProviderListener providerListener) {
088        if (instance == null) {
089            instance = new ConfigCrypto(context, providerListener);
090        }
091        return instance;
092    }

094    /**
095     * Check each provider for meta-data for a given pid until the first match is found
096     * Requested each time configuration is changed so that meta data providers can handle additional plug-ins
097     * 
098     * @param pidOrFactory the pid or factory pid 
099     * @param factoryAlias the alias of the factory configuration instance
100     * @return the list of properties to encrypt
101     */
102    public List<JsonPointer> getPropertiesToEncrypt(String pidOrFactory, String factoryAlias, JsonValue parsed)
103            throws WaitForMetaData {
104        Collection<MetaDataProvider> providers = providerTracker.getProviders();
105        WaitForMetaData lastWaitException = null;
106        for (MetaDataProvider provider : providers) {
107            try {
108                List<JsonPointer> result = provider.getPropertiesToEncrypt(pidOrFactory, factoryAlias, parsed);
109                if (result != null) {
110                    return result;
111                }
112            } catch (WaitForMetaData ex) {
113                // Continue to check if another meta data provider can resolve the meta data
114                lastWaitException = ex;
115            } catch (NotConfiguration e) {
116                logger.error("Error getting additional properties to encrypt", e);
117            }
118        }
119        if (lastWaitException != null) {
120            throw lastWaitException;
121        }
122        
123        return null;
124    }

126    /**
127     * Encrypt properties in the configuration if necessary
128     * Also results in pretty print formatting of the JSON configuration.
129     * 
130     * @param pidOrFactory the PID of either the managed service; or for factory configuration the PID of the Managed Service Factory
131     * @param instanceAlias null for plain managed service, or the subname (alias) for the managed factory configuration instance
132     * @param config The OSGi configuration 
133     * @return The configuration with any properties encrypted that a component's meta data marks as encrypted
134     * @throws InvalidException if the configuration was not valid JSON and could not be parsed
135     * @throws InternalErrorException if parsing or encryption failed for technical, possibly transient reasons
136     */
137    public Dictionary encrypt(String pidOrFactory, String instanceAlias, Dictionary config)
138            throws InvalidException, InternalErrorException, WaitForMetaData {
139
140        JsonValue parsed = parse(config, pidOrFactory);
141        return encrypt(pidOrFactory, instanceAlias, config, parsed);
142    }
143    
144    public Dictionary encrypt(String pidOrFactory, String instanceAlias, Dictionary existingConfig, JsonValue newConfig) 
145            throws WaitForMetaData {
146        
147        JsonValue parsed = newConfig;
148        Dictionary encrypted = (existingConfig == null ? new Hashtable() : existingConfig); // Default to existing
149        
150        List<JsonPointer> props = getPropertiesToEncrypt(pidOrFactory, instanceAlias, parsed);
151        if (logger.isTraceEnabled()) {
152            logger.trace("Properties to encrypt for {} {}: {}", new Object[] {pidOrFactory, instanceAlias, props}); 
153        }
154        if (props != null && !props.isEmpty()) {
155            boolean modified = false;
156            CryptoService crypto = getCryptoService(context);
157            for (JsonPointer pointer : props) {
158                logger.trace("Handling property to encrypt {}", pointer);
159
160                JsonValue valueToEncrypt = parsed.get(pointer);
161                if (null != valueToEncrypt && !valueToEncrypt.isNull() && !crypto.isEncrypted(valueToEncrypt)) {
162
163                    if (logger.isTraceEnabled()) {
164                        logger.trace("Encrypting {} with cipher {} and alias {}", new Object[] {pointer,
165                                ServerConstants.SECURITY_CRYPTOGRAPHY_DEFAULT_CIPHER, alias});
166                    }
167                    
168                    // Encrypt and replace value
169                    try {
170                        JsonValue encryptedValue = crypto.encrypt(valueToEncrypt,
171                                ServerConstants.SECURITY_CRYPTOGRAPHY_DEFAULT_CIPHER, alias);
172                        parsed.put(pointer, encryptedValue.getObject());
173                        modified = true;
174                    } catch (JsonCryptoException ex) {
175                        throw new InternalErrorException("Failure during encryption of configuration " 
176                                + pidOrFactory + "-" + instanceAlias + " for property " + pointer.toString()
177                                + " : " + ex.getMessage(), ex);
178                    }
179                }
180            }
181        }
182        String value = null;
183        try {
184            ObjectWriter writer = prettyPrint.getWriter();
185            value = writer.writeValueAsString(parsed.asMap());
186        } catch (Exception ex) {
187            throw new InternalErrorException("Failure in writing formatted and encrypted configuration "
188                    + pidOrFactory + "-" + instanceAlias + " : " + ex.getMessage(), ex);
189        }
190
191        encrypted.put(JSONConfigInstaller.JSON_CONFIG_PROPERTY, value); 
192        
193        if (logger.isDebugEnabled()) {
194            logger.debug("Config with senstiive data encrypted {} {} : {}", 
195                    new Object[] {pidOrFactory, instanceAlias, encrypted});
196        }
197        
198        return encrypted;
199    }

201    /**
202     * Parse the OSGi configuration in JSON format
203     * 
204     * @param dict the OSGi configuration
205     * @param serviceName a name for the configuration getting parsed for logging purposes
206     * @return The parsed JSON structure
207     * @throws InvalidException if the configuration was not valid JSON and could not be parsed
208     * @throws InternalErrorException if parsing failed for technical, possibly transient reasons
209     */
210    public JsonValue parse(Dictionary<String, Object> dict, String serviceName)
211            throws InvalidException, InternalErrorException {
212        JsonValue jv = new JsonValue(new HashMap<String, Object>());
213        
214        if (dict != null) {
215            Map<String, Object> parsedConfig = null;
216            String jsonConfig = (String) dict.get(JSONConfigInstaller.JSON_CONFIG_PROPERTY);
217
218            try {
219                if (jsonConfig != null && jsonConfig.trim().length() > 0) {
220                    parsedConfig = mapper.readValue(jsonConfig, Map.class);
221                }
222            } catch (Exception ex) {
223                throw new InvalidException("Configuration for " + serviceName
224                                + " could not be parsed and may not be valid JSON : " + ex.getMessage(), ex);
225            }
226
227            try {
228                jv = new JsonValue(parsedConfig);
229            } catch (JsonValueException ex) {
230                throw new InvalidException("Component configuration for " + serviceName
231                                + " is invalid: " + ex.getMessage(), ex);
232            }
233        }
234        logger.debug("Parsed configuration for {}", serviceName);
235
236        return jv;
237    }
238
239    private CryptoService getCryptoService(BundleContext context)
240            throws InternalErrorException {
241        CryptoService crypto = null;
242
243        try {
244            synchronized (JSONEnhancedConfig.class) {
245                if (cryptoTracker == null) {
246                    Filter cryptoFilter = context.createFilter("("
247                            + Constants.OBJECTCLASS + "="
248                            + CryptoService.class.getName() + ")");
249                    cryptoTracker = new ServiceTracker(context, cryptoFilter,
250                            null);
251                    cryptoTracker.open();
252                }
253            }
254
255            crypto = (CryptoService) cryptoTracker.waitForService(5000);
256            if (crypto != null) {
257                logger.trace("Obtained crypto service");
258            } else {
259                logger.warn("Failed to get crypto service to handle configuration encryption");
260                if (logger.isTraceEnabled()) {
261                    logger.trace("List of available service {}", 
262                            Arrays.asList(context.getAllServiceReferences(null, null)));
263                }
264                throw new InternalErrorException(
265                        "Configuration handling could not locate cryptography service to encrypt configuration." 
266                        + " Cryptography service is not registered..");
267            }
268        } catch (Exception ex) {
269            logger.warn("Exception in getting crypto service to handle configuration encryption", ex);
270            throw new InternalErrorException("Exception in getting cryptography service to encrypt configuration "
271                            + ex.getMessage(), ex);
272        }
273        return crypto;
274    }
