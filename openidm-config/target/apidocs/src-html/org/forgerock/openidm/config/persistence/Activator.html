024package org.forgerock.openidm.config.persistence;
025
026import java.util.Hashtable;
027
028import org.apache.felix.cm.PersistenceManager;
029import org.apache.felix.fileinstall.ArtifactInstaller;
030import org.forgerock.openidm.config.installer.JSONConfigInstaller;
031import org.forgerock.openidm.core.IdentityServer;
032import org.forgerock.openidm.logging.OsgiLogHandler;
033import org.osgi.framework.BundleActivator;
034import org.osgi.framework.BundleContext;
035import org.osgi.service.cm.ConfigurationListener;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039/**
040 * OSGi bundle activator
041 * @author aegloff
042 */
043public class Activator implements BundleActivator {
044    final static Logger logger = LoggerFactory.getLogger(Activator.class);
045
046    JSONConfigInstaller installer;
047    
048    public void start(BundleContext context) {
049        logger.debug("Config Bundle starting");
050        // Re-direct OSGi logging to the same openidm log
051        Hashtable<String, String> logHandlerProp = new Hashtable<String, String>();
052        logHandlerProp.put("service.description", "OpenIDM OSGi log handler");
053        logHandlerProp.put("service.vendor", "ForgeRock AS");
054        OsgiLogHandler logHandler = new OsgiLogHandler(context);
055        context.registerService(OsgiLogHandler.class.getName(), logHandler, logHandlerProp);
056        logger.debug("Log handler service registered");
057
058        // Initiate configuration bootstrapping by registering the repository based
059        // persistence manager plug-in to store and manipulate configuration
060        Hashtable<String, String> persistenceProp = new Hashtable<String, String>();
061        persistenceProp.put("service.cmRanking", "0");
062        RepoPersistenceManager persistenceMgr = new RepoPersistenceManager(context);
063        context.registerService(new String[] {PersistenceManager.class.getName(), ConfigPersisterMarker.class.getName()},
064                persistenceMgr, persistenceProp);
065        logger.debug("Repository persistence manager service registered");
066
067        // Register the optional "file view" handling of configuration
068        // It will not start polling configuration files until configuration for it is set
069        JSONConfigInstaller installer = new JSONConfigInstaller();
070        installer.start(context);
071        Hashtable<String, String> installerProp = new Hashtable<String, String>();
072        installerProp.put("service.description", "Config installer for JSON files");
073        context.registerService(new String[] {ArtifactInstaller.class.getName(), ConfigurationListener.class.getName()}, 
074                 installer, installerProp);
075        logger.debug("JSON configuration installer service registered");
076        logger.info("OpenIDM is starting from {}", IdentityServer.getInstance().getServerRoot());
077    }
078
079    public void stop(BundleContext context) {
080        if (installer != null) {
081            installer.stop(context);
082        }
083        logger.debug("Config Bundle stopped");
084    }
085}
