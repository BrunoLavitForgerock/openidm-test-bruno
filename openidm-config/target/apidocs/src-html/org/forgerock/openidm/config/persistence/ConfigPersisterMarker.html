016package org.forgerock.openidm.config.persistence;
017
018/**
019 * Interface implemented by OpenIDM 
020 * PersistenceManager extension/implementation.
021 * 
022 * Allows the system to recognize when the extension is ready
023 * to process configuration.
024 */
025public interface ConfigPersisterMarker {
026    
027    /**
028     * Notifies the extension when the system determined that 
029     * all necessary services are ready, and checks if the 
030     * extension itself is ready
031     * @throws an exception if the extension could not initiate
032     */
033    void checkReady() throws BootstrapFailure;
034}
