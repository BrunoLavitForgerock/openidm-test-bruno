046/**
047 * Helper class for bootstrapping the configuration, and in turn the system
048 * 
049 * The boostrapping mechanism works in the following order:
050 * 1. Repository bundle activators register a bootstrap repository that knows how to access configuration.
051 *    The basic info to bootstrap comes from system properties or configuration files
052 * 2. A repo persistence plug-in for the configuration admin is registered for configuration to get loaded/stored in the repository
053 * 3. When the OSGi configuration administration service comes up, proceed with handling configuration in files (if enabled)
054 *    via the felix file install mechanism
055 * 
056 * @author aegloff
057 */
058public class ConfigBootstrapHelper {
059
060    // Properties to set bootstrapping behavior
061    public static final String PREFIX_OPENIDM_REPO = "openidm.repo.";
062    public static final String OPENIDM_REPO_TYPE = "openidm.repo.type";
063    
064    // Properties to set configuration file handling behavior
065    public static final String OPENIDM_FILEINSTALL_BUNDLES_NEW_START = "openidm.fileinstall.bundles.new.start";
066    public static final String OPENIDM_FILEINSTALL_FILTER = "openidm.fileinstall.filter";
067    public static final String OPENIDM_FILEINSTALL_DIR = "openidm.fileinstall.dir";
068    public static final String OPENIDM_FILEINSTALL_POLL = "openidm.fileinstall.poll";
069    public static final String OPENIDM_FILEINSTALL_ENABLED = "openidm.fileinstall.enabled";
070    
071    public static final String OPENIDM_UI_FILEINSTALL_ENABLED = "openidm.ui.fileinstall.enabled";
072    public static final String OPENIDM_UI_FILEINSTALL_DIR = "openidm.ui.fileinstall.dir";
073    public static final String OPENIDM_UI_FILEINSTALL_POLL = "openidm.ui.fileinstall.poll";
074
075    public static final String FELIX_FILEINSTALL_PID = "org.apache.felix.fileinstall";
076
077    // Filename prefix for repository configuration
078    public static final String REPO_FILE_PREFIX = "repo.";
079    public static final String JSON_CONFIG_FILE_EXT = ".json";
080    
081    // Default prefix for OpenIDM OSGi services
082    public static final String DEFAULT_SERVICE_RDN_PREFIX = "org.forgerock.openidm.";
083    
084    final static Logger logger = LoggerFactory.getLogger(ConfigBootstrapHelper.class);
085
086    static boolean warnMissingConfig = true;
087    
088    /**
089     * Get the configured bootstrap information for a repository
090     * 
091     * Currently only one bootstrap repository is selected.
092     * 
093     * Bootstrap information in system properties takes precedence over configuration files
094     * 
095     * Configuration keys returned are lower case, whether they originate from system properties or
096     * configuration files.
097     * 
098     * @param repoType the type of the bootstrap repository, 
099     * equivalent to the last part of its PID 
100     * @return The relevant bootstrap configuration if this repository should be bootstraped, null if not
101     */
102    public static JsonValue getRepoBootConfig(String repoType, BundleContext bundleContext) {
103        JsonValue result = new JsonValue(new TreeMap<String, Object>(String.CASE_INSENSITIVE_ORDER));
104        result.put(OPENIDM_REPO_TYPE, repoType);
105        
106        // System properties take precedence over configuration files
107        String sysPropType = System.getProperty(OPENIDM_REPO_TYPE);
108        if (sysPropType != null && sysPropType.toLowerCase().equals(repoType)) {
109            for (Object entry : System.getProperties().keySet()) {
110                String key = (String) entry;
111                if (key.startsWith(PREFIX_OPENIDM_REPO)) {
112                    result.put(key.substring(PREFIX_OPENIDM_REPO.length()).toLowerCase(), System.getProperty(key));
113                }
114            }
115            logger.info("Bootstrapping with settings from system properties {}", result);
116            return result;
117        }
118
119        // If bootstrap info not found in system properties, check for configuration files
120        String confDir = getConfigFileInstallDir();
121        File unqualified = new File(confDir, REPO_FILE_PREFIX + repoType.toLowerCase() + JSON_CONFIG_FILE_EXT);
122        File qualified = new File(confDir, ConfigBootstrapHelper.DEFAULT_SERVICE_RDN_PREFIX + REPO_FILE_PREFIX 
123                + repoType.toLowerCase() + JSON_CONFIG_FILE_EXT);
124        
125        File loadedFile = null;
126        try {
127            Dictionary rawConfig = null;
128            if (unqualified.exists()) {
129                rawConfig = JSONConfigInstaller.loadConfigFile(unqualified);
130                loadedFile = unqualified;
131            } else if (qualified.exists()) {
132                rawConfig = JSONConfigInstaller.loadConfigFile(qualified);
133                loadedFile = qualified;
134            } else {
135                logger.debug("No configuration to bootstrap {}", repoType);
136                
137                // Check at least one configuration exists
138                String[] repoConfigFiles = getRepoConfigFiles(confDir);
139                if (warnMissingConfig && (repoConfigFiles == null || repoConfigFiles.length == 0)) {
140                    logger.error("No configuration to bootstrap the repository found.");
141                    warnMissingConfig = false;
142                }
143                
144                return null;
145            }
146            JsonValue jsonCfg = new JSONEnhancedConfig().getConfiguration(rawConfig, bundleContext, repoType);
147            Map<String, Object> cfg = jsonCfg.asMap();
148            for (Entry<String, Object> entry : cfg.entrySet()) {
149                result.put(entry.getKey(), entry.getValue());
150            }
151        } catch (Exception ex) {
152            logger.warn("Failed to load configuration file to bootstrap repository " + ex.getMessage(), ex);
153            throw new RuntimeException("Failed to load configuration file to bootstrap repository " 
154                    + ex.getMessage(), ex);
155        }
156        logger.info("Bootstrapping with settings from configuration file {}", loadedFile);
157        
158        return result;
159    }
160    
161    // A list of repository configuration files 
162    static String[] getRepoConfigFiles(final String confDir) {
163        FilenameFilter repoConfFilter = new FilenameFilter() {
164            @Override
165            public boolean accept(File dir, String name) {
166                return name.startsWith(REPO_FILE_PREFIX);
167            }
168        };
169
170        return IdentityServer.getFileForProjectPath(confDir).list(repoConfFilter);
171    }
172    
173    /**
174     * Get the directory for configuration file view
175     * 
176     * @return config dir
177     */
178    public static String getConfigFileInstallDir() {
179        // Default the configuration directory if not declared
180        String dir = System.getProperty(OPENIDM_FILEINSTALL_DIR, "conf");
181        dir =  IdentityServer.getFileForProjectPath(dir).getAbsolutePath();
182        logger.debug("Configuration file directory {}", dir);
183        return dir;
184    }

186    /**
187     * Configure to process all JSON configuration files (if enabled)
188     * 
189     * @param configAdmin the OSGi configuration admin service
190     * @throws java.io.IOException
191     */
192    public static void installAllConfig(ConfigurationAdmin configAdmin) throws IOException {
193        IdentityServer identityServer = IdentityServer.getInstance();
194        
195        String enabled = System.getProperty(OPENIDM_FILEINSTALL_ENABLED, "true");
196        String poll = System.getProperty(OPENIDM_FILEINSTALL_POLL, "2000");
197        String dir = getConfigFileInstallDir();
198        String filter = System.getProperty(OPENIDM_FILEINSTALL_FILTER, ".*\\.cfg|.*\\.json");
199        String start = System.getProperty(OPENIDM_FILEINSTALL_BUNDLES_NEW_START, "false");
200
201        String uiConfigEnabled = identityServer.getProperty(OPENIDM_UI_FILEINSTALL_ENABLED, "true");
202        String uiPoll = identityServer.getProperty(OPENIDM_UI_FILEINSTALL_POLL, "2000");
203        String uiDir = identityServer.getProperty(OPENIDM_UI_FILEINSTALL_DIR, "ui/default");
204
205        Configuration config = configAdmin.createFactoryConfiguration(FELIX_FILEINSTALL_PID, null);
206        
207        Configuration uiConfig = configAdmin.createFactoryConfiguration(FELIX_FILEINSTALL_PID, null);
208    
209        Dictionary props = config.getProperties();
210        if (props == null) {
211            props = new Hashtable();
212        }
213        
214        Dictionary uiProps = uiConfig.getProperties();
215        if (uiProps == null) {
216            uiProps = new Hashtable();
217        }
218        
219        if ("true".equals(enabled)) {
220            // Apply the latest configuration changes from file
221            props.put("felix.fileinstall.poll", poll);
222            props.put("felix.fileinstall.noInitialDelay", "true");
223            props.put("felix.fileinstall.dir", dir);
224            props.put("felix.fileinstall.filter", filter);
225            props.put("felix.fileinstall.bundles.new.start", start);
226            props.put("config.factory-pid","openidm");
227            config.update(props);
228            logger.info("Configuration from file enabled");
229        } else {
230            logger.info("Configuration from file disabled");
231        }
232        
233        if ("true".equals(uiConfigEnabled)) {
234            uiProps.put("felix.fileinstall.poll", uiPoll);
235            uiProps.put("felix.fileinstall.dir", uiDir);
236            uiProps.put("config.factory-pid", "ui");
237            uiConfig.update(uiProps);
238            logger.info("UI file installer enabled");
239        } else {
240            logger.info("UI file installer disabled");
241        }
242        
243    }

245    /**
246     * Prefixes unqualified PIDs with the default RDN qualifier
247     * I.e. file names can be unqualified and will be prefixed
248     * with the default. 
249     * Configuring services with PIDs that are not qualified 
250     * by org. or com. is currently not supported.
251     * @param fileNamePid
252     * @return
253     */
254    public static String qualifyPid(String fileNamePid) {
255        String qualifiedPid = fileNamePid;
256        // Prefix unqualified pid names with the default.
257        if (fileNamePid != null && !(fileNamePid.startsWith("org.") || fileNamePid.startsWith("com."))) {
258            qualifiedPid = DEFAULT_SERVICE_RDN_PREFIX + fileNamePid;
259        }
260        return qualifiedPid;
261    }
262    
263    /**
264     * Removes the default RDN prefix if this is a qualified name 
265     * int the default namespace. IF not, it STAYS qualified.
266     * 
267     * Configuring services with PIDs that are not qualified 
268     * by org. or com. is currently not supported.
269     * @param qualifiedPid
270     * @return
271     */
272    public static String unqualifyPid(String qualifiedPid) {
273        if (qualifiedPid != null && qualifiedPid.startsWith(DEFAULT_SERVICE_RDN_PREFIX)) {
274            return qualifiedPid.substring(DEFAULT_SERVICE_RDN_PREFIX.length());
275        } else {
276            return qualifiedPid;
277        }
278    }
