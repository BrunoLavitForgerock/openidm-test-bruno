017package org.forgerock.openidm.metadata.impl;
018
019import java.io.InputStream;
020import java.net.URL;
021import java.util.Collection;
022import java.util.Enumeration;
023import java.util.Map;
024import java.util.concurrent.ConcurrentSkipListMap;
025
026import org.codehaus.jackson.map.ObjectMapper;
027import org.forgerock.openidm.metadata.MetaDataProvider;
028import org.forgerock.openidm.metadata.MetaDataProviderCallback;
029import org.forgerock.openidm.osgi.ServiceTrackerListener;
030import org.forgerock.openidm.osgi.ServiceTrackerNotifier;
031import org.osgi.framework.Bundle;
032import org.osgi.framework.BundleContext;
033import org.osgi.framework.Constants;
034import org.osgi.framework.ServiceReference;
035import org.osgi.util.tracker.ServiceTracker;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039/**
040 * Keep track of meta data providers, either declared in bundle meta-data, or 
041 * registered as services
042 * 
043 * @author aegloff
044 * @author ckienle
045 *
046 */
047public class ProviderTracker implements ServiceTrackerListener {
048    final static Logger logger = LoggerFactory.getLogger(ProviderTracker.class);
049
050    static ServiceTracker providerTracker;
051    ProviderListener providerListener;
052    BundleContext context;
053    ObjectMapper mapper = new ObjectMapper();
054
055    // Map from origin identifiers to MetaDataProvider
056    // Long type key are bundle identifiers
057    // String keys are service pids
058    // This map MUST be thread safe to avoid the java.util.ConcurrentModificationException
059    Map<String, MetaDataProvider> providers = new ConcurrentSkipListMap<String, MetaDataProvider>();
060
061    /**
062     * Constructor
063     * @param context a bundle context to access OSGi
064     * @param listener the listener to notify when a provider change was detecte
065     * @param notifyDuringInit whether to notify the listener during the ProviderTracker construction. 
066     * Setting it to false allows to query the providers acquired during init with getProviders, and to process additional 
067     * providers via the listener.
068     */
069    public ProviderTracker(BundleContext context, ProviderListener listener, boolean notifyDuringInit) {
070        this.context = context;
071        providerListener = listener;
072        initBundleProviders(context, notifyDuringInit);
073        providerTracker = initServiceTracker(context);
074        // TODO: add bundle listeners to track new installs and remove uninstalls
075    }
076
077    private void initBundleProviders(BundleContext context, boolean notifyDuringInit) { 
078        Bundle[] bundles = context.getBundles();
079        for (Bundle bundle : bundles) {
080            if (logger.isTraceEnabled()) {
081                logger.trace("Scanning bundle {} for metadata file", bundle.getBundleId());
082            }
083            Enumeration<URL> entries = bundle.findEntries("org/forgerock/metadata", "bundle.json", true);
084            try { 
085                if (entries != null && entries.hasMoreElements()) {
086                    URL entryUrl = entries.nextElement();
087                    logger.trace("Found metadata file, load and parse {}", entryUrl);
088                    InputStream in = entryUrl.openStream();
089                    Map metaConfig = mapper.readValue(in, Map.class);
090                    in.close();
091                    String providerClazzName = (String) metaConfig.get("metaDataProvider");
092                    logger.trace("Loading declared MetaDataProvider {}", providerClazzName);
093                    if (providerClazzName == null) {
094                        logger.trace("No MetaDataProvider class declared in meta data file {} for {}", entryUrl, bundle.getSymbolicName());
095                    } else {
096                        logger.trace("Loading declared MetaDataProvider {}", providerClazzName);
097                        Class providerClazz = bundle.loadClass(providerClazzName);
098                        MetaDataProvider provider = (MetaDataProvider) providerClazz.newInstance();
099                        String id = Long.valueOf(bundle.getBundleId()).toString();
100                        // Instantiate and set the provider callback
101                        provider.setCallback(new ProviderTrackerCallback(provider, id));
102                        // Add the provider to the listener
103                        addProvider(id, provider, notifyDuringInit);
104                        logger.debug("Registered MetaDataProvider {} for {}", providerClazzName, bundle.getSymbolicName());
105                    }
106                }
107            } catch (Exception ex) {
108                logger.warn("Failed to obtain meta-data on handling configuration for {}", bundle.getSymbolicName(), ex);
109            }
110        }
111    }
112
113    private void addProvider(String originId, MetaDataProvider provider, boolean notify) {
114        providers.put(originId, provider);
115        if (providerListener != null && notify) {
116            logger.debug("Notifying listener of added MetaDataProvider {}", originId);
117            providerListener.addedProvider(originId, provider);
118        }
119    }
120
121    private ServiceTracker initServiceTracker(BundleContext context) {
122        ServiceTracker tracker = new ServiceTrackerNotifier(context, MetaDataProvider.class.getName(), null, this);
123        tracker.open();
124        return tracker;
125    }
126
127    public void addedService(ServiceReference reference, Object service) {
128        String pid = (String) reference.getProperty(Constants.SERVICE_PID);
129        MetaDataProvider provider = (MetaDataProvider) service;
130        // Instantiate and set the provider callback
131        provider.setCallback(new ProviderTrackerCallback(provider, pid));
132        // Add the provider to the listener
133        addProvider(pid, provider, true);
134    }
135
136    public void removedService(ServiceReference reference, Object service) {
137        String pid = (String) reference.getProperty(Constants.SERVICE_PID);
138        providers.remove(pid);
139    }
140
141    public void modifiedService(ServiceReference reference, Object service) {
142        String pid = (String) reference.getProperty(Constants.SERVICE_PID);
143        modifiedProvider(pid, (MetaDataProvider) service, true);
144    }
145    
146    public void modifiedProvider(String pid, MetaDataProvider provider, boolean notify) {
147        addProvider(pid, provider, notify);
148    }
149
150    /**
151     * Get the current registered providers
152     * Providers can be backed by an OSGi services, 
153     * or created from a bundle factory
154     * @return meta data providers
155     */
156    public Collection<MetaDataProvider> getProviders() {
157        //The returned value MUST be thread safe!
158        return providers.values();
159    }
160
161    /**
162     * A MetaDataProviderCallback implementation that is used to notify the ProvicerListener
163     * when a MetaDataProvider calls refresh()
164     */
165    private class ProviderTrackerCallback implements MetaDataProviderCallback {
166
167        private MetaDataProvider provider = null;
168        private String originId = null;
169
170        /**
171         * Constructor
172         * @param provider the MetaDataProvider instance
173         * @param originId the MetaDataProvider id
174         */
175        public ProviderTrackerCallback(MetaDataProvider provider, String originId) {
176            this.provider = provider;
177            this.originId = originId;
178        }
179
180        @Override
181        public void refresh() {
182            modifiedProvider(originId, provider, true);
183        }
184
185    }
186}
