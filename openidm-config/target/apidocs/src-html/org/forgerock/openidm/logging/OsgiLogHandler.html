017package org.forgerock.openidm.logging;
018
019import java.util.Collections;
020import java.util.HashMap;
021import java.util.Map;
022
023import org.osgi.framework.BundleContext;
024import org.osgi.framework.ServiceReference;
025import org.osgi.service.log.LogEntry;
026import org.osgi.service.log.LogListener;
027import org.osgi.service.log.LogReaderService;
028import org.osgi.service.log.LogService;
029import org.osgi.util.tracker.ServiceTracker;
030import org.osgi.util.tracker.ServiceTrackerCustomizer;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034/**
035 * Directs OSGi logging service entries into the slf4j logger
036 *
037 * @author aegloff
038 */
039public class OsgiLogHandler {
040    
041    final static Logger logger = LoggerFactory.getLogger(OsgiLogHandler.class);
042
043    ServiceTracker tracker;
044    
045    public OsgiLogHandler(final BundleContext context) {
046        tracker = new LogServiceTracker(context, LogReaderService.class.getName(), null);
047        tracker.open();
048        logger.debug("Log service tracker opened");
049    }
050}
051
052/**
053 * Attach log listeners to log reader services to re-direct to slf4j
054 */
055class LogServiceTracker extends ServiceTracker {
056    
057    final static Logger logger = LoggerFactory.getLogger(LogServiceTracker.class);
058    
059    // Logger to use for OSGi log entries 
060    final static Logger defaultEntryLogger = LoggerFactory.getLogger("org.forgerock.openidm.Framework");
061    
062    Map<LogReaderService, LogListener> logReaderServices = 
063        Collections.synchronizedMap(new HashMap<LogReaderService, LogListener>()); 
064    
065    public LogServiceTracker(BundleContext context, String clazz, ServiceTrackerCustomizer customizer) {
066        super(context, clazz, customizer);
067    }
068    
069    @Override
070    public Object addingService(ServiceReference reference) {
071        LogReaderService svc = (LogReaderService)context.getService(reference);
072        addLogReaderService(svc);
073        return reference;
074    }
075    
076    @Override
077    public void removedService(ServiceReference reference, Object service) {
078        LogReaderService svc = (LogReaderService)context.getService(reference);
079        removeLogReaderService(svc);
080    }
081    
082    private void addLogReaderService(LogReaderService reader) {
083        logger.trace("Adding log reader service");
084        LogListener existing = logReaderServices.get(reader);
085        if (existing == null) {
086            LogListener logListener = new LogListener() {
087                @Override
088                public void logged(LogEntry entry) {
089                    logEntry(entry);
090                }
091            };
092            reader.addLogListener(logListener);
093            
094            // This dumps the existing log entries. 
095            // These entries may appear out of chronological order
096            // if services already started logging to slf4j directly
097            java.util.Enumeration entries = reader.getLog();
098            while (entries.hasMoreElements()) {
099                logEntry((LogEntry) entries.nextElement());
100            }
101            
102            // Redirects all new log entries
103            logReaderServices.put(reader, logListener);
104        }
105    }
106
107    /**
108     * Log the OSGi entry using the slf4j logger
109     * @param entry
110     */
111    void logEntry(LogEntry entry) {
112
113        StringBuilder logMessage = new StringBuilder("Bundle: ");
114        logMessage.append(entry.getBundle());
115        if (entry.getServiceReference() != null) {
116            logMessage.append(" - ");
117            logMessage.append(entry.getServiceReference());
118        }
119        logMessage.append(" ");
120        logMessage.append(entry.getMessage());
121       
122        Throwable ex = entry.getException();
123        
124        switch (entry.getLevel()) {
125            case LogService.LOG_ERROR: {
126                defaultEntryLogger.error(logMessage.toString(), ex);
127                break;
128            }
129            case LogService.LOG_WARNING: {
130                defaultEntryLogger.warn(logMessage.toString(), ex);
131                break;
132            }
133            case LogService.LOG_INFO: {
134                defaultEntryLogger.info(logMessage.toString(), ex);
135                break;
136            }
137            case LogService.LOG_DEBUG: {
138                defaultEntryLogger.debug(logMessage.toString(), ex);
139                break;
140            }
141            default: {
142                defaultEntryLogger.warn("Unknown OSGi log level [" + entry.getLevel() + "] for" + logMessage.toString(), ex);
143            }
144        }
145    }
146    
147    private void removeLogReaderService(LogReaderService reader) {
148        logger.trace("Removing log reader service");
149        LogListener logListener = logReaderServices.remove(reader);
150        if (logListener != null) {
151            reader.removeLogListener(logListener);
152        }
153    }
154}
