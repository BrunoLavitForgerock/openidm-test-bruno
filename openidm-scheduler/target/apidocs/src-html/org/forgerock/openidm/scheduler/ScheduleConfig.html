045    private Boolean enabled = null;
046    private Boolean persisted = null;
047    private String misfirePolicy = null;
048    private String scheduleType = null;
049    private Date startTime = null;
050    private Date endTime = null;
051    private String cronSchedule = null;
052    private TimeZone timeZone = null;
053    private String invokeService = null;
054    private Object invokeContext = null;
055    private String invokeLogLevel = null;
056    private Boolean concurrentExecution = null;
057
058    public ScheduleConfig(JsonValue config) throws ResourceException {
059        JsonValue enabledValue = config.get(SchedulerService.SCHEDULE_ENABLED);
060        if (enabledValue.isString()) {
061            enabled = Boolean.parseBoolean(enabledValue.defaultTo("true").asString());
062        } else {
063            enabled = enabledValue.defaultTo(Boolean.TRUE).asBoolean();
064        }
065        JsonValue persistedValue = config.get(SchedulerService.SCHEDULE_PERSISTED);
066        if (persistedValue.isString()) {
067            persisted = Boolean.parseBoolean(persistedValue.defaultTo("false").asString());
068        } else {
069            persisted = persistedValue.defaultTo(Boolean.FALSE).asBoolean();
070        }
071        JsonValue concurrentExecutionValue = config.get(SchedulerService.SCHEDULE_CONCURRENT_EXECUTION);
072        if (concurrentExecutionValue.isString()) {
073            concurrentExecution = Boolean.parseBoolean(concurrentExecutionValue.defaultTo("false").asString());
074        } else {
075            concurrentExecution = concurrentExecutionValue.defaultTo(Boolean.FALSE).asBoolean();
076        }
077        misfirePolicy = config.get(SchedulerService.SCHEDULE_MISFIRE_POLICY).defaultTo(SchedulerService.MISFIRE_POLICY_FIRE_AND_PROCEED).asString();
078        if (!misfirePolicy.equals(SchedulerService.MISFIRE_POLICY_FIRE_AND_PROCEED) &&
079                !misfirePolicy.equals(SchedulerService.MISFIRE_POLICY_DO_NOTHING)) {
080            throw new BadRequestException(new StringBuilder("Invalid misfire policy: ").append(misfirePolicy).toString());
081        }
082        cronSchedule = config.get(SchedulerService.SCHEDULE_CRON_SCHEDULE).asString();
083        scheduleType = config.get(SchedulerService.SCHEDULE_TYPE).asString();
084        invokeService = config.get(SchedulerService.SCHEDULE_INVOKE_SERVICE).asString();
085        if (!StringUtils.isNotBlank(invokeService)) {
086            throw new BadRequestException("Invalid scheduler configuration, the "
087                    + SchedulerService.SCHEDULE_INVOKE_SERVICE
088                    + " property needs to be set but is empty. "
089                    + "Complete config:" + config);
090        } else {
091            // service PIDs fragments are prefixed with openidm qualifier
092            if (!invokeService.contains(".")) {
093                String fragment = invokeService;
094                invokeService = SchedulerService.SERVICE_RDN_PREFIX + fragment;
095            }
096        }
097        invokeContext = config.get(SchedulerService.SCHEDULE_INVOKE_CONTEXT).getObject();
098        invokeLogLevel = config.get(SchedulerService.SCHEDULE_INVOKE_LOG_LEVEL).defaultTo("info").asString();
099        String timeZoneString = config.get(SchedulerService.SCHEDULE_TIME_ZONE).asString();
100        String startTimeString = config.get(SchedulerService.SCHEDULE_START_TIME).asString();
101        String endTimeString = config.get(SchedulerService.SCHEDULE_END_TIME).asString();
102        if (StringUtils.isNotBlank(timeZoneString)) {
103            timeZone = TimeZone.getTimeZone(timeZoneString);
104            // JDK has fall-back behavior to GMT if it doesn't understand timezone passed
105            if (!timeZoneString.equals(timeZone.getID())) {
106                throw new BadRequestException("Scheduler configured timezone is not understood: " + timeZoneString);
107            }
108        }
109        if (StringUtils.isNotBlank(startTimeString)) {
110            Calendar parsed = DatatypeConverter.parseDateTime(startTimeString);
111            startTime = parsed.getTime();
112            // TODO: enhanced logging for failure
113        }
114
115        if (StringUtils.isNotBlank(endTimeString)) {
116            Calendar parsed = DatatypeConverter.parseDateTime(endTimeString);
117            endTime = parsed.getTime();
118            // TODO: enhanced logging for failure
119        }
120
121        if (StringUtils.isNotBlank(scheduleType)) {
122            if (!scheduleType.equals(SchedulerService.SCHEDULE_TYPE_CRON)) {
123                throw new BadRequestException("Scheduler configuration contains unknown schedule type "
124                        + scheduleType + ". Known types include " + SchedulerService.SCHEDULE_TYPE_CRON);
125            }
126        }
127    }
128
129    public ScheduleConfig(CronTrigger trigger, JobDataMap map, boolean persisted, boolean concurrentExecution) {
130        this.persisted = persisted;
131        this.concurrentExecution = concurrentExecution;
132        this.enabled = true;
133        this.cronSchedule = trigger.getCronExpression();
134        this.startTime = trigger.getStartTime();
135        this.endTime = trigger.getEndTime();
136        this.timeZone = trigger.getTimeZone();
137        this.invokeService = (String)map.get(ScheduledService.CONFIGURED_INVOKE_SERVICE);
138        this.invokeLogLevel = (String)map.get(ScheduledService.CONFIGURED_INVOKE_LOG_LEVEL);
139        this.invokeContext = map.get(ScheduledService.CONFIGURED_INVOKE_CONTEXT);
140        this.scheduleType = SchedulerService.SCHEDULE_TYPE_CRON;
141        int mp = trigger.getMisfireInstruction();
142        if (mp == CronTrigger.MISFIRE_INSTRUCTION_DO_NOTHING) {
143            this.misfirePolicy = SchedulerService.MISFIRE_POLICY_DO_NOTHING;
144        } else {
145            this.misfirePolicy = SchedulerService.MISFIRE_POLICY_FIRE_AND_PROCEED;
146        }
147    }
148
149    public JsonValue getConfig() {
150        HashMap<String, Object> map = new HashMap<String, Object>();
151        map.put(SchedulerService.SCHEDULE_ENABLED, getEnabled());
152        map.put(SchedulerService.SCHEDULE_PERSISTED, getPersisted());
153        map.put(SchedulerService.SCHEDULE_MISFIRE_POLICY, getMisfirePolicy());
154        map.put(SchedulerService.SCHEDULE_CRON_SCHEDULE, getCronSchedule());
155        map.put(SchedulerService.SCHEDULE_TYPE, getScheduleType());
156        map.put(SchedulerService.SCHEDULE_INVOKE_SERVICE, getInvokeService());
157        map.put(SchedulerService.SCHEDULE_INVOKE_CONTEXT, getInvokeContext());
158        map.put(SchedulerService.SCHEDULE_INVOKE_LOG_LEVEL, getInvokeLogLevel());
159        map.put(SchedulerService.SCHEDULE_TIME_ZONE, getTimeZone());
160        map.put(SchedulerService.SCHEDULE_START_TIME, getStartTime());
161        map.put(SchedulerService.SCHEDULE_END_TIME, getEndTime());
162        map.put(SchedulerService.SCHEDULE_CONCURRENT_EXECUTION, getConcurrentExecution());
163        return new JsonValue(map);
164    }
