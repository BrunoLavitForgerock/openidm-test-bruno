075@Component(name = "org.forgerock.openidm.taskscanner", policy = ConfigurationPolicy.IGNORE, immediate = true)
076@Properties({
077    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM TaskScanner Service"),

081@Service
082public class TaskScannerService implements RequestHandler, ScheduledService {
083    private final static Logger logger = LoggerFactory.getLogger(TaskScannerService.class);
084
085    private final static String INVOKE_CONTEXT = "invokeContext";
086
087    private int maxCompletedRuns;
088
089    /**
090     * Map from TaskScanID ID to the run itself
091     * In historical start order, oldest first.
092     */
093    Map<String, TaskScannerContext> taskScanRuns =
094            Collections.synchronizedMap(new LinkedHashMap<String, TaskScannerContext>());
095
096    /** The Connection Factory */
097    @Reference(policy = ReferencePolicy.STATIC, target="(service.pid=org.forgerock.openidm.internal)")
098    protected ConnectionFactory connectionFactory;

101    private ScriptRegistry scopeFactory;

103    @Activate
104    public void activate(ComponentContext context) {
105        String maxCompletedStr =
106                IdentityServer.getInstance().getProperty("openidm.taskscanner.maxcompletedruns", "100");
107        maxCompletedRuns = Integer.parseInt(maxCompletedStr);
108    }

113    @Override
114    public void execute(ServerContext context, Map<String, Object> contextMap) throws ExecutionException {
115        String invokerName = (String) contextMap.get(INVOKER_NAME);
116        String scriptName = (String) contextMap.get(CONFIG_NAME);
117        JsonValue params = new JsonValue(contextMap).get(CONFIGURED_INVOKE_CONTEXT);
118        startTaskScanJob(context, invokerName, scriptName, params);
119    }

121    @Override
122    public void handleRead(ServerContext context, ReadRequest request, ResultHandler<Resource> handler) {
123        try {
124            String id = request.getResourceName();
125            Map<String, Object> result = new LinkedHashMap<String, Object>();
126            if (request.getResourceNameObject().isEmpty()) {
127                List<Map<String, Object>> taskList = new ArrayList<Map<String, Object>>();
128                for (TaskScannerContext entry : taskScanRuns.values()) {
129                    Map<String, Object> taskData = buildTaskData(entry);
130                    taskList.add(taskData);
131                }
132                result.put("tasks", taskList);
133            } else {
134                TaskScannerContext foundRun = taskScanRuns.get(request.getResourceName());
135                if (foundRun == null) {
136                    throw new NotFoundException("Task with id '" + request.getResourceName() + "' not found." );
137                }
138                result = buildTaskData(foundRun);
139            }
140            handler.handleResult(new Resource(id, null, new JsonValue(result)));
141        } catch (Throwable t) {
142            handler.handleError(ResourceUtil.adapt(t));
143        }
144    }
145
146    // TODO maybe move this into TaskScannerContext?
147    private Map<String, Object> buildTaskData(TaskScannerContext entry) {
148        Map<String, Object> result = new LinkedHashMap<String, Object>();
149        result.put("_id", entry.getTaskScanID());
150        result.put("progress", entry.getProgress());
151        result.put("started", entry.getStatistics().getJobStartTime());
152        result.put("ended", entry.getStatistics().getJobEndTime());
153        return result;
154    }

156    @Override
157    public void handleAction(ServerContext context, ActionRequest request, ResultHandler<JsonValue> handler) {
158        try {
159            Map<String, String> params = request.getAdditionalParameters();
160            Map<String, Object> result = new LinkedHashMap<String, Object>();
161
162            String action = request.getAction();
163            if (request.getResourceNameObject().isEmpty()) {
164                try {
165                    if ("execute".equalsIgnoreCase(action)) {
166                        try {
167                            result.put("_id", onExecute(context, request.getResourceName(), params));
168                        } catch (JsonProcessingException e) {
169                            throw new InternalServerErrorException(e);
170                        } catch (IOException e) {
171                            throw new InternalServerErrorException(e);
172                        }
173                    } else {
174                        throw new BadRequestException("Unknown action: " + action);
175                    }
176                } catch (ExecutionException e) {
177                    logger.warn(e.getMessage());
178                    throw new BadRequestException(e.getMessage(), e);
179                }
180            } else {
181                // operation on individual resource
182                TaskScannerContext foundRun = taskScanRuns.get(request.getResourceName());
183                if (foundRun == null) {
184                    throw new NotFoundException("Task with id '" + request.getResourceName() + "' not found." );
185                }
186
187                if ("cancel".equalsIgnoreCase(action)) {
188                    foundRun.cancel();
189                    result.put("_id", foundRun.getTaskScanID());
190                    result.put("action", action);
191                    result.put("status", "SUCCESS");
192                } else {
193                    throw new BadRequestException("Action '" + action + "' on Task '" + request.getResourceName() + "' not supported " + params);
194                }
195            }
196
197            handler.handleResult(new JsonValue(result));
198        } catch (Throwable t) {
199            handler.handleError(ResourceUtil.adapt(t));
200        }
201    }
202
203    /**
204     * Performs the "execute" action, executing a supplied configuration
205     *
206     * Expects a field "name" containing the name of some config object that can be found via
207     * a read on "config/" + name <br><br>
208     *
209     * <b><i>e.g.</b></i> "taskscan/sunset" => "config/taskscan/sunset" => "[openidm-directory]/conf/taskscan-sunset.json"<br>
210     *
211     * @param id the id to perform the action on
212     * @param params field contaning the parameters of execution
213     * @return the set of parameters supplied
214     * @throws ExecutionException
215     * @throws JsonProcessingException
216     * @throws IOException
217     */
218    private String onExecute(ServerContext context, String id, Map<String, String> params)
219            throws ExecutionException, JsonProcessingException, IOException, ScriptException {
220        String name = params.get("name");
221        JsonValue config;
222        try {
223            config = connectionFactory.getConnection().read(context, Requests.newReadRequest("config/" + name)).getContent();
224        } catch (ResourceException e) {
225            throw new ExecutionException("Error obtaining named config: '" + name + "'", e);
226        }
227        JsonValue invokeContext = config.get(INVOKE_CONTEXT);
228
229        return startTaskScanJob(context, "REST", name, invokeContext);
230    }
231
232    private String startTaskScanJob(ServerContext context, String invokerName, String scriptName, JsonValue params) throws ExecutionException {
233        TaskScannerContext taskScannerContext = new TaskScannerContext(invokerName, scriptName, params, context);
234        addTaskScanRun(taskScannerContext);
235        TaskScannerJob taskScanJob = null;
236        try {
237            taskScanJob = new TaskScannerJob(connectionFactory, taskScannerContext, scopeFactory);
238        } catch (ScriptException e) {
239            throw new ExecutionException(e);
240        }
241        return taskScanJob.startTask();
242    }
243
244    private void addTaskScanRun(TaskScannerContext context) {
245        // Clean out run history if needed
246        // Since it only checks for completed runs when a new run is started this
247        // only provides for approximate adherence to maxCompleteRuns
248        synchronized(taskScanRuns) {
249            if (taskScanRuns.size() > maxCompletedRuns) {
250                int completedCount = 0;
251                // Since oldest runs are first in the list, inspect backwards
252                ListIterator<String> iter = new ArrayList<String>(taskScanRuns.keySet())
253                        .listIterator(taskScanRuns.size());
254                while (iter.hasPrevious()) {
255                    String key = iter.previous();
256                    TaskScannerContext aRun = taskScanRuns.get(key);
257                    if (aRun.isCompleted()) {
258                        ++completedCount;
259                        if (completedCount > maxCompletedRuns) {
260                            taskScanRuns.remove(key);
261                        }
262                    }
263                }
264            }
265            taskScanRuns.put(context.getTaskScanID(), context);
266        }
267    }
