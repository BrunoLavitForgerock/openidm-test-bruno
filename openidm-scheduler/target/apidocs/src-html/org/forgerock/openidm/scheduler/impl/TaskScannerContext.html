024package org.forgerock.openidm.scheduler.impl;
025
026import java.util.LinkedHashMap;
027import java.util.Map;
028
029import org.forgerock.json.fluent.JsonPointer;
030import org.forgerock.json.fluent.JsonValue;
031import org.forgerock.json.resource.ServerContext;
032import org.forgerock.openidm.util.ConfigMacroUtil;
033import org.joda.time.Days;
034import org.joda.time.ReadablePeriod;
035
036// TODO: cache components that are fetched frequently
037public class TaskScannerContext {
038
039    enum TaskScannerState {
040        INITIALIZED,
041        ACTIVE,
042        COMPLETED,
043        CANCELLED,
044        ERROR
045    }
046
047    private String invokerName;
048    private String scriptName;
049    private JsonValue params;
050    private ServerContext context;
051    private boolean canceled = false;
052    private TaskScannerStatistic statistics;
053    private TaskScannerState state;
054
055    public TaskScannerContext(String invokerName, String scriptName, JsonValue params, ServerContext context) {
056        this.invokerName = invokerName;
057        this.scriptName = scriptName;
058        this.params = params;
059        this.context = context;
060        this.statistics = new TaskScannerStatistic();
061        this.state = TaskScannerState.INITIALIZED;
062    }
063
064    public void startJob() {
065        state = TaskScannerState.ACTIVE;
066        statistics.jobStart();
067    }
068
069    public void endJob() {
070        state = TaskScannerState.COMPLETED;
071        statistics.jobEnd();
072    }
073
074    public void startQuery() {
075        statistics.queryStart();
076    }
077
078    public void endQuery() {
079        statistics.queryEnd();
080    }
081
082    public void cancel() {
083        state = TaskScannerState.CANCELLED;
084        this.canceled = true;
085    }
086
087    public void interrupted() {
088        state = TaskScannerState.ERROR;
089    }
090
091    public boolean isStarted() {
092        return state == TaskScannerState.ACTIVE;
093    }
094
095    public boolean isCompleted() {
096        return state == TaskScannerState.COMPLETED;
097    }
098
099    public boolean hasErorr() {
100        return state == TaskScannerState.ERROR;
101    }
102
103    public boolean isInactive() {
104        return !isStarted();
105    }
106
107    public boolean isCanceled() {
108        return this.canceled;
109    }
110
111    public String getState() {
112        return state.toString();
113    }
114
115    public void setNumberOfTasksToProcess(int number) {
116        statistics.setNumberOfTasksToProcess(number);
117    }
118
119    public ServerContext getContext() {
120        return this.context;
121    }
122
123    public JsonValue getParams() {
124        return this.params;
125    }
126
127    public String getInvokerName() {
128        return this.invokerName;
129    }
130
131    public String getScriptName() {
132        return this.scriptName;
133    }
134
135    public String getTaskScanID() {
136        return context.getId();
137    }
138
139    /**
140     * Retrieve the timeout period from the supplied config
141     * @param value JsonValue configuration containing "recovery/timeout"
142     * @return the timeout period taken from the config
143     */
144    public ReadablePeriod getRecoveryTimeout() {
145        String timeoutStr = getScanValue().get("recovery").get("timeout").asString();
146
147        // Default to a 0-Day period if there's nothing to add
148        if (timeoutStr == null) {
149            return Days.days(0);
150        }
151
152        return ConfigMacroUtil.getTimePeriod(timeoutStr);
153    }
154
155    public Integer getMaxRecords() {
156        return params.get("maxRecords").asInteger();
157    }
158
159    public JsonValue getScriptValue() {
160        return params.get("task").expect(Map.class).get("script").expect(Map.class);
161    }
162
163    public JsonValue getScanValue() {
164        return params.get("scan").expect(Map.class);
165    }
166
167    public String getObjectID() {
168        return getScanValue().get("object").required().asString();
169    }
170
171    public JsonPointer getStartField() {
172        return getTaskStatePointer(getScanValue(), "started");
173    }
174
175    public JsonPointer getCompletedField() {
176        return getTaskStatePointer(getScanValue(), "completed");
177    }
178
179    public boolean getWaitForCompletion() {
180        JsonValue waitParam = params.get("waitForCompletion").defaultTo("false");
181        Boolean waitForCompletion = Boolean.FALSE;
182        if (waitParam.isBoolean()) {
183            waitForCompletion = waitParam.asBoolean();
184        } else {
185            waitForCompletion = Boolean.parseBoolean(waitParam.asString());
186        }
187        return waitForCompletion.booleanValue();
188    }
189
190    public int getNumberOfThreads() {
191        JsonValue numParams = params.get("numberOfThreads").defaultTo(10);
192        return numParams.asInteger();
193    }
194
195    public TaskScannerStatistic getStatistics() {
196        return this.statistics;
197    }
198
199    public Map<String, Object> getProgress() {
200        Map<String, Object> progress = new LinkedHashMap<String, Object>();
201        progress.put("state", state);
202        progress.put("processed", statistics.getNumberOfTasksProcessed());
203        progress.put("total", statistics.getNumberOfTasksToProcess());
204        progress.put("successes", statistics.getNumberOfTasksSucceeded());
205        progress.put("failures", statistics.getNumberOfTasksFailed());
206        return progress;
207    }
208
209    /**
210     * Fetches a JsonPointer from the "taskState" object in value
211     * @param value JsonValue to fetch the pointer from
212     * @param field the subfield to fetch from the taskState object
213     * @return the JsonPointer contained within the "taskState/${field}" object
214     */
215    private JsonPointer getTaskStatePointer(JsonValue value, String field) {
216        return value.get("taskState").required().get(field).required().asPointer();
217    }
218}
