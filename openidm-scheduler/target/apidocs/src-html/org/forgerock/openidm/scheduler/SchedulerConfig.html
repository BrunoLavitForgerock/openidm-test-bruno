026package org.forgerock.openidm.scheduler;
027
028import java.util.Properties;
029import java.util.Set;
030
031import org.forgerock.json.fluent.JsonValue;
032import org.forgerock.openidm.quartz.impl.RepoJobStore;
033import org.quartz.impl.StdSchedulerFactory;
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036
037/**
038 * Holds a Scheduler's configuration
039 * 
040 * @author ckienle
041 */
042public class SchedulerConfig {
043    final static Logger logger = LoggerFactory.getLogger(SchedulerConfig.class);
044    
045    // Top level objects
046    private final static String ADVANCED_CONFIG = "advancedProperties";
047    private final static String THREADPOOL = "threadPool";
048    private final static String SCHEDULER = "scheduler";
049    
050    // Threadpool objects
051    private final static String THREAD_COUNT = "threadCount";
052    
053    // Scheduler objects
054    private final static String EXECUTE_PERSISTENT_SCHEDULES = "executePersistentSchedules";
055    
056    private String threadCount = null;
057    private String instanceId = null;
058    private boolean executePersistentSchedules = true;
059    
060    private Properties props;
061    
062    public SchedulerConfig(JsonValue config, String instanceId) {
063        if (config != null && !config.isNull()) {
064            setThreadpoolConfig(config.get(THREADPOOL));
065            setSchedulerConfig(config.get(SCHEDULER));
066            props = getAdvancedProps(config);
067        } else {
068            logger.debug("Config is null, defaults will be used");
069            props = new Properties();
070        }
071        setInstanceId(instanceId);
072    }
073
074    private void setThreadpoolConfig(JsonValue config) {
075        if (!config.isNull()) {
076            JsonValue value = config.get(THREAD_COUNT);
077            if (!value.isNull()) {
078                threadCount = value.asString();
079            }
080        }
081    }
082
083    private void setSchedulerConfig(JsonValue config) {
084        if (!config.isNull()) {
085            JsonValue value = config.get(EXECUTE_PERSISTENT_SCHEDULES);
086            if (!value.isNull()) {
087                if (value.isString()) {
088                    executePersistentSchedules = Boolean.parseBoolean(value.asString());
089                } else {
090                    executePersistentSchedules = value.asBoolean();
091                }
092            }
093        }
094    }
095
096    private Properties getAdvancedProps(JsonValue config) {
097        Properties props = new Properties();
098        JsonValue advancedConfig = config.get(ADVANCED_CONFIG);
099        if (!advancedConfig.isNull()) {
100            Set<String> keys = advancedConfig.keys();
101            for (String key : keys) {
102                props.put(key, advancedConfig.get(key).asString());
103            }
104        }
105        return props;
106    }
107
108    public boolean executePersistentSchedulesEnabled() {
109        return executePersistentSchedules;
110    }
111    
112    public String getInstanceName() {
113        return props.getProperty(StdSchedulerFactory.PROP_SCHED_INSTANCE_NAME, "DefaultQuartzScheduler");
114    }
115    
116    public void setInstanceName(String instanceName) {
117        props.setProperty(StdSchedulerFactory.PROP_SCHED_INSTANCE_NAME, instanceName);
118    }
119    
120    public String getInstanceId() {
121        if (instanceId != null) {
122            return instanceId;
123        }
124        return props.getProperty(StdSchedulerFactory.PROP_SCHED_INSTANCE_ID, "AUTO");
125    }
126    
127    public void setInstanceId(String instanceId) {
128        this.instanceId = instanceId;
129    }
130    
131    public String getRmiExport() {
132        return props.getProperty(StdSchedulerFactory.PROP_SCHED_RMI_EXPORT, "false");
133    }
134    
135    public void setRmiExport(String rmiExport) {
136        props.setProperty(StdSchedulerFactory.PROP_SCHED_RMI_EXPORT, rmiExport);
137    }
138    
139    public String getRmiProxy() {
140        return props.getProperty(StdSchedulerFactory.PROP_SCHED_RMI_PROXY, "false");
141    }
142    
143    public void setRmiProxy(String rmiProxy) {
144        props.setProperty(StdSchedulerFactory.PROP_SCHED_RMI_PROXY, rmiProxy);
145    }
146    
147    public String getWrapJobExecutionInUserTransaction() {
148        return props.getProperty(StdSchedulerFactory.PROP_SCHED_WRAP_JOB_IN_USER_TX, "false");
149    }
150    
151    public void setWrapJobExecutionInUserTransaction(
152            String wrapJobExecutionInUserTransaction) {
153        props.setProperty(StdSchedulerFactory.PROP_SCHED_WRAP_JOB_IN_USER_TX, wrapJobExecutionInUserTransaction);
154    }
155    
156    public String getThreadPoolClass() {
157        return props.getProperty(StdSchedulerFactory.PROP_THREAD_POOL_CLASS, "org.quartz.simpl.SimpleThreadPool");
158    }
159    
160    public void setThreadPoolClass(String threadPoolClass) {
161        StringBuilder prop = new StringBuilder(StdSchedulerFactory.PROP_THREAD_POOL_PREFIX).append(".threadCount");
162        props.setProperty(prop.toString(), threadPoolClass);
163    }
164    
165    public String getThreadPoolThreadCount() {
166        if (threadCount != null) {
167            return threadCount;
168        }
169        StringBuilder prop = new StringBuilder(StdSchedulerFactory.PROP_THREAD_POOL_PREFIX).append(".threadCount");
170        return props.getProperty(prop.toString(), "10");
171    }
172    
173    public void setThreadPoolThreadCount(String threadPoolThreadCount) {
174        threadCount = threadPoolThreadCount;
175    }
176    
177    public String getThreadPoolThreadPriority() {
178        StringBuilder prop = new StringBuilder(StdSchedulerFactory.PROP_THREAD_POOL_PREFIX).append(".threadPriority");
179        return props.getProperty(prop.toString(), "5");
180    }
181    
182    public void setThreadPoolThreadPriority(String threadPoolThreadPriority) {
183        StringBuilder prop = new StringBuilder(StdSchedulerFactory.PROP_THREAD_POOL_PREFIX).append(".threadPriority");
184        props.setProperty(prop.toString(), threadPoolThreadPriority);
185    }
186    
187    public String getThreadPoolThreadsInheritContextClassLoaderOfInitializingThread() {
188        return props.getProperty(
189                StdSchedulerFactory.PROP_SCHED_SCHEDULER_THREADS_INHERIT_CONTEXT_CLASS_LOADER_OF_INITIALIZING_THREAD, 
190                "true");
191    }
192    
193    public void setThreadPoolThreadsInheritCtxtCLOfInitThread(
194            String threadPoolThreadsInheritCtxtCLOfInitThread) {
195        props.setProperty(
196                StdSchedulerFactory.PROP_SCHED_SCHEDULER_THREADS_INHERIT_CONTEXT_CLASS_LOADER_OF_INITIALIZING_THREAD, 
197                threadPoolThreadsInheritCtxtCLOfInitThread);
198    }
199    
200    public String getJobStoreClass() {
201        return props.getProperty(StdSchedulerFactory.PROP_JOB_STORE_CLASS, "org.forgerock.openidm.quartz.impl.RepoJobStore");
202    }
203    
204    public void setJobStoreClass(String jobStoreClass) {
205        props.setProperty(StdSchedulerFactory.PROP_JOB_STORE_CLASS, jobStoreClass);
206    }
207
208    public Properties toProps() {
209        StringBuilder propThreadPoolPriority = new StringBuilder(StdSchedulerFactory.PROP_THREAD_POOL_PREFIX).append(".threadPriority");
210        StringBuilder propThreadPoolCount = new StringBuilder(StdSchedulerFactory.PROP_THREAD_POOL_PREFIX).append(".threadCount");
211        Properties props = new Properties(this.props);
212        props.put(StdSchedulerFactory.PROP_SCHED_INSTANCE_NAME, getInstanceName());
213        props.put(StdSchedulerFactory.PROP_SCHED_INSTANCE_ID, getInstanceId());
214        props.put(StdSchedulerFactory.PROP_SCHED_RMI_EXPORT, getRmiExport());
215        props.put(StdSchedulerFactory.PROP_SCHED_RMI_PROXY, getRmiProxy());
216        props.put(StdSchedulerFactory.PROP_SCHED_WRAP_JOB_IN_USER_TX, getWrapJobExecutionInUserTransaction());
217        props.put(StdSchedulerFactory.PROP_THREAD_POOL_CLASS, getThreadPoolClass());
218        props.put(propThreadPoolCount.toString(), getThreadPoolThreadCount());
219        props.put(propThreadPoolPriority.toString(), getThreadPoolThreadPriority());
220        props.put(StdSchedulerFactory.PROP_SCHED_SCHEDULER_THREADS_INHERIT_CONTEXT_CLASS_LOADER_OF_INITIALIZING_THREAD, 
221                getThreadPoolThreadsInheritContextClassLoaderOfInitializingThread());
222        props.put(StdSchedulerFactory.PROP_JOB_STORE_CLASS, getJobStoreClass());
223        return props;
224    }
225}
