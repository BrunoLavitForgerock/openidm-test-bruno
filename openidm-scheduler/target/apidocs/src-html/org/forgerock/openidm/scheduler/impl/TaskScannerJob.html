070    public TaskScannerJob(ConnectionFactory connectionFactory, TaskScannerContext context, ScriptRegistry scopeFactory)
071            throws ExecutionException, ScriptException {
072        this.connectionFactory = connectionFactory;
073        this.taskScannerContext = context;
074        this.scopeFactory = scopeFactory;
075
076        JsonValue scriptValue = context.getScriptValue();
077        if (!scriptValue.isNull()) {
078            this.script = scopeFactory.takeScript(scriptValue);
079        } else {
080            throw new ExecutionException("No valid script '" + scriptValue + "' configured in task scanner.");
081        }
082    }

084    /**
085     * Starts the task associated with a task scanner event.
086     * This method may run synchronously or launch a new thread depending upon the settings in the TaskScannerContext
087     * @return identifier associated with this task scan job
088     * @throws ExecutionException
089     */
090    public String startTask() throws ExecutionException {
091        int numberOfThreads = taskScannerContext.getNumberOfThreads();
092        final ExecutorService executor = Executors.newFixedThreadPool(numberOfThreads);
093        
094        if (taskScannerContext.getWaitForCompletion()) {
095            try {
096                performTask(executor);
097            } catch (ExecutionException ex) {
098                throw ex;
099            } finally {
100                executor.shutdown();
101            }
102        } else {
103            // Launch a new thread for the whole taskscan process
104            final ServerContext context = taskScannerContext.getContext();
105            Runnable command = new Runnable() {
106                @Override
107                public void run() {
108                    try {
109                        performTask(executor);
110                    } catch (Exception ex) {
111                        logger.warn("Taskscanner failed with unexpected exception", ex);
112                    } finally {
113                        executor.shutdown();
114                    }
115                }
116            };
117            new Thread(command).start();
118            // Shouldn't need to keep ahold of this, I don't think? Can just start it and let it go
119        }
120        return taskScannerContext.getTaskScanID();
121    }

123    /**
124     * Performs the task associated with the task scanner event.
125     * Runs the query and executes the script across each resulting object.
126     *
127     * @param invokerName name of the invoker
128     * @param scriptName name of the script associated with the task scanner event
129     * @param params parameters necessary for the execution of the script
130     * @throws ExecutionException
131     */
132    private void performTask(ExecutorService executor)
133            throws ExecutionException {
134        taskScannerContext.startJob();
135        logger.info("Task {} started from {} with script {}",
136                new Object[] { taskScannerContext.getTaskScanID(), taskScannerContext.getInvokerName(), taskScannerContext.getScriptName() });
137
138        JsonValue results;
139        taskScannerContext.startQuery();
140        try {
141            results = fetchAllObjects();
142        } catch (ResourceException e1) {
143            throw new ExecutionException("Error during query", e1);
144        }
145        taskScannerContext.endQuery();
146        Integer maxRecords = taskScannerContext.getMaxRecords();
147        if (maxRecords == null) {
148            taskScannerContext.setNumberOfTasksToProcess(results.size());
149        } else {
150            taskScannerContext.setNumberOfTasksToProcess(Math.min(results.size(), maxRecords));
151        }
152        logger.debug("TaskScan {} query results: {}", taskScannerContext.getInvokerName(), results.size());
153        // TODO jump out early if it's empty?
154
155        // Split and prune the result set according to our max and if we're synchronous or not
156        List<JsonValue> resultSets = splitResultsOverThreads(results, taskScannerContext.getNumberOfThreads(), maxRecords);
157        logger.debug("Split result set into {} units", resultSets.size());
158
159        List<Callable<Object>> todo = new ArrayList<Callable<Object>>();
160        for (final JsonValue result : resultSets) {
161            Runnable command = new Runnable() {
162                @Override
163                public void run() {
164                    try {
165                        performTaskOverSet(result);
166                    } catch (Exception ex) {
167                        logger.warn("Taskscanner failed with unexpected exception", ex);
168                    }
169                }
170            };
171            todo.add(Executors.callable(command));
172        }
173
174        try {
175            executor.invokeAll(todo);
176        } catch (InterruptedException e) {
177            // Mark it interrupted
178            taskScannerContext.interrupted();
179            logger.warn("Task scan '" + taskScannerContext.getTaskScanID() + "' interrupted");
180        }
181        // Don't mark the job as completed if its been deactivated
182        if (!taskScannerContext.isInactive()) {
183            taskScannerContext.endJob();
184        }
185
186        logger.info("Task '{}' completed. Total time: {}ms. Query time: {}ms. Progress: {}",
187                new Object[] { taskScannerContext.getTaskScanID(),
188                taskScannerContext.getStatistics().getJobDuration(),
189                taskScannerContext.getStatistics().getQueryDuration(),
190                taskScannerContext.getProgress()
191        });
192    }
193
194    private List<JsonValue> splitResultsOverThreads(JsonValue results, int numberOfThreads, Integer max) {
195        List<List<Object>> resultSets = new ArrayList<List<Object>>();
196        for (int i = 0; i < numberOfThreads; i++) {
197            resultSets.add(new ArrayList<Object>());
198        }
199
200        int i = 0;
201        for (JsonValue obj : results) {
202            if (max != null && i >= max) {
203                break;
204            }
205            resultSets.get(i % numberOfThreads).add(obj.getObject());
206            i++;
207        }
208
209        List<JsonValue> jsonSets = new ArrayList<JsonValue>();
210        for (List<Object> set : resultSets) {
211            jsonSets.add(new JsonValue(set));
212        }
213
214        return jsonSets;
215    }
216
217    private void performTaskOverSet(JsonValue results)
218                    throws ExecutionException {
219        for (JsonValue input : results) {
220            if (taskScannerContext.isCanceled()) {
221                logger.info("Task '" + taskScannerContext.getTaskScanID() + "' cancelled. Terminating execution.");
222                break; // Jump out quick since we've cancelled the job
223            }
224            // Check if this object has a STARTED time already
225            JsonValue startTime = input.get(taskScannerContext.getStartField());
226            String startTimeString = null;
227            if (startTime != null && !startTime.isNull()) {
228                startTimeString = startTime.asString();
229                DateTime startedTime = DATE_UTIL.parseTimestamp(startTimeString);
230
231                // Skip if the startTime + interval has not been passed
232                ReadablePeriod period = taskScannerContext.getRecoveryTimeout();
233                DateTime expirationDate = startedTime.plus(period);
234                if (expirationDate.isAfterNow()) {
235                    logger.debug("Object already started and has not expired. Started at: {}. Timeout: {}. Expires at: {}",
236                            new Object[] {
237                            DATE_UTIL.formatDateTime(startedTime),
238                            period,
239                            DATE_UTIL.formatDateTime(expirationDate)});
240                    continue;
241                }
242            }
243
244            try {
245                claimAndExecScript(input, startTimeString);
246            } catch (ResourceException e) {
247                throw new ExecutionException("Error during claim and execution phase", e);
248            }
249        }
250    }

252    /**
253     * Flatten a list of parameters and perform a query to fetch all objects from storage
254     * @param id the identifier of the resource to query
255     * @param params the parameters of the query
256     * @return JsonValue containing a list of all the retrieved objects
257     * @throws JsonResourceException
258     */
259    private JsonValue fetchAllObjects() throws ResourceException {
260        JsonValue flatParams = flattenJson(taskScannerContext.getScanValue());
261        ConfigMacroUtil.expand(flatParams);
262        return performQuery(taskScannerContext.getObjectID(), flatParams);
263    }

265    /**
266     * Performs a query on a resource and returns the result set
267     * @param resourceID the identifier of the resource to query
268     * @param params parameters to supply to the query
269     * @return the set of results from the performed query
270     * @throws JsonResourceException
271     */
272    private JsonValue performQuery(String resourceID, JsonValue params) throws ResourceException {
273        final JsonValue queryResults = new JsonValue(new ArrayList<Map<String, Object>>());;
274        QueryRequest queryRequest = Requests.newQueryRequest(resourceID);
275        queryRequest.setQueryId(params.get("_queryId").asString());
276        queryRequest.setQueryExpression(params.get("_queryExpression").asString());
277        for (Map.Entry<String, Object> e: params.asMap().entrySet()){
278            queryRequest.getAdditionalParameters().put(e.getKey(), String.valueOf(e.getValue()));
279        }
280        connectionFactory.getConnection().query(taskScannerContext.getContext(), queryRequest, new QueryResultHandler() {
281            @Override
282            public void handleError(ResourceException error) {
283                // Ignore
284            }
285
286            @Override
287            public boolean handleResource(Resource resource) {
288                queryResults.add(resource.getContent().getObject());
289                return true;
290            }
291
292            @Override
293            public void handleResult(QueryResult result) {
294                // Ignore
295            }
296        });
297        return queryResults;
298    }

313    /**
314     * Adds an object to a JsonValue and performs an update
315     * @param resourceID the resource identifier that the updated value belongs to
316     * @param value value to perform the update with
317     * @param path JsonPointer to the updated/added field
318     * @param obj object to add to the field
319     * @return the updated JsonValue
320     * @throws JsonResourceException
321     */
322    private JsonValue updateValueWithObject(String resourceID, JsonValue value, JsonPointer path, Object obj) throws ResourceException {
323        ensureJsonPointerExists(path, value);
324        value.put(path, obj);
325        return performUpdate(resourceID, value);
326    }

328    /**
329     * Performs an update on a given resource with a supplied JsonValue
330     * @param resourceID the resource identifier to perform the update on
331     * @param value the object to update with
332     * @return the updated object
333     * @throws JsonResourceException
334     */
335    private JsonValue performUpdate(String resourceID, JsonValue value) throws ResourceException {
336        String id = value.get("_id").required().asString();
337        String fullID = retrieveFullID(resourceID, value);
338        String rev = value.get("_rev").required().asString();
339        UpdateRequest updateRequest = Requests.newUpdateRequest(fullID, value);
340        updateRequest.setRevision(rev);
341
342        connectionFactory.getConnection().update(taskScannerContext.getContext(), updateRequest);
343        return retrieveObject(resourceID, id);
344    }

390    private void claimAndExecScript(JsonValue input, String expectedStartDateStr)
391            throws ExecutionException, ResourceException {
392        String id = input.get("_id").required().asString();
393        boolean claimedTask = false;
394        boolean retryClaimTask = false;
395
396        JsonPointer startField = taskScannerContext.getStartField();
397        JsonPointer completedField = taskScannerContext.getCompletedField();
398        String resourceID = taskScannerContext.getObjectID();
399
400        JsonValue _input = input;
401        do {
402            try {
403                retryClaimTask = false;
404                _input = updateValueWithObject(resourceID, _input, startField, DATE_UTIL.now());
405                _input = updateValueWithObject(resourceID, _input, completedField, null);
406                logger.debug("Claimed task and updated StartField: {}", _input);
407                claimedTask = true;
408            } catch (PreconditionFailedException ex) {
409                    // If the object changed since we queried, get the latest
410                    // and check if it's still in a state we want to process the task.
411                    _input = retrieveObject(resourceID, id);
412                    String currentStartDateStr = (_input.get(startField) == null)  ? null : _input.get(startField).asString();
413                    String currentCompletedDateStr = (_input.get(completedField) == null)  ? null : _input.get(completedField).asString();
414                    if (currentCompletedDateStr == null && (currentStartDateStr == null || currentStartDateStr.equals(expectedStartDateStr))) {
415                        retryClaimTask = true;
416                    } else {
417                        // Someone else managed to update the started field first,
418                        // claimed the task. Do not execute it here this run.
419                        logger.debug("Task for {} {} was already claimed, ignore.", resourceID, id);
420                    }
421            }
422        } while (retryClaimTask && !taskScannerContext.isCanceled());
423        if (claimedTask) {
424            execScript(_input);
425        }
426    }

428    /**
429     * Performs the individual executions of the supplied script
430     *
431     * Passes <b>"input"</b> and <b>"objectID"</b> to the script.<br>
432     *   <b>"objectID"</b> contains the full ID of the supplied object (including resource identifier).
433     *      Useful for performing updates.<br>
434     *   <b>"input"</b> contains the supplied object
435     *
436     * @param scriptName name of the script
437     * @param invokerName name of the invoker
438     * @param script the script to execute
439     * @param resourceID the resource identifier for the object that the script will be performed on
440     * @param startField JsonPointer to the field that will be marked at script start
441     * @param completedField JsonPointer to the field that will be marked at script completion
442     * @param input value to input to the script
443     * @throws ExecutionException
444     * @throws JsonResourceException
445     */
446    private void execScript(JsonValue input)
447            throws ExecutionException, ResourceException {
448        if (script != null) {
449            String resourceID = taskScannerContext.getObjectID();
450            ServerContext context = taskScannerContext.getContext();
451
452            try {
453                Script scope = script.getScript(context);
454                scope.put("input", input.getObject());
455                scope.put("objectID", retrieveFullID(resourceID, input));
456
457                Object returnedValue = scope.eval();
458                JsonValue _input = retrieveUpdatedObject(resourceID, input);
459                logger.debug("After script execution: {}", _input);
460
461                if (returnedValue == Boolean.TRUE) {
462                   _input = updateValueWithObject(resourceID, _input, taskScannerContext.getCompletedField(), DATE_UTIL.now());
463                   taskScannerContext.getStatistics().taskSucceded();
464                   logger.debug("Updated CompletedField: {}", _input);
465                } else {
466                    taskScannerContext.getStatistics().taskFailed();
467                }
468
469            } catch (ScriptException se) {
470                taskScannerContext.getStatistics().taskFailed();
471                String msg = taskScannerContext.getScriptName() + " script invoked by " +
472                        taskScannerContext.getInvokerName() + " encountered exception";
473                logger.debug(msg, se);
474                throw new ExecutionException(msg, se);
475            }
476        }
477    }

485        return flattenJson("", original);

488    /**
489     * Flattens JsonValue into a one-level-deep object
490     * @param parent name of the parent object (for nested objects)
491     * @param original original JsonValue object
492     * @return flattened JsonValue
493     */
494    private static JsonValue flattenJson(String parent, JsonValue original) {
495        JsonValue flattened = new JsonValue(new HashMap<String, Object>());
496        Iterator<String> iter = original.keys().iterator();
497        while (iter.hasNext()) {
498            String oKey = iter.next();
499            String key = (parent.isEmpty() ? "" : parent + ".") + oKey;
500
501            JsonValue value = original.get(oKey);
502            if (value.isMap()) {
503                addAllToJson(flattened, flattenJson(key, value));
504            } else {
505                flattened.put(key, value.getObject());
506            }
507        }
508        return flattened;
509    }

511    /**
512     * Adds all objects from one JsonValue to another (performs a merge).
513     * Any values contained in both objects will be overwritten to reflect the values in <b>from</b>
514     * <br><br>
515     * <i><b>NOTE:</b> this should be a part of JsonValue itself (so we can support merging two JsonValue objects)</i>
516     * @param to JsonValue that will have objects added to it
517     * @param from JsonValue that will be used as reference for updating
518     */
519    private static void addAllToJson(JsonValue to, JsonValue from) {
520        Iterator<String> iter = from.keys().iterator();
521        while (iter.hasNext()) {
522            String key = iter.next();
523            to.put(key, from.get(key).getObject());
524        }
525    }

527    /**
528     * Ensure that some JsonPointer exists within a supplied object so that some object can be placed in that field
529     * @param ptr JsonPointer to ensure exists at each level
530     * @param obj object to ensure the JsonPointer exists within
531     */
532    private static void ensureJsonPointerExists(JsonPointer ptr, JsonValue obj) {
533        JsonValue refObj = obj;
534
535        for (String p : ptr) {
536            if (!refObj.isDefined(p)) {
537                refObj.put(p, new JsonValue(new HashMap<String, Object>()));
538            }
539            refObj = refObj.get(p);
540        }
