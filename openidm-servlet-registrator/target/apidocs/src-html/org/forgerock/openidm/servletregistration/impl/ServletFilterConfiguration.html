017package org.forgerock.openidm.servletregistration.impl;
018
019import java.util.HashMap;
020import java.util.Map;
021
022import org.apache.felix.scr.annotations.Activate;
023import org.apache.felix.scr.annotations.Component;
024import org.apache.felix.scr.annotations.ConfigurationPolicy;
025import org.apache.felix.scr.annotations.Deactivate;
026import org.apache.felix.scr.annotations.Reference;
027import org.apache.felix.scr.annotations.ReferenceCardinality;
028import org.apache.felix.scr.annotations.ReferencePolicy;
029import org.forgerock.json.fluent.JsonValue;
030import org.forgerock.openidm.config.enhanced.JSONEnhancedConfig;
031import org.forgerock.openidm.servletregistration.RegisteredFilter;
032import org.forgerock.openidm.servletregistration.ServletRegistration;
033import static org.forgerock.openidm.servletregistration.ServletRegistration.SERVLET_FILTER_SYSTEM_PROPERTIES;
034import org.osgi.service.component.ComponentContext;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037
038/**
039 * Takes configuration to register and de-register configured servlet filters,
040 * with support to load the filter or supporting classes off a defined class path.
041 *
042 * @author aegloff
043 * @author ckienle
044 */
045@Component(
046        name = "org.forgerock.openidm.servletfilter",
047        immediate = true,
048        policy = ConfigurationPolicy.REQUIRE,
049        configurationFactory=true
050)
051public class ServletFilterConfiguration {
052    private final static Logger logger = LoggerFactory.getLogger(ServletFilterConfiguration.class);
053
054    // Handle to registered servlet filter
055    private RegisteredFilter registeredFilter;
056
057    // Original setting of system properties
058    Map<String, String> origSystemProperties = new HashMap<String, String>();
059
060    @Reference(
061            name = "ref_ServletFilterRegistration",
062            referenceInterface = ServletRegistration.class,
063            policy = ReferencePolicy.DYNAMIC,
064            cardinality = ReferenceCardinality.MANDATORY_UNARY,
065            bind = "bindServletFilterRegistration",
066            unbind = "unbindServletFilterRegistration"
067    )
068    private ServletRegistration servletFilterRegistration;
069    private void bindServletFilterRegistration(ServletRegistration servletFilterRegistration) {
070        this.servletFilterRegistration = servletFilterRegistration;
071    }
072    private void unbindServletFilterRegistration(ServletRegistration servletFilterRegistration) {
073        this.servletFilterRegistration = null;
074    }
075
076    /**
077     * Parses the servlet filter configuration and registers a servlet filter in OSGi.
078     *
079     * @param context The ComponentContext.
080     * @throws Exception If a problem occurs whilst registering the filter.
081     */
082    @Activate
083    protected synchronized void activate(ComponentContext context) throws Exception {
084        logger.info("Activating servlet registrator with configuration {}", context.getProperties());
085        JsonValue config = new JSONEnhancedConfig().getConfigurationAsJson(context);
086
087        logger.debug("Parsed servlet filter config: {}", config);
088
089        registeredFilter = servletFilterRegistration.registerFilter(config);
090        logger.info("Successfully registered servlet filter {}", context.getProperties());
091
092        origSystemProperties = new HashMap<String, String>();
093        JsonValue rawSystemProperties = config.get(SERVLET_FILTER_SYSTEM_PROPERTIES);
094        for (String key : rawSystemProperties.keys()) {
095            String prev = System.setProperty(key, rawSystemProperties.get(key).asString());
096            // null value is used to keep track of properties that weren't set before
097            origSystemProperties.put(key, prev);
098        }
099    }
100
101    /**
102     * De-registers the corresponding servlet filter.
103     *
104     * @param context The ComponentContext.
105     */
106    @Deactivate
107    protected synchronized void deactivate(ComponentContext context) {
108        if (registeredFilter != null) {
109            try {
110                servletFilterRegistration.unregisterFilter(registeredFilter);
111                logger.info("Unregistered servlet filter {}.", context.getProperties());
112            } catch (Exception ex) {
113                logger.warn("Failure reported during unregistering of servlet filter {}: {}",
114                        new Object[]{context.getProperties(), ex.getMessage(), ex});
115            }
116        }
117
118        // Restore the system properties before this config was applied
119        for (String key : origSystemProperties.keySet()) {
120            String val = origSystemProperties.get(key);
121            if (val == null) {
122                System.clearProperty(key);
123            } else {
124                System.setProperty(key, val);
125            }
126        }
127        logger.debug("Deactivated {}", context);
128    }
129}
