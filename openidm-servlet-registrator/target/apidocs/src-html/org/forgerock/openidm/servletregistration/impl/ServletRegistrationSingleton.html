067/**
068 * Takes configuration to register and de-register configured servlet filters, 
069 * with support to load the filter or supporting classes off a defined class path

075@Component(

077    immediate = true,

133    /**
134     * {@inheritDoc}
135     */
136    @Override
137    public RegisteredFilter registerFilter(JsonValue config) throws Exception {
138        synchronized (registrationLock) {
139            RegisteredFilterImpl newFilter = new RegisteredFilterImpl(config);
140            // Unregister existing filters if necessary
141            for (RegisteredFilter regFilter : filters) {
142                unregisterFilterWithWebContainer(regFilter.getFilter());
143            }
144            // Add new filter
145            filters.add(newFilter);
146            // Sort the filters
147            Collections.sort(filters);
148            // Register all filters
149            Exception failedRegistrationException = null;
150            for (RegisteredFilterImpl regFilter : filters) {
151                try {
152                    regFilter.setFilter(registerFilterWithWebContainer(regFilter.getConfig()));
153                } catch (Exception e) {
154                    logger.error("Error registering filter {}", config);
155                    if (regFilter.equals(newFilter)) {
156                        failedRegistrationException = e;
157                    }
158                }
159            }
160            // Throw the exception if the new filter failed to register
161            if (failedRegistrationException != null) {
162                filters.remove(newFilter);
163                throw failedRegistrationException;
164            }
165            logger.info("Successfully registered servlet filter {}", config);
166            // Register ServletFilterRegistrator service
167            registerFilterService(newFilter.getConfig());
168            return newFilter;
169        }
170    }
171    
172    /**
173     * Registers a servlet filter configuration
174     * @param config the filter configuration
175     * @return the registered Filter
176     * @throws Exception
177     */
178    private Filter registerFilterWithWebContainer(JsonValue config) throws Exception {
179        // Get required info from config
180        String filterClass = config.get(SERVLET_FILTER_CLASS).required().asString();
181        logger.info("Using filter class: {}", filterClass);
182        List<URL> urls = config.get(SERVLET_FILTER_CLASS_PATH_URLS).asList(
183                new Function<JsonValue, URL, JsonValueException>() {
184                    @Override
185                    public URL apply(JsonValue jsonValue) throws JsonValueException {
186                        return jsonValue.asURL();
187                    }
188                });
189        logger.info("Added URLs { {} })) to filter classpath", StringUtils.join(urls, ", "));
190
191        Map<String, Object> preInvokeReqAttributes = config.get(SERVLET_FILTER_PRE_INVOKE_ATTRIBUTES).asMap();
192        
193        // Servlet names this filter should apply to, e.g. one could also add "OpenIDM Web"
194        List<String> servletNames = config.get(SERVLET_FILTER_SERVLET_NAMES)
195                .defaultTo(Arrays.asList(DEFAULT_SERVLET_NAME))
196                .asList(String.class);
197
198        // URL patterns to apply the filter to, e.g. one could also add "/openidmui/*");
199        List<String> urlPatterns = config.get(SERVLET_FILTER_URL_PATTERNS)
200                .defaultTo(Arrays.asList(DEFAULT_SERVLET_URL_PATTERNS))
201                .asList(String.class);
202
203        // Filter init params, a string to string map
204        JsonValue rawInitParams = config.get("initParams");
205        Map<String, String> initParams = null;
206        if (!rawInitParams.isNull()) {
207            initParams = new HashMap<String, String>();
208        }
209        for (String initParamKey : rawInitParams.keys()) {
210            initParams.put(initParamKey, rawInitParams.get(initParamKey).asString());
211        }
212
213        // Create a classloader and dynamically create the requested filter
214        Filter filter = null; 
215        ClassLoader filterCL = null;
216        ClassLoader origCL = Thread.currentThread().getContextClassLoader();
217        try {
218            filterCL = new URLClassLoader(urls.toArray(new URL[0]), this.getClass().getClassLoader());
219            Thread.currentThread().setContextClassLoader(filterCL);
220            filter = (Filter)(Class.forName(filterClass, true, filterCL).newInstance());
221        } catch (Exception ex) {
222            logger.warn("Configured class {} failed to load from configured class path URLs {}", new Object[] {filterClass, urls, ex});
223            throw ex;
224        } finally {
225            Thread.currentThread().setContextClassLoader(origCL);
226        }
227
228        // Create filter
229        Filter proxiedFilter = (Filter) Proxy.newProxyInstance(
230                filter.getClass().getClassLoader(),
231                new Class[] { Filter.class },
232                new FilterProxy(filter, filterCL, preInvokeReqAttributes));
233
234        // Register filter
235        webContainer.registerFilter(proxiedFilter,
236                urlPatterns.toArray(new String[urlPatterns.size()]),
237                servletNames.toArray(new String[servletNames.size()]),
238                new Hashtable<String, Object>(initParams),
239                webContainer.getDefaultSharedHttpContext());
240        return proxiedFilter;
241    }
242    
243    private void registerFilterService(JsonValue config) {
244        // Register ServletFilterRegistrator service
245        ServletFilterRegistrator servletFilterRegistrator = new ServletFilterRegistratorSvc(config);
246        bundleContext.registerService(ServletFilterRegistrator.class.getName(), servletFilterRegistrator, null);
247    }
248
249    @Override
250    public void unregisterFilter(RegisteredFilter filter) throws Exception {
251        synchronized (registrationLock) {
252            unregisterFilterWithWebContainer(filter.getFilter());
253            filters.remove(filter);
254            logger.info("Successfully unregistered servlet filter {}", filter);
255        }
256    }
257    
258    public void unregisterFilterWithWebContainer(Filter filter) {
259        webContainer.unregisterFilter(filter);
260    }

262    /**
263     * Wraps Filter to set the thread context classloader
264     * @author aegloff
265     */
266    private static class FilterProxy implements InvocationHandler {
267        Filter filter;
268        ClassLoader threadCtxClassloader;
269        Map<String, Object> preInvokeReqAttributes;
270        /**
271         * Sets the thread context for a filter invocation to the desired value
272         * @param filter filter to wrap
273         * @param threadCtxClassloader classloader to set as thread context classloader
274         * @param preInvokeReqAttributes request attributes to set before the filter doFilter is invoked 
275         */
276        public FilterProxy(Filter filter, ClassLoader threadCtxClassloader, 
277                Map<String, Object> preInvokeReqAttributes) { 
278            this.filter = filter;
279            this.threadCtxClassloader = threadCtxClassloader;
280            this.preInvokeReqAttributes = preInvokeReqAttributes;
281        }
282
283        public Object invoke(Object proxy, Method m, Object[] args) throws Throwable {
284            ClassLoader origCL = Thread.currentThread().getContextClassLoader();
285            try {
286                Thread.currentThread().setContextClassLoader(threadCtxClassloader);
287                if (preInvokeReqAttributes != null) {
288                    if (m.getName().equals("doFilter") && args != null && args.length == 3 && args[0] instanceof ServletRequest) {
289                        ServletRequest request = (ServletRequest) args[0];
290                        for (Map.Entry<String, Object> entry : preInvokeReqAttributes.entrySet()) {
291                            logger.trace("Adding attribute to doFilter request {}", entry);
292                            request.setAttribute(entry.getKey(), entry.getValue());
293                        }
294                    }
295                }
296                Object val = m.invoke(filter, args);
297                return val;
298            } catch (InvocationTargetException e) {
299                logger.debug("Filter invocation InvocationTargetException", e.getTargetException());
300                throw e.getTargetException();
301            } catch (Exception e) {
302                logger.debug("Filter invocation Exception", e);
303                throw e;
304            } finally {
305                Thread.currentThread().setContextClassLoader(origCL);
306            }
307        }
308    }
