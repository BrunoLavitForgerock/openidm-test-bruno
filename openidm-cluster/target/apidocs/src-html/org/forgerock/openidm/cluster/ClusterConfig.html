001package org.forgerock.openidm.cluster;
002
003import java.util.Properties;
004import java.util.Set;
005
006import org.forgerock.json.fluent.JsonValue;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010public class ClusterConfig {
011    final static Logger logger = LoggerFactory.getLogger(ClusterConfig.class);
012
013    
014    // Scheduler objects
015    private final static String INSTANCE_ID = "instanceId";
016    private final static String INSTANCE_TIMEOUT = "instanceTimeout";
017    private final static String INSTANCE_RECOVERY_TIMEOUT = "instanceRecoveryTimeout";
018    private final static String INSTANCE_CHECK_IN_INTERVAL = "instanceCheckInInterval";
019    private final static String INSTANCE_CHECK_IN_OFFSET = "instanceCheckInOffset";
020    private final static String ENABLED = "enabled";
021    
022    private String instanceId = "instance0";
023    private long instanceTimeout = 30000;
024    private long instanceRecoveryTimeout = 30000;
025    private long instanceCheckInInterval = 5000;
026    private long instanceCheckInOffset = 0;
027    private boolean enabled = true;
028    
029    public ClusterConfig(JsonValue config) {
030        if (!config.isNull()) {
031            JsonValue value = config.get(INSTANCE_ID);
032            if (!value.isNull()) {
033                instanceId = value.asString();
034            }
035            value = config.get(INSTANCE_TIMEOUT);
036            if (!value.isNull()) {
037                setInstanceTimeout(Long.parseLong(value.asString()));
038            }
039            value = config.get(INSTANCE_RECOVERY_TIMEOUT);
040            if (!value.isNull()) {
041                setInstanceRecoveryTimeout(Long.parseLong(value.asString()));
042            }
043            value = config.get(INSTANCE_CHECK_IN_INTERVAL);
044            if (!value.isNull()) {
045                setInstanceCheckInInterval(Long.parseLong(value.asString()));
046            }
047            value = config.get(INSTANCE_CHECK_IN_OFFSET);
048            if (!value.isNull()) {
049                setInstanceCheckInOffset(Long.parseLong(value.asString()));
050            }
051            value = config.get(ENABLED);
052            if (!value.isNull() && value.isBoolean()) {
053                setEnabled(value.asBoolean());
054            } else if (!value.isNull() && value.isString()) {
055                setEnabled(Boolean.parseBoolean(value.asString()));
056            }
057        }
058    }
059    
060    public String getInstanceId() {
061            return instanceId;
062    }
063    
064    public void setInstanceId(String instanceId) {
065        this.instanceId = instanceId;
066    }
067
068    public Properties toProps() {
069        Properties props = new Properties();
070        props.put(INSTANCE_TIMEOUT, getInstanceTimeout());
071        props.put(INSTANCE_RECOVERY_TIMEOUT, getInstanceRecoveryTimeout());
072        props.put(INSTANCE_CHECK_IN_INTERVAL, getInstanceCheckInInterval());
073        props.put(INSTANCE_CHECK_IN_OFFSET, getInstanceCheckInOffset());
074        props.put(INSTANCE_ID, getInstanceId());
075        return props;
076    }
077
078    public long getInstanceTimeout() {
079        return instanceTimeout;
080    }
081
082    public void setInstanceTimeout(long instanceTimeout) {
083        this.instanceTimeout = instanceTimeout;
084    }
085
086    public long getInstanceRecoveryTimeout() {
087        return instanceRecoveryTimeout;
088    }
089
090    public void setInstanceRecoveryTimeout(long instanceRecoveryTimeout) {
091        this.instanceRecoveryTimeout = instanceRecoveryTimeout;
092    }
093
094    public long getInstanceCheckInInterval() {
095        return instanceCheckInInterval;
096    }
097
098    public void setInstanceCheckInInterval(long instanceCheckInInterval) {
099        this.instanceCheckInInterval = instanceCheckInInterval;
100    }
101
102    public long getInstanceCheckInOffset() {
103        return instanceCheckInOffset;
104    }
105
106    public void setInstanceCheckInOffset(long instanceCheckInOffset) {
107        this.instanceCheckInOffset = instanceCheckInOffset;
108    }
109
110    public boolean isEnabled() {
111        return enabled;
112    }
113
114    public void setEnabled(boolean enabled) {
115        this.enabled = enabled;
116    }
117}
