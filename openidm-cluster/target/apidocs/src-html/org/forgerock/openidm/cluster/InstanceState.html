001package org.forgerock.openidm.cluster;
002
003import java.util.HashMap;
004import java.util.Map;
005
006public class InstanceState {
007    
008    public final static int STATE_RUNNING = 1;
009    public final static int STATE_PROCESSING_DOWN = 2;
010    public final static int STATE_DOWN = 3;
011
012    public final static String PROP_INSTANCE_ID                 = "instanceId";
013    public final static String PROP_RECOVERING_INSTANCE_ID      = "recoveringInstanceId";
014    public final static String PROP_STATE                       = "state";
015    public final static String PROP_TIMESTAMP_LEASE             = "timestamp";
016    public final static String PROP_TIMESTAMP_STARTUP           = "startup";
017    public final static String PROP_TIMESTAMP_SHUTDOWN          = "shutdown";
018    public final static String PROP_TIMESTAMP_DETECTED_DOWN     = "detectedDown";
019    public final static String PROP_TIMESTAMP_RECOVERY          = "recoveringTimestamp";
020    public final static String PROP_TIMESTAMP_RECOVERY_STARTED  = "recoveryStarted";
021    public final static String PROP_TIMESTAMP_RECOVERY_FINISHED = "recoveryFinished";
022    public final static String PROP_RECOVERY_ATTEMPTS           = "recoveryAttempts";
023    public final static String PROP_REV                         = "_rev";
024    public final static String PROP_ID                          = "_id";
025    
026    private String instanceId;
027    private int state;
028    private int recoveryAttempts;
029    private long startup;
030    private long shutdown;
031    private long timestamp;
032    private long detectedDown;
033    private long recoveryStarted;
034    private long recoveryFinished;
035    private long recoveringTimestamp;
036    private String recoveringInstanceId;
037    private String rev;
038    private String id;
039    
040    public InstanceState(String instanceId, Map<String, Object> map) {
041        this.instanceId = instanceId;
042        this.recoveringInstanceId = (String)map.get(PROP_RECOVERING_INSTANCE_ID);
043        this.state = ((map.get(PROP_STATE) == null) ? STATE_RUNNING : (Integer)map.get(PROP_STATE));
044        this.timestamp = ((map.get(PROP_TIMESTAMP_LEASE) == null) ? System.currentTimeMillis() : 
045            Long.parseLong((String)map.get(PROP_TIMESTAMP_LEASE)));
046        this.startup = ((map.get(PROP_TIMESTAMP_STARTUP) == null) ? System.currentTimeMillis() : 
047            Long.parseLong((String)map.get(PROP_TIMESTAMP_STARTUP)));
048        this.shutdown = ((map.get(PROP_TIMESTAMP_SHUTDOWN) == null) ? 0L : 
049            Long.parseLong((String)map.get(PROP_TIMESTAMP_SHUTDOWN)));
050        this.detectedDown = ((map.get(PROP_TIMESTAMP_DETECTED_DOWN) == null) ? 0L : 
051            Long.parseLong((String)map.get(PROP_TIMESTAMP_DETECTED_DOWN)));
052        this.recoveringTimestamp = ((map.get(PROP_TIMESTAMP_RECOVERY) == null) ? 0L : 
053            Long.parseLong((String)map.get(PROP_TIMESTAMP_RECOVERY)));
054        this.recoveryStarted = ((map.get(PROP_TIMESTAMP_RECOVERY_STARTED) == null) ? 0L : 
055            Long.parseLong((String)map.get(PROP_TIMESTAMP_RECOVERY_STARTED)));
056        this.recoveryFinished = ((map.get(PROP_TIMESTAMP_RECOVERY_FINISHED) == null) ? 0L : 
057            Long.parseLong((String)map.get(PROP_TIMESTAMP_RECOVERY_FINISHED)));
058        this.recoveryAttempts = ((map.get(PROP_RECOVERY_ATTEMPTS) == null) ? 0 : 
059            (Integer)map.get(PROP_RECOVERY_ATTEMPTS));
060        this.rev = (String)map.get(PROP_REV);
061        this.id = (String)map.get(PROP_ID);
062    }
063    
064    public InstanceState(String instanceId) {
065        this.instanceId = instanceId;
066        this.state = STATE_RUNNING;
067        this.timestamp = System.currentTimeMillis();
068        this.startup = System.currentTimeMillis();
069        this.shutdown = 0L;
070        this.recoveringTimestamp = 0L;
071        this.recoveringInstanceId = null;
072        this.recoveryStarted = 0L;
073        this.recoveryFinished = 0L;
074        this.detectedDown = 0L;
075        this.recoveryAttempts = 0;
076    }
077
078    public Map<String, Object> toMap() {
079        Map<String, Object> map = new HashMap<String, Object>();
080        map.put(PROP_INSTANCE_ID, getInstanceId());
081        map.put(PROP_STATE, getState());
082        map.put(PROP_TIMESTAMP_LEASE, pad(new Long(getTimestamp())));
083        map.put(PROP_TIMESTAMP_STARTUP, pad(new Long(getStartup())));
084        map.put(PROP_TIMESTAMP_SHUTDOWN, pad(new Long(getShutdown())));
085        map.put(PROP_TIMESTAMP_DETECTED_DOWN, pad(new Long(getDetectedDown())));
086        map.put(PROP_RECOVERING_INSTANCE_ID, getRecoveringInstanceId());
087        map.put(PROP_TIMESTAMP_RECOVERY, pad(new Long(getRecoveringTimestamp())));
088        map.put(PROP_TIMESTAMP_RECOVERY_STARTED, pad(getRecoveryStarted()));
089        map.put(PROP_TIMESTAMP_RECOVERY_FINISHED, pad(getRecoveryFinished()));
090        map.put(PROP_RECOVERY_ATTEMPTS, getRecoveryAttempts());
091        map.put(PROP_REV, getRevision());
092        map.put(PROP_ID, id);
093        return map;
094    }
095    
096    public boolean hasFailed(long timeout) {
097        if (System.currentTimeMillis() - timestamp > timeout) {
098            return true;
099        }
100        return false;
101    }
102    
103    public boolean hasRecoveringFailed(long timout) {
104        if (state == STATE_PROCESSING_DOWN && 
105                (System.currentTimeMillis() - recoveringTimestamp > timout)) {
106            return true;
107        }
108        return false;
109    }
110    
111    public boolean hasShutdown() {
112        if (shutdown == 0L) {
113            return false;
114        }
115        return true;
116    }
117
118    public void updateStartup() {
119        this.startup = System.currentTimeMillis();
120    }
121
122    public void updateShutdown() {
123        this.shutdown = System.currentTimeMillis();
124    }
125
126    public void updateTimestamp() {
127        timestamp = System.currentTimeMillis();
128    }
129    
130    public void updateRecoveringTimestamp() {
131        recoveringTimestamp = System.currentTimeMillis();
132    }
133    
134    public void updateDetectedDown() {
135        detectedDown = System.currentTimeMillis();
136    }
137
138    public String getInstanceId() {
139        return instanceId;
140    }
141    
142    public int getState() {
143        return state;
144    }
145
146    public long getTimestamp() {
147        return timestamp;
148    }
149
150    public long getStartup() {
151        return startup;
152    }
153
154    public long getShutdown() {
155        return shutdown;
156    }
157    
158    public long getDetectedDown() {
159        return detectedDown;
160    }
161
162    public long getRecoveringTimestamp() {
163        return recoveringTimestamp;
164    }
165
166    public String getRecoveringInstanceId() {
167        return recoveringInstanceId;
168    }
169
170    public int getRecoveryAttempts() {
171        return recoveryAttempts;
172    }
173    
174    public long getUptime() {
175        return System.currentTimeMillis() - startup;
176    }
177    
178    public String getRevision() {
179        return rev;
180    }
181    
182    public void clearShutdown() {
183        shutdown = 0L;
184    }
185    
186    public void clearDetectedDown() {
187        detectedDown = 0L;
188    }
189    
190    public void clearRecoveryAttempts() {
191        recoveryAttempts = 0;
192    }
193
194    public void setState(int state) {
195        this.state = state;
196    }
197
198    public void setInstanceId(String instanceId) {
199        this.instanceId = instanceId;
200    }
201
202    public void setRecoveringInstanceId(String recoveringInstanceId) {
203        this.recoveringInstanceId = recoveringInstanceId;
204    }
205
206    public long getRecoveryStarted() {
207        return recoveryStarted;
208    }
209
210    public void startRecovery() {
211        recoveryStarted = System.currentTimeMillis();
212        recoveryAttempts++;
213    }
214
215    public long getRecoveryFinished() {
216        return recoveryFinished;
217    }
218
219    public void finishRecovery() {
220        this.recoveryFinished = System.currentTimeMillis();
221    }
222    
223    public static String pad(long l) {
224        return String.format("%019d", l);
225    }
226}
