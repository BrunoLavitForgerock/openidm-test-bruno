037import org.slf4j.LoggerFactory;
038
039/**
040 * Apply Jetty security constraints of applicable
041 *
042 * @author aegloff
043 */
044public final class JettySecurityConfigurator implements SecurityConfigurator {
045    final static Logger logger = LoggerFactory.getLogger(JettySecurityConfigurator.class);
046
047    static String enabled = null;
048    static {
049         enabled = System.getProperty("openidm.jetty.jaas.enabled", "false");
050    }
051    
052    public JettySecurityConfigurator() {
053    }
054
055    public void activate(HttpService httpService, HttpContext httpContext,  ComponentContext context) {
056        
057        // TODO: consider making configurable
058
059        // If embedded Jetty is used, enable security
060        if (enabled.equals("true") && httpService instanceof org.ops4j.pax.web.service.WebContainer) {
061            org.ops4j.pax.web.service.WebContainer webContainer = (org.ops4j.pax.web.service.WebContainer) httpService;
062/*
063            String authMethod = "FORM";
064            String realmName = "openidm";
065            String formLoginPage = "/login/login";
066            String formErrorPage = "/login/error";
067*/
068            String authMethod = "BASIC";
069            String realmName = "openidm";
070            String formLoginPage = null;
071            String formErrorPage = null;
072            webContainer.registerLoginConfig(authMethod, realmName, formLoginPage, formErrorPage, httpContext);
073    
074            String constraintName="openidm-core-login";
075            String url = "/openidm/*";
076            String mapping = null;
077            String dataConstraint = null;
078            boolean authentication = true;
079            java.util.List<String> roles = new java.util.ArrayList<String>();
080            roles.add("openidm-authorized");
081            //roles.add("*");
082            
083            webContainer.registerConstraintMapping(constraintName, url, mapping, dataConstraint,
084                    authentication, roles, httpContext); 
085            logger.debug("Enabled Jetty security for OpenIDM shared context");
086        }
087    }
088
089    public void deactivate(HttpService httpService, HttpContext httpContext,  ComponentContext context) {
090    }
091}
