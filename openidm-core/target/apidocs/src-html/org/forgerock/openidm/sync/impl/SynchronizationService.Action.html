134    @Activate
135    protected void activate(ComponentContext context) {
136        JsonValue config = new JsonValue(new JSONEnhancedConfig().getConfiguration(context));
137        try {
138            mappings = new ArrayList<ObjectMapping>();
139            initMappings(mappings, config);
140        } catch (JsonValueException jve) {
141            throw new ComponentException("Configuration error: " + jve.getMessage(), jve);
142        }
143    }

145    @Deactivate
146    protected void deactivate(ComponentContext context) {
147        mappings = null;
148    }
149
150    @Modified
151    protected void modified(ComponentContext context) {
152        JsonValue config = new JsonValue(new JSONEnhancedConfig().getConfiguration(context));
153        ArrayList<ObjectMapping> newMappings = new ArrayList<ObjectMapping>();
154        try {
155            initMappings(newMappings, config);
156            mappings = newMappings;
157        } catch (JsonValueException jve) {
158            throw new ComponentException("Configuration error: " + jve.getMessage(), jve);
159        }
160    }

162    private void initMappings(ArrayList<ObjectMapping> mappingList, JsonValue config) {
163        for (JsonValue jv : config.get("mappings").expect(List.class)) {
164            mappingList.add(new ObjectMapping(this, jv)); // throws JsonValueException
165        }
166        for (ObjectMapping mapping : mappingList) {
167            mapping.initRelationships(this, mappingList);
168        }
169    }

171    /**
172     * Return the {@link ObjectMapping} for a the mapping {@code name}.
173     *
174     * @param name the mapping name
175     * @return the ObjectMapping
176     * @throws SynchronizationException if no mapping exists by the given name.
177     */
178    public ObjectMapping getMapping(String name) throws SynchronizationException {
179        for (ObjectMapping mapping : mappings) {
180            if (mapping.getName().equals(name)) {
181                return mapping;
182            }
183        }
184        throw new SynchronizationException("No such mapping: " + name);
185    }

189     *
190     * @param mappingConfig the mapping configuration
191     * @return the created ObjectMapping
192     */
193    public ObjectMapping createMapping(JsonValue mappingConfig) {
194        ObjectMapping createdMapping = new ObjectMapping(this, mappingConfig);
195        List<ObjectMapping> augmentedMappings = new ArrayList<ObjectMapping>(mappings);
196        augmentedMappings.add(createdMapping);
197        createdMapping.initRelationships(this, augmentedMappings);
198        return createdMapping;

210    /**
211     * Results we can expect from synchronizing a specific source object to a given mapping.
212     */
213    private enum MappingSyncResult {
214        SUCCESSFUL, SKIPPED, FAILED
215    }
216
217    /**
218     * Local interface to encapsulate the notifyCreate/notifyUpdate/notifyDelete ObjectMapping synchronization
219     * across all mappings.
220     *
221     * @see #syncAllMappings(org.forgerock.openidm.sync.impl.SynchronizationService.SyncAction, String, String)
222     */
223    private interface SyncAction {
224        JsonValue sync(ObjectMapping mapping) throws SynchronizationException;
225    }
226
227    /**
228     * Synchronize all mappings; keeping track of success/failure conditions.
229     *
230     * @param action the {@code SyncAction} to perform
231     * @param resourceContainer the source object set
232     * @param resourceId the source object id
233     * @returns a JsonValue list of ObjectMappings' sync results
234     * @throws SynchronizationException on failure to sync one of the mappings
235     */
236    private JsonValue syncAllMappings(SyncAction action, final String resourceContainer, final String resourceId)
237            throws SynchronizationException {
238        final JsonValue syncDetails = new JsonValue(new ArrayList<Object>());
239        SynchronizationException exceptionPending = null;
240
241        // cannot sync anything with empty resourceId
242        if (resourceId.isEmpty()) {
243            return syncDetails;
244        }
245
246        // mappings that should be synced are those which are enabled and whose
247        // source object set matches the resource container
248        final Predicate<ObjectMapping> thatMatchSource = new Predicate<ObjectMapping>() {
249            public boolean apply(ObjectMapping objectMapping) {
250                return objectMapping.isSyncEnabled()
251                        && objectMapping.isSourceObject(resourceContainer, resourceId);
252            }
253        };
254
255        for (final ObjectMapping mapping : filter(mappings, thatMatchSource)) {
256            JsonValue mappingResult = new JsonValue(new HashMap<String,Object>());
257            MappingSyncResult result = MappingSyncResult.SUCCESSFUL;
258            try {
259                if (exceptionPending == null) {
260                    // no failures yet, sync this one
261                    mappingResult = action.sync(mapping);
262                } else {
263                    // we've already failed, skip the sync attempt
264                    result = MappingSyncResult.SKIPPED;
265                }
266            } catch (SynchronizationException e) {
267                // failed to sync; store the exception and mark as failed
268                exceptionPending = new SynchronizationException(e.getMessage(), e.getCause());
269                // the exception detail contains the mapping result
270                mappingResult = e.getDetail();
271                mappingResult.put("cause", exceptionPending.toJsonValue().getObject());
272                result = MappingSyncResult.FAILED;
273            } finally {
274                mappingResult.put("result", result.name());
275                mappingResult.put("mapping", mapping.getName());
276                mappingResult.put("targetObjectSet", mapping.getTargetObjectSet());
277                syncDetails.add(mappingResult);
278            }
279        }
280
281        // If there was a failure, send the synchronization results along in the exception...
282        if (exceptionPending != null) {
283            exceptionPending.setDetail(syncDetails);
284            throw exceptionPending;
285        }
286
287        // ...otherwise, return them
288        return syncDetails;
289    }

291    private JsonValue notifyCreate(final String resourceContainer, final String resourceId, final JsonValue object)
292            throws SynchronizationException {
293        PendingLink.handlePendingLinks(mappings, resourceContainer, resourceId, object);
294        return syncAllMappings(new SyncAction() {
295            @Override
296            public JsonValue sync(ObjectMapping mapping) throws SynchronizationException {
297                return mapping.notifyCreate(resourceContainer, resourceId, object);
298            }
299        }, resourceContainer, resourceId);
300    }
301
302    private JsonValue notifyUpdate(final String resourceContainer, final String resourceId, final JsonValue oldValue, final JsonValue newValue)
303            throws SynchronizationException {
304        return syncAllMappings(new SyncAction() {
305            @Override
306            public JsonValue sync(ObjectMapping mapping) throws SynchronizationException {
307                return mapping.notifyUpdate(resourceContainer, resourceId, oldValue, newValue);
308            }
309        }, resourceContainer, resourceId);
310    }
311
312    private JsonValue notifyDelete(final String resourceContainer, final String resourceId, final JsonValue oldValue)
313            throws SynchronizationException {
314        return syncAllMappings(new SyncAction() {
315            @Override
316            public JsonValue sync(ObjectMapping mapping) throws SynchronizationException {
317                return mapping.notifyDelete(resourceContainer, resourceId, oldValue);
318            }
319        }, resourceContainer, resourceId);
320    }

322    /**
323     * ScheduledService interface for supporting scheduled recon.
324     */
325    @Override
326    public void execute(ServerContext context, Map<String, Object> scheduledContext) throws ExecutionException {
327        try {
328            JsonValue params = new JsonValue(scheduledContext).get(CONFIGURED_INVOKE_CONTEXT);
329            String action = params.get("action").asString();
330
331            // "reconcile" in schedule config is the legacy equivalent of the action "recon"
332            if ("reconcile".equals(action)
333                    || ReconciliationService.ReconAction.isReconAction(action)) {
334                JsonValue mapping = params.get("mapping");
335                ObjectSetContext.push(context);
336
337                // Legacy support for spelling recon action as reconcile
338                if ("reconcile".equals(action)) {
339                    params.put("_action", ReconciliationService.ReconAction.recon.toString());
340                } else {
341                    params.put("_action", action);
342                }
343
344                try {
345                    reconService.reconcile(ReconciliationService.ReconAction.recon, mapping, Boolean.TRUE, params, null);
346                } finally {
347                    ObjectSetContext.pop();
348                }
349            } else {
350                throw new ExecutionException("Action '" + action +
351                        "' configured in schedule not supported.");
352            }
353        } catch (JsonValueException jve) {
354            throw new ExecutionException(jve);
355        } catch (ResourceException re) {
356            throw new ExecutionException(re);
357        }
358    }

360    @Override
361    public void actionInstance(ServerContext context, ActionRequest request, ResultHandler<JsonValue> handler) {
362        try {
363            ObjectSetContext.push(context);
364            JsonValue _params = new JsonValue(request.getAdditionalParameters(), new JsonPointer("params"));
365            String resourceContainer;
366            String resourceId;
367            switch (request.getActionAsEnum(Action.class)) {
368                case notifyCreate:
369                    resourceContainer = _params.get(ACTION_PARAM_RESOURCE_CONTAINER).required().asString();
370                    resourceId = _params.get(ACTION_PARAM_RESOURCE_ID).required().asString();
371                    logger.debug("Synchronization action=notifyCreate, resourceContainer={}, resourceId={} ", resourceContainer, resourceId);
372                    handler.handleResult(notifyCreate(resourceContainer, resourceId, request.getContent().get("newValue")));
373                    break;
374                case notifyUpdate:
375                    resourceContainer = _params.get(ACTION_PARAM_RESOURCE_CONTAINER).required().asString();
376                    resourceId = _params.get(ACTION_PARAM_RESOURCE_ID).required().asString();
377                    logger.debug("Synchronization action=notifyUpdate, resourceContainer={}, resourceId={}", resourceContainer, resourceId);
378                    handler.handleResult(notifyUpdate(resourceContainer, resourceId, request.getContent().get("oldValue"), request.getContent().get("newValue")));
379                    break;
380                case notifyDelete:
381                    resourceContainer = _params.get(ACTION_PARAM_RESOURCE_CONTAINER).required().asString();
382                    resourceId = _params.get(ACTION_PARAM_RESOURCE_ID).required().asString();
383                    logger.debug("Synchronization action=notifyDelete, resourceContainer={}, resourceId={}", resourceContainer, resourceId);
384                    handler.handleResult(notifyDelete(resourceContainer, resourceId, request.getContent().get("oldValue")));
385                    break;
386                case recon:
387                    JsonValue result = new JsonValue(new HashMap<String, Object>());
388                    JsonValue mapping = _params.get("mapping").required();
389                    logger.debug("Synchronization action=recon, mapping={}", mapping);
390                    String reconId = reconService.reconcile(ReconciliationService.ReconAction.recon, mapping, Boolean.TRUE, _params, request.getContent());
391                    result.put("reconId", reconId);
392                    result.put("_id", reconId);
393                    result.put("comment1", "Deprecated API on sync service. Call recon action on recon service instead.");
394                    result.put("comment2", "Deprecated return property reconId, use _id instead.");
395                    handler.handleResult(result);
396                    break;
397                case performAction:
398                    logger.debug("Synchronization action=performAction, params={}", _params);
399                    ObjectMapping objectMapping = getMapping(_params.get("mapping").required().asString());
400                    objectMapping.performAction(_params);
401                    //result.put("status", performAction(_params));
402                    handler.handleResult(new JsonValue(new HashMap<String, Object>()));
403                    break;
404                default:
405                    throw new BadRequestException("Action" + request.getAction() + " is not supported.");
406            }
407        } catch (ResourceException e) {
408            handler.handleError(e);
409        } catch (IllegalArgumentException e) { // from getActionAsEnum
410            handler.handleError(new BadRequestException(e.getMessage(), e));
411        } catch (Throwable t) {
412            handler.handleError(ResourceUtil.adapt(t));
413        } finally {
414            ObjectSetContext.pop();
415        }
416    }
