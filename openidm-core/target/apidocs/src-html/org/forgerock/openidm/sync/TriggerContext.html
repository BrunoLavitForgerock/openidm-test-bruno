024package org.forgerock.openidm.sync;
025
026import static org.forgerock.util.Reject.checkNotNull;
027
028import org.forgerock.json.fluent.JsonValue;
029import org.forgerock.json.resource.Context;
030import org.forgerock.json.resource.ResourceException;
031import org.forgerock.json.resource.ServerContext;
032import org.forgerock.json.resource.PersistenceConfig;
033
034
035/**
036 * A ServerContext that stores the source of a trigger during a sync operation.
037 *
038 * @author brmiller
039 */
040public class TriggerContext extends ServerContext {
041
042    private static final String CONTEXT_NAME = "trigger";
043
044    // persisted attribute namon
045    private static final String ATTR_TRIGGER_SOURCE = "trigger-source";
046
047    /**
048     * Create a new trigger context from an existing (parent) context and the source of the trigger.
049     *
050     * @param parent the parent server context
051     * @param trigger the trigger source
052     */
053    public TriggerContext(final Context parent, final String trigger) {
054        super(checkNotNull(parent, "Cannot instantiate TriggerContext with null parent Context"));
055        data.put(ATTR_TRIGGER_SOURCE, trigger);
056    }
057
058    /**
059     * Restore from JSON representation.
060     *
061     * @param savedContext
062     *            The JSON representation from which this context's attributes
063     *            should be parsed.
064     * @param config
065     *            The persistence configuration.
066     * @throws ResourceException
067     *             If the JSON representation could not be parsed.
068     */
069    public TriggerContext(final JsonValue savedContext, final PersistenceConfig config)
070            throws ResourceException {
071        super(savedContext, config);
072    }
073
074    /**
075     * Get this Context's name.
076     *
077     * @return this object's name
078     */
079    public String getContextName() {
080        return CONTEXT_NAME;
081    }
082
083    /**
084     * Retrieves the trigger source.
085     * @return the trigger
086     */
087    public String getTrigger() {
088        return data.get(ATTR_TRIGGER_SOURCE).asString();
089    }
090
091}
