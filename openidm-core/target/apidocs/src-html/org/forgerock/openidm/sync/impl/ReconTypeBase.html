047/**
048 * A base class for reconciliation type handling
049 *
050 * @author aegloff
051 */
052public abstract class ReconTypeBase implements ReconTypeHandler {
053    private static final Logger logger = LoggerFactory.getLogger(ReconTypeBase.class);
054
055    ReconciliationContext reconContext;
056    final boolean runTargetPhase;
057    final boolean allowEmptySourceSet;
058
059    public ReconTypeBase(ReconciliationContext reconContext, boolean defaultRunTargetPhase) {
060        this.reconContext = reconContext;
061        this.allowEmptySourceSet = calcEffectiveConfig("allowEmptySourceSet").defaultTo(false).asBoolean();
062        logger.debug("allowEmptySourceSet: {}", allowEmptySourceSet);
063        this.runTargetPhase = calcEffectiveConfig("runTargetPhase").defaultTo(defaultRunTargetPhase).asBoolean();
064        logger.debug("runTargetPhase: {}", runTargetPhase);
065    }
066
067    public boolean isRunTargetPhase() {
068        return runTargetPhase;
069    }
070    
071    public boolean allowEmptySourceSet() {
072        return allowEmptySourceSet;
073    }
074
075    /**
076     * Calculate the effective configuration for the given configuration property
077     * Properties passed with the request body are given precedence, they override the default configuration
078     * If not overridden by request body, the static configuration is used
079     * @param configPropertyName The name of the configuration property
080     * @param reconContext The reconciliation context
081     * @return the effective configuration; may be null if neither an override or static configuration is provided
082     */
083    protected JsonValue calcEffectiveConfig(String configPropertyName) {
084        JsonValue overridingConfig = reconContext.getOverridingConfig();
085        // Precedence to config supplied in the request body
086        JsonValue effectiveConfig = overridingConfig == null ? new JsonValue(null) : overridingConfig.get(configPropertyName);
087        
088        if (effectiveConfig.isNull()) {
089            // Use regular configuration when not overridden in request body
090            JsonValue cfg = reconContext.getObjectMapping().getConfig().get(configPropertyName);
091            logger.debug("Using settings from mapping configuration for {} : {}", configPropertyName, cfg);
092            return cfg;
093        } else {
094            logger.debug("Using settings supplied in call for {} : {}", configPropertyName, effectiveConfig);
095        }
096        return effectiveConfig;
097    }
098
099    /**
100     * Calculate the effective query, taking into account config overrides in the request, as well
101     * as defaults
102     * @param queryConfigPropertyName The property name in the configuration for this query
103     * @param mappingResource the resource name in the mapping that this query relates to.
104     * Used to default the resource the query will be sent to
105     * @return the effecive query
106     */
107    protected JsonValue calcEffectiveQuery(String queryConfigPropertyName, String mappingResource) {
108        JsonValue queryCfg = calcEffectiveConfig(queryConfigPropertyName);
109
110        if (queryCfg.isNull()) {
111            queryCfg = new JsonValue(new LinkedHashMap());
112        }
113        // If not defined in the query config itself, default the query resource to the mapping source
114        if (!queryCfg.isDefined("resourceName")) {
115            queryCfg.put("resourceName", mappingResource);
116            logger.debug("Default {} resource to query to {}", queryConfigPropertyName, mappingResource);
117        }
118
119        // If config doesn't explicitly specify the query, default to query all ids
120        if (!specifiesQuery(queryCfg)) {
121            queryCfg.put(QueryRequest.FIELD_QUERY_ID, ServerConstants.QUERY_ALL_IDS);
122            logger.debug("Default {} query to {}", queryConfigPropertyName, ServerConstants.QUERY_ALL_IDS);
123        }
124        logger.debug("Effective query for {}: {}", queryConfigPropertyName, queryCfg);
125
126        return queryCfg;
127    }
128
129    /**
130     * @param queryCfg The effective query configuration
131     * @return true if the effective query configuration explicitly defines the query to execute,
132     * false if not
133     */
134    protected boolean specifiesQuery(JsonValue queryCfg) {
135        // Check if there is a property that defines what query to execute
136        boolean specifiesQuery =
137                queryCfg.isDefined(QueryRequest.FIELD_QUERY_ID)
138                || queryCfg.isDefined(QueryRequest.FIELD_QUERY_EXPRESSION)
139                || queryCfg.isDefined(QueryRequest.FIELD_QUERY_FILTER)
140                || queryCfg.isDefined("query");
141        // OpenICF provisioner uses an inconsistent "query" param - to be deprecated
142
143        if (logger.isDebugEnabled()) {
144            if (specifiesQuery) {
145                logger.debug("Explicit query was specified");
146            } else {
147                logger.debug("No explicit query specified");
148            }
149        }
150
151        return specifiesQuery;
152    }
153
154    /**
155     * Execute the specified query
156     *
157     * @param objectSet the object set to query
158     * @param query the query parameters
159     * @param collectionToPopulate the collection to populate with results
160     * @param caseSensitive whether the collection should be populated in case
161     * sensitive fashion, or if false it populates as lower case only
162     * @return the collection of (unqualified) ids
163     * @throws SynchronizationException if retrieving or processing the ids failed
164     */
165    protected Collection<String> query(final String objectSet, JsonValue query, final ReconciliationContext reconContext,
166            Collection<String> collectionToPopulate, final boolean caseSensitive) throws SynchronizationException {
167        final Collection<String> ids = collectionToPopulate;
168
169        try {
170            QueryRequest r = Requests.newQueryRequest(objectSet);
171            r.setQueryId(query.get(QueryRequest.FIELD_QUERY_ID).asString());
172            r.setQueryExpression(query.get(QueryRequest.FIELD_QUERY_EXPRESSION).asString());
173            JsonValue queryFilter = query.get(QueryRequest.FIELD_QUERY_FILTER);
174            if (!queryFilter.isNull()) {
175                r.setQueryFilter(QueryFilter.valueOf(queryFilter.asString()));
176            }
177            for (Map.Entry<String, Object> e: query.asMap().entrySet()) {
178                r.setAdditionalParameter(e.getKey(), String.valueOf(e.getValue()));
179            }
180            reconContext.getService().getConnectionFactory().getConnection().query(reconContext.getService().getRouter(), r,
181                    new QueryResultHandler() {
182                        @Override
183                        public void handleError(ResourceException error) {
184                            // ignore
185                        }
186
187                        @Override
188                        public boolean handleResource(Resource resource) {
189                            if (resource.getId() == null) {
190                                // do not add null values to collection
191                                logger.warn("Resource {0} id is null!", resource.toString());
192                            }
193                            else {
194                                ids.add(
195                                    caseSensitive
196                                    ? resource.getId()
197                                    : reconContext.getObjectMapping().getLinkType().normalizeId(resource.getId()));
198                            }
199                            return true;
200                        }
201
202                        @Override
203                        public void handleResult(QueryResult result) {
204                            //ignore
205                        }
206                    });
207        } catch (JsonValueException jve) {
208            throw new SynchronizationException(jve);
209        } catch (ResourceException ose) {
210            throw new SynchronizationException(ose);
211        }
212
213        reconContext.checkCanceled(); // Throws an exception if reconciliation was canceled
214        return ids;
215    }
216
217    /**
218     * @inheritDoc
219     */
220    public abstract List<String> querySourceIds() throws SynchronizationException;
221
222    /**
223     * @inheritDoc
224     */
225    public abstract List<String> queryTargetIds() throws SynchronizationException;
