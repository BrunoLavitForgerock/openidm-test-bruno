024package org.forgerock.openidm.sync.impl;
025
026import java.text.SimpleDateFormat;
027import java.util.ArrayList;
028import java.util.Collections;
029import java.util.Date;
030import java.util.EnumMap;
031import java.util.HashMap;
032import java.util.List;
033import java.util.Map;
034import java.util.Map.Entry;
035
036/**
037 *
038 * @author gael
039 */
040public class ReconStats {
041
042    private String reconId = "";
043    private String reconName = "";
044    private long startTime = 0;
045    private long endTime = 0;
046    private long allIdsStartTime = 0;
047    private long allIdsEndTime = 0;
048    private String startTimestamp = "";
049    private String endTimestamp = "";
050    private Map<Situation, List<String>> ids = new EnumMap<Situation, List<String>>(Situation.class);
051    private SimpleDateFormat sdformat = new SimpleDateFormat();
052    public long entries = 0;
053    public List<String> notValid = null;
054
055    public ReconStats(String reconId, String reconName) {
056        this.reconId = reconId;
057        this.reconName = reconName;
058        ids.put(Situation.CONFIRMED, Collections.synchronizedList(new ArrayList<String>()));
059        ids.put(Situation.FOUND, Collections.synchronizedList(new ArrayList<String>()));
060        ids.put(Situation.ABSENT, Collections.synchronizedList(new ArrayList<String>()));
061        ids.put(Situation.AMBIGUOUS, Collections.synchronizedList(new ArrayList<String>()));
062        ids.put(Situation.MISSING, Collections.synchronizedList(new ArrayList<String>()));
063        ids.put(Situation.UNQUALIFIED, Collections.synchronizedList(new ArrayList<String>()));
064        ids.put(Situation.UNASSIGNED, Collections.synchronizedList(new ArrayList<String>()));
065        ids.put(Situation.SOURCE_MISSING, Collections.synchronizedList(new ArrayList<String>()));
066        ids.put(Situation.SOURCE_IGNORED, Collections.synchronizedList(new ArrayList<String>()));
067        ids.put(Situation.TARGET_IGNORED, Collections.synchronizedList(new ArrayList<String>()));
068        notValid = new ArrayList<String>();
069    }
070
071    public void start() {
072        startTime = System.currentTimeMillis();
073        startTimestamp = sdformat.format(new Date());
074    }
075    
076    public void end() {
077        endTime = System.currentTimeMillis();
078        endTimestamp = sdformat.format(new Date());
079    }
080
081    public void startAllIds() {
082        allIdsStartTime = System.currentTimeMillis();
083    }
084
085    public void endAllIds() {
086        allIdsEndTime = System.currentTimeMillis();
087    }
088
089    public void addSituation(String id, Situation situation) {
090        if (situation != null) {
091            List<String> situationIds = ids.get(situation);
092            if (situationIds != null) {
093                situationIds.add(id);
094            }
095        }
096    }
097
098    // TODO: phase out notValid and replace with source ignored, target ignored, unqualified
099    // situation processing
100    public void addNotValid(String id) {
101        notValid.add(id);
102    }
103
104    public void addAction(String id, Action action) {
105    }
106
107    public Map<String, Object> asMap() {
108        Map<String, Object> results = new HashMap();
109
110        results.put("startTime", startTimestamp);
111        results.put("endTime", endTimestamp);
112        results.put("duration", endTime - startTime);
113        results.put("allIds", allIdsEndTime - allIdsStartTime);
114        results.put("entries", entries);
115        results.put("reconId", reconId);
116        results.put("reconName", reconName);
117
118        Map<String, Object> nv = new HashMap();
119        nv.put("count", notValid.size());
120        nv.put("ids", notValid);
121        results.put("NOTVALID", nv);
122
123        for (Entry<Situation, List<String>> e : ids.entrySet()) {
124            Map<String, Object> res = new HashMap();
125            res.put("count", e.getValue().size());
126            res.put("ids", e.getValue());
127            results.put(e.getKey().name(), res);
128        }
129
130        return results;
131    }
132    
133    public static String simpleSummary(ReconStats globalStats, ReconStats sourceStats, ReconStats targetStats) {
134        Map<String, Integer> simpleSummary = new HashMap<String, Integer>();
135        updateSummary(simpleSummary, globalStats);
136        updateSummary(simpleSummary, sourceStats);
137        updateSummary(simpleSummary, targetStats);
138        
139        StringBuilder sb = new StringBuilder();
140        for (Entry<String, Integer> e : simpleSummary.entrySet()) {
141            sb.append(e.getKey());
142            sb.append(": ");
143            sb.append(e.getValue());
144            sb.append(" ");
145        }
146        return sb.toString();
147    }
148    
149    private static void updateSummary(Map<String, Integer> simpleSummary, ReconStats stat) {
150        if (stat != null) {
151            for (Entry<Situation, List<String>> e : stat.ids.entrySet()) {
152                String key = e.getKey().name();
153                Integer existing = simpleSummary.get(key);
154                if (existing == null) {
155                    existing = Integer.valueOf(0);
156                }
157                Integer updated = existing + e.getValue().size();
158                simpleSummary.put(key, updated);
159            }
160        }
161    }
162}
