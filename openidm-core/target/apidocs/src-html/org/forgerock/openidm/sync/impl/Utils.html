044/**
045 * A Utils class is a collection of common methods used in the script library.
046 *
047 * @author Laszlo Hordos
048 */
049public class Utils {
050
051    public static class Pair<A, B> {
052        public final A fst;
053        public final B snd;
054
055        public Pair(final A a, final B b) {
056            fst = a;
057            snd = b;
058        }
059
060        public String toString() {
061            return "Pair[" + fst + "," + snd + "]";
062        }
063
064        private static boolean equals(Object x, Object y) {
065            return (x == null && y == null) || (x != null && x.equals(y));
066        }
067
068        public boolean equals(Object other) {
069            return other instanceof Pair<?, ?> && equals(fst, ((Pair<?, ?>) other).fst)
070                    && equals(snd, ((Pair<?, ?>) other).snd);
071        }
072
073        public int hashCode() {
074            int result = fst != null ? fst.hashCode() : 0;
075            result = 31 * result + (snd != null ? snd.hashCode() : 0);
076            return result;
077        }
078
079        public static <A, B> Pair<A, B> of(final A a, final B b) {
080            return new Pair<A, B>(a, b);
081        }
082    }
083
084    private Utils() {
085    }
086
087    public static void copyURLToFile(URL in, File out) throws IOException {
088        ReadableByteChannel inChannel = Channels.newChannel(in.openStream());
089        FileChannel outChannel = new FileOutputStream(out).getChannel();
090        try {
091            outChannel.transferFrom(inChannel, 0, 1 << 24);
092        } catch (IOException e) {
093            throw e;
094        } finally {
095            if (inChannel != null)
096                inChannel.close();
097            if (outChannel != null)
098                outChannel.close();
099        }
100    }
101
102    public static <T> T deepCopy(final T source) {
103        if (source instanceof JsonValue) {
104            return (T) new JsonValue(deepCopy(((JsonValue) source).getObject()));
105        } else if (source instanceof Collection || source instanceof Map) {
106            return (T) deepCopy(source, new Stack<Pair<Object, Object>>());
107        } else {
108            return source;
109        }
110    }
111
112    @SuppressWarnings({ "unchecked" })
113    private static Object deepCopy(Object source, final Stack<Pair<Object, Object>> valueStack) {
114        Iterator<Pair<Object, Object>> i = valueStack.iterator();
115        while (i.hasNext()) {
116            Pair<Object, Object> next = i.next();
117            if (next.fst == source) {
118                return next.snd;
119            }
120        }
121
122        if (source instanceof JsonValue) {
123            return new JsonValue(deepCopy(((JsonValue) source).getObject(), valueStack));
124        } else if (source instanceof Collection) {
125            List<Object> copy = new ArrayList<Object>(((Collection) source).size());
126            valueStack.push(Pair.of(source, (Object) copy));
127            for (Object o : (Collection) source) {
128                copy.add(deepCopy(o, valueStack));
129            }
130            // valueStack.pop();
131            return copy;
132        } else if (source instanceof Map) {
133            Map copy = new LinkedHashMap(((Map) source).size());
134            valueStack.push(Pair.of(source, (Object) copy));
135            for (Map.Entry<Object, Object> entry : ((Map<Object, Object>) source).entrySet()) {
136                copy.put(entry.getKey(), deepCopy(entry.getValue(), valueStack));
137            }
138            // valueStack.pop();
139            return copy;
140        } else {
141            return source;
142        }
143    }
144}
