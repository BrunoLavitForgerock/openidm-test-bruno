077    /**
078     * Creates the instance with info from the current call context
079     * @param reconAction the recon action
080     * @param mapping the mapping configuration
081     * @param callingContext The resource call context
082     * @param reconParams configuration options for the recon
083     */
084    public ReconciliationContext(
085            ReconciliationService.ReconAction reconAction,
086            ObjectMapping mapping,
087            ServerContext callingContext,
088            JsonValue reconParams,
089            JsonValue overridingConfig,
090            ReconciliationService service)
091        throws BadRequestException {
092
093        this.mapping = mapping;
094        this.reconId = callingContext.getId();
095        this.reconStat = new ReconciliationStatistic(this);
096        this.reconParams = reconParams;
097        this.overridingConfig = overridingConfig;
098        this.service = service;
099        
100        reconTypeHandler = createReconTypeHandler(reconAction);
101        
102        // Initialize the executor for this recon, or null if no executor should be used
103        int noOfThreads = mapping.getTaskThreads();
104        if (noOfThreads > 0) {
105            executor = Executors.newFixedThreadPool(noOfThreads);
106        } else {
107            executor = null;
108        }
109    }
110
111    /**
112     * Factory method for the recon type handlers
113     * @param reconAction the recon action
114     * @return the handler appropriate for this recon action type
115     */
116    private ReconTypeHandler createReconTypeHandler(ReconciliationService.ReconAction reconAction) throws BadRequestException {
117        switch (reconAction) {
118        case recon :
119            return new ReconTypeByQuery(this);
120        case reconById :
121            return new ReconTypeById(this);
122        default:
123            throw new BadRequestException("Unknown action " + reconAction.toString());
124        }
125    }

212    /**
213     * @return the populated run progress structure
214     */
215    public Map<String, Object> getProgress() {
216        // Unknown total entries are currently represented via question mark string.
217        String totalSourceEntriesStr = (totalSourceEntries == null ? "?" : Integer.toString(totalSourceEntries));
218        String totalTargetEntriesStr = (totalTargetEntries == null ? "?" : Integer.toString(totalTargetEntries));
219
220        String totalLinkEntriesStr = "?";
221        if (totalLinkEntries == null) {
222            if (getStage() == ReconStage.COMPLETED_SUCCESS) {
223                totalLinkEntriesStr = Integer.toString(getStatistics().getLinkProcessed());
224            }
225        } else {
226            totalLinkEntriesStr = Integer.toString(totalLinkEntries);
227        }
228
229        Map<String, Object> progressDetail = new LinkedHashMap<String, Object>();
230        Map<String, Object> sourceDetail = new LinkedHashMap<String, Object>();
231        Map<String, Object> sourceExisting = new LinkedHashMap<String, Object>();
232        Map<String, Object> targetDetail = new LinkedHashMap<String, Object>();
233        Map<String, Object> targetExisting = new LinkedHashMap<String, Object>();
234        Map<String, Object> linkDetail = new LinkedHashMap<String, Object>();
235        Map<String, Object> linkExisting = new LinkedHashMap<String, Object>();
236
237        sourceExisting.put("processed", getStatistics().getSourceProcessed());
238        sourceExisting.put("total", totalSourceEntriesStr);
239        sourceDetail.put("existing", sourceExisting);
240        progressDetail.put("source", sourceDetail);
241
242        targetExisting.put("processed", getStatistics().getTargetProcessed());
243        targetExisting.put("total", totalTargetEntriesStr);
244        targetDetail.put("existing", targetExisting);
245        targetDetail.put("created", getStatistics().getTargetCreated());
246        progressDetail.put("target", targetDetail);
247
248        linkExisting.put("processed", getStatistics().getLinkProcessed());
249        linkExisting.put("total", totalLinkEntriesStr);
250        linkDetail.put("existing", linkExisting);
251        linkDetail.put("created", getStatistics().getLinkCreated());
252        progressDetail.put("links", linkDetail);
253
254        return progressDetail;
255    }

264    /**
265     * @return the source ids to reconcile in this recon scope
266     * @throws SynchronizationException if getting the ids to reconcile failed
267     */
268    Iterator<String> querySourceIdsIter() throws SynchronizationException {
269        List<String> sourceIds = getReconHandler().querySourceIds();
270        setSourceIds(sourceIds);
271        return sourceIds.iterator();
272    }
273
274    /**
275     * @param sourceIds the list of all source object ids in the reconciliation scope
276     */
277    void setSourceIds(List<String> sourceIds) {
278        // Choose a hash based collection as we need fast "contains" handling
279        this.sourceIds = Collections.newSetFromMap(new ConcurrentHashMap<String, Boolean>());
280        this.sourceIds.addAll(sourceIds);
281        this.totalSourceEntries = Integer.valueOf(sourceIds.size());
282    }
283
284    /**
285     * @param targetIds the list of all ids in the target object set
286     * If the target system IDs are case insensitive, the ids are kept in normalized (lower case) form
287     */
288    void setTargetIds(List<String> targetIds) {
289        // Choose a hash based collection as we need fast "contains" handling
290        this.targetIds = Collections.newSetFromMap(new ConcurrentHashMap<String, Boolean>());
291        if (targetIds != null) {
292            this.targetIds.addAll(targetIds);
293        }
294        this.totalTargetEntries = Integer.valueOf(targetIds.size());
295    }
296
297    /**
298     * Set all pre-fetched links
299     * Since pre-fetching all links is optional, links may be gotten individually rather than
300     * this getting set.
301     * @param allLinks the list of all links for a given mapping
302     */
303    void setAllLinks(Map<String, Link> allLinks) {
304        if (allLinks != null) {
305            this.totalLinkEntries = Integer.valueOf(allLinks.size());
306        }

327    /**
328     * @param stage Sets the current state and stage in the reconciliation process
329     */
330    public void setStage(ReconStage newStage) {
331        // If there is already a stage in progress, end it first
332        if (this.stage != ReconStage.ACTIVE_INITIALIZED) {
333            reconStat.endStage(this.stage);
334        }
335        if (canceled) {
336            if (newStage.isComplete()) {
337                this.stage = ReconStage.COMPLETED_CANCELED;
338            } else {
339                this.stage = ReconStage.ACTIVE_CANCELING;
340            }
341        } else {
342            this.stage = newStage;
343        }
344        if (newStage.isComplete()) {
345            cleanupState();
346        } else {
347            reconStat.startStage(newStage);
348        }
349    }
350
351    /**
352     * Remove any state from memory that should not be kept
353     * past the completion of the reconciliation run
354     */
355    private synchronized void cleanupState() {
356        sourceIds = null;
357        targetIds = null;
358        if (executor != null) {
359            executor.shutdown();
360            executor = null;
361        }
362    }
363
364    /**
365     * Returns a summary of the reconciliation run.
366     */
367    public Map<String, Object> getSummary() {
368        Map<String, Object> reconSummary = new LinkedHashMap<String, Object>();
369        reconSummary.put("_id", getReconId());
370        reconSummary.put("mapping", getMapping());
371        reconSummary.put("state", getState());
372        reconSummary.put("stage", getStage().toString());
373        reconSummary.put("stageDescription", getStage().getDescription());
374        reconSummary.put("progress", getProgress());
375        reconSummary.put("situationSummary", getStatistics().getSituationSummary());
376        reconSummary.put("started", getStatistics().getStarted());
377        reconSummary.put("ended", getStatistics().getEnded());
378        return reconSummary;
379    }
