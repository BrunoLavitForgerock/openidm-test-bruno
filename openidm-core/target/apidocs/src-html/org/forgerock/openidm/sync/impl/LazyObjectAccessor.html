025package org.forgerock.openidm.sync.impl;
026
027import java.util.Map;
028
029import org.forgerock.json.fluent.JsonValue;
030import org.forgerock.json.resource.ConnectionFactory;
031import org.forgerock.json.resource.NotFoundException;
032import org.forgerock.json.resource.ReadRequest;
033import org.forgerock.json.resource.Requests;
034import org.forgerock.json.resource.ResourceException;
035import org.forgerock.json.resource.ServerContext;
036import org.forgerock.openidm.smartevent.EventEntry;
037import org.forgerock.openidm.smartevent.Name;
038import org.forgerock.openidm.smartevent.Publisher;
039
040import org.slf4j.Logger;
041import org.slf4j.LoggerFactory;
042
043/**
044 * Loads and caches the object only once on demand.
045 * This class is not thread safe.
046 *
047 * @author Laszlo
048 * @author aegloff
049 */
050public class LazyObjectAccessor {
051    private static final Logger logger = LoggerFactory.getLogger(ObjectMapping.class);
052
053    public static final Name EVENT_READ_OBJ = Name.get("openidm/internal/discovery-engine/sync/read-object");
054
055    private SynchronizationService service;
056    private JsonValue object = null;       // The object once loaded, or null if not found
057    private boolean loaded = false;        // Whether it considers its state as loaded/initialized
058    private final String componentContext; // The qualifier for the id
059    private final String localId;          // The local part of the id
060
061    /**
062     * Construct with a known value of the object. The object is considered loaded.
063     * @param service the sync service
064     * @param componentContext the component qualifier of the object id
065     * @param localId the unqualified part of the object id
066     * @param value the object value
067     */
068    public LazyObjectAccessor(SynchronizationService service, String componentContext,
069                              String localId, JsonValue value) {
070        this.service = service;
071        this.object = value;
072        this.componentContext = componentContext;
073        this.localId = localId;
074        this.loaded = true;
075    }
076
077    /**
078     * Construct with just the identifier of the object. The object is not yet considered loaded.
079     * @param service the sync service
080     * @param componentContext the component qualifier of the object id
081     * @param localId the unqualified part of the object id
082     */
083    public LazyObjectAccessor(SynchronizationService service, String componentContext,
084                              String localId) {
085        this.service = service;
086        this.componentContext = componentContext;
087        this.localId = localId;
088    }
089
090    /**
091     * @return the object value, null if the object does not exist.
092     * This may trigger a load on demand
093     * if it has not been loaded already.
094     *
095     * @throws SynchronizationException if loading the object failed
096     * for reasons other than just not finding the object.
097     */
098    public JsonValue getObject() throws SynchronizationException {
099        if (!loaded) {
100            try {
101                // If not found, the object will be null
102                object = rawReadObject(service.getServerContext(), service.getConnectionFactory(), componentContext, localId);
103            } catch (SynchronizationException ex) {
104                throw ex; // being explicit that this would not be considered loaded
105            }
106            loaded = true;
107        }
108        return object;
109    }
110
111    /**
112     * @return whether the object state has been loaded/initialized.
113     * This also is true if an object is not found, and the object value hence is null.
114     */
115    public boolean isLoaded() {
116        return loaded;
117    }
118
119    /**
120     * @return The object representation as a map
121     * @throws SynchronizationException if loading the object on demand failed
122     */
123    public Map<String, Object> asMap() throws SynchronizationException {
124        return null != getObject() ? getObject().asMap() : null;
125    }
126
127    /**
128     * @return The unqualified part of the object id
129     */
130    public String getLocalId() {
131        return localId;
132    }
133
134    /**
135     * Tries to retrieve the object directly from where it is stored.
136     * Does not take into account any cached/lazy loaded object in this class.
137     *
138     * Typically access to objects should be through {@code getObject()} instead,
139     * which can support the lazy/cached loading.
140     *
141     * @param resourceContainer location where the object is stored
142     * @param resourceId the object identifier
143     * @throws NullPointerException if {@code targetId} is {@code null}.
144     * @throws SynchronizationException if retrieving the object failed
145     * @return the object value if found, null if not found
146     */
147    public static JsonValue rawReadObject(ServerContext router, ConnectionFactory connectionFactory, String resourceContainer, String resourceId) throws SynchronizationException {
148        if (resourceId == null) {
149            throw new NullPointerException("Identifier passed to readObject is null");
150        }
151        EventEntry measure = Publisher.start(EVENT_READ_OBJ, null, resourceId);
152        try {
153            ReadRequest r = Requests.newReadRequest(resourceContainer, resourceId);
154            JsonValue result = connectionFactory.getConnection().read(router,r).getContent();
155            measure.setResult(result);
156            return result;
157        } catch (NotFoundException nfe) { // target not found results in null
158            return null;
159        } catch (ResourceException ose) {
160            logger.warn("Failed to read target object", ose);
161            throw new SynchronizationException(ose);
162        } finally {
163            measure.end();
164        }
165    }
166
167    /**
168     * @param componentContext the resource component name the localId is relative to
169     * @param localId local identifier within the component context
170     * @return The qualified identifier, qualified with the component context
171     */
172    public static String qualifiedId(String componentContext, String localId) {
173        StringBuilder sb = new StringBuilder();
174        sb.append(componentContext);
175        if (componentContext != null) {
176            sb.append('/').append(localId);
177        }
178        return sb.toString();
179    }
180}
