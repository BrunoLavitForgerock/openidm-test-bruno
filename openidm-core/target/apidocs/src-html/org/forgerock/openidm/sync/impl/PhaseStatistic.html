025package org.forgerock.openidm.sync.impl;
026
027//import java.text.SimpleDateFormat;
028import java.util.ArrayList;
029import java.util.Collections;
030import java.util.Date;
031import java.util.EnumMap;
032import java.util.HashMap;
033import java.util.List;
034import java.util.Map;
035import java.util.Map.Entry;
036import java.util.concurrent.atomic.AtomicLong;
037
038/**
039 * Holds the (source/target) Phase specific statistics
040 *
041 * @author gael
042 * @author aegloff
043 */
044public class PhaseStatistic {
045
046    public enum Phase {SOURCE, TARGET};
047    
048    private ReconciliationStatistic parentStat;
049    Phase phase;
050    private Map<Situation, List<String>> ids = Collections.synchronizedMap(new EnumMap<Situation, List<String>>(Situation.class));
051    private AtomicLong processedEntries = new AtomicLong();
052    private List<String> notValid;
053
054    public PhaseStatistic(ReconciliationStatistic parentStat, Phase phase) {
055        this.parentStat = parentStat;
056        this.phase = phase;
057        ids.put(Situation.CONFIRMED, Collections.synchronizedList(new ArrayList<String>()));
058        ids.put(Situation.FOUND, Collections.synchronizedList(new ArrayList<String>()));
059        ids.put(Situation.ABSENT, Collections.synchronizedList(new ArrayList<String>()));
060        ids.put(Situation.AMBIGUOUS, Collections.synchronizedList(new ArrayList<String>()));
061        ids.put(Situation.MISSING, Collections.synchronizedList(new ArrayList<String>()));
062        ids.put(Situation.UNQUALIFIED, Collections.synchronizedList(new ArrayList<String>()));
063        ids.put(Situation.UNASSIGNED, Collections.synchronizedList(new ArrayList<String>()));
064        ids.put(Situation.SOURCE_MISSING, Collections.synchronizedList(new ArrayList<String>()));
065        ids.put(Situation.SOURCE_IGNORED, Collections.synchronizedList(new ArrayList<String>()));
066        ids.put(Situation.TARGET_IGNORED, Collections.synchronizedList(new ArrayList<String>()));
067        notValid = Collections.synchronizedList(new ArrayList<String>());
068    }
069
070    /**
071     * Handle the processed notification to update the statistics appropriately
072     * @param sourceId The source id processed, or null if none
073     * @param targetId The target id processed, or null if none
074     * @param linkExisted indication if the link existed before the operation
075     * @param linkId the link identifier, if available. For created links this may not currently be available.
076     * @param linkCreated indication if a new link was created during the operation
077     * @param situation the assessed situation
078     * @param action the action that was processed
079     */
080    public void processed(String sourceId, String targetId, boolean linkExisted, String linkId, boolean linkCreated, 
081            Situation situation, Action action) {
082        
083        String id = null;
084        if (phase == Phase.SOURCE) {
085            id = sourceId;
086        } else {
087            id = targetId;
088        }
089        parentStat.processed(sourceId, targetId, linkExisted, linkId, linkCreated, situation, action);
090        if (id != null) {
091            processedEntries.incrementAndGet();
092            if (situation != null) {
093                List<String> situationIds = ids.get(situation);
094                if (situationIds != null) {
095                    situationIds.add(id); // TODO: option to not keep all results in memory
096                }
097            }
098        }
099    }
100    
101    // TODO: phase out notValid and replace with source ignored, target ignored, unqualified
102    // situation processing
103    public void addNotValid(String id) {
104        notValid.add(id);
105    }
106
107    public long getProcessed() {
108        return processedEntries.get();
109    }
110    
111    public Map<String, Object> asMap() {
112        Map<String, Object> results = new HashMap();
113
114        results.put("processed", processedEntries);
115
116        Map<String, Object> nv = new HashMap();
117        nv.put("count", notValid.size());
118        nv.put("ids", notValid);
119        results.put("NOTVALID", nv);
120
121        for (Entry<Situation, List<String>> e : ids.entrySet()) {
122            Map<String, Object> res = new HashMap();
123            res.put("count", e.getValue().size());
124            res.put("ids", e.getValue());
125            results.put(e.getKey().name(), res);
126        }
127
128        return results;
129    }
130
131    public void updateSummary(Map<String, Integer> simpleSummary) {
132        for (Entry<Situation, List<String>> e : ids.entrySet()) {
133            String key = e.getKey().name();
134            Integer existing = simpleSummary.get(key);
135            if (existing == null) {
136                existing = Integer.valueOf(0);
137            }
138            Integer updated = existing + e.getValue().size();
139            simpleSummary.put(key, updated);
140        }
141    }
142}
