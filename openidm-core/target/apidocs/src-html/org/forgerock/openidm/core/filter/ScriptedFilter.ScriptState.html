058    /**
059     * Indicates that the request processing should continue normally. If
060     * returned from the {@link #evaluateOnRequest} method, the filter then
061     * invokes the
062     * {@link CrossCutFilterResultHandler#handleContinue(org.forgerock.json.resource.ServerContext, Object)}
063     * method.
064     */
065    public static final int CONTINUE = 0;
066
067    /**
068     * Indicates that after the {@link #evaluateOnRequest} method, the request
069     * processing should skip the
070     * {@link CrossCutFilterResultHandler#handleContinue(org.forgerock.json.resource.ServerContext, Object)}
071     * method to continue with the {@link RequestHandler} method.
072     */
073    public static final int SKIP = 1;
074
075    /**
076     * Indicates that the request processing should stop and return the current
077     * response from the filter.
078     */
079    public static final int STOP = 2;
080
081    /**
082     * Setup logging for the {@link ScriptedFilter}.
083     */
084    private static final Logger logger = LoggerFactory.getLogger(ScriptedFilter.class);

086    public static class ScriptState {

088        private Object state = null;

090        private Request request;
091
092        public ScriptState(final Request request) {
093            this.request = request;
094        }
095    }

098    private final Pair<JsonPointer, ScriptEntry> onResponse;
099    private final Pair<JsonPointer, ScriptEntry> onFailure;

101    public ScriptedFilter(
102            Pair<JsonPointer, ScriptEntry> onRequest,
103            Pair<JsonPointer, ScriptEntry> onResponse,
104            Pair<JsonPointer, ScriptEntry> onFailure) {
105        this.onRequest = onRequest;
106        this.onResponse = onResponse;
107        this.onFailure = onFailure;

110    // ----- Filter Action Request
111
112    @Override
113    public void filterActionRequest(final ServerContext context, final ActionRequest request,
114            final RequestHandler next,
115            final CrossCutFilterResultHandler<ScriptState, JsonValue> handler) {
116        final ScriptState state = new ScriptState(request);
117        try {
118            switch (evaluateOnRequest(context, state)) {
119            case CONTINUE: {
120                handler.handleContinue(context, state);
121                break;
122            }
123            case SKIP: {
124                break;
125            }
126            case STOP: {
127                break;
128            }
129            }
130        } catch (ResourceException e) {
131            logger.debug("Failed the execute the onRequest script", e);
132            handler.handleError(e);
133        }
134    }
135
136    @Override
137    public void filterActionResult(final ServerContext context, final ScriptState state,
138            final JsonValue result, final ResultHandler<JsonValue> handler) {
139        if (null != onResponse) {
140            handler.handleResult(result);
141        } else {
142            handler.handleResult(result);
143        }
144    }
145
146    @Override
147    public void filterActionError(final ServerContext context, final ScriptState state,
148            final ResourceException error, final ResultHandler<JsonValue> handler) {
149        if (null != onFailure) {
150            logger.info("Filter GenericError response: {}.", context.getId());
151            try {
152                evaluateOnFailure(context, state, error, handler);
153            } catch (ResourceException e) {
154                logger.debug("Failed the execute the onFailure script", e);
155                // TODO combine the messages
156                // handler.handleError(e);
157            } finally {
158                handler.handleError(error);
159            }
160        } else {
161            handler.handleError(error);
162        }
163    }

307    // ----- Evaluate Script methods
308
309    protected int evaluateOnRequest(final ServerContext context, final ScriptState state)
310            throws ResourceException {
311        if (onRequest != null) {
312            ScriptEntry scriptEntry = onRequest.getRight();
313            if (!scriptEntry.isActive()) {
314                throw new ServiceUnavailableException("Failed to execute inactive script: "
315                        + onRequest.getRight().getName());
316            }
317            Script script = populateScript(scriptEntry, context, state.request);
318            try {
319                //TODO Add function to support SKIP/STOP/CONTINUE
320                state.state = script.eval();
321            } catch (Throwable t) {
322                logger.debug("Filter/{} script {} encountered exception at {}", onRequest.getRight()
323                        .getName(), onRequest.getLeft(), t);
324                ResourceException re = Utils.adapt(t);
325                logger.debug("ResourceException detail: " + re.getDetail());
326                throw re;
327            }
328        }
329        return CONTINUE;
330    }
331
332    public boolean evaluateOnResponse(final ServerContext context, final ScriptState state,
333                                     final Resource resource) throws ResourceException {
334        if (onResponse != null) {
335            ScriptEntry scriptEntry = onResponse.getRight();
336            if (!scriptEntry.isActive()) {
337                throw new ServiceUnavailableException("Failed to execute inactive script: "
338                        + onResponse.getRight().getName());
339            }
340            Script script = populateScript(scriptEntry, context, state.request);
341            script.put("response", resource);
342            try {
343                state.state = script.eval();
344            } catch (Throwable t) {
345                logger.debug("Filter/{} script {} encountered exception at {}", onResponse.getRight()
346                        .getName(), onResponse.getLeft(), t);
347                throw Utils.adapt(t);
348            }
349        }
350        return true;
351    }
352
353    public void evaluateOnFailure(final ServerContext context, final ScriptState state,
354            final ResourceException error, final ResultHandler<?> handler) throws ResourceException {
355        if (onFailure != null) {
356            ScriptEntry scriptEntry = onFailure.getRight();
357            if (!scriptEntry.isActive()) {
358                throw new ServiceUnavailableException("Failed to execute inactive script: "
359                        + onFailure.getRight().getName());
360            }
361            Script script = populateScript(scriptEntry, context, state.request);
362            script.put("exception", error.includeCauseInJsonValue().toJsonValue().asMap());
363            try {
364                state.state = script.eval();
365            } catch (Throwable t) {
366                logger.debug("Filter/{} script {} encountered exception at {}", onFailure.getRight()
367                        .getName(), onFailure.getLeft(), t);
368                throw Utils.adapt(t);
369            }
370        }
371    }
372
373    private Script populateScript(final ScriptEntry scriptEntry, final ServerContext context, final Request request) {
374        final Script script = scriptEntry.getScript(context);
375
376        if (ScriptUtil.REQUEST_BINDING_2_1.equals(scriptEntry.getName().getRequestBinding())) {
377            // old, 2.1.x, deprecated script object bindings
378            script.put("request", ScriptUtil.getRequestMap(request, context));
379            script.put("_context", ScriptUtil.getLazyContext(context));
380            script.put("context", context);
381        } else {
382            script.put("request", request);
383            script.put("context", context);
384        }
385
386        return script;
387    }
