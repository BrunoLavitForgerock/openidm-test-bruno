074/**
075 * Provides access to managed objects.
076 * 
077 * @author Paul C. Bryan
078 */
079@Component(name = ManagedObjectService.PID, immediate = true,
080        policy = ConfigurationPolicy.REQUIRE)
081@Service
082@Properties({
083    @Property(name = Constants.SERVICE_DESCRIPTION, value = "OpenIDM managed objects service"),
084    @Property(name = Constants.SERVICE_VENDOR, value = ServerConstants.SERVER_VENDOR_NAME),
085    @Property(name = ServerConstants.ROUTER_PREFIX, value = "/managed*") })
086public class ManagedObjectService implements RequestHandler {
087
088    public static final String PID = "org.forgerock.openidm.managed";
089
090    /**
091     * Setup logging for the {@link ManagedObjectService}.
092     */
093    private final static Logger logger = LoggerFactory.getLogger(ManagedObjectService.class);
094
095    /** Cryptographic service. */
096    @Reference(policy = ReferencePolicy.DYNAMIC)
097    protected CryptoService cryptoService;
098
099    /** Script Registry service. */
100    @Reference(policy = ReferencePolicy.DYNAMIC)
101    protected ScriptRegistry scriptRegistry;
102
103    /** Route service. */
104    @Reference(referenceInterface = RouteService.class,
105            policy = ReferencePolicy.DYNAMIC,
106            bind = "bindSyncRoute",
107            unbind = "unbindSyncRoute",
108            cardinality = ReferenceCardinality.OPTIONAL_UNARY,
109            target = "(" + ServerConstants.ROUTER_PREFIX + "=/sync*)")
110    private final AtomicReference<RouteService> syncRoute = new AtomicReference<RouteService>();
111
112    private void bindSyncRoute(final RouteService service) {
113        syncRoute.set(service);
114    }
115
116    private void unbindSyncRoute(final RouteService service) {
117        syncRoute.set(null);
118    }
119
120    /* The Connection Factory */
121    @Reference(policy = ReferencePolicy.STATIC, target="(service.pid=org.forgerock.openidm.internal)")
122    protected ConnectionFactory connectionFactory;
123
124    private final ConcurrentMap<String, Route> managedRoutes = new ConcurrentHashMap<String, Route>();
125
126    private final Router managedRouter = new Router();
127
128
129    /**
130     * TODO: Description.
131     * 
132     * @param context
133     *            TODO.
134     */
135    @Activate
136    protected void activate(ComponentContext context) throws Exception {
137        JsonValue configuration = JSONEnhancedConfig.newInstance().getConfigurationAsJson(context);
138        for (JsonValue value : configuration.get("objects").expect(List.class)) {
139            ManagedObjectSet objectSet = new ManagedObjectSet(scriptRegistry, cryptoService, syncRoute, connectionFactory, value);
140            if (managedRoutes.containsKey(objectSet.getName())) {
141                throw new ComponentException("Duplicate definition of managed object type: " + objectSet.getName());
142            }
143            managedRoutes.put(objectSet.getName(),managedRouter.addRoute(objectSet.getTemplate(), objectSet));
144        }
145    }
146
147    @Modified
148    protected void modified(ComponentContext context) throws Exception {
149        JsonValue configuration = JSONEnhancedConfig.newInstance().getConfigurationAsJson(context);
150
151        Set<String> tempRoutes = new HashSet<String>();
152        for (JsonValue value : configuration.get("objects").expect(List.class)) {
153            ManagedObjectSet objectSet = new ManagedObjectSet(scriptRegistry, cryptoService, syncRoute, connectionFactory, value);
154            if (tempRoutes.contains(objectSet.getName())) {
155                throw new ComponentException("Duplicate definition of managed object type: " + objectSet.getName());
156            }
157            Route oldRoute = managedRoutes.get(objectSet.getName());
158            if (null != oldRoute) {
159                managedRouter.removeRoute(oldRoute);
160            }
161            managedRoutes.put(objectSet.getName(),managedRouter.addRoute(objectSet.getTemplate(), objectSet));
162            tempRoutes.add(objectSet.getName());
163        }
164        for (Map.Entry<String, Route> entry : managedRoutes.entrySet()){
165           //Use ConcurrentMap to avoid ConcurrentModificationException with this iteration
166            if (tempRoutes.contains(entry.getKey())) {
167                continue;
168            }
169            managedRouter.removeRoute(managedRoutes.remove(entry.getKey()));
170        }
171    }
172
173    /**
174     * TODO: Description.
175     * 
176     * @param context
177     *            TODO.
178     */
179    @Deactivate
180    protected void deactivate(ComponentContext context) {
181        managedRouter.removeAllRoutes();
182        managedRoutes.clear();
183    }
184
185    @Override
186    public void handleAction(ServerContext context, ActionRequest request,
187            ResultHandler<JsonValue> handler) {
188        managedRouter.handleAction(context, request, handler);
189    }
190

192    public void handleCreate(ServerContext context, CreateRequest request,
193            ResultHandler<Resource> handler) {
194        managedRouter.handleCreate(context, request, handler);
195    }
196

200        managedRouter.handleDelete(context, request, handler);
201    }
202

206        managedRouter.handlePatch(context, request, handler);
207    }
208
209    @Override
210    public void handleQuery(ServerContext context, QueryRequest request, QueryResultHandler handler) {
211        managedRouter.handleQuery(context, request, handler);
212    }
213
214    @Override

218    }

221    public void handleUpdate(ServerContext context, UpdateRequest request,
222            ResultHandler<Resource> handler) {
223        managedRouter.handleUpdate(context, request, handler);
224    }
